351     private static class TestErrorObservable implements Observable.OnSubscribe<String> {
352 
353         String[] valuesToReturn;
354 
355         TestErrorObservable(String... values) {
356             valuesToReturn = values;
357         }
358 
359         @Override
360         public void call(Subscriber<? super String> observer) {
361             boolean errorThrown = false;
362             for (String s : valuesToReturn) {
363                 if (s == null) {
364                     System.out.println("throwing exception");
365                     observer.onError(new NullPointerException());
366                     errorThrown = true;
367                     // purposefully not returning here so it will continue calling onNext
368                     // so that we also test that we handle bad sequences like this
369                 } else {
370                     observer.onNext(s);
371                 }
372             }
373             if (!errorThrown) {
374                 observer.onCompleted();
375             }
376         }
377     }

437     @Test
438     public void testMergeSourceWhichDoesntPropagateExceptionBack() {
439         Observable<Integer> source = Observable.create(new OnSubscribe<Integer>() {
440             @Override
441             public void call(Subscriber<? super Integer> t1) {
442                 try {
443                     t1.onNext(0);
444                 } catch (Throwable swallow) {
445                     
446                 }
447                 t1.onNext(1);
448                 t1.onCompleted();
449             }
450         });
451         
452         Observable<Integer> result = Observable.mergeDelayError(source, Observable.just(2));
453         
454         @SuppressWarnings("unchecked")
455         final Observer<Integer> o = mock(Observer.class);
456         InOrder inOrder = inOrder(o);
457         
458         result.unsafeSubscribe(new Subscriber<Integer>() {
459             int calls;
460             @Override
461             public void onNext(Integer t) {
462                 if (calls++ == 0) {
463                     throw new TestException();
464                 }
465                 o.onNext(t);
466             }
467 
468             @Override
469             public void onError(Throwable e) {
470                 o.onError(e);
471             }
472 
473             @Override
474             public void onCompleted() {
475                 o.onCompleted();
476             }
477             
478         });
479         
480         /*
481          * If the child onNext throws, why would we keep accepting values from
482          * other sources?
483          */
484         inOrder.verify(o).onNext(2);
485         inOrder.verify(o, never()).onNext(0);
486         inOrder.verify(o, never()).onNext(1);
487         inOrder.verify(o, never()).onNext(anyInt());
488         inOrder.verify(o).onError(any(TestException.class));
489         verify(o, never()).onCompleted();
