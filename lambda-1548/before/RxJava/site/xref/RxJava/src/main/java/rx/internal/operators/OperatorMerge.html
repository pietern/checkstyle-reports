29  /**
30   * Flattens a list of {@link Observable}s into one {@code Observable}, without any transformation.
31   * <p>
32   * <img width="640" height="380" src="https://raw.githubusercontent.com/wiki/ReactiveX/RxJava/images/rx-operators/merge.png" alt="">
33   * <p>
34   * You can combine the items emitted by multiple {@code Observable}s so that they act like a single {@code Observable}, by using the merge operation.
35   * <p>
36   * The {@code instance(true)} call behaves like {@link OperatorMerge} except that if any of the merged Observables notify of
37   * an error via {@code onError}, {@code mergeDelayError} will refrain from propagating that error
38   * notification until all of the merged Observables have finished emitting items.
39   * <p>
40   * <img width="640" src="https://github.com/ReactiveX/RxJava/wiki/images/rx-operators/mergeDelayError.png" alt="">
41   * <p>
42   * Even if multiple merged Observables send {@code onError} notifications, {@code mergeDelayError} will
43   * only invoke the {@code onError} method of its Observers once.
44   * <p>
45   * This operation allows an Observer to receive all successfully emitted items from all of the
46   * source Observables without being interrupted by an error notification from one of them.
47   * <p>
48   * <em>Note:</em> If this is used on an Observable that never completes, it will never call {@code onError} and will effectively swallow errors.
49  
50   * @param <T>
51   *            the type of the items emitted by both the source and merged {@code Observable}s
52   */
53  public final class OperatorMerge<T> implements Operator<T, Observable<? extends T>> {
54      /** Lazy initialization via inner-class holder. */
55      private static final class HolderNoDelay {
56          /** A singleton instance. */
57          static final OperatorMerge<Object> INSTANCE = new OperatorMerge<Object>(false, Integer.MAX_VALUE);
58      }
59      /** Lazy initialization via inner-class holder. */
60      private static final class HolderDelayErrors {
61          /** A singleton instance. */
62          static final OperatorMerge<Object> INSTANCE = new OperatorMerge<Object>(true, Integer.MAX_VALUE);
63      }
64      /**
65       * @param delayErrors should the merge delay errors?
66       * @return a singleton instance of this stateless operator.
67       */
68      @SuppressWarnings("unchecked")
69      public static <T> OperatorMerge<T> instance(boolean delayErrors) {
70          if (delayErrors) {
71              return (OperatorMerge<T>)HolderDelayErrors.INSTANCE;
72          }
73          return (OperatorMerge<T>)HolderNoDelay.INSTANCE;
74      }
75      /**
76       * Creates a new instance of the operator with the given delayError and maxConcurrency settings.
77       * @param delayErrors
78       * @param maxConcurrent the maximum number of concurrent subscriptions or Integer.MAX_VALUE for unlimited
79       * @return
80       */
81      public static <T> OperatorMerge<T> instance(boolean delayErrors, int maxConcurrent) {
82          if (maxConcurrent == Integer.MAX_VALUE) {
83              return instance(delayErrors);
84          }
85          return new OperatorMerge<T>(delayErrors, maxConcurrent);
86      }
87  
88      final boolean delayErrors;
89      final int maxConcurrent;
90  
91      private OperatorMerge(boolean delayErrors, int maxConcurrent) {
92          this.delayErrors = delayErrors;
93          this.maxConcurrent = maxConcurrent;
94      }
95  
96      @Override
97      public Subscriber<Observable<? extends T>> call(final Subscriber<? super T> child) {
98          MergeSubscriber<T> subscriber = new MergeSubscriber<T>(child, delayErrors, maxConcurrent);
99          MergeProducer<T> producer = new MergeProducer<T>(subscriber);
100         subscriber.producer = producer;
101         
102         child.add(subscriber);
103         child.setProducer(producer);
104         
105         return subscriber;
106     }

108     static final class MergeProducer<T> extends AtomicLong implements Producer {
109         /** */
110         private static final long serialVersionUID = -1214379189873595503L;

112         final MergeSubscriber<T> subscriber;
113         
114         public MergeProducer(MergeSubscriber<T> subscriber) {
115             this.subscriber = subscriber;
116         }

118         @Override
119         public void request(long n) {
120             if (n > 0) {
121                 if (get() == Long.MAX_VALUE) {
122                     return;
123                 }
124                 BackpressureUtils.getAndAddRequest(this, n);
125                 subscriber.emit();
126             } else 
127             if (n < 0) {
128                 throw new IllegalArgumentException("n >= 0 required");
129             }
130         }
131         public long produced(int n) {
132             return addAndGet(-n);
133         }

136     /**
137      * The subscriber that observes Observables. 
138      * @param <T> the value type
139      */
140     static final class MergeSubscriber<T> extends Subscriber<Observable<? extends T>> {
141         final Subscriber<? super T> child;
142         final boolean delayErrors;
143         final int maxConcurrent;
144         
145         MergeProducer<T> producer;
146         
147         volatile RxRingBuffer queue;
148         
149         /** Tracks the active subscriptions to sources. */
150         volatile CompositeSubscription subscriptions;
151         /** Due to the emission loop, we need to store errors somewhere if !delayErrors. */
152         volatile ConcurrentLinkedQueue<Throwable> errors;
153         
154         final NotificationLite<T> nl;
155         
156         volatile boolean done;
157         
158         /** Guarded by this. */
159         boolean emitting;
160         /** Guarded by this. */
161         boolean missed;
162         
163         final Object innerGuard;
164         /** Copy-on-write array, guarded by innerGuard. */
165         volatile InnerSubscriber<?>[] innerSubscribers;
166         
167         /** Used to generate unique InnerSubscriber IDs. Modified from onNext only. */
168         long uniqueId;
169         
170         /** Which was the last InnerSubscriber that emitted? Accessed if emitting == true. */
171         long lastId;
172         /** What was its index in the innerSubscribers array? Accessed if emitting == true. */
173         int lastIndex;
174         
175         /** An empty array to avoid creating new empty arrays in removeInner. */ 
176         static final InnerSubscriber<?>[] EMPTY = new InnerSubscriber<?>[0];
177 
178         public MergeSubscriber(Subscriber<? super T> child, boolean delayErrors, int maxConcurrent) {
179             this.child = child;
180             this.delayErrors = delayErrors;
181             this.maxConcurrent = maxConcurrent;
182             this.nl = NotificationLite.instance();
183             this.innerGuard = new Object();
184             this.innerSubscribers = EMPTY;
185             long r = Math.min(maxConcurrent, RxRingBuffer.SIZE);
186             request(r);
187         }
188         
189         Queue<Throwable> getOrCreateErrorQueue() {
190             ConcurrentLinkedQueue<Throwable> q = errors;
191             if (q == null) {
192                 synchronized (this) {
193                     q = errors;
194                     if (q == null) {
195                         q = new ConcurrentLinkedQueue<Throwable>();
196                         errors = q;
197                     }
198                 }
199             }
200             return q;
201         }
202         CompositeSubscription getOrCreateComposite() {
203             CompositeSubscription c = subscriptions;
204             if (c == null) {
205                 boolean shouldAdd = false;
206                 synchronized (this) {
207                     c = subscriptions;
208                     if (c == null) {
209                         c = new CompositeSubscription();
210                         subscriptions = c;
211                         shouldAdd = true;
212                     }
213                 }
214                 if (shouldAdd) {
215                     add(c);
216                 }
217             }
218             return c;
219         }
220         
221         @Override
222         public void onNext(Observable<? extends T> t) {
223             if (t == null) {
224                 return;
225             }
226             if (t instanceof ScalarSynchronousObservable) {
227                 tryEmit(((ScalarSynchronousObservable<? extends T>)t).get());
228             } else {
229                 InnerSubscriber<T> inner = new InnerSubscriber<T>(this, uniqueId++);
230                 addInner(inner);
231                 t.unsafeSubscribe(inner);
232                 emit();
233             }
234         }
235         
236         private void reportError() {
237             List<Throwable> list = new ArrayList<Throwable>(errors);
238             if (list.size() == 1) {
239                 child.onError(list.get(0));
240             } else {
241                 child.onError(new CompositeException(list));
242             }
243         }
244         
245         @Override
246         public void onError(Throwable e) {
247             getOrCreateErrorQueue().offer(e);
248             done = true;
249             emit();
250         }
251         @Override
252         public void onCompleted() {
253             done = true;
254             emit();
255         }
256         
257         void addInner(InnerSubscriber<T> inner) {
258             getOrCreateComposite().add(inner);
259             synchronized (innerGuard) {
260                 InnerSubscriber<?>[] a = innerSubscribers;
261                 int n = a.length;
262                 InnerSubscriber<?>[] b = new InnerSubscriber<?>[n + 1];
263                 System.arraycopy(a, 0, b, 0, n);
264                 b[n] = inner;
265                 innerSubscribers = b;
266             }
267         }
268         void removeInner(InnerSubscriber<T> inner) {
269             RxRingBuffer q = inner.queue;
270             if (q != null) {
271                 q.release();
272             }
273             // subscription is non-null here because the very first addInner will create it before
274             // this can be called
275             subscriptions.remove(inner);
276             synchronized (innerGuard) {
277                 InnerSubscriber<?>[] a = innerSubscribers;
278                 int n = a.length;
279                 int j = -1;
280                 // locate the inner
281                 for (int i = 0; i < n; i++) {
282                     if (inner.equals(a[i])) {
283                         j = i;
284                         break;
285                     }
286                 }
287                 if (j < 0) {
288                     return;
289                 }
290                 if (n == 1) {
291                     innerSubscribers = EMPTY;
292                     return;
293                 }
294                 InnerSubscriber<?>[] b = new InnerSubscriber<?>[n - 1];
295                 System.arraycopy(a, 0, b, 0, j);
296                 System.arraycopy(a, j + 1, b, j, n - j - 1);
297                 innerSubscribers = b;
298             }
299         }

301         /**
302          * Tries to emit the value directly to the child if
303          * no concurrent emission is happening at the moment.
304          * <p>
305          * Since the scalar-value queue optimization applies
306          * to both the main source and the inner subscribers,
307          * we handle things in a shared manner.
308          * 
309          * @param subscriber
310          * @param value
311          */
312         void tryEmit(InnerSubscriber<T> subscriber, T value) {
313             boolean success = false;
314             long r = producer.get();
315             if (r != 0L) {
316                 synchronized (this) {
317                     // if nobody is emitting and child has available requests
318                     r = producer.get();
319                     if (!emitting && r != 0L) {
320                         emitting = true;
321                         success = true;
322                     }
323                 }
324             }
325             if (success) {
326                 emitScalar(subscriber, value, r);
327             } else {
328                 queueScalar(subscriber, value);
329             }
330         }
331 
332         protected void queueScalar(InnerSubscriber<T> subscriber, T value) {
333             /*
334              * If the attempt to make a fast-path emission failed
335              * due to lack of requests or an ongoing emission,
336              * enqueue the value and try the slow emission path.
337              */
338             RxRingBuffer q = subscriber.queue;
339             if (q == null) {
340                 q = RxRingBuffer.getSpscInstance();
341                 subscriber.add(q);
342                 subscriber.queue = q;
343             }
344             try {
345                 q.onNext(nl.next(value));
346             } catch (MissingBackpressureException ex) {
347                 subscriber.unsubscribe();
348                 subscriber.onError(ex);
349                 return;
350             } catch (IllegalStateException ex) {
351                 if (!subscriber.isUnsubscribed()) {
352                     subscriber.unsubscribe();
353                     subscriber.onError(ex);
354                 }
355                 return;
356             }
357             emit();
358         }
359 
360         protected void emitScalar(InnerSubscriber<T> subscriber, T value, long r) {
361             boolean skipFinal = false;
362             try {
363                 try {
364                     child.onNext(value);
365                 } catch (Throwable t) {
366                     if (!delayErrors) {
367                         Exceptions.throwIfFatal(t);
368                         skipFinal = true;
369                         subscriber.unsubscribe();
370                         subscriber.onError(t);
371                         return;
372                     }
373                     getOrCreateErrorQueue().offer(t);
374                 }
375                 if (r != Long.MAX_VALUE) {
376                     producer.produced(1);
377                 }
378                 subscriber.requestMore(1);
379                 // check if some state changed while emitting
380                 synchronized (this) {
381                     skipFinal = true;
382                     if (!missed) {
383                         emitting = false;
384                         return;
385                     }
386                     missed = false;
387                 }
388             } finally {
389                 if (!skipFinal) {
390                     synchronized (this) {
391                         emitting = false;
392                     }
393                 }
394             }
395             /*
396              * In the synchronized block below request(1) we check
397              * if there was a concurrent emission attempt and if there was,
398              * we stay in emission mode and enter the emission loop
399              * which will take care all the queued up state and 
400              * emission possibilities.
401              */
402             emitLoop();
403         }

410          * Tries to emit the value directly to the child if
411          * no concurrent emission is happening at the moment.
412          * <p>
413          * Since the scalar-value queue optimization applies
414          * to both the main source and the inner subscribers,
415          * we handle things in a shared manner.

420         void tryEmit(T value) {

422             long r = producer.get();
423             if (r != 0L) {
424                 synchronized (this) {
425                     // if nobody is emitting and child has available requests
426                     r = producer.get();
427                     if (!emitting && r != 0L) {
428                         emitting = true;
429                         success = true;
430                     }
431                 }
432             }
433             if (success) {
434                 emitScalar(value, r);
435             } else {
436                 queueScalar(value);
437             }
438         }

440         protected void queueScalar(T value) {
441             /*
442              * If the attempt to make a fast-path emission failed
443              * due to lack of requests or an ongoing emission,
444              * enqueue the value and try the slow emission path.
445              */
446             RxRingBuffer q = this.queue;
447             if (q == null) {
448                 q = RxRingBuffer.getSpscInstance();
449                 this.add(q);
450                 this.queue = q;
451             }
452             try {
453                 q.onNext(nl.next(value));
454             } catch (MissingBackpressureException ex) {
455                 this.unsubscribe();
456                 this.onError(ex);
457                 return;
458             } catch (IllegalStateException ex) {
459                 if (!this.isUnsubscribed()) {
460                     this.unsubscribe();
461                     this.onError(ex);
462                 }
463                 return;
464             }
465             emit();
466         }

468         protected void emitScalar(T value, long r) {
469             boolean skipFinal = false;
470             try {
471                 try {
472                     child.onNext(value);
473                 } catch (Throwable t) {
474                     if (!delayErrors) {
475                         Exceptions.throwIfFatal(t);
476                         skipFinal = true;
477                         this.unsubscribe();
478                         this.onError(t);
479                         return;
480                     }
481                     getOrCreateErrorQueue().offer(t);
482                 }
483                 if (r != Long.MAX_VALUE) {
484                     producer.produced(1);
485                 }
486                 this.requestMore(1);
487                 // check if some state changed while emitting
488                 synchronized (this) {
489                     skipFinal = true;
490                     if (!missed) {
491                         emitting = false;
492                         return;
493                     }
494                     missed = false;
495                 }
496             } finally {
497                 if (!skipFinal) {
498                     synchronized (this) {
499                         emitting = false;
500                     }
501                 }
502             }
503             /*
504              * In the synchronized block below request(1) we check
505              * if there was a concurrent emission attempt and if there was,
506              * we stay in emission mode and enter the emission loop
507              * which will take care all the queued up state and 
508              * emission possibilities.
509              */
510             emitLoop();
511         }
512         
513         void emit() {
514             synchronized (this) {
515                 if (emitting) {
516                     missed = true;
517                     return;
518                 }
519                 emitting = true;
520             }
521             emitLoop();
522         }
523         /**
524          * The standard emission loop serializing events and requests.
525          */
526         void emitLoop() {
527             boolean skipFinal = false;
528             try {
529                 final Subscriber<? super T> child = this.child;
530                 for (;;) {
531                     // eagerly check if child unsubscribed or we reached a terminal state.
532                     if (checkTerminate()) {
533                         skipFinal = true;
534                         return;
535                     }
536                     RxRingBuffer svq = queue;
537                     
538                     long r = producer.get();
539                     boolean unbounded = r == Long.MAX_VALUE;
540                     
541                     // count the number of 'completed' sources to replenish them in batches
542                     int replenishMain = 0;
543 
544                     // try emitting as many scalars as possible
545                     if (svq != null) {
546                         for (;;) {
547                             int scalarEmission = 0;
548                             Object o = null;
549                             while (r > 0) {
550                                 o = svq.poll();
551                                 // eagerly check if child unsubscribed or we reached a terminal state.
552                                 if (checkTerminate()) {
553                                     skipFinal = true;
554                                     return;
555                                 }
556                                 if (o == null) {
557                                     break;
558                                 }
559                                 T v = nl.getValue(o);
560                                 // if child throws, report bounce it back immediately
561                                 try {
562                                     child.onNext(v);
563                                 } catch (Throwable t) {
564                                     if (!delayErrors) {
565                                         Exceptions.throwIfFatal(t);
566                                         skipFinal = true;
567                                         unsubscribe();
568                                         child.onError(t);
569                                         return;
570                                     }
571                                     getOrCreateErrorQueue().offer(t);
572                                 }
573                                 replenishMain++;
574                                 scalarEmission++;
575                                 r--;
576                             }
577                             if (scalarEmission > 0) {
578                                 if (unbounded) {
579                                     r = Long.MAX_VALUE;
580                                 } else {
581                                     r = producer.produced(scalarEmission);
582                                 }
583                             }
584                             if (r == 0L || o == null) {
585                                 break;
586                             }
587                         }
588                     }
589 
590                     /*
591                      * We need to read done before innerSubscribers because innerSubcribers are added
592                      * before done is set to true. If it were the other way around, we could read an empty
593                      * innerSubscribers, get paused and then read a done flag but an async producer
594                      * might have added more subscribers between the two.
595                      */
596                     boolean d = done;
597                     // re-read svq because it could have been created
598                     // asynchronously just before done was set to true.
599                     svq = queue;
600                     // read the current set of inner subscribers
601                     InnerSubscriber<?>[] inner = innerSubscribers;
602                     int n = inner.length;
603                     
604                     // check if upstream is done, there are no scalar values 
605                     // and no active inner subscriptions
606                     if (d && (svq == null || svq.isEmpty()) && n == 0) {
607                         Queue<Throwable> e = errors;
608                         if (e == null || e.isEmpty()) {
609                             child.onCompleted();
610                         } else {
611                             reportError();
612                         }
613                         if (svq != null) {
614                             svq.release();
615                         }
616                         skipFinal = true;
617                         return;
618                     }
619                     
620                     boolean innerCompleted = false;
621                     if (n > 0) {
622                         // let's continue the round-robin emission from last location
623                         long startId = lastId;
624                         int index = lastIndex;
625                         
626                         // in case there were changes in the array or the index
627                         // no longer points to the inner with the cached id
628                         if (n <= index || inner[index].id != startId) {
629                             if (n <= index) {
630                                 index = 0;
631                             }
632                             // try locating the inner with the cached index
633                             int j = index;
634                             for (int i = 0; i < n; i++) {
635                                 if (inner[j].id == startId) {
636                                     break;
637                                 }
638                                 // wrap around in round-robin fashion
639                                 j++;
640                                 if (j == n) {
641                                     j = 0;
642                                 }
643                             }
644                             // if we found it again, j will point to it
645                             // otherwise, we continue with the replacement at j
646                             index = j;
647                             lastIndex = j;
648                             lastId = inner[j].id;
649                         }
650                         
651                         int j = index;
652                         // loop through all sources once to avoid delaying any new sources too much
653                         for (int i = 0; i < n; i++) {
654                             // eagerly check if child unsubscribed or we reached a terminal state.
655                             if (checkTerminate()) {
656                                 skipFinal = true;
657                                 return;
658                             }
659                             @SuppressWarnings("unchecked")
660                             InnerSubscriber<T> is = (InnerSubscriber<T>)inner[j];
661                             
662                             Object o = null;
663                             for (;;) {
664                                 int produced = 0;
665                                 while (r > 0) {
666                                     // eagerly check if child unsubscribed or we reached a terminal state.
667                                     if (checkTerminate()) {
668                                         skipFinal = true;
669                                         return;
670                                     }
671                                     RxRingBuffer q = is.queue;
672                                     if (q == null) {
673                                         break;
674                                     }
675                                     o = q.poll();
676                                     if (o == null) {
677                                         break;
678                                     }
679                                     T v = nl.getValue(o);
680                                     // if child throws, report bounce it back immediately
681                                     try {
682                                         child.onNext(v);
683                                     } catch (Throwable t) {
684                                         skipFinal = true;
685                                         Exceptions.throwIfFatal(t);
686                                         try {
687                                             child.onError(t);
688                                         } finally {
689                                             unsubscribe();
690                                         }
691                                         return;
692                                     }
693                                     r--;
694                                     produced++;
695                                 }
696                                 if (produced > 0) {
697                                     if (!unbounded) {
698                                         r = producer.produced(produced);
699                                     } else {
700                                         r = Long.MAX_VALUE;
701                                     }
702                                     is.requestMore(produced);
703                                 }
704                                 // if we run out of requests or queued values, break
705                                 if (r == 0 || o == null) {
706                                     break;
707                                 }
708                             }
709                             boolean innerDone = is.done;
710                             RxRingBuffer innerQueue = is.queue;
711                             if (innerDone && (innerQueue == null || innerQueue.isEmpty())) {
712                                 removeInner(is);
713                                 if (checkTerminate()) {
714                                     skipFinal = true;
715                                     return;
716                                 }
717                                 replenishMain++;
718                                 innerCompleted = true;
719                             }
720                             // if we run out of requests, don't try the other sources
721                             if (r == 0) {
722                                 break;
723                             }
724                             
725                             // wrap around in round-robin fashion
726                             j++;
727                             if (j == n) {
728                                 j = 0;
729                             }
730                         }
731                         // if we run out of requests or just completed a round, save the index and id
732                         lastIndex = j;
733                         lastId = inner[j].id;
734                     }
735                     
736                     if (replenishMain > 0) {
737                         request(replenishMain);
738                     }
739                     // if one or more inner completed, loop again to see if we can terminate the whole stream
740                     if (innerCompleted) {
741                         continue;
742                     }
743                     // in case there were updates to the state, we loop again
744                     synchronized (this) {
745                         if (!missed) {
746                             skipFinal = true;
747                             emitting = false;
748                             break;
749                         }
750                         missed = false;
751                     }
752                 }
753             } finally {
754                 if (!skipFinal) {
755                     synchronized (this) {
756                         emitting = false;
757                     }
758                 }
759             }
760         }

762         /**
763          * Check if the operator reached some terminal state: child unsubscribed,
764          * an error was reported and we don't delay errors.
765          * @return true if the child unsubscribed or there are errors available and merge doesn't delay errors.
766          */
767         boolean checkTerminate() {
768             if (child.isUnsubscribed()) {
769                 return true;
770             }
771             Queue<Throwable> e = errors;
772             if (!delayErrors && (e != null && !e.isEmpty())) {
773                 try {
774                     reportError();
775                 } finally {
776                     unsubscribe();
777                 }
778                 return true;
779             }
780             return false;
781         }
782     }
783     static final class InnerSubscriber<T> extends Subscriber<T> {
784         final MergeSubscriber<T> parent;
785         final long id;
786         volatile boolean done;
787         volatile RxRingBuffer queue;
788         int outstanding;
789         static final int limit = RxRingBuffer.SIZE / 4;
790         
791         public InnerSubscriber(MergeSubscriber<T> parent, long id) {
792             this.parent = parent;
793             this.id = id;
794         }
795         @Override
796         public void onStart() {
797             outstanding = RxRingBuffer.SIZE;
798             request(RxRingBuffer.SIZE);
799         }
800         @Override
801         public void onNext(T t) {
802             parent.tryEmit(this, t);
803         }
804         @Override
805         public void onError(Throwable e) {
806             done = true;
807             parent.getOrCreateErrorQueue().offer(e);
808             parent.emit();
809         }
810         @Override
811         public void onCompleted() {
812             done = true;
813             parent.emit();
814         }
815         public void requestMore(long n) {
816             int r = outstanding - (int)n;
817             if (r > limit) {
818                 outstanding = r;
819                 return;
820             }
821             outstanding = RxRingBuffer.SIZE;
822             int k = RxRingBuffer.SIZE - r;
823             if (k > 0) {
824                 request(k);
825             }
826         }
827     }}
