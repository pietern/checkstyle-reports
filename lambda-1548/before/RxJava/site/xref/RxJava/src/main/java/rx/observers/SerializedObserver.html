16  package rx.observers;
17  
18  import rx.Observer;
19  import rx.exceptions.*;
20  import rx.internal.operators.NotificationLite;
21  
22  /**
23   * Enforces single-threaded, serialized, ordered execution of {@link #onNext}, {@link #onCompleted}, and
24   * {@link #onError}.
25   * <p>
26   * When multiple threads are emitting and/or notifying they will be serialized by:
27   * </p><ul>
28   * <li>Allowing only one thread at a time to emit</li>
29   * <li>Adding notifications to a queue if another thread is already emitting</li>
30   * <li>Not holding any locks or blocking any threads while emitting</li>
31   * </ul>
32   * 
33   * @param <T>
34   *          the type of items expected to be observed by the {@code Observer}
35   */
36  public class SerializedObserver<T> implements Observer<T> {
37      private final Observer<? super T> actual;
38  
39      private boolean emitting;
40      /** Set to true if a terminal event was received. */
41      private volatile boolean terminated;
42      /** If not null, it indicates more work. */
43      private FastList queue;
44      private final NotificationLite<T> nl = NotificationLite.instance();
45  
46      /** Number of iterations without additional safepoint poll in the drain loop. */
47      private static final int MAX_DRAIN_ITERATION = 1024;
48  
49      static final class FastList {
50          Object[] array;
51          int size;
52  
53          public void add(Object o) {
54              int s = size;
55              Object[] a = array;
56              if (a == null) {
57                  a = new Object[16];
58                  array = a;
59              } else if (s == a.length) {
60                  Object[] array2 = new Object[s + (s >> 2)];
61                  System.arraycopy(a, 0, array2, 0, s);
62                  a = array2;
63                  array = a;
64              }
65              a[s] = o;
66              size = s + 1;
67          }
68      }
69  
70      public SerializedObserver(Observer<? super T> s) {
71          this.actual = s;
72      }
73  
74      @Override
75      public void onNext(T t) {
76          if (terminated) {
77              return;
78          }
79          synchronized (this) {
80              if (terminated) {
81                  return;
82              }
83              if (emitting) {
84                  FastList list = queue;
85                  if (list == null) {
86                      list = new FastList();
87                      queue = list;
88                  }
89                  list.add(nl.next(t));
90                  return;
91              }
92              emitting = true;
93          }
94          try {
95              actual.onNext(t);
96          } catch (Throwable e) {
97              terminated = true;
98              Exceptions.throwIfFatal(e);
99              actual.onError(OnErrorThrowable.addValueAsLastCause(e, t));
100             return;
101         }
102         for (;;) {
103             for (int i = 0; i < MAX_DRAIN_ITERATION; i++) {
104                 FastList list;
105                 synchronized (this) {
106                     list = queue;
107                     if (list == null) {
108                         emitting = false;
109                         return;
110                     }
111                     queue = null;
112                 }
113                 for (Object o : list.array) {
114                     if (o == null) {
115                         break;
116                     }
117                     try {
118                         if (nl.accept(actual, o)) {
119                             terminated = true;
120                             return;
121                         }
122                     } catch (Throwable e) {
123                         terminated = true;
124                         Exceptions.throwIfFatal(e);
125                         actual.onError(OnErrorThrowable.addValueAsLastCause(e, t));
126                         return;
127                     }
128                 }
129             }
130         }
131     }
132     
133     @Override
134     public void onError(final Throwable e) {
135         Exceptions.throwIfFatal(e);
136         if (terminated) {
137             return;
138         }
139         synchronized (this) {
140             if (terminated) {
141                 return;
142             }
143             terminated = true;
144             if (emitting) {
145                 /* 
146                  * FIXME: generally, errors jump the queue but this wasn't true 
147                  * for SerializedObserver and may break existing expectations. 
148                  */
149                 FastList list = queue;
150                 if (list == null) {
151                     list = new FastList();
152                     queue = list;
153                 }
154                 list.add(nl.error(e));
155                 return;
156             }
157             emitting = true;
158         }
159         actual.onError(e);
160     }
161 
162     @Override
163     public void onCompleted() {
164         if (terminated) {
165             return;
166         }
167         synchronized (this) {
168             if (terminated) {
169                 return;
170             }
171             terminated = true;
172             if (emitting) {
173                 FastList list = queue;
174                 if (list == null) {
175                     list = new FastList();
176                     queue = list;
177                 }
178                 list.add(nl.completed());
179                 return;
180             }
181             emitting = true;
182         }
183         actual.onCompleted();
184     }
185 }
