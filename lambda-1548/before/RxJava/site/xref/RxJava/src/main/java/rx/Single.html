47  /**
48   * The Single class implements the Reactive Pattern for a single value response. See {@link Observable} for the
49   * implementation of the Reactive Pattern for a stream or vector of values.
50   * <p>
51   * {@code Single} behaves the same as {@link Observable} except that it can only emit either a single successful
52   * value, or an error (there is no "onComplete" notification as there is for {@link Observable})
53   * <p>
54   * Like an {@link Observable}, a {@code Single} is lazy, can be either "hot" or "cold", synchronous or
55   * asynchronous.
56   * <p>
57   * The documentation for this class makes use of marble diagrams. The following legend explains these diagrams:
58   * <p>
59   * <img width="605" height="285" src="https://raw.github.com/wiki/ReactiveX/RxJava/images/rx-operators/Single.legend.png" alt="">
60   * <p>
61   * For more information see the <a href="http://reactivex.io/documentation/observable.html">ReactiveX
62   * documentation</a>.
63   * 
64   * @param <T>
65   *            the type of the item emitted by the Single
66   * @since (If this class graduates from "Experimental" replace this parenthetical with the release number)
67   */

73      /**
74       * Creates a Single with a Function to execute when it is subscribed to (executed).
75       * <p>
76       * <em>Note:</em> Use {@link #create(OnSubscribe)} to create a Single, instead of this constructor,
77       * unless you specifically have a need for inheritance.
78       * 
79       * @param f
80       *            {@code OnExecute} to be executed when {@code execute(SingleSubscriber)} or
81       *            {@code subscribe(Subscriber)} is called
82       */
83      protected Single(final OnSubscribe<T> f) {
84          // bridge between OnSubscribe (which all Operators and Observables use) and OnExecute (for Single)
85          this.onSubscribe = new Observable.OnSubscribe<T>() {
86  
87              @Override
88              public void call(final Subscriber<? super T> child) {
89                  final SingleDelayedProducer<T> producer = new SingleDelayedProducer<T>(child);
90                  child.setProducer(producer);
91                  SingleSubscriber<T> ss = new SingleSubscriber<T>() {
92  
93                      @Override
94                      public void onSuccess(T value) {
95                          producer.setValue(value);
96                      }
97  
98                      @Override
99                      public void onError(Throwable error) {
100                         child.onError(error);
101                     }
102 
103                 };
104                 child.add(ss);
105                 f.call(ss);
106             }
107 
108         };
109     }

111     private Single(final Observable.OnSubscribe<T> f) {
112         this.onSubscribe = f;
113     }

154     /**
155      * Lifts a function to the current Single and returns a new Single that when subscribed to will pass the
156      * values of the current Single through the Operator function.
157      * <p>
158      * In other words, this allows chaining TaskExecutors together on a Single for acting on the values within
159      * the Single.
160      * <p>
161      * {@code task.map(...).filter(...).lift(new OperatorA()).lift(new OperatorB(...)).subscribe() }
162      * <p>
163      * If the operator you are creating is designed to act on the item emitted by a source Single, use
164      * {@code lift}. If your operator is designed to transform the source Single as a whole (for instance, by
165      * applying a particular set of existing RxJava operators to it) use {@link #compose}.
166      * <dl>
167      * <dt><b>Scheduler:</b></dt>
168      * <dd>{@code lift} does not operate by default on a particular {@link Scheduler}.</dd>
169      * </dl>
170      * 
171      * @param lift
172      *            the Operator that implements the Single-operating function to be applied to the source Single
173      * @return a Single that is the result of applying the lifted Operator to the source Single
174      * @see <a href="https://github.com/ReactiveX/RxJava/wiki/Implementing-Your-Own-Operators">RxJava wiki: Implementing Your Own Operators</a>
175      */
176     private final <R> Single<R> lift(final Operator<? extends R, ? super T> lift) {
177         // This method is private because not sure if we want to expose the Observable.Operator in this public API rather than a Single.Operator 
178 
179         return new Single<R>(new Observable.OnSubscribe<R>() {
180             @Override
181             public void call(Subscriber<? super R> o) {
182                 try {
183                     final Subscriber<? super T> st = hook.onLift(lift).call(o);
184                     try {
185                         // new Subscriber created and being subscribed with so 'onStart' it
186                         st.onStart();
187                         onSubscribe.call(st);
188                     } catch (Throwable e) {
189                         // localized capture of errors rather than it skipping all operators 
190                         // and ending up in the try/catch of the subscribe method which then
191                         // prevents onErrorResumeNext and other similar approaches to error handling
192                         if (e instanceof OnErrorNotImplementedException) {
193                             throw (OnErrorNotImplementedException) e;
194                         }
195                         st.onError(e);
196                     }
197                 } catch (Throwable e) {
198                     if (e instanceof OnErrorNotImplementedException) {
199                         throw (OnErrorNotImplementedException) e;
200                     }
201                     // if the lift function failed all we can do is pass the error to the final Subscriber
202                     // as we don't have the operator available to us
203                     o.onError(e);
204                 }
205             }
206         });
207     }
