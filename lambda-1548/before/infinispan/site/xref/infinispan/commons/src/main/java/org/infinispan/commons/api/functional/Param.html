3   import org.infinispan.commons.util.Experimental;

5   import java.util.concurrent.CompletableFuture;

7   /**
8    * An easily extensible parameter that allows functional map operations to be
9    * tweaked. Apart from {@link org.infinispan.commons.api.functional.Param.FutureMode}, examples
10   * would include local-only parameter, skip-cache-store parameter and others.
11   *
12   * <p>What makes {@link Param} different from {@link MetaParam} is that {@link Param}
13   * values are never stored in the functional map. They merely act as ways to
14   * tweak how operations are executed.
15   *
16   * <p>Since {@link Param} instances control how the internals work, only
17   * {@link Param} implementations by Infinispan will be supported.
18   *
19   * <p>This interface is equivalent to Infinispan's Flag, but it's more
20   * powerful because it allows to pass a flag along with a value. Infinispan's
21   * Flag are enum based which means no values can be passed along with value.
22   *
23   * <p>Since each param is an independent entity, it's easy to create
24   * public versus private parameter distinction. When parameters are stored in
25   * enums, it's more difficult to make such distinction.
26   *
27   * @param <P> type of parameter
28   * @since 8.0
29   */
30  @Experimental
31  public interface Param<P> {
32  
33     /**
34      * A parameter's identifier. Each parameter must have a different id.
35      *
36      * <p>A numeric id makes it flexible enough to be stored in collections that
37      * take up low resources, such as arrays.
38      */
39     int id();
40  
41     /**
42      * Parameter's value.
43      */
44     P get();
45  
46     /**
47      * When a method defines {@link CompletableFuture} as a return type, it
48      * implies the method called will be called asynchronously and that the
49      * {@link CompletableFuture} returned will be completed once the method's
50      * work is complete.
51      *
52      * <p>So, calling a method that returns {@link CompletableFuture} normally
53      * implies that the method will allocate a thread to do that job. However,
54      * there are situations when the user calls a method that returns
55      * {@link CompletableFuture} and immediately calls {@link CompletableFuture#get()}
56      * or similar methods to get the result. Calling such methods result in the
57      * caller thread blocking in which case, having such method spawn another
58      * thread is a waste of resources. So, for such situations, the caller can
59      * pass in the {@link #COMPLETED} param so that the internal logic avoids
60      * creating a separate thread, since the caller thread will block to get
61      * the result immediately.
62      *
63      * <p>By default, all methods returning {@link CompletableFuture} are
64      * asynchronous, hence using the {@link #ASYNC} future mode.
65      *
66      * @since 8.0
67      */
68     @Experimental
69     enum FutureMode implements Param<FutureMode> {
70        ASYNC, COMPLETED;
71  
72        public static final int ID = ParamIds.FUTURE_MODE_ID;
73  
74        @Override
75        public int id() {
76           return ID;
77        }
78  
79        @Override
80        public FutureMode get() {
81           return this;
82        }
83  
84        /**
85         * Provides default future mode.
86         */
87        public static FutureMode defaultValue() {
88           return ASYNC;
89        }
90     }
91  
92     /**
93      * When a persistence store is attached to a cache, by default all write
94      * operations, regardless of whether they are inserts, updates or removes,
95      * are persisted to the store. Using {@link #SKIP}, the write operations
96      * can skip the persistence store modification, applying the effects of
97      * the write operation only in the in-memory contents of the caches in
98      * the cluster.
99      *
100     * @apiNote Amongst the old flags, there's one that allows cache store
101     * to be skipped for loading or reading. There's no need for such
102     * per-invocation parameter here, because to avoid loading or reading from
103     * the store, {@link org.infinispan.commons.api.functional.FunctionalMap.WriteOnlyMap}
104     * operations can be called which do not read previous values from the
105     * persistence store.
106     *
107     * @since 8.0
108     */
109    @Experimental
110    enum PersistenceMode implements Param<PersistenceMode> {
111       PERSIST, SKIP;
112 
113       public static final int ID = ParamIds.PERSISTENCE_MODE_ID;
114 
115       @Override
116       public int id() {
117          return ID;
118       }
119 
120       @Override
121       public PersistenceMode get() {
122          return this;
123       }
124 
125       /**
126        * Provides default persistence mode.
127        */
128       public static PersistenceMode defaultValue() {
129          return PERSIST;
130       }
131    }
