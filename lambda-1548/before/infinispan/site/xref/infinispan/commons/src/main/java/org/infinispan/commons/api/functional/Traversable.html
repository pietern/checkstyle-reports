1   package org.infinispan.commons.api.functional;
2   
3   import org.infinispan.commons.util.Experimental;
4   
5   import java.util.Comparator;
6   import java.util.Optional;
7   import java.util.function.BiConsumer;
8   import java.util.function.BiFunction;
9   import java.util.function.BinaryOperator;
10  import java.util.function.Consumer;
11  import java.util.function.Function;
12  import java.util.function.Predicate;
13  import java.util.function.Supplier;
14  import java.util.stream.Collector;
15  
16  /**
17   * Unsorted traversable stream for sequential and aggregating operations.
18   *
19   * <p>Traversable contains two type of operations:
20   * <ol>
21   *    <li>Intermediate operations which transform a traversable, into another,
22   *    e.g. {@link #filter(Predicate)}.
23   *    </li>
24   *    <li>Terminal operations which produce a side effect, e.g. {@link #forEach(Consumer)}.
25   *    Once a terminal operation is completed, the resources taken by the
26   *    traversable are released.
27   *    </li>
28   * </ol>
29   *
30   * <p>Traversable cannot be reused and hence is designed to be used only once
31   * via its intermediate and terminal operations.
32   *
33   * <p>In distributed environments, unless individually specified, all lambdas
34   * passed to methods are executed where data is located. For example, if
35   * executing {@link #forEach(Consumer)}, the {@link Consumer} function is
36   * executed wherever a particular key resides. To execute a for-each operation
37   * where the side effects are executed locally, all the {@link Traversable}'s
38   * data needs to be collected and iterated over manually.
39   *
40   * @param <T>
41   * @since 8.0
42   */
43  @Experimental
44  public interface Traversable<T> {
45  
46     /**
47      * An intermediate operation that returns a traversable containing elements
48      * matching the given predicate.
49      */
50     Traversable<T> filter(Predicate<? super T> p);
51  
52     /**
53      * An intermediate operation that returns a traversable containing the
54      * results of applying the given function over the elements of the
55      * traversable.
56      */
57     <R> Traversable<R> map(Function<? super T, ? extends R> f);
58  
59     /**
60      * An intermediate operation that returns a traversable containing the
61      * results of replacing each element of this traversable with the contents
62      * of a traversable produced by applying the provided function to each element.
63      *
64      * <p>From a functional map perspective, this operation is particularly handy
65      * when the values are collections.
66      */
67     <R> Traversable<R> flatMap(Function<? super T, ? extends Traversable<? extends R>> f);
68  
69     /**
70      * A terminal operation that applies an operation to all elements of this
71      * traversable.
72      */
73     void forEach(Consumer<? super T> c);
74  
75     /**
76      * A terminal operation that applies a binary folding operation to a start
77      * value and all elements of this traversable.
78      */
79     T reduce(T z, BinaryOperator<T> folder);
80  
81     /**
82      * A terminal operation that applies a binary folding operation to all
83      * elements of this traversable, and wraps the result in an optional.
84      * If the traversable is empty, it returns an empty optional.
85      */
86     Optional<T> reduce(BinaryOperator<T> folder);
87  
88     /**
89      * A terminal operation that applies a binary folding operation to a start
90      * value and the result of each element having a mapping function applied.
91      *
92      * <p>This is a combined map/reduce which could potentially be done more
93      * efficiently than if a map is executed and then reduce.
94      */
95     <U> U reduce(U z, BiFunction<U, ? super T, U> mapper, BinaryOperator<U> folder);
96  
97     /**
98      * A terminal operation that transforms the traversable into a result
99      * container, first constructed with the given supplier, and then
100     * accumulating elements over it with the given consumer.
101     *
102     * <p>The combiner can be used to combine accumulated results executed in
103     * parallel or coming from different nodes in a distributed environment.
104     *
105     * <p>In distributed environments where some keys are remote, the
106     * {@link Supplier} and {@link BiConsumer} instances passed in are sent to
107     * other nodes and hence they need to be marshallable. If the collect
108     * operation can be defined using the helper methods in
109     * {@link java.util.stream.Collectors}, it is recommended that those are
110     * used, which can easily be made marshalled using the
111     * {@code org.infinispan.stream.CacheCollectors#serializableCollector} method.
112     */
113    <R> R collect(Supplier<R> s, BiConsumer<R, ? super T> accumulator, BiConsumer<R, R> combiner);
114 
115    /**
116     * A terminal operation that transforms the traversable into a result
117     * container using a {@code Collector}.
118     *
119     * <p>In distributed environments where some keys are remote, the
120     * {@link Collector} instance passed in is sent other nodes and hence it
121     * needs to be marshallable. This can easily be made achieved using the
122     * {@code org.infinispan.stream.CacheCollectors#serializableCollector} method.
123     */
124    <R, A> R collect(Collector<? super T, A, R> collector);
125 
126    /**
127     * A terminal operation that returns an optional containing the minimum
128     * element of this traversable based on the comparator passed in.
129     * If the traversable is empty, it returns an empty optional.
130     */
131    default Optional<T> min(Comparator<? super T> comparator) {
132       return reduce(BinaryOperator.minBy(comparator));
133    }
134 
135    /**
136     * A terminal operation that returns an optional containing the maximum
137     * element of this traversable based on the comparator passed in.
138     * If the traversable is empty, it returns an empty optional.
139     */
140    default Optional<T> max(Comparator<? super T> comparator) {
141       return reduce(BinaryOperator.maxBy(comparator));
142    }
143 
144    /**
145     * A terminal operation that returns the number of elements in the traversable.
146     */
147    long count();
148 
149    /**
150     * A terminal operation that returns whether any elements of this
151     * traversable match the provided predicate.
152     *
153     * <p>An important reason to keep this method is the fact as opposed
154     * to a reduction which must evaluate all elements in the traversable, this
155     * method could stop as soon as it has found an element that matches.
156     */
157    boolean anyMatch(Predicate<? super T> p);
158 
159    /**
160     * A terminal operation that returns whether all elements of this
161     * traversable match the provided predicate.
162     *
163     * <p>An important reason to keep this method is the fact as opposed
164     * to a reduction which must evaluate all elements in the traversable, this
165     * method could stop as soon as it has found an element that does not match
166     * the predicate.
167     */
168    boolean allMatch(Predicate<? super T> p);
169 
170    /**
171     * A terminal operation that returns whether no elements of this
172     * traversable match the provided predicate.
173     *
174     * <p>An important reason to keep this method is the fact as opposed
175     * to a reduction which must evaluate all elements in the traversable, this
176     * method could stop as soon as it has found an element that does matches
177     * the predicate.
178     */
179    boolean noneMatch(Predicate<? super T> predicate);
180 
181    /**
182     * A terminal operation that returns an optional containing an element of
183     * the traversable, or an empty optional if empty.
184     */
185    Optional<T> findAny();
186 
187 }
