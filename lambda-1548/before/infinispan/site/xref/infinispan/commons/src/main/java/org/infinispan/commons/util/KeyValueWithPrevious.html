3   import java.io.IOException;
4   import java.io.ObjectInput;
5   import java.io.ObjectOutput;

7   import org.infinispan.commons.marshall.Externalizer;
8   import org.infinispan.commons.marshall.SerializeWith;
9   
10  @SerializeWith(KeyValueWithPrevious.KeyValueWithPreviousExternalizer.class)
11  public class KeyValueWithPrevious<K, V> {
12     /**
13      * The serialVersionUID
14      */
15     private static final long serialVersionUID = -7875910676423622104L;
16  
17     private final K key;
18     private final V value;
19     private final V prev;
20  
21     public KeyValueWithPrevious(K key, V value, V prev) {
22        this.key = key;
23        this.value = value;
24        this.prev = prev;
25     }
26  
27     public K getKey() {
28        return key;
29     }
30  
31     public V getValue() {
32        return value;
33     }
34  
35     public V getPrev() {
36        return prev;
37     }
38  
39     @Override
40     public boolean equals(Object o) {
41        if (this == o) return true;
42        if (o == null || getClass() != o.getClass()) return false;
43  
44        KeyValueWithPrevious keyValueWithPrevious = (KeyValueWithPrevious) o;
45  
46        if (key != null ? !key.equals(keyValueWithPrevious.key) : keyValueWithPrevious.key != null) return false;
47        if (prev != null ? !prev.equals(keyValueWithPrevious.prev) : keyValueWithPrevious.prev != null) return false;
48        if (value != null ? !value.equals(keyValueWithPrevious.value) : keyValueWithPrevious.value != null) return false;
49  
50        return true;
51     }
52  
53     @Override
54     public int hashCode() {
55        int result = key != null ? key.hashCode() : 0;
56        result = 31 * result + (value != null ? value.hashCode() : 0);
57        result = 31 * result + (prev != null ? prev.hashCode() : 0);
58        return result;
59     }
60  
61     @Override
62     public String toString() {
63        return "KeyValueWithPrevious{" +
64              "key=" + key +
65              ", value=" + value +
66              ", prev=" + prev +
67              '}';
68     }
69  
70     @SuppressWarnings("unchecked")
71     public static class KeyValueWithPreviousExternalizer implements Externalizer<KeyValueWithPrevious> {
72  
73        @Override
74        public void writeObject(ObjectOutput output, KeyValueWithPrevious kvPair) throws IOException {
75           output.writeObject(kvPair.getKey());
76           output.writeObject(kvPair.getValue());
77           output.writeObject(kvPair.getPrev());
78        }
79  
80        @Override
81        public KeyValueWithPrevious readObject(ObjectInput input) throws IOException, ClassNotFoundException {
82           return new KeyValueWithPrevious(input.readObject(), input.readObject(), input.readObject());
83        }
84  
85     }
