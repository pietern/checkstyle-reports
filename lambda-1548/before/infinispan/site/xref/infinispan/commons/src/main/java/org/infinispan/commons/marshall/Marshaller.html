1   package org.infinispan.commons.marshall;
2   
3   import net.jcip.annotations.ThreadSafe;
4   import org.infinispan.commons.io.ByteBuffer;
5   
6   import java.io.IOException;
7   
8   /**
9    * A marshaller is a class that is able to marshall and unmarshall objects efficiently.
10   * <p/>
11   * This interface is used to marshall {@link org.infinispan.commands.ReplicableCommand}s, their parameters and their
12   * response values, as well as any other arbitraty Object <--> byte[] conversions, such as those used in client/server
13   * communications.
14   * <p/>
15   * A single instance of any implementation is shared by multiple threads, so implementations <i>need</i> to be threadsafe,
16   * and preferably immutable.
17   *
18   * @author Manik Surtani
19   * @version 4.1
20   */
21  @ThreadSafe
22  public interface Marshaller {
23  
24     /**
25      * Marshalls an object to a byte array.  The estimatedSize parameter is a hint that can be passed in to allow for
26      * efficient sizing of the byte array before attempting to marshall the object.  The more accurate this estimate is,
27      * the less likely byte[]s will need to be resized to hold the byte stream generated by marshalling the object.
28      *
29      * @param obj           object to convert to a byte array.  Must not be null.
30      * @param estimatedSize an estimate of how large the resulting byte array may be
31      * @return a byte array with the marshalled form of the object
32      * @throws IOException if marshalling cannot complete due to some I/O error
33      * @throws InterruptedException if the marshalling was interrupted. Clients should take this as a sign that
34      * the marshaller is no longer available, maybe due to shutdown, and so no more unmarshalling should be attempted.
35      */
36     byte[] objectToByteBuffer(Object obj, int estimatedSize) throws IOException, InterruptedException;
37  
38     /**
39      * Marshalls an object to a byte array.
40      *
41      * @param obj object to convert to a byte array.  Must not be null.
42      * @return a byte array
43      * @throws IOException if marshalling cannot complete due to some I/O error
44      * @throws InterruptedException if the marshalling process was interrupted. Clients should take this as a sign that
45      * the marshaller is no longer available, maybe due to shutdown, and so no more marshalling should be attempted.
46      */
47     byte[] objectToByteBuffer(Object obj) throws IOException, InterruptedException;
48  
49     /**
50      * Unmarshalls an object from a byte array.
51      *
52      * @param buf byte array containing the binary representation of an object.  Must not be null.
53      * @return an object
54      * @throws IOException if unmarshalling cannot complete due to some I/O error
55      * @throws ClassNotFoundException if the class of the object trying to unmarshall is unknown
56      */
57     Object objectFromByteBuffer(byte[] buf) throws IOException, ClassNotFoundException;
58  
59     /**
60      * Unmarshalls an object from a specific portion of a byte array.
61      *
62      * @param buf    byte array containing the binary representation of an object.  Must not be null.
63      * @param offset point in buffer to start reading
64      * @param length number of bytes to consider
65      * @return an object
66      * @throws IOException if unmarshalling cannot complete due to some I/O error
67      * @throws ClassNotFoundException if the class of the object trying to unmarshall is unknown
68      */
69     Object objectFromByteBuffer(byte[] buf, int offset, int length) throws IOException, ClassNotFoundException;
70  
71     /**
72      * A method that returns an instance of {@link org.infinispan.commons.io.ByteBuffer}, which allows direct access to the byte
73      * array with minimal array copying
74      *
75      * @param o object to marshall
76      * @throws IOException if marshalling cannot complete due to some I/O error
77      * @throws InterruptedException if the marshalling process was interrupted. Clients should take this as a sign that
78      * the marshaller is no longer available, maybe due to shutdown, and so no more marshalling should be attempted.
79      */
80     ByteBuffer objectToBuffer(Object o) throws IOException, InterruptedException;
81  
82     /**
83      * A method that checks whether the given object is marshallable as per the rules of this marshaller.
84      *
85      * @param o object to verify whether it's marshallable or not
86      * @return true if the object is marshallable, otherwise false
87      * @throws Exception if while checking whether the object was serializable or not, an exception arose
88      */
89     boolean isMarshallable(Object o) throws Exception;
90  
91     /**
92      * Returns a marshalled payload size predictor for a particular type.
93      * Accurate prediction of a type's serialized payload size helps avoid
94      * unnecessary copying and speeds up application performance.
95      *
96      * @param o Object for which serialized predictor will be returned
97      * @return an instance of {@link BufferSizePredictor}
98      * @throws NullPointerException if o is null
99      */
100    BufferSizePredictor getBufferSizePredictor(Object o);
101 
102 }
