1   package org.infinispan.commons.util;
2   
3   import java.util.Iterator;
4   import java.util.function.Function;
5   
6   /**
7    * A iterator that maps each value to the output of the Function.  Note that the remove is supported if the iterator
8    * originally supported remove.
9    * @author William Burns
10   * @since 8.0
11   */
12  public class IteratorMapper<E, S> implements Iterator<S> {
13     private final Iterator<? extends E> iterator;
14     private final Function<? super E, ? extends S> function;
15  
16     public IteratorMapper(Iterator<? extends E> iterator, Function<? super E, ? extends S> function) {
17        if (iterator == null || function == null) {
18           throw new NullPointerException();
19        }
20        this.iterator = iterator;
21        this.function = function;
22     }
23  
24     @Override
25     public boolean hasNext() {
26        return iterator.hasNext();
27     }
28  
29     @Override
30     public S next() {
31        E value = iterator.next();
32        return function.apply(value);
33     }
34  
35     @Override
36     public void remove() {
37        iterator.remove();
38     }
39  }
