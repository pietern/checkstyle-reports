1   package org.infinispan.jcache.remote;
2   
3   import org.infinispan.client.hotrod.annotation.ClientCacheEntryCreated;
4   import org.infinispan.client.hotrod.annotation.ClientCacheEntryModified;
5   import org.infinispan.client.hotrod.annotation.ClientCacheEntryRemoved;
6   import org.infinispan.client.hotrod.annotation.ClientListener;
7   import org.infinispan.client.hotrod.event.ClientCacheEntryCustomEvent;
8   import org.infinispan.commons.util.KeyValueWithPrevious;
9   import org.infinispan.jcache.AbstractJCache;
10  import org.infinispan.jcache.AbstractJCacheListenerAdapter;
11  import org.infinispan.jcache.AbstractJCacheNotifier;
12  
13  @ClientListener(converterFactoryName = "key-value-with-previous-converter-factory")
14  public class JCacheListenerAdapter<K, V> extends AbstractJCacheListenerAdapter<K, V> {
15     public JCacheListenerAdapter(AbstractJCache<K, V> jcache, AbstractJCacheNotifier<K, V> notifier) {
16        super(jcache, notifier);
17     }
18  
19     @ClientCacheEntryCreated
20     @ClientCacheEntryModified
21     @ClientCacheEntryRemoved
22     public void handleCacheEntryEvent(ClientCacheEntryCustomEvent<KeyValueWithPrevious<K, V>> e) {
23        KeyValueWithPrevious<K, V> event = e.getEventData();
24        switch (e.getType()) {
25        case CLIENT_CACHE_ENTRY_CREATED: {
26           notifier.notifyEntryCreated(jcache, event.getKey(), event.getValue());
27           break;
28        }
29        case CLIENT_CACHE_ENTRY_REMOVED: {
30           notifier.notifyEntryRemoved(jcache, event.getKey(), event.getPrev());
31           break;
32        }
33        case CLIENT_CACHE_ENTRY_MODIFIED: {
34           notifier.notifyEntryUpdated(jcache, event.getKey(), event.getValue());
35           break;
36        }
37        case CLIENT_CACHE_FAILOVER:
38           // No JSR107 correspondent.
39           break;
40        default:
41           break;
42        }
43     }
44  }
