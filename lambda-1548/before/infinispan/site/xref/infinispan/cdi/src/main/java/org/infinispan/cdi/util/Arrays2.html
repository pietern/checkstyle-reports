17      /**
18       * Create a set from an array. If the array contains duplicate objects, the
19       * last object in the array will be placed in resultant set.
20       *
21       * @param <T>   the type of the objects in the set
22       * @param array the array from which to create the set
23       * @return the created sets
24       */
25      public static <T> Set<T> asSet(T... array) {
26          Set<T> result = new HashSet<T>();
27          for (T a : array) {
28              result.add(a);
29          }
30          return result;
31      }
