1   package org.infinispan.cdi.util;

3   import javax.enterprise.context.spi.CreationalContext;
4   import javax.enterprise.inject.spi.Bean;
5   
6   /**
7    * Callbacks used by {@link BeanBuilder} and {@link ImmutableBean} to allow control
8    * of the creation and destruction of a custom bean.
9    *
10   * @param <T> the class of the bean instance
11   * @author Stuart Douglas
12   */
13  public interface ContextualLifecycle<T> {
14      /**
15       * Callback invoked by a created bean when
16       * {@link Bean#create(CreationalContext)} is called.
17       *
18       * @param bean              the bean initiating the callback
19       * @param creationalContext the context in which this instance was created
20       */
21      public T create(Bean<T> bean, CreationalContext<T> creationalContext);
22  
23      /**
24       * Callback invoked by a created bean when
25       * {@link Bean#destroy(Object, CreationalContext)} is called.
26       *
27       * @param bean              the bean initiating the callback
28       * @param instance          the contextual instance to destroy
29       * @param creationalContext the context in which this instance was created
30       */
31      public void destroy(Bean<T> bean, T instance, CreationalContext<T> creationalContext);
32  
33  }
