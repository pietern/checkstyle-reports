3   import java.lang.reflect.AccessibleObject;
4   import java.security.PrivilegedAction;
5   
6   /**
7    * A {@link PrivilegedAction} that calls
8    * {@link AccessibleObject#setAccessible(boolean)}
9    *
10   * @author Stuart Douglas
11   */
12  public class SetAccessiblePriviligedAction implements PrivilegedAction<Void> {
13  
14      private final AccessibleObject member;
15  
16      public SetAccessiblePriviligedAction(AccessibleObject member) {
17          this.member = member;
18      }
19  
20      public Void run() {
21          member.setAccessible(true);
22          return null;
23      }
24  
25  }
