11  /**
12   * The JBoss Logging interface which defined the logging methods for the CDI integration. The id range for the CDI
13   * integration is 17001-18000
14   *
15   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
16   */
17  @MessageLogger(projectCode = "ISPN")
18  public interface Log extends BasicLogger {
19  
20     @LogMessage(level = INFO)
21     @Message(value = "Infinispan CDI extension version: %s", id = 17001)
22     void version(String version);
23  
24     @LogMessage(level = INFO)
25     @Message(value = "Configuration for cache '%s' has been defined in cache manager '%s'", id = 17002)
26     void cacheConfigurationDefined(String cacheName, EmbeddedCacheManager cacheManager);
27  
28     @Message(value = "%s parameter must not be null", id = 17003)
29     IllegalArgumentException parameterMustNotBeNull(String parameterName);
30  
31  }
