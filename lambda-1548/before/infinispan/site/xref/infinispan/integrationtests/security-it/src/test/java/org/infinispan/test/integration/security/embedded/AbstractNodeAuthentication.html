83     protected Cache<String, String> getReplicatedCache(EmbeddedCacheManager manager) throws Exception {
84        ConfigurationBuilder cacheConfig = new ConfigurationBuilder();
85        cacheConfig.transaction().lockingMode(LockingMode.PESSIMISTIC);
86        cacheConfig.invocationBatching().enable();
87        cacheConfig.jmxStatistics().disable();
88        cacheConfig.clustering().cacheMode(CacheMode.REPL_SYNC);
89  
90        manager.defineConfiguration(CACHE_NAME, cacheConfig.build());
91        Cache<String, String> replicatedCache = manager.getCache(CACHE_NAME);
92  
93        return replicatedCache;
94     }
95  
96     protected EmbeddedCacheManager getCacheManager(String jgrousConfigFile) {
97        GlobalConfigurationBuilder globalConfig = new GlobalConfigurationBuilder();
98        globalConfig.globalJmxStatistics().disable();
99        globalConfig.globalJmxStatistics().mBeanServerLookup(null); //TODO remove once WFLY-3124 is fixed, for now fail JMX registration
100       globalConfig.transport().defaultTransport().addProperty("configurationFile", jgrousConfigFile);
101       EmbeddedCacheManager manager = new DefaultCacheManager(globalConfig.build());
102       return manager;
103    }

147    @Test
148    @InSequence(4)
149    public void stopJoiningNodes() throws Exception {
150       deployer.undeploy(getJoiningNodeName());
151       deployer.undeploy(COORDINATOR_NODE);
152 
153       if (krbProvided) {
154          for (ManagementClientParams params : getManagementClientListParams()) {
155             ModelControllerClient client = getModelControllerClient(params);
156             ManagementClient managementClient = new ManagementClient(client, params.getHostname(),
157                                                                      params.getPort(), "http-remoting");
158 
159             KerberosSystemPropertiesSetupTask.INSTANCE.tearDown(managementClient, null);
160             SecurityTraceLoggingServerSetupTask.INSTANCE.tearDown(managementClient, null);
161             SecurityDomainsSetupTask.INSTANCE.tearDown(managementClient, null);
162          }
163       }
164 
165       try {
166          controller.stop(getJoiningNodeName());
167       } catch (Exception e) {
168          LOG.warn("Joining node stop failed with %s", e.getCause());
169          controller.kill(getJoiningNodeName());
170       }
171       try {
172          controller.stop(COORDINATOR_NODE);
173       } catch (Exception e) {
174          LOG.warn("Coordinator node stop failed with %s", e.getCause());
175          controller.kill(COORDINATOR_NODE);
176       }
177       assertFalse(controller.isStarted(getJoiningNodeName()));
178       assertFalse(controller.isStarted(COORDINATOR_NODE));
179    }

255       public static final String SECURITY_DOMAIN_PREFIX = "krb-";
256       private static final String KEYTABS_DIR = "${java.io.tmpdir}" + File.separator + "keytabs" + File.separator;
257 
258       public static final SecurityDomainsSetupTask INSTANCE = new SecurityDomainsSetupTask();
259 
260       /**
261        * Returns SecurityDomains configuration for this testcase.
262        */
263       @Override
264       protected SecurityDomain[] getSecurityDomains() {
265 
266          final SecurityDomain krbNode0 = getKrbSecurityDomain("node0", KEYTABS_DIR + "jgroups_node0_clustered.keytab", "jgroups/node0/clustered@INFINISPAN.ORG");
267          final SecurityDomain krbNode1 = getKrbSecurityDomain("node1", KEYTABS_DIR + "jgroups_node1_clustered.keytab", "jgroups/node1/clustered@INFINISPAN.ORG");
268          final SecurityDomain krbFail = getKrbSecurityDomain("node1-fail", KEYTABS_DIR + "jgroups_node0_fail_clustered.keytab", "jgroups/node1/clustered2@INFINISPAN.ORG");
269 
270          return new SecurityDomain[]{krbNode0, krbNode1, krbFail};
271       }
272 
273       private SecurityDomain getKrbSecurityDomain(String name, String path, String principal) {
274          SecurityModule.Builder smBuilder = new SecurityModule.Builder();
275          if (Utils.IBM_JDK) {
276             smBuilder.name("com.ibm.security.auth.module.Krb5LoginModule").flag("required")
277                   .putOption("useKeytab", path)
278                   .putOption("credsType", "both") //
279                   .putOption("forwardable", TRUE) //
280                   .putOption("proxiable", TRUE) //
281                   .putOption("noAddress", TRUE);
282          } else {
283             smBuilder.name("Kerberos").flag("required") //
284                   .putOption("storeKey", "true")
285                   .putOption("useKeyTab", "true")
286                   .putOption("refreshKrb5Config", "true")
287                   .putOption("doNotPrompt", "true")
288                   .putOption("keyTab", path); //
289          }
290 
291          smBuilder
292                .putOption("principal", principal + "@INFINISPAN.ORG")
293                .putOption("debug", TRUE);
294 
295          return new SecurityDomain.Builder()
296                .name(SECURITY_DOMAIN_PREFIX + name).cacheType("default")
297                .loginModules(
298                      smBuilder.build()
299                ).build();
300       }

304     * A Kerberos/Ldap server setup task. Starts Kerberos/Ldap server
305     *
306     * @author <a href="mailto:jcacek@redhat.com">Josef Cacek</a>
307     * @author <a href="mailto:vchepeli@redhat.com">Vitalii Chepeliuk</a>

309    static class KrbLdapServerSetupTask implements ServerSetupTask {
310       private static ApacheDsKrbLdap krbLdapServer;
311       private static boolean krbStarted = false;
312 
313       @Override
314       public void setup(ManagementClient managementClient, String s) throws Exception {
315          final String hostname = Utils.getCannonicalHost(managementClient);
316          System.setProperty("java.security.krb5.conf", System.getProperty("java.io.tmpdir") + File.separator + "krb5.conf");
317          if (!krbStarted) {
318             krbLdapServer = new ApacheDsKrbLdap(hostname);
319             krbLdapServer.start();
320             krbStarted = true;
321          }
322       }
323 
324       @Override
325       public void tearDown(ManagementClient managementClient, String s) throws Exception {
326          if (krbStarted) {
327             krbLdapServer.stop();
328             krbStarted = false;
329          }
330       }
331    }
