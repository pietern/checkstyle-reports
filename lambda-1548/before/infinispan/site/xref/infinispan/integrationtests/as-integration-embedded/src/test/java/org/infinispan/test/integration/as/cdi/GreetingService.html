3   import javax.cache.annotation.CacheResult;

5   /**
6    * <p>This is the Greeting Service class.</p>
7    *
8    * <p>Each call to the {@link GreetingService#greet(String)} method will be cached in the greeting-cache (in this case
9    * the {@linkplain javax.cache.annotation.CacheKey CacheKey} will be the name). If this method has been already called
10   * with the same name the cached value will be returned and this method will not be called.</p>
11   *
12   * @author Kevin Pollet <pollet.kevin@gmail.com> (C) 2011
13   * @see CacheResult
14   */
15  public class GreetingService {
16  
17     @CacheResult(cacheName = "greeting-cache")
18     public java.lang.String greet(String name) {
19        return "Hello " + name + " :)";
20     }
21  
22  }
