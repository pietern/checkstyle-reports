13  public class SizeOperation extends RetryOnFailureOperation<Integer> {
14  
15     protected SizeOperation(Codec codec, TransportFactory transportFactory,
16           byte[] cacheName, AtomicInteger topologyId, Flag[] flags) {
17        super(codec, transportFactory, cacheName, topologyId, flags);
18     }
19  
20     @Override
21     protected Transport getTransport(int retryCount, Set<SocketAddress> failedServers) {
22        return transportFactory.getTransport(failedServers, cacheName);
23     }
24  
25     @Override
26     protected Integer executeOperation(Transport transport) {
27        HeaderParams params = writeHeader(transport, SIZE_REQUEST);
28        transport.flush();
29        readHeaderAndValidate(transport, params);
30        return transport.readVInt();
31     }
32  
33  }
