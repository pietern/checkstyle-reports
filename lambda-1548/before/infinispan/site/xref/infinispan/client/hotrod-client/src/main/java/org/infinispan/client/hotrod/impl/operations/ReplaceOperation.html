13  /**
14   * Implements "Replace" operation as defined by  <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod
15   * protocol specification</a>.
16   *
17   * @author Mircea.Markus@jboss.com
18   * @since 4.1
19   */
20  @Immutable
21  public class ReplaceOperation extends AbstractKeyValueOperation<byte[]> {
22  
23     public ReplaceOperation(Codec codec, TransportFactory transportFactory,
24              byte[] key, byte[] cacheName, AtomicInteger topologyId,
25              Flag[] flags, byte[] value, long lifespan, TimeUnit lifespanTimeUnit, long maxIdle, TimeUnit maxIdleTimeUnit) {
26        super(codec, transportFactory, key, cacheName, topologyId, flags, value, lifespan, lifespanTimeUnit, maxIdle, maxIdleTimeUnit);
27     }
28  
29     @Override
30     protected byte[] executeOperation(Transport transport) {
31        byte[] result = null;
32        short status = sendPutOperation(transport, REPLACE_REQUEST, REPLACE_RESPONSE);
33        result = returnPossiblePrevValue(transport, status);
34        return result;
35     }
36  }
