17     @Override
18     public void testGetNearCache() {
19        assertClient.expectNoNearEvents();
20        assertClient.get(1, null).expectNearGetNull(1);
21        assertClient.put(1, "v1").expectNearRemove(1);
22        assertClient.get(1, "v1").expectNearGetNull(1).expectNearPutIfAbsent(1, "v1");
23        assertClient.get(1, "v1").expectNearGetValue(1, "v1");
24        assertClient.remove(1).expectNearRemove(1);
25        assertClient.get(1, null).expectNearGetNull(1);
26     }
