20  /**
21   * Base class for all the operations that need retry logic: if the operation fails due to connection problems, try with 
22   * another available connection.
23   *
24   * @author Mircea.Markus@jboss.com
25   * @since 4.1
26   * @param T the return type of this operation
27   */
28  @Immutable
29  public abstract class RetryOnFailureOperation<T> extends HotRodOperation {
30  
31     private static final Log log = LogFactory.getLog(RetryOnFailureOperation.class, Log.class);
32  
33     protected final TransportFactory transportFactory;
34  
35     private boolean triedCompleteRestart = false;
36  
37     protected RetryOnFailureOperation(Codec codec, TransportFactory transportFactory,
38              byte[] cacheName, AtomicInteger topologyId, Flag[] flags) {
39        super(codec, flags, cacheName, topologyId);
40        this.transportFactory = transportFactory;
41     }
42  
43     @Override
44     public T execute() {
45        int retryCount = 0;
46        Set<SocketAddress> failedServers = null;
47        while (shouldRetry(retryCount)) {
48           Transport transport = null;
49           try {
50              // Transport retrieval should be retried
51              transport = getTransport(retryCount, failedServers);
52              return executeOperation(transport);
53           } catch (TransportException te) {
54              SocketAddress address = te.getServerAddress();
55              failedServers = updateFailedServers(address, failedServers);
56              // Invalidate transport since this exception means that this
57              // instance is no longer usable and should be destroyed.
58              invalidateTransport(transport, address);
59              retryCount = logTransportErrorAndThrowExceptionIfNeeded(retryCount, te);
60           } catch (RemoteIllegalLifecycleStateException e) {
61              SocketAddress address = e.getServerAddress();
62              failedServers = updateFailedServers(address, failedServers);
63              // Invalidate transport since this exception means that this
64              // instance is no longer usable and should be destroyed.
65              invalidateTransport(transport, address);
66              retryCount = logTransportErrorAndThrowExceptionIfNeeded(retryCount, e);
67           } catch (RemoteNodeSuspectException e) {
68              // Do not invalidate transport because this exception is caused
69              // as a result of a server finding out that another node has
70              // been suspected, so there's nothing really wrong with the server
71              // from which this node was received.
72              logErrorAndThrowExceptionIfNeeded(retryCount, e);
73           } finally {
74              releaseTransport(transport);
75           }
76  
77           retryCount++;
78        }
79        throw new IllegalStateException("We should not reach here!");
80     }
81  
82     private void invalidateTransport(Transport transport, SocketAddress address) {
83        if (transport != null) {
84           if (log.isTraceEnabled())
85              log.tracef("Invalidating transport %s as a result of transport exception", transport);
86  
87           transportFactory.invalidateTransport(address, transport);
88        }
89     }
90  
91     private Set<SocketAddress> updateFailedServers(SocketAddress address, Set<SocketAddress> failedServers) {
92        if (failedServers == null) {
93           failedServers = new HashSet<SocketAddress>();
94        }
95  
96        if (log.isTraceEnabled())
97           log.tracef("Add %s to failed servers", address);
98  
99        failedServers.add(address);
100       return failedServers;
101    }
102 
103    protected boolean shouldRetry(int retryCount) {
104       return retryCount <= transportFactory.getMaxRetries();
105    }
106 
107    protected int logTransportErrorAndThrowExceptionIfNeeded(int i, HotRodClientException e) {
108       String message = "Exception encountered. Retry %d out of %d";
109       if (i >= transportFactory.getMaxRetries() || transportFactory.getMaxRetries() < 0) {
110          if (!triedCompleteRestart) {
111             log.debug("Cluster might have completely shut down, try resetting transport layer and topology id", e);
112             transportFactory.reset(cacheName);
113             triedCompleteRestart = true;
114             return -1; // reset retry count
115          } else {
116             log.exceptionAndNoRetriesLeft(i,transportFactory.getMaxRetries(), e);
117             throw e;
118          }
119       } else {
120          log.tracef(e, message, i, transportFactory.getMaxRetries());
121          return i;
122       }
123    }
124 
125    protected void logErrorAndThrowExceptionIfNeeded(int i, HotRodClientException e) {
126       String message = "Exception encountered. Retry %d out of %d";
127       if (i >= transportFactory.getMaxRetries() || transportFactory.getMaxRetries() < 0) {
128          log.exceptionAndNoRetriesLeft(i,transportFactory.getMaxRetries(), e);
129          throw e;
130       } else {
131          log.tracef(e, message, i, transportFactory.getMaxRetries());
132       }
133    }
134 
135    protected void releaseTransport(Transport transport) {
136       if (transport != null)
137          transportFactory.releaseTransport(transport);
138    }
139 
140    protected abstract Transport getTransport(int retryCount, Set<SocketAddress> failedServers);
141 
142    protected abstract T executeOperation(Transport transport);
143 }
