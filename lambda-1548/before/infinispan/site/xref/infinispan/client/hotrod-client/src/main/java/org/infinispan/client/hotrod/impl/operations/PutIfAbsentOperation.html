16  /**
17   * Implements "putIfAbsent" operation as described in  <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod
18   * protocol specification</a>.
19   *
20   * @author Mircea.Markus@jboss.com
21   * @since 4.1
22   */
23  @Immutable
24  public class PutIfAbsentOperation extends AbstractKeyValueOperation<byte[]> {
25  
26     private static final BasicLogger log = BasicLogFactory.getLog(PutIfAbsentOperation.class);
27  
28     public PutIfAbsentOperation(Codec codec, TransportFactory transportFactory,
29                                 byte[] key, byte[] cacheName, AtomicInteger topologyId,
30                                 Flag[] flags, byte[] value, long lifespan,TimeUnit lifespanTimeUnit, long maxIdleTime, TimeUnit maxIdleTimeUnit) {
31        super(codec, transportFactory, key, cacheName, topologyId, flags, value, lifespan, lifespanTimeUnit, maxIdleTime, maxIdleTimeUnit);
32     }
33  
34     @Override
35     protected byte[] executeOperation(Transport transport) {
36        short status = sendPutOperation(transport, PUT_IF_ABSENT_REQUEST, PUT_IF_ABSENT_RESPONSE);
37        byte[] previousValue = null;
38        if (status == NO_ERROR_STATUS || status == NOT_PUT_REMOVED_REPLACED_STATUS || status == NOT_EXECUTED_WITH_PREVIOUS) {
39           previousValue = returnPossiblePrevValue(transport, status);
40           if (log.isTraceEnabled()) {
41              log.tracef("Returning from putIfAbsent: %s", Util.printArray(previousValue, false));
42           }
43        }
44        return previousValue;
45     }
46  }
