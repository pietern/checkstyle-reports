12   * @author gustavonalle
13   * @since 8.0

16  
17     private Set<byte[]> keys = CollectionFactory.makeSet(ByteArrayEquivalence.INSTANCE);
18  
19     @Override
20     public boolean track(byte[] key) {
21        return keys.add(key);
22     }
23  
24     @Override
25     public void segmentsFinished(byte[] finishedSegments) {
26     }
27  
28     @Override
29     public Set<Integer> missedSegments() {
30        return null;
31     }
32  }
