9   /**
10   * Type-aware properties.  Extends the JDK {@link Properties} class to provide accessors that convert values to certain
11   * types, using default values if a conversion is not possible.
12   *
13   *
14   * @configRef name="Properties to add to the enclosing component."
15   *
16   * @author Manik Surtani
17   * @since 4.0
18   */
19  public class TypedProperties extends Properties {
20  
21     /** The serialVersionUID */
22     private static final long serialVersionUID = 3799321248100686287L;
23  
24     private static final Log log = LogFactory.getLog(TypedProperties.class);
25  
26     /**
27      * Copy constructor
28      *
29      * @param p properties instance to from.  If null, then it is treated as an empty Properties instance.
30      */
31     public TypedProperties(Properties p) {
32        if (p != null && !p.isEmpty()) putAll(p);
33     }
34  
35     /**
36      * Default constructor that returns an empty instance
37      */
38     public TypedProperties() {
39  
40     }
41  
42     /**
43      * Factory method that converts a JDK {@link Properties} instance to an instance of TypedProperties, if needed.
44      *
45      * @param p properties to convert.
46      * @return A TypedProperties object.  Returns an empty TypedProperties instance if p is null.
47      */
48     public static TypedProperties toTypedProperties(Properties p) {
49        if (p instanceof TypedProperties) return (TypedProperties) p;
50        return new TypedProperties(p);
51     }
52  
53     public int getIntProperty(String key, int defaultValue) {
54        return getIntProperty(key, defaultValue, false);
55     }
56  
57     public int getIntProperty(String key, int defaultValue, boolean doStringReplace) {
58        String value = getProperty(key);
59        if (value == null) return defaultValue;
60        value = value.trim();
61        if (value.length() == 0) return defaultValue;
62  
63        if (doStringReplace)
64           value = StringPropertyReplacer.replaceProperties(value);
65  
66        try {
67           return Integer.parseInt(value);
68        }
69        catch (NumberFormatException nfe) {
70           log.unableToConvertStringPropertyToInt(value, defaultValue);
71           return defaultValue;
72        }
73     }
74  
75     public long getLongProperty(String key, long defaultValue) {
76        return getLongProperty(key, defaultValue, false);
77     }

97     public boolean getBooleanProperty(String key, boolean defaultValue) {

101    public boolean getBooleanProperty(String key, boolean defaultValue, boolean doStringReplace) {
102       String value = getProperty(key);
103       if (value == null) return defaultValue;
104       value = value.trim();
105       if (value.length() == 0) return defaultValue;
106 
107       if (doStringReplace)
108          value = StringPropertyReplacer.replaceProperties(value);
109 
110       try {
111          return Boolean.parseBoolean(value);
112       }
113       catch (Exception e) {
114          log.unableToConvertStringPropertyToBoolean(value, defaultValue);
115          return defaultValue;
116       }
117    }
118 
119    /**
120     * Get the property associated with the key, optionally applying string property replacement as defined in
121     * {@link StringPropertyReplacer#replaceProperties} to the result.
122     *
123     * @param   key               the hashtable key.
124     * @param   defaultValue      a default value.
125     * @param   doStringReplace   boolean indicating whether to apply string property replacement
126     * @return  the value in this property list with the specified key valu after optionally being inspected for String property replacement
127     */
128    public String getProperty(String key, String defaultValue, boolean doStringReplace) {
129       if (doStringReplace)
130          return StringPropertyReplacer.replaceProperties(getProperty(key, defaultValue));
131       else
132          return getProperty(key, defaultValue);
133    }
134 
135    @Override
136    public synchronized TypedProperties setProperty(String key, String value) {
137       super.setProperty(key, value);
138       return this;
139    }
