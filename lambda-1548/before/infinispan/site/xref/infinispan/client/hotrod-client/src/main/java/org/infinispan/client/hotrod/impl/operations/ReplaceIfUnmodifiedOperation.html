1   package org.infinispan.client.hotrod.impl.operations;

3   import org.infinispan.client.hotrod.Flag;
4   import org.infinispan.client.hotrod.impl.VersionedOperationResponse;
5   import org.infinispan.client.hotrod.impl.protocol.Codec;
6   import org.infinispan.client.hotrod.impl.protocol.HeaderParams;
7   import org.infinispan.client.hotrod.impl.transport.Transport;
8   import org.infinispan.client.hotrod.impl.transport.TransportFactory;
9   
10  import java.util.concurrent.TimeUnit;
11  import java.util.concurrent.atomic.AtomicInteger;
12  
13  /**
14   * Implement "replaceIfUnmodified" as defined by  <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod
15   * protocol specification</a>.
16   *
17   * @author Mircea.Markus@jboss.com
18   * @since 4.1
19   */
20  public class ReplaceIfUnmodifiedOperation extends AbstractKeyValueOperation<VersionedOperationResponse> {
21     private final long version;
22  
23     public ReplaceIfUnmodifiedOperation(Codec codec, TransportFactory transportFactory, byte[] key, byte[] cacheName,
24                                         AtomicInteger topologyId, Flag[] flags, byte[] value,
25                                         long lifespan, TimeUnit lifespanTimeUnit, long maxIdle, TimeUnit maxIdleTimeUnit, long version) {
26        super(codec, transportFactory, key, cacheName, topologyId, flags, value, lifespan, lifespanTimeUnit, maxIdle, maxIdleTimeUnit);
27        this.version = version;
28     }
29  
30     @Override
31     protected VersionedOperationResponse executeOperation(Transport transport) {
32        // 1) write header
33        HeaderParams params = writeHeader(transport, REPLACE_IF_UNMODIFIED_REQUEST);
34  
35        //2) write message body
36        transport.writeArray(key);
37        codec.writeExpirationParams(transport, lifespan, lifespanTimeUnit, maxIdle, maxIdleTimeUnit);
38        transport.writeLong(version);
39        transport.writeArray(value);
40        transport.flush();
41  
42        return returnVersionedOperationResponse(transport, params);
43     }
44  }
