1   package org.infinispan.client.hotrod.marshall;
2   
3   import org.infinispan.protostream.MessageMarshaller;
4   import org.infinispan.query.dsl.embedded.testdomain.hsearch.AccountHS;
5   
6   import java.io.IOException;
7   import java.util.Date;
8   
9   /**
10   * @author anistor@redhat.com
11   * @since 6.0
12   */
13  public class EmbeddedAccountMarshaller implements MessageMarshaller<AccountHS> {
14  
15     @Override
16     public String getTypeName() {
17        return "sample_bank_account.Account";
18     }
19  
20     @Override
21     public Class<AccountHS> getJavaClass() {
22        return AccountHS.class;
23     }
24  
25     @Override
26     public AccountHS readFrom(ProtoStreamReader reader) throws IOException {
27        int id = reader.readInt("id");
28        String description = reader.readString("description");
29        long creationDate = reader.readLong("creationDate");
30  
31        AccountHS account = new AccountHS();
32        account.setId(id);
33        account.setDescription(description);
34        account.setCreationDate(new Date(creationDate));
35        return account;
36     }
37  
38     @Override
39     public void writeTo(ProtoStreamWriter writer, AccountHS account) throws IOException {
40        writer.writeInt("id", account.getId());
41        writer.writeString("description", account.getDescription());
42        writer.writeLong("creationDate", account.getCreationDate().getTime());
43     }
44  }
