18  @Test(groups = "functional", testName = "client.hotrod.iteration.MultiServerDistRemoteIteratorTest")
19  public class MultiServerDistRemoteIteratorTest extends BaseMultiServerRemoteIteratorTest {
20  
21     private static final int NUM_SERVERS = 3;
22  
23     @Override
24     protected void createCacheManagers() throws Throwable {
25        createHotRodServers(NUM_SERVERS, getCacheConfiguration());
26     }
27  
28     private ConfigurationBuilder getCacheConfiguration() {
29        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, false);
30        builder.clustering().hash().numOwners(2);
31        return builder;
32     }
33  
34     @Override
35     protected org.infinispan.client.hotrod.configuration.ConfigurationBuilder createHotRodClientConfigurationBuilder(int serverPort) {
36        org.infinispan.client.hotrod.configuration.ConfigurationBuilder clientBuilder = new org.infinispan.client.hotrod.configuration.ConfigurationBuilder();
37        clientBuilder.addServer()
38                .host("localhost")
39                .port(serverPort)
40                .maxRetries(maxRetries())
41                .balancingStrategy(new PreferredServerBalancingStrategy(new InetSocketAddress("localhost", serverPort)))
42                .pingOnStartup(false);
43        return clientBuilder;
44     }
45  
46     @Test
47     public void testIterationRouting() throws Exception {
48        for (int i = 0; i < clients.size(); i++) {
49           RemoteCacheManager client = client(i);
50           try (CloseableIterator<Map.Entry<Object, Object>> ignored = client.getCache().retrieveEntries(null, 10)) {
51              assertIterationActiveOnlyOnServer(i);
52           }
53        }
54     }
55  
56     private void assertIterationActiveOnlyOnServer(int index) {
57        for (int i = 0; i < servers.size(); i++) {
58           int activeIterations = server(i).iterationManager().activeIterations();
59           if (i == index) {
60              assertEquals(1L, activeIterations);
61           } else {
62              assertEquals(0L, activeIterations);
63           }
64        }
65     }
