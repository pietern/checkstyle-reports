29  /**
30   * Maintains topology information about caches.
31   *
32   * @author gustavonalle
33   */
34  public final class TopologyInfo {
35  
36     private static final Log log = LogFactory.getLog(TopologyInfo.class, Log.class);
37  
38     private Collection<SocketAddress> servers = new ArrayList<>();
39     private Map<byte[], ConsistentHash> consistentHashes = CollectionFactory.makeMap(ByteArrayEquivalence.INSTANCE, AnyEquivalence.getInstance());
40     private Map<byte[], Integer> segmentsByCache = CollectionFactory.makeMap(ByteArrayEquivalence.INSTANCE, AnyEquivalence.getInstance());
41     private volatile AtomicInteger topologyId;
42     private final ConsistentHashFactory hashFactory = new ConsistentHashFactory();
43  
44     public TopologyInfo(AtomicInteger topologyId, Collection<SocketAddress> servers, Configuration configuration) {
45        this.topologyId = topologyId;
46        this.servers = servers;
47        hashFactory.init(configuration);
48     }
49  
50     private Map<SocketAddress, Set<Integer>> getSegmentsByServer(byte[] cacheName) {
51        ConsistentHash consistentHash = consistentHashes.get(cacheName);
52        if (consistentHash != null) {
53           return consistentHash.getSegmentsByServer();
54        } else {
55           Optional<Integer> numSegments = Optional.ofNullable(segmentsByCache.get(cacheName));
56           Optional<Set<Integer>> segments = numSegments.map(n -> range(0, n).boxed().collect(Collectors.toSet()));
57           return Immutables.immutableMapWrap(
58                 servers.stream().collect(toMap(identity(), s -> segments.orElse(emptySet())))
59           );
60        }
61     }
62  
63     public Collection<SocketAddress> getServers() {
64        return servers;
65     }
66  
67     public void updateTopology(Map<SocketAddress, Set<Integer>> servers2Hash, int numKeyOwners, short hashFunctionVersion, int hashSpace, byte[] cacheName) {
68        ConsistentHash hash = hashFactory.newConsistentHash(hashFunctionVersion);
69        if (hash == null) {
70           log.noHasHFunctionConfigured(hashFunctionVersion);
71        } else {
72           hash.init(servers2Hash, numKeyOwners, hashSpace);
73        }
74        consistentHashes.put(cacheName, hash);
75  
76     }
77  
78     public void updateTopology(SocketAddress[][] segmentOwners, int numSegments, short hashFunctionVersion, byte[] cacheName) {
79        if (hashFunctionVersion > 0) {
80           SegmentConsistentHash hash = hashFactory.newConsistentHash(hashFunctionVersion);
81           if (hash == null) {
82              log.noHasHFunctionConfigured(hashFunctionVersion);
83           } else {
84              hash.init(segmentOwners, numSegments);
85           }
86           consistentHashes.put(cacheName, hash);
87        }
88        segmentsByCache.put(cacheName, numSegments);
89  
90     }
91  
92     public Optional<SocketAddress> getHashAwareServer(byte[] key, byte[] cacheName) {
93        Optional<SocketAddress> server = Optional.empty();
94        ConsistentHash consistentHash = consistentHashes.get(cacheName);
95        if (consistentHash != null) {
96           server = Optional.of(consistentHash.getServer(key));
97           if (log.isTraceEnabled()) {
98              log.tracef("Using consistent hash for determining the server: " + server);
99           }
100       }
101       return server;
102    }
103 
104    public void updateServers(Collection<SocketAddress> updatedServers) {
105       servers = updatedServers;
106    }
107 
108    public ConsistentHash getConsistentHash(byte[] cacheName) {
109       return consistentHashes.get(cacheName);
110    }
111 
112    public ConsistentHashFactory getConsistentHashFactory() {
113       return hashFactory;
114    }
115 
116    public void setTopologyId(int topologyId) {
117       this.topologyId.set(topologyId);
118    }
119 
120    public CacheTopologyInfo getCacheTopologyInfo(byte[] cacheName) {
121       return new CacheTopologyInfoImpl(getSegmentsByServer(cacheName), segmentsByCache.get(cacheName), topologyId.get());
122    }
123 
124 }
