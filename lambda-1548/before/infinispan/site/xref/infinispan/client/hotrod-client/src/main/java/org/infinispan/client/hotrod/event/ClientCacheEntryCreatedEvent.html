3   /**
4    * Client side cache entry created events provide information on the created
5    * key, and the version of the entry. This version can be used to invoke conditional
6    * operations on the server, such as
7    * {@link org.infinispan.client.hotrod.RemoteCache#replaceWithVersion(Object, Object, long)}
8    * or {@link org.infinispan.client.hotrod.RemoteCache#removeWithVersion(Object, long)}
9    *
10   * @param <K> type of key created.
11   */
12  public interface ClientCacheEntryCreatedEvent<K> extends ClientEvent {
13  
14     /**
15      * Created cache entry's key.
16      * @return an instance of the key with which a cache entry has been
17      * created in the remote server(s).
18      */
19     K getKey();
20  
21     /**
22      * Provides access to the version of the created cache entry. This version
23      * can be used to invoke conditional operations on the server, such as
24      * {@link org.infinispan.client.hotrod.RemoteCache#replaceWithVersion(Object, Object, long)}
25      * or {@link org.infinispan.client.hotrod.RemoteCache#removeWithVersion(Object, long)}
26      *
27      * @return a long containing the version of the created cache entry.
28      */
29     long getVersion();
30  
31     /**
32      * This will be true if the write command that caused this had to be retried
33      * again due to a topology change.  This could be a sign that this event
34      * has been duplicated or another event was dropped and replaced
35      * (eg: ModifiedEvent replaced CreateEvent)
36      *
37      * @return Whether the command that caused this event was retried
38      */
39     boolean isCommandRetried();
40  
41  }
