12  import java.util.concurrent.atomic.AtomicInteger;
13  
14  /**
15   * Corresponds to clear operation as defined by <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod protocol specification</a>.
16   *
17   * @author Mircea.Markus@jboss.com
18   * @since 4.1
19   */
20  @Immutable
21  public class ClearOperation extends RetryOnFailureOperation<Void> {
22  
23     public ClearOperation(Codec codec, TransportFactory transportFactory,
24              byte[] cacheName, AtomicInteger topologyId, Flag[] flags) {
25        super(codec, transportFactory, cacheName, topologyId, flags);
26     }
27  
28     @Override
29     protected Transport getTransport(int retryCount, Set<SocketAddress> failedServers) {
30        return transportFactory.getTransport(failedServers, cacheName);
31     }
32  
33     @Override
34     protected Void executeOperation(Transport transport) {
35        HeaderParams params = writeHeader(transport, CLEAR_REQUEST);
36        transport.flush();
37  
38        readHeaderAndValidate(transport, params);
39        return null;
40     }
41  }
