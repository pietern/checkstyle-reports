51     @Override
52     public void writeExpirationParams(Transport transport, long lifespan, TimeUnit lifespanTimeUnit, long maxIdle, TimeUnit maxIdleTimeUnit) {
53        if (!CodecUtils.isIntCompatible(lifespan)) {
54           log.warn("Lifespan value greater than the max supported size (Integer.MAX_VALUE), this can cause precision loss");
55        }
56        if (!CodecUtils.isIntCompatible(maxIdle)) {
57           log.warn("MaxIdle value greater than the max supported size (Integer.MAX_VALUE), this can cause precision loss");
58        }
59        int lifespanSeconds = CodecUtils.toSeconds(lifespan, lifespanTimeUnit);
60        int maxIdleSeconds = CodecUtils.toSeconds(maxIdle, maxIdleTimeUnit);
61        transport.writeVInt(lifespanSeconds);
62        transport.writeVInt(maxIdleSeconds);
63     }

65     protected HeaderParams writeHeader(
66              Transport transport, HeaderParams params, byte version) {
67        transport.writeByte(HotRodConstants.REQUEST_MAGIC);
68        transport.writeVLong(params.messageId(MSG_ID.incrementAndGet()).messageId);
69        transport.writeByte(version);
70        transport.writeByte(params.opCode);
71        transport.writeArray(params.cacheName);
72  
73        int flagInt = 0;
74        if (params.flags != null) {
75           for (Flag flag : params.flags) {
76              if (flag.equals(Flag.FORCE_RETURN_VALUE)) // 1.0 / 1.1 servers only understand this flag
77                 flagInt = flag.getFlagInt();
78           }
79        }
80        transport.writeVInt(flagInt);
81        transport.writeByte(params.clientIntel);
82        transport.writeVInt(params.topologyId.get());
83        //todo change once TX support is added
84        transport.writeByte(params.txMarker);
85        getLog().tracef("Wrote header for message %d. Operation code: %#04x. Flags: %#x",
86                   params.messageId, params.opCode, flagInt);
87        return params;
88     }

90     @Override
91     public short readHeader(Transport transport, HeaderParams params) {
92        short magic = transport.readByte();
93        final Log localLog = getLog();
94        boolean isTrace = localLog.isTraceEnabled();
95        if (magic != HotRodConstants.RESPONSE_MAGIC) {
96           String message = "Invalid magic number. Expected %#x and received %#x";
97           localLog.invalidMagicNumber(HotRodConstants.RESPONSE_MAGIC, magic);
98           if (isTrace)
99              localLog.tracef("Socket dump: %s", hexDump(transport.dumpStream()));
100          throw new InvalidResponseException(String.format(message, HotRodConstants.RESPONSE_MAGIC, magic));
101       }
102       long receivedMessageId = transport.readVLong();
103       // If received id is 0, it could be that a failure was noted before the
104       // message id was detected, so don't consider it to a message id error
105       if (receivedMessageId != params.messageId && receivedMessageId != 0) {
106          String message = "Invalid message id. Expected %d and received %d";
107          localLog.invalidMessageId(params.messageId, receivedMessageId);
108          if (isTrace)
109             localLog.tracef("Socket dump: %s", hexDump(transport.dumpStream()));
110          throw new InvalidResponseException(String.format(message, params.messageId, receivedMessageId));
111       }
112       localLog.tracef("Received response for message id: %d", receivedMessageId);
113 
114       short receivedOpCode = transport.readByte();
115       // Read both the status and new topology (if present),
116       // before deciding how to react to error situations.
117       short status = transport.readByte();
118       readNewTopologyIfPresent(transport, params);
119 
120       // Now that all headers values have been read, check the error responses.
121       // This avoids situatiations where an exceptional return ends up with
122       // the socket containing data from previous request responses.
123       if (receivedOpCode != params.opRespCode) {
124          if (receivedOpCode == HotRodConstants.ERROR_RESPONSE) {
125             checkForErrorsInResponseStatus(transport, params, status);
126          }
127          throw new InvalidResponseException(String.format(
128                "Invalid response operation. Expected %#x and received %#x",
129                params.opRespCode, receivedOpCode));
130       }
131       localLog.tracef("Received operation code is: %#04x", receivedOpCode);
132 
133       return status;
134    }

146    @Override
147    public byte[] returnPossiblePrevValue(Transport transport, short status, Flag[] flags) {
148       if (hasForceReturn(flags)) {
149          byte[] bytes = transport.readArray();
150          if (log.isTraceEnabled()) log.tracef("Previous value bytes is: %s", Util.printArray(bytes, false));
151          //0-length response means null
152          return bytes.length == 0 ? null : bytes;
153       } else {
154          return null;
155       }
156    }

158    private boolean hasForceReturn(Flag[] flags) {
159       if (flags == null) return false;
160       for (Flag flag : flags) {
161          if (flag == Flag.FORCE_RETURN_VALUE) return true;
162       }
163       return false;
164    }

171    protected void checkForErrorsInResponseStatus(Transport transport, HeaderParams params, short status) {
172       final Log localLog = getLog();
173       boolean isTrace = localLog.isTraceEnabled();
174       if (isTrace) localLog.tracef("Received operation status: %#x", status);
175 
176       try {
177          switch (status) {
178             case HotRodConstants.INVALID_MAGIC_OR_MESSAGE_ID_STATUS:
179             case HotRodConstants.REQUEST_PARSING_ERROR_STATUS:
180             case HotRodConstants.UNKNOWN_COMMAND_STATUS:
181             case HotRodConstants.SERVER_ERROR_STATUS:
182             case HotRodConstants.COMMAND_TIMEOUT_STATUS:
183             case HotRodConstants.UNKNOWN_VERSION_STATUS: {
184                // If error, the body of the message just contains a message
185                String msgFromServer = transport.readString();
186                if (status == HotRodConstants.COMMAND_TIMEOUT_STATUS && isTrace) {
187                   localLog.tracef("Server-side timeout performing operation: %s", msgFromServer);
188                } if (msgFromServer.contains("SuspectException")
189                      || msgFromServer.contains("SuspectedException")) {
190                   // Handle both Infinispan's and JGroups' suspicions
191                   if (isTrace)
192                      localLog.tracef("A remote node was suspected while executing messageId=%d. " +
193                         "Check if retry possible. Message from server: %s", params.messageId, msgFromServer);
194                   // TODO: This will be better handled with its own status id in version 2 of protocol
195                   throw new RemoteNodeSuspectException(msgFromServer, params.messageId, status);
196                } else {
197                   localLog.errorFromServer(msgFromServer);
198                }
199                throw new HotRodClientException(msgFromServer, params.messageId, status);
200             }
201             default: {
202                throw new IllegalStateException(String.format("Unknown status: %#04x", status));
203             }
204          }
205       } finally {
206          // Errors related to protocol parsing are odd, and they can sometimes
207          // be the consequence of previous errors, so whenever these errors
208          // occur, invalidate the underlying transport instance so that a
209          // brand new connection is established next time around.
210          switch (status) {
211             case HotRodConstants.INVALID_MAGIC_OR_MESSAGE_ID_STATUS:
212             case HotRodConstants.REQUEST_PARSING_ERROR_STATUS:
213             case HotRodConstants.UNKNOWN_COMMAND_STATUS:
214             case HotRodConstants.UNKNOWN_VERSION_STATUS: {
215                transport.invalidate();
216             }
217          }
218       }
219    }

221    protected void readNewTopologyIfPresent(Transport transport, HeaderParams params) {
222       short topologyChangeByte = transport.readByte();
223       if (topologyChangeByte == 1)
224          readNewTopologyAndHash(transport, params.topologyId, params.cacheName);
225    }

227    protected void readNewTopologyAndHash(Transport transport, AtomicInteger topologyId, byte[] cacheName) {
228       final Log localLog = getLog();
229       int newTopologyId = transport.readVInt();
230       topologyId.set(newTopologyId);
231       int numKeyOwners = transport.readUnsignedShort();
232       short hashFunctionVersion = transport.readByte();
233       int hashSpace = transport.readVInt();
234       int clusterSize = transport.readVInt();
235 
236       Map<SocketAddress, Set<Integer>> servers2Hash = computeNewHashes(
237             transport, localLog, newTopologyId, numKeyOwners,
238             hashFunctionVersion, hashSpace, clusterSize);
239 
240       Set<SocketAddress> socketAddresses = servers2Hash.keySet();
241       if (localLog.isInfoEnabled()) {
242          localLog.newTopology(transport.getRemoteSocketAddress(), newTopologyId,
243                socketAddresses.size(), socketAddresses);
244       }
245       transport.getTransportFactory().updateServers(socketAddresses, cacheName, false);
246       if (hashFunctionVersion == 0) {
247          localLog.trace("Not using a consistent hash function (hash function version == 0).");
248       } else {
249          transport.getTransportFactory().updateHashFunction(
250                servers2Hash, numKeyOwners, hashFunctionVersion, hashSpace, cacheName);
251       }
252    }

254    protected Map<SocketAddress, Set<Integer>> computeNewHashes(Transport transport,
255          Log localLog, int newTopologyId, int numKeyOwners,
256          short hashFunctionVersion, int hashSpace, int clusterSize) {
257       if (localLog.isTraceEnabled()) {
258          localLog.tracef("Topology change request: newTopologyId=%d, numKeyOwners=%d, " +
259                        "hashFunctionVersion=%d, hashSpaceSize=%d, clusterSize=%d",
260                  newTopologyId, numKeyOwners, hashFunctionVersion, hashSpace, clusterSize);
261       }
262 
263       Map<SocketAddress, Set<Integer>> servers2Hash = new LinkedHashMap<SocketAddress, Set<Integer>>();
264 
265       for (int i = 0; i < clusterSize; i++) {
266          String host = transport.readString();
267          int port = transport.readUnsignedShort();
268          int hashCode = transport.read4ByteInt();
269          localLog.tracef("Server read: %s:%d - hash code is %d", host, port, hashCode);
270          SocketAddress address = new InetSocketAddress(host, port);
271          Set<Integer> hashes = servers2Hash.get(address);
272          if (hashes == null) {
273             hashes = new HashSet<Integer>();
274             servers2Hash.put(address, hashes);
275          }
276          hashes.add(hashCode);
277          localLog.tracef("Hash code is: %d", hashCode);
278       }
279       return servers2Hash;
280    }
