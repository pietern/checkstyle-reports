3   /**
4    * Client side cache entry removed events provide information on the removed key.
5    *
6    * @param <K> type of key created.
7    */
8   public interface ClientCacheEntryRemovedEvent<K> extends ClientEvent {
9   
10     /**
11      * Created cache entry's key.
12      * @return an instance of the key with which a cache entry has been
13      * created in remote server.
14      */
15     K getKey();
16  
17     /**
18      * This will be true if the write command that caused this had to be retried
19      * again due to a topology change.  This could be a sign that this event
20      * has been duplicated or another event was dropped and replaced
21      * (eg: ModifiedEvent replaced CreateEvent)
22      *
23      * @return Whether the command that caused this event was retried
24      */
25     boolean isCommandRetried();
26  
27  }
