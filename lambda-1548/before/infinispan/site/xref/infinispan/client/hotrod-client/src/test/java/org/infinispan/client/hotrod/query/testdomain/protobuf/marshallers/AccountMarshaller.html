3   import org.infinispan.client.hotrod.query.testdomain.protobuf.AccountPB;
4   import org.infinispan.protostream.MessageMarshaller;

6   import java.io.IOException;
7   import java.util.Date;
8   
9   /**
10   * @author anistor@redhat.com
11   * @since 7.0
12   */
13  public class AccountMarshaller implements MessageMarshaller<AccountPB> {
14  
15     @Override
16     public String getTypeName() {
17        return "sample_bank_account.Account";
18     }
19  
20     @Override
21     public Class<AccountPB> getJavaClass() {
22        return AccountPB.class;
23     }
24  
25     @Override
26     public AccountPB readFrom(ProtoStreamReader reader) throws IOException {
27        int id = reader.readInt("id");
28        String description = reader.readString("description");
29        long creationDate = reader.readLong("creationDate");
30  
31        AccountPB account = new AccountPB();
32        account.setId(id);
33        account.setDescription(description);
34        account.setCreationDate(new Date(creationDate));
35        return account;
36     }
37  
38     @Override
39     public void writeTo(ProtoStreamWriter writer, AccountPB account) throws IOException {
40        writer.writeInt("id", account.getId());
41        writer.writeString("description", account.getDescription());
42        writer.writeLong("creationDate", account.getCreationDate().getTime());
43     }
44  }
