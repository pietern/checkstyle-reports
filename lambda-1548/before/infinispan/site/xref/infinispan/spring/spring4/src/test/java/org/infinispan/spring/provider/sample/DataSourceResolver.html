3   import javax.sql.XADataSource;
4   import com.arjuna.ats.internal.jdbc.DynamicClass;
5   import org.h2.jdbcx.JdbcDataSource;
6   
7   /**
8    * Required by JBoss Transactions for DataSource resolving.
9    *
10   * @author Matej Cimbora (mcimbora@redhat.com)
11   */
12  public class DataSourceResolver implements DynamicClass {
13  
14     @Override
15     public XADataSource getDataSource(String url) {
16        JdbcDataSource dataSource = new JdbcDataSource();
17        dataSource.setURL(String.format("jdbc:%s;DB_CLOSE_DELAY=-1", url));
18        dataSource.setUser("sa");
19        dataSource.setPassword("");
20        return dataSource;
21     }
22  }
