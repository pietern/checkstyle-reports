1   package org.infinispan.query.dsl;
2   
3   /**
4    * The context of a complete filter. Provides operations to allow connecting multiple filters together with boolean
5    * operators.
6    *
7    * @author anistor@redhat.com
8    * @since 6.0
9    */
10  public interface FilterConditionContext {
11  
12     /**
13      * Creates a new context and connects it with the current one using boolean AND. The new context is added after the
14      * current one. The two conditions are not grouped so operator precedence in the resulting condition might change.
15      * <p/>
16      * The effect is: a AND b
17      *
18      * @return the new context
19      */
20     FilterConditionBeginContext and();
21  
22     /**
23      * Connects a given context with the current one using boolean AND. The new context is added after the current one
24      * and is grouped. Operator precedence will be unaffected due to grouping.
25      * <p/>
26      * The effect is: a AND (b)
27      *
28      * @param rightCondition the second condition
29      * @return the new context
30      */
31     FilterConditionContext and(FilterConditionContext rightCondition);
32  
33     /**
34      * Creates a new context and connects it with the current one using boolean OR. The new context is added after the
35      * current one.
36      * <p/>
37      * The effect is: a OR b
38      *
39      * @return the new context
40      */
41     FilterConditionBeginContext or();
42  
43     /**
44      * Connects a given context with the current one using boolean OR. The new context is added after the current one and
45      * is grouped.
46      * <p/>
47      * The effect is: a OR (b)
48      *
49      * @param rightCondition the second condition
50      * @return the new context
51      */
52     FilterConditionContext or(FilterConditionContext rightCondition);
53  
54     /**
55      * Get the {@link QueryBuilder} that created this context.
56      *
57      * @return the parent builder
58      */
59     <T extends Query> QueryBuilder<T> toBuilder();
60  }
