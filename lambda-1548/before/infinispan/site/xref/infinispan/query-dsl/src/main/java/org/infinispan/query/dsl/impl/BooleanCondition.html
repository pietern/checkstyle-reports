1   package org.infinispan.query.dsl.impl;
2   
3   import org.infinispan.query.dsl.QueryBuilder;
4   import org.infinispan.query.dsl.QueryFactory;
5   
6   /**
7    * Unary or binary boolean condition (NOT, AND, OR).
8    *
9    * @author anistor@redhat.com
10   * @since 6.0
11   */
12  abstract class BooleanCondition extends BaseCondition {
13  
14     private BaseCondition leftCondition;
15  
16     private BaseCondition rightCondition;
17  
18     public BooleanCondition(QueryFactory queryFactory, BaseCondition leftCondition, BaseCondition rightCondition) {
19        super(queryFactory);
20        this.leftCondition = leftCondition;
21        this.rightCondition = rightCondition;
22     }
23  
24     public BaseCondition getFirstCondition() {
25        return leftCondition;
26     }
27  
28     public BaseCondition getSecondCondition() {
29        return rightCondition;
30     }
31  
32     public void replaceChildCondition(BaseCondition oldChild, BaseCondition newChild) {
33        if (leftCondition == oldChild) {
34           leftCondition = newChild;
35        } else if (rightCondition == oldChild) {
36           rightCondition = newChild;
37        } else {
38           throw new IllegalStateException("Old child condition not found in parent");
39        }
40        newChild.setParent(this);
41     }
42  
43     @Override
44     void setQueryBuilder(QueryBuilder queryBuilder) {
45        super.setQueryBuilder(queryBuilder);
46        if (leftCondition != null) {
47           leftCondition.setQueryBuilder(queryBuilder);
48        }
49        if (rightCondition != null) {
50           rightCondition.setQueryBuilder(queryBuilder);
51        }
52     }
53  }
