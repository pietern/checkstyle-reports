3   import org.hibernate.search.annotations.Analyze;
4   import org.hibernate.search.annotations.Norms;
5   import org.hibernate.search.annotations.Store;
6   import org.hibernate.search.cfg.SearchMapping;
7   import org.infinispan.Cache;
8   import org.infinispan.query.remote.impl.indexing.ProtobufValueWrapper;
9   import org.infinispan.query.remote.impl.indexing.ProtobufValueWrapperFieldBridge;
10  import org.infinispan.query.spi.ProgrammaticSearchMappingProvider;
11  import org.kohsuke.MetaInfServices;
12  
13  /**
14   * @author anistor@redhat.com
15   * @since 6.0
16   */
17  @MetaInfServices
18  public final class ProgrammaticSearchMappingProviderImpl implements ProgrammaticSearchMappingProvider {
19  
20     @Override
21     public void defineMappings(Cache cache, SearchMapping searchMapping) {
22        searchMapping.entity(ProtobufValueWrapper.class)
23              .indexed()
24              .classBridgeInstance(new ProtobufValueWrapperFieldBridge(cache))
25              .norms(Norms.NO)
26              .analyze(Analyze.NO)
27              .store(Store.NO);
28     }
29  }
