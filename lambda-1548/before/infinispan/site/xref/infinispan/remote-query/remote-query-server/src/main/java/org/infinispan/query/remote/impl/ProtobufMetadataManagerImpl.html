44  @MBean(objectName = ProtobufMetadataManagerConstants.OBJECT_NAME,
45        description = "Component that acts as a manager and container for Protocol Buffers message type definitions in the scope of a CacheManger.")
46  public final class ProtobufMetadataManagerImpl implements ProtobufMetadataManager {
47  
48     private Cache<String, String> protobufSchemaCache;
49  
50     private ObjectName objectName;
51  
52     private final SerializationContext serCtx;
53  
54     private EmbeddedCacheManager cacheManager;
55  
56     public ProtobufMetadataManagerImpl() {
57        org.infinispan.protostream.config.Configuration.Builder configBuilder = new org.infinispan.protostream.config.Configuration.Builder();
58        configBuilder
59              .messageAnnotation(IndexingMetadata.INDEXED_ANNOTATION)
60                 .attribute(AnnotationElement.Annotation.DEFAULT_ATTRIBUTE)
61                    .booleanType()
62                    .defaultValue(true)
63                 .annotationMetadataCreator(new IndexingMetadataCreator())
64              .fieldAnnotation(IndexingMetadata.INDEXED_FIELD_ANNOTATION)
65                 .attribute("index")
66                    .booleanType()
67                    .defaultValue(true)
68                 .attribute("store")
69                    .booleanType()
70                    .defaultValue(true);
71        serCtx = ProtobufUtil.newSerializationContext(configBuilder.build());
72        try {
73           MarshallerRegistration.registerMarshallers(serCtx);
74        } catch (IOException | DescriptorParserException e) {
75           throw new CacheException("Failed to initialise the Protobuf serialization context", e);
76        }
77     }
78  
79     @Inject
80     protected void init(EmbeddedCacheManager cacheManager, InternalCacheRegistry internalCacheRegistry) {
81        this.cacheManager = cacheManager;
82        internalCacheRegistry.registerInternalCache(PROTOBUF_METADATA_CACHE_NAME,
83                                                    getProtobufMetadataCacheConfig().build(),
84                                                    EnumSet.of(InternalCacheRegistry.Flag.USER, InternalCacheRegistry.Flag.PERSISTENT));
85     }
86  
87     /**
88      * Obtain the cache, lazily.
89      */
90     Cache<String, String> getCache() {
91        if (protobufSchemaCache == null) {
92           protobufSchemaCache = cacheManager.getCache(PROTOBUF_METADATA_CACHE_NAME);
93        }
94        return protobufSchemaCache;
95     }

97     private ConfigurationBuilder getProtobufMetadataCacheConfig() {
98        GlobalConfiguration globalConfiguration = cacheManager.getGlobalComponentRegistry().getGlobalConfiguration();
99        CacheMode cacheMode = globalConfiguration.isClustered() ? CacheMode.REPL_SYNC : CacheMode.LOCAL;
100 
101       ConfigurationBuilder cfg = new ConfigurationBuilder();
102       cfg.transaction()
103             .transactionMode(TransactionMode.TRANSACTIONAL).invocationBatching().enable()
104             .transaction().lockingMode(LockingMode.PESSIMISTIC).syncCommitPhase(true).syncRollbackPhase(true)
105             .locking().isolationLevel(IsolationLevel.READ_COMMITTED).useLockStriping(false)
106             .clustering().cacheMode(cacheMode).sync()
107             .stateTransfer().fetchInMemoryState(true).awaitInitialTransfer(false)
108             .compatibility().enable().marshaller(new CompatibilityProtoStreamMarshaller())
109             .customInterceptors().addInterceptor()
110             .interceptor(new ProtobufMetadataManagerInterceptor()).after(PessimisticLockingInterceptor.class);
111       return cfg;
112    }

129    @ManagedOperation(description = "Registers a Protobuf definition file", displayName = "Register Protofile")
130    @Override
131    public void registerProtofile(@Parameter(name = "fileName", description = "the name of the .proto file") String fileName,
132                                  @Parameter(name = "contents", description = "contents of the file") String contents) {
133       getCache().put(fileName, contents);
134    }
135 
136    @ManagedOperation(description = "Registers a set of Protobuf definition files", displayName = "Register Protofiles")
137    @Override
138    public void registerProtofiles(@Parameter(name = "fileNames", description = "names of the protofiles") String[] fileNames,
139                                   @Parameter(name = "fileContents", description = "content of the files") String[] contents) throws Exception {
140       if (fileNames.length != contents.length) {
141          throw new MBeanException(new IllegalArgumentException("invalid parameter sizes"));
142       }
143       Map<String, String> files = new HashMap<String, String>(fileNames.length);
144       for (int i = 0; i < fileNames.length; i++) {
145          files.put(fileNames[i], contents[i]);
146       }
147       getCache().putAll(files);
148    }
149 
150    @ManagedAttribute(description = "The names of all Protobuf files", displayName = "Protofile Names")
151    @Override
152    public String[] getProtofileNames() {
153       Set<String> fileNames = new HashSet<String>();
154       for (String k : getCache().keySet()) {
155          if (k.endsWith(PROTO_KEY_SUFFIX)) {
156             fileNames.add(k);
157          }
158       }
159       return fileNames.toArray(new String[fileNames.size()]);
160    }
161 
162    @ManagedOperation(description = "Get the contents of a protobuf definition file", displayName = "Get Protofile")
163    @Override
164    public String getProtofile(@Parameter(name = "fileName", description = "the name of the .proto file") String fileName) {
165       if (!fileName.endsWith(PROTO_KEY_SUFFIX)) {
166          throw new IllegalArgumentException("The file name must have \".proto\" suffix");
167       }
168       return getCache().get(fileName);
169    }
170 
171    @ManagedAttribute(description = "The names of the files that have errors, if any", displayName = "Files With Errors")
172    @Override
173    public String[] getFilesWithErrors() {
174       String filesWithErrors = getCache().get(ERRORS_KEY_SUFFIX);
175       if (filesWithErrors == null) {
176          return null;
177       }
178       return filesWithErrors.split("\n");
179    }
180 
181    @ManagedOperation(description = "Obtains the errors associated with a protobuf definition file", displayName = "Get Errors For A File")
182    @Override
183    public String getFileErrors(@Parameter(name = "fileName", description = "the name of the .proto file") String fileName) {
184       if (!fileName.endsWith(PROTO_KEY_SUFFIX)) {
185          throw new IllegalArgumentException("The file name must have \".proto\" suffix");
186       }
187       return getCache().get(fileName + ERRORS_KEY_SUFFIX);
188    }

194    /**
195     * Obtains the ProtobufMetadataManagerImpl instance associated to a cache manager.
196     */
197    private static ProtobufMetadataManagerImpl getProtobufMetadataManager(EmbeddedCacheManager cacheManager) {
198       if (cacheManager == null) {
199          throw new IllegalArgumentException("cacheManager cannot be null");
200       }
201       ProtobufMetadataManagerImpl metadataManager = (ProtobufMetadataManagerImpl) cacheManager.getGlobalComponentRegistry().getComponent(ProtobufMetadataManager.class);
202       if (metadataManager == null) {
203          throw new IllegalStateException("ProtobufMetadataManager not initialised yet!");
204       }
205       return metadataManager;
206    }

209       return new DelegatingSerializationContext(getProtobufMetadataManager(cacheManager));
