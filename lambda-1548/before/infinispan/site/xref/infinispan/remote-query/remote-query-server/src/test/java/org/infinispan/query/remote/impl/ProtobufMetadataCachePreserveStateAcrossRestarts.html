15  @Test(groups = "functional", testName = "query.remote.impl.ProtobufMetadataCachePreserveStateAcrossRestarts")
16  public class ProtobufMetadataCachePreserveStateAcrossRestarts extends AbstractInfinispanTest {
17  
18     protected EmbeddedCacheManager createCacheManager(String persistentStateLocation) throws Exception {
19        GlobalConfigurationBuilder global = new GlobalConfigurationBuilder();
20        global.statePersistence().enable().location(persistentStateLocation);
21        EmbeddedCacheManager cacheManager = TestCacheManagerFactory.createCacheManager(global, new ConfigurationBuilder());
22        cacheManager.getCache();
23        return cacheManager;
24     }
25  
26     public void testStatePreserved() throws Exception {
27        String persistentStateLocation = TestingUtil.tmpDirectory(this.getClass());
28        TestingUtil.recursiveFileRemove(persistentStateLocation);
29  
30        TestingUtil.withCacheManager(new CacheManagerCallable(createCacheManager(persistentStateLocation)) {
31           @Override
32           public void call() throws Exception {
33              Cache<String, String> protoBufMetadaCache = cm.getCache(ProtobufMetadataManagerConstants.PROTOBUF_METADATA_CACHE_NAME);
34              protoBufMetadaCache.put("test.proto", "package X;");
35           }
36        });
37  
38        TestingUtil.withCacheManager(new CacheManagerCallable(createCacheManager(persistentStateLocation)) {
39           @Override
40           public void call() throws Exception {
41              Cache<String, String> protoBufMetadaCache = cm.getCache(ProtobufMetadataManagerConstants.PROTOBUF_METADATA_CACHE_NAME);
42              assertTrue(protoBufMetadaCache.containsKey("test.proto"));
43           }
44        });
45     }
46  }
