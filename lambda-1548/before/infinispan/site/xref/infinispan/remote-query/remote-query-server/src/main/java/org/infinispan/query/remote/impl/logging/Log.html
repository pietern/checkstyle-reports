18     @Message(value = "Unknown field %s in type %s", id = 18001)

20  
21     @Message(value = "Field %s from type %s is not indexed", id = 18002)
22     IllegalArgumentException fieldIsNotIndexed(String fieldName, String fullyQualifiedTypeName);
23  
24     @Message(value = "An exception has occurred during query execution", id = 18003)
