9   import java.io.IOException;

11  import javax.transaction.TransactionManager;

13  /**
14   * <p>Factory for locks obtained in <code>InfinispanDirectory</code>, this factory produces instances of
15   * <code>TransactionalSharedLuceneLock</code>.</p> <p>Usually Lucene acquires the lock when creating an IndexWriter and
16   * releases it when closing it; these open-close operations are mapped to transactions as begin-commit, so all changes
17   * are going to be effective at IndexWriter close. The advantage is that a transaction rollback will be able to undo all
18   * changes applied to the index, but this requires enough memory to hold all the changes until the commit.</p> <p>Using
19   * a TransactionalSharedLuceneLock is not compatible with Lucene's default MergeScheduler: use an in-thread
20   * implementation like SerialMergeScheduler <code>indexWriter.setMergeScheduler( new SerialMergeScheduler()
21   * );</code></p>
22   *
23   * @author Sanne Grinovero
24   * @author Lukasz Moren
25   * @see TransactionalSharedLuceneLock
26   * @see org.apache.lucene.index.SerialMergeScheduler
27   * @since 4.0
28   */
29  public class TransactionalLockFactory extends LockFactory {
30  
31     public static final TransactionalLockFactory INSTANCE = new TransactionalLockFactory();
32  
33     /**
34      * {@inheritDoc}
35      */
36     @Override
37     public TransactionalSharedLuceneLock obtainLock(Directory dir, String lockName) throws IOException {
38        if (!(dir instanceof DirectoryLucene)) {
39           throw new UnsupportedOperationException("TransactionalSharedLuceneLock can only be used with DirectoryLucene, got: " + dir);
40        }
41        DirectoryLucene infinispanDirectory = (DirectoryLucene) dir;
42        Cache cache = infinispanDirectory.getDistLockCache();
43        String indexName = infinispanDirectory.getIndexName();
44        TransactionManager tm = cache.getAdvancedCache().getTransactionManager();
45        if (tm == null) {
46           ComponentStatus status = cache.getAdvancedCache().getComponentRegistry().getStatus();
47           if (status.equals(ComponentStatus.RUNNING)) {
48              throw new CacheException(
49                      "Failed looking up TransactionManager. Check if any transaction manager is associated with Infinispan cache: \'"
50                              + cache.getName() + "\'");
51           } else {
52              throw new CacheException("Failed looking up TransactionManager: the cache is not running");
53           }
54        }
55  
56        TransactionalSharedLuceneLock lock = new TransactionalSharedLuceneLock(cache, indexName, lockName, tm);
57        CommonLockObtainUtils.attemptObtain(lock);
58        return lock;
59     }

61  }
