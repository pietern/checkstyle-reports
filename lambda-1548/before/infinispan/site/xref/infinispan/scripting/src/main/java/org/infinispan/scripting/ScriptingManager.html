1   package org.infinispan.scripting;
2   
3   import javax.script.Bindings;
4   import javax.script.ScriptEngine;
5   
6   import org.infinispan.Cache;
7   import org.infinispan.commons.marshall.Marshaller;
8   import org.infinispan.commons.util.concurrent.NotifyingFuture;
9   
10  /**
11   * ScriptingManager. Defines the operations that can be performed on scripts. Scripts are stored in
12   * a dedicated cache.
13   *
14   * @author Tristan Tarrant
15   * @since 7.2
16   */
17  public interface ScriptingManager {
18  
19     /**
20      * Adds a new named script.
21      *
22      * @param name
23      *           the name of the script. The name should contain an extension identifying its
24      *           language
25      * @param script
26      *           the source of the script
27      * @param mimeType
28      *           the mimeType of the source which selects the {@link ScriptEngine} to use to compile
29      *           the script
30      */
31     void addScript(String name, String script);
32  
33     /**
34      * Removes a script.
35      *
36      * @param name
37      *           the name of the script ro remove
38      */
39     void removeScript(String name);
40  
41     /**
42      * Configures a custom marshaller to be exposed to scripts
43      * as a binding variable
44      *
45      * @param marshaller
46      */
47     void setMarshaller(Marshaller marshaller);
48  
49     /**
50      * @return configured marshaller
51      */
52     Marshaller getMarshaller();
53  
54     /**
55      * Runs a named script
56      *
57      * @param scriptName The name of the script to run. Use {@link #addScript(String, String)} to add a script
58      * @return a {@link NotifyingFuture} which will return the result of the script execution
59      */
60     <T> NotifyingFuture<T> runScript(String scriptName);
61  
62     /**
63      * Runs a named script using the specified user bindings
64      *
65      * @param scriptName The name of the script to run. Use {@link #addScript(String, String)} to add a script
66      * @param parameters The user parameters that will be combined with the system bindings and made available to the script
67      * @return a {@link NotifyingFuture} which will return the result of the script execution
68      */
69     <T> NotifyingFuture<T> runScript(String scriptName, Bindings parameters);
70  
71     /**
72      * Runs a named script using the specified cache as a "driver". Use this form when invoking clustered scripts (i.e. distributed, map/reduce)
73      *
74      * @param scriptName The name of the script to run. Use {@link #addScript(String, String)} to add a script
75      * @param cache A cache to add to the system bindings. The parameter is compulsory if running clustered scripts
76      * @return a {@link NotifyingFuture} which will return the result of the script execution
77      */
78     <T> NotifyingFuture<T> runScript(String scriptName, Cache<?, ?> cache);
79  
80     /**
81      * Runs a named script using the specified user bindings and the specified cache as a "driver". Use this form when invoking clustered scripts (i.e. distributed, map/reduce)
82      *
83      * @param scriptName The name of the script to run. Use {@link #addScript(String, String)} to add a script
84      * @param cache A cache to add to the system bindings. The parameter is compulsory if running clustered scripts
85      * @param parameters The user parameters that will be combined with the system bindings and made available to the script
86      * @return a {@link NotifyingFuture} which will return the result of the script execution
87      */
88     <T> NotifyingFuture<T> runScript(String scriptName, Cache<?, ?> cache, Bindings parameters);
89  
90  }
