12     /**
13      * Get a callback handler for the given mechanism name.
14      *
15      * This method is called each time a mechanism is selected for the connection and the resulting
16      * AuthorizingCallbackHandler will be cached and used multiple times for this connection,
17      * AuthorizingCallbackHandler should either be thread safe or the ServerAuthenticationProvider
18      * should provide a new instance each time called.
19      *
20      * @param mechanismName
21      *           the SASL mechanism to get a callback handler for
22      * @param mechanismProperties
23      *           the mechanism properties that might need to be adjusted to support the specific mechanism / callbackhandler combination
24      * @return the callback handler or {@code null} if the mechanism is not supported
25      */
26     AuthorizingCallbackHandler getCallbackHandler(String mechanismName, Map<String, String> mechanismProperties);
