23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import org.infinispan.configuration.cache.XSiteStateTransferConfigurationBuilder;
26  import org.jboss.as.clustering.infinispan.subsystem.CacheConfigOperationHandlers.CacheConfigAdd;
27  import org.jboss.as.controller.AttributeDefinition;
28  import org.jboss.as.controller.OperationStepHandler;
29  import org.jboss.as.controller.PathElement;
30  import org.jboss.as.controller.ReloadRequiredRemoveStepHandler;
31  import org.jboss.as.controller.ReloadRequiredWriteAttributeHandler;
32  import org.jboss.as.controller.SimpleAttributeDefinition;
33  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
34  import org.jboss.as.controller.SimpleResourceDefinition;
35  import org.jboss.as.controller.registry.AttributeAccess;
36  import org.jboss.as.controller.registry.ManagementResourceRegistration;
37  import org.jboss.dmr.ModelNode;
38  import org.jboss.dmr.ModelType;
39  
40  /**
41   * Cross-Site state transfer resource.
42   *
43   * @author Pedro Ruivo
44   * @since 7.0
45   */
46  public class BackupSiteStateTransferConfigurationResource extends SimpleResourceDefinition {
47  
48     private static final PathElement PATH = PathElement.pathElement(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME);
49  static final SimpleAttributeDefinition STATE_TRANSFER_CHUNK_SIZE = new SimpleAttributeDefinitionBuilder(ModelKeys.CHUNK_SIZE, ModelType.INT, true)
50           .setXmlName(Attribute.CHUNK_SIZE.getLocalName())
51           .setAllowExpression(true)
52           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
53           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_CHUNK_SIZE))
54           .build();
55     static final SimpleAttributeDefinition STATE_TRANSFER_TIMEOUT = new SimpleAttributeDefinitionBuilder(ModelKeys.TIMEOUT, ModelType.LONG, true)
56           .setXmlName(Attribute.TIMEOUT.getLocalName())
57           .setAllowExpression(true)
58           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
59           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_TIMEOUT))
60           .build();
61     static final SimpleAttributeDefinition STATE_TRANSFER_WAIT_TIME = new SimpleAttributeDefinitionBuilder(ModelKeys.WAIT_TIME, ModelType.LONG, true)
62           .setXmlName(Attribute.WAIT_TIME.getLocalName())
63           .setAllowExpression(true)
64           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
65           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_WAIT_TIME))
66           .build();
67     static final SimpleAttributeDefinition STATE_TRANSFER_MAX_RETRIES = new SimpleAttributeDefinitionBuilder(ModelKeys.MAX_RETRIES, ModelType.INT, true)
68           .setXmlName(Attribute.MAX_RETRIES.getLocalName())
69           .setAllowExpression(true)
70           .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
71           .setDefaultValue(new ModelNode().set(XSiteStateTransferConfigurationBuilder.DEFAULT_MAX_RETRIES))
72           .build();
73  
74     static final AttributeDefinition[] ATTRIBUTES = new AttributeDefinition[]{STATE_TRANSFER_CHUNK_SIZE, STATE_TRANSFER_MAX_RETRIES, STATE_TRANSFER_TIMEOUT, STATE_TRANSFER_WAIT_TIME};
75  
76     private final boolean runtimeRegistration;
77  
78     BackupSiteStateTransferConfigurationResource(boolean runtimeRegistration) {
79        super(PATH, new InfinispanResourceDescriptionResolver(ModelKeys.BACKUP, ModelKeys.STATE_TRANSFER), new CacheConfigAdd(ATTRIBUTES), ReloadRequiredRemoveStepHandler.INSTANCE);
80        this.runtimeRegistration = runtimeRegistration;
81     }
82  
83     @Override
84     public void registerAttributes(ManagementResourceRegistration registration) {
85        final OperationStepHandler writeHandler = new ReloadRequiredWriteAttributeHandler(ATTRIBUTES);
86        for (AttributeDefinition attribute : ATTRIBUTES) {
87           registration.registerReadWriteAttribute(attribute, null, writeHandler);
88        }
89     }
90  
91     public boolean isRuntimeRegistration() {
92        return runtimeRegistration;
93     }
94  }
