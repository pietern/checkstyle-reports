68  public class ProtocolResourceRegistrationHandler implements OperationStepHandler, ProtocolMetricsHandler.ProtocolLocator {
69  
70      private final String name;
71      private final PathAddress stackAddress;
72  
73      public ProtocolResourceRegistrationHandler(String name, PathAddress stackAddress) {
74          this.name = name;
75          this.stackAddress = stackAddress;
76      }
77  
78      @Override
79      public Protocol findProtocol(ServiceRegistry registry, PathAddress address) throws ClassNotFoundException, ModuleLoadException {
80          String channelName = address.getElement(address.size() - 2).getValue();
81          String protocolName = address.getElement(address.size() - 1).getValue();
82  
83          ServiceController<?> controller = registry.getService(ChannelServiceName.CHANNEL.getServiceName(channelName));
84          if (controller != null) {
85              Channel channel = (Channel) controller.getValue();
86              if (channel != null) {
87                  controller = registry.getService(ChannelServiceName.FACTORY.getServiceName(channelName));
88                  ChannelFactory factory = (ChannelFactory) controller.getValue();
89                  if (factory != null) {
90                      ProtocolStackConfiguration configuration = factory.getProtocolStackConfiguration();
91                      for (ProtocolConfiguration protocol : configuration.getProtocols()) {
92                          if (protocol.getName().equals(protocolName)) {
93                              Class<? extends Protocol> protocolClass = configuration.getModuleLoader().loadModule(protocol.getModule()).getClassLoader().loadClass(protocol.getProtocolClassName()).asSubclass(Protocol.class);
94                              return channel.getProtocolStack().findProtocol(protocolClass);
95                          }
96                      }
97                  }
98              }
99          }
100         return null;
101     }
102 
103     @Override
104     public void execute(OperationContext context, ModelNode operation) throws OperationFailedException {
105 
106         OverrideDescriptionProvider provider = new OverrideDescriptionProvider() {
107             @Override
108             public Map<String, ModelNode> getAttributeOverrideDescriptions(Locale locale) {
109                 return Collections.emptyMap();
110             }
111 
112             @Override
113             public Map<String, ModelNode> getChildTypeOverrideDescriptions(Locale locale) {
114                 StandardResourceDescriptionResolver resolver = new JGroupsResourceDescriptionResolver();
115                 String description = resolver.getChildTypeDescription(ProtocolResourceDefinition.WILDCARD_PATH.getKey(), locale, resolver.getResourceBundle(locale));
116                 ModelNode result = new ModelNode();
117                 result.get(ModelDescriptionConstants.DESCRIPTION).set(description);
118                 return Collections.singletonMap(ProtocolResourceDefinition.WILDCARD_PATH.getKey(), result);
119             }
120         };
121 
122         Resource resource = context.readResourceForUpdate(PathAddress.EMPTY_ADDRESS);
123         ManagementResourceRegistration registration = context.getResourceRegistrationForUpdate().registerOverrideModel(this.name, provider);
124 
125         Resource stackResource = context.readResourceFromRoot(this.stackAddress, false);
126 
127         for (String name: stackResource.getChildrenNames(TransportResourceDefinition.WILDCARD_PATH.getKey())) {
128             PathAddress transportAddress = this.stackAddress.append(TransportResourceDefinition.pathElement(name));
129             ModelNode transport = context.readResourceFromRoot(transportAddress, false).getModel();
130             ModuleIdentifier module = ModelNodes.asModuleIdentifier(ProtocolResourceDefinition.MODULE.resolveModelAttribute(context, transport));
131             Class<? extends Protocol> transportClass = findProtocolClass(context, name, module);
132             registration.registerSubModel(this.createProtocolResourceDefinition(name, transportClass)).setRuntimeOnly(true);
133             resource.registerChild(ProtocolResourceDefinition.pathElement(name), PlaceholderResource.INSTANCE);
134         }
135 
136         for (String name: stackResource.getChildrenNames(ProtocolResourceDefinition.WILDCARD_PATH.getKey())) {
137             Resource protocolResource = context.readResourceFromRoot(this.stackAddress.append(ProtocolResourceDefinition.pathElement(name)), false);
138             ModuleIdentifier module = ModelNodes.asModuleIdentifier(ProtocolResourceDefinition.MODULE.resolveModelAttribute(context, protocolResource.getModel()));
139             Class<? extends Protocol> protocolClass = findProtocolClass(context, name, module);
140             registration.registerSubModel(this.createProtocolResourceDefinition(name, protocolClass)).setRuntimeOnly(true);
141             resource.registerChild(ProtocolResourceDefinition.pathElement(name), PlaceholderResource.INSTANCE);
142         }
143 
144         if (stackResource.hasChild(RelayResourceDefinition.PATH)) {
145             registration.registerSubModel(this.createProtocolResourceDefinition(RelayConfiguration.PROTOCOL_NAME, RELAY2.class)).setRuntimeOnly(true);
146             resource.registerChild(ProtocolResourceDefinition.pathElement(RelayConfiguration.PROTOCOL_NAME), PlaceholderResource.INSTANCE);
147         }
148 
149         if (stackResource.hasChild(SaslResourceDefinition.PATH)) {
150             registration.registerSubModel(this.createProtocolResourceDefinition(SaslConfiguration.PROTOCOL_NAME, SASL.class)).setRuntimeOnly(true);
151             resource.registerChild(ProtocolResourceDefinition.pathElement(SaslConfiguration.PROTOCOL_NAME), PlaceholderResource.INSTANCE);
152         }
153     }
154 
155     private ResourceDefinition createProtocolResourceDefinition(String protocolName, Class<? extends Protocol> protocolClass) {
156 
157         SimpleResourceDescriptionResolver resolver = new SimpleResourceDescriptionResolver(protocolName, protocolClass.getSimpleName());
158         ResourceBuilder builder = ResourceBuilder.Factory.create(ProtocolResourceDefinition.pathElement(protocolName), resolver);
159         ProtocolMetricsHandler handler = new ProtocolMetricsHandler(this);
160 
161         for (Map.Entry<String, Attribute> entry: ProtocolMetricsHandler.findProtocolAttributes(protocolClass).entrySet()) {
162             String name = entry.getKey();
163             Attribute attribute = entry.getValue();
164             FieldType type = FieldType.valueOf(attribute.getType());
165             resolver.addDescription(name, attribute.getDescription());
166             builder.addMetric(new SimpleAttributeDefinitionBuilder(name, type.getModelType()).setStorageRuntime().build(), handler);
167         }
168 
169         return builder.build();
170     }
171 
172     static Class<? extends Protocol> findProtocolClass(OperationContext context, String protocolName, ModuleIdentifier module) throws OperationFailedException {
173         String className = protocolName;
174         if (module.equals(ProtocolConfiguration.DEFAULT_MODULE) && !protocolName.startsWith(org.jgroups.conf.ProtocolConfiguration.protocol_prefix)) {
175             className = org.jgroups.conf.ProtocolConfiguration.protocol_prefix + "." + protocolName;
176         }
177 
178         try {
179             ModuleLoader loader = (ModuleLoader) context.getServiceRegistry(false).getRequiredService(Services.JBOSS_SERVICE_MODULE_LOADER).getValue();
180             return loader.loadModule(module).getClassLoader().loadClass(className).asSubclass(Protocol.class);
181         } catch (ClassNotFoundException | ModuleLoadException e) {
182             throw JGroupsLogger.ROOT_LOGGER.unableToLoadProtocolClass(className);
183         }
184     }
