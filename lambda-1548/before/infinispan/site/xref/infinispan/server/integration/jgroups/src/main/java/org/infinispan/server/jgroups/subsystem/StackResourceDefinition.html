60      public static final PathElement WILDCARD_PATH = pathElement(PathElement.WILDCARD_VALUE);

63          return PathElement.pathElement(ModelKeys.STACK, name);
64      }

87      static void buildTransformation(ModelVersion version, ResourceTransformationDescriptionBuilder parent) {
88          ResourceTransformationDescriptionBuilder builder = parent.addChildResource(WILDCARD_PATH);
89  
90          if (JGroupsModel.VERSION_3_0_0.requiresTransformation(version)) {
91              // Create legacy "protocols" attributes, which lists protocols by name
92              ResourceTransformer transformer = new ResourceTransformer() {
93                  @Override
94                  public void transformResource(ResourceTransformationContext context, PathAddress address, Resource resource) throws OperationFailedException {
95                      for (String name : resource.getChildrenNames(ProtocolResourceDefinition.WILDCARD_PATH.getKey())) {
96                          resource.getModel().get(PROTOCOLS.getName()).add(name);
97                      }
98                      context.addTransformedResource(PathAddress.EMPTY_ADDRESS, resource).processChildren(resource);
99                  }
100             };
101             builder.setCustomResourceTransformer(transformer);
102         }
103 
104         if (JGroupsModel.VERSION_2_0_0.requiresTransformation(version)) {
105             builder.rejectChildResource(RelayResourceDefinition.PATH);
106         } else {
107             RelayResourceDefinition.buildTransformation(version, builder);
108         }
109         SaslResourceDefinition.buildTransformation(version, builder);
110         TransportResourceDefinition.buildTransformation(version, builder);
111         ProtocolResourceDefinition.buildTransformation(version, builder);
112     }

114     // registration
115     public StackResourceDefinition(boolean allowRuntimeOnlyRegistration) {
116         super(WILDCARD_PATH, new JGroupsResourceDescriptionResolver(ModelKeys.STACK), null, new StackRemoveHandler());
117         this.allowRuntimeOnlyRegistration = allowRuntimeOnlyRegistration;
118     }

120     @Override
121     public void registerOperations(ManagementResourceRegistration registration) {
122         super.registerOperations(registration);
123 
124         OperationDefinition addOperation = new SimpleOperationDefinitionBuilder(ModelDescriptionConstants.ADD, this.getResourceDescriptionResolver())
125                 .addParameter(TRANSPORT) // Deprecated
126                 .addParameter(PROTOCOLS) // Deprecated
127                 .setAttributeResolver(new JGroupsResourceDescriptionResolver(ModelKeys.STACK, ModelDescriptionConstants.ADD))
128                 .build();
129 
130         // Transform deprecated ADD parameters into individual add operations
131         OperationStepHandler addHandler = new StackAddHandler() {
132             @Override
133             protected void populateModel(OperationContext context, ModelNode operation, Resource resource) throws OperationFailedException {
134                 PathAddress address = context.getCurrentAddress();
135                 ModelNode transport = null;
136                 if (operation.hasDefined(TRANSPORT.getName())) {
137                     transport = operation.remove(TRANSPORT.getName());
138                 }
139                 List<ModelNode> protocols = Collections.emptyList();
140                 if (operation.hasDefined(PROTOCOLS.getName())) {
141                     protocols = operation.remove(PROTOCOLS.getName()).asList();
142                 }
143 
144                 if (transport != null) {
145                     String type = ProtocolResourceDefinition.TYPE.resolveModelAttribute(context, transport).asString();
146                     PathAddress transportAddress = address.append(TransportResourceDefinition.pathElement(type));
147                     ModelNode transportOperation = Util.createAddOperation(transportAddress);
148                     for (AttributeDefinition attribute : TransportResourceDefinition.ATTRIBUTES) {
149                         String name = attribute.getName();
150                         if (transport.hasDefined(name)) {
151                             transportOperation.get(name).set(transport.get(name));
152                         }
153                     }
154                     context.addStep(transportOperation, new ReloadRequiredAddStepHandler(TransportResourceDefinition.ATTRIBUTES), OperationContext.Stage.MODEL);
155 
156                     if (transport.hasDefined(ProtocolResourceDefinition.PROPERTIES.getName())) {
157                         for (Property property : operation.get(ProtocolResourceDefinition.PROPERTIES.getName()).asPropertyList()) {
158                             ModelNode propertyOperation = Util.createAddOperation(transportAddress.append(property.getName()));
159                             propertyOperation.set(PropertyResourceDefinition.VALUE.getName()).set(property.getValue());
160                             context.addStep(propertyOperation, new ReloadRequiredAddStepHandler(PropertyResourceDefinition.VALUE), OperationContext.Stage.MODEL);
161                         }
162                     }
163                 }
164                 if (!protocols.isEmpty()) {
165                     for (ModelNode protocol : protocols) {
166                         String type = ProtocolResourceDefinition.TYPE.resolveModelAttribute(context, protocol).asString();
167                         PathAddress protocolAddress = address.append(ProtocolResourceDefinition.pathElement(type));
168                         ModelNode protocolOperation = Util.createAddOperation(protocolAddress);
169                         for (AttributeDefinition attribute : ProtocolResourceDefinition.ATTRIBUTES) {
170                             String name = attribute.getName();
171                             if (protocol.hasDefined(name)) {
172                                 protocolOperation.get(name).set(protocol.get(name));
173                             }
174                         }
175                         context.addStep(protocolOperation, new ReloadRequiredAddStepHandler(ProtocolResourceDefinition.ATTRIBUTES), OperationContext.Stage.MODEL);
176 
177                         if (protocol.hasDefined(ProtocolResourceDefinition.PROPERTIES.getName())) {
178                             for (Property property : operation.get(ProtocolResourceDefinition.PROPERTIES.getName()).asPropertyList()) {
179                                 ModelNode propertyOperation = Util.createAddOperation(protocolAddress.append(property.getName()));
180                                 propertyOperation.set(PropertyResourceDefinition.VALUE.getName()).set(property.getValue());
181                                 context.addStep(propertyOperation, new ReloadRequiredAddStepHandler(PropertyResourceDefinition.VALUE), OperationContext.Stage.MODEL);
182                             }
183                         }
184                     }
185                 }
186             }
187         };
188         registration.registerOperationHandler(addOperation, addHandler);
189 
190         OperationDefinition legacyAddProtocolOperation = new SimpleOperationDefinitionBuilder(ModelKeys.ADD_PROTOCOL, this.getResourceDescriptionResolver())
191                 .setParameters(ProtocolResourceDefinition.SOCKET_BINDING)
192                 .addParameter(ProtocolResourceDefinition.TYPE)
193                 .addParameter(ProtocolResourceDefinition.PROPERTIES)
194                 .setDeprecated(JGroupsModel.VERSION_3_0_0.getVersion())
195                 .build();
196         // Transform legacy /subsystem=jgroups/stack=*:add-protocol() operation -> /subsystem=jgroups/stack=*/protocol=*:add()
197         OperationStepHandler legacyAddProtocolHandler = new OperationStepHandler() {
198             @Override
199             public void execute(OperationContext context, ModelNode operation) {
200                 PathAddress address = context.getCurrentAddress();
201                 String protocol = operation.require(ProtocolResourceDefinition.TYPE.getName()).asString();
202                 PathAddress protocolAddress = address.append(ProtocolResourceDefinition.pathElement(protocol));
203                 ModelNode protocolOperation = Util.createAddOperation(protocolAddress);
204                 for (AttributeDefinition attribute : ProtocolResourceDefinition.ATTRIBUTES) {
205                     String name = attribute.getName();
206                     if (operation.hasDefined(name)) {
207                         protocolOperation.get(name).set(operation.get(name));
208                     }
209                 }
210                 context.addStep(protocolOperation, new ReloadRequiredAddStepHandler(ProtocolResourceDefinition.ATTRIBUTES), OperationContext.Stage.MODEL);
211                 if (operation.hasDefined(ProtocolResourceDefinition.PROPERTIES.getName())) {
212                     for (Property property : operation.get(ProtocolResourceDefinition.PROPERTIES.getName()).asPropertyList()) {
213                         ModelNode addPropertyOperation = Util.createAddOperation(protocolAddress.append(PropertyResourceDefinition.pathElement(property.getName())));
214                         addPropertyOperation.get(PropertyResourceDefinition.VALUE.getName()).set(property.getValue());
215                         context.addStep(addPropertyOperation, new ReloadRequiredAddStepHandler(PropertyResourceDefinition.VALUE), OperationContext.Stage.MODEL);
216                     }
217                 }
218             }
219         };
220         registration.registerOperationHandler(legacyAddProtocolOperation, legacyAddProtocolHandler);
221 
222         OperationDefinition legacyRemoveProtocolOperation = new SimpleOperationDefinitionBuilder(ModelKeys.REMOVE_PROTOCOL, this.getResourceDescriptionResolver())
223                 .setParameters(ProtocolResourceDefinition.TYPE)
224                 .setDeprecated(JGroupsModel.VERSION_3_0_0.getVersion())
225                 .build();
226         // Transform legacy /subsystem=jgroups/stack=*:remove-protocol() operation -> /subsystem=jgroups/stack=*/protocol=*:remove()
227         OperationStepHandler legacyRemoveProtocolHandler = new OperationStepHandler() {
228             @Override
229             public void execute(OperationContext context, ModelNode operation) {
230                 PathAddress address = context.getCurrentAddress();
231                 String protocol = operation.require(ProtocolResourceDefinition.TYPE.getName()).asString();
232                 PathAddress protocolAddress = address.append(ProtocolResourceDefinition.pathElement(protocol));
233                 ModelNode removeOperation = Util.createRemoveOperation(protocolAddress);
234                 context.addStep(removeOperation, ReloadRequiredRemoveStepHandler.INSTANCE, context.getCurrentStage());
235             }
236         };
237         registration.registerOperationHandler(legacyRemoveProtocolOperation, legacyRemoveProtocolHandler);
238 
239         // register export-native-configuration
240         if (this.allowRuntimeOnlyRegistration) {
241             OperationDefinition exportOperation = new SimpleOperationDefinitionBuilder(ModelKeys.EXPORT_NATIVE_CONFIGURATION, this.getResourceDescriptionResolver()).setReplyType(ModelType.STRING).build();
242             registration.registerOperationHandler(exportOperation, new ExportNativeConfiguration());
243         }
244     }
245 
246     @Override
247     public void registerChildren(ManagementResourceRegistration registration) {
248         registration.registerSubModel(new TransportResourceDefinition());
249         registration.registerSubModel(new ProtocolResourceDefinition());
250         registration.registerSubModel(new RelayResourceDefinition());
251         registration.registerSubModel(new SaslResourceDefinition());
252     }
