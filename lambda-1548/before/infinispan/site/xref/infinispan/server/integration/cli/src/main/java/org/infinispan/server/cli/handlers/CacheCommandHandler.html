1   package org.infinispan.server.cli.handlers;
2   
3   import org.infinispan.server.cli.util.CliCommandBuffer;
4   import org.infinispan.server.cli.util.InfinispanUtil;
5   import org.jboss.as.cli.CommandContext;
6   import org.jboss.as.cli.CommandHandler;
7   import org.jboss.as.cli.CommandHandlerProvider;
8   import org.jboss.as.cli.CommandLineException;
9   import org.jboss.dmr.ModelNode;
10  
11  /**
12   * The {@link CacheCommand#CACHE} handler.
13   *
14   * @author Pedro Ruivo
15   * @since 6.1
16   */
17  public class CacheCommandHandler extends CacheNameArgumentCommandHandler {
18  
19     public CacheCommandHandler(CliCommandBuffer buffer) {
20        super(CacheCommand.CACHE, buffer);
21     }
22  
23     @Override
24     protected void printResult(ModelNode result, CommandContext context) throws CommandLineException {
25        InfinispanUtil.changeToCache(context, InfinispanUtil.getCacheInfo(context).getContainer(),
26              cacheName.getValue(context.getParsedCommandLine()));
27        super.printResult(result, context);
28     }
29  
30     public static class Provider implements CommandHandlerProvider {
31  
32        @Override
33        public CommandHandler createCommandHandler(CommandContext ctx) {
34           return new CacheCommandHandler(CliCommandBuffer.INSTANCE);
35        }
36  
37        @Override
38        public boolean isTabComplete() {
39           return true;
40        }
41  
42        @Override
43        public String[] getNames() {
44           return new String[] { CacheCommand.CACHE.getName() };
45        }
46  
47     }
48  }
