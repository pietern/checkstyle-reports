16  public abstract class ClusteringSubsystemTest extends AbstractSubsystemBaseTest {
17      private final String path;
18  
19      protected ClusteringSubsystemTest(String name, Extension extension, String path) {
20          super(name, extension);
21          this.path = path;
22      }
23  
24      @Override
25      protected String getSubsystemXml() throws IOException {
26          return readResource(path);
27      }
28  
29      /**
30       * Need to override the XML comparison in the case where the input xsd and the output xsd differ.
31       *
32       * @param configId   the id of the xml configuration
33       * @param original   the original subsystem xml
34       * @param marshalled the marshalled subsystem xml
35       * @throws Exception
36       */
37      @Override
38      protected void compareXml(String configId, final String original, final String marshalled) throws Exception {
39  
40          final XMLStreamReader originalReader = XMLInputFactory.newInstance().createXMLStreamReader(new StringReader(original));
41          final XMLStreamReader marshalledReader = XMLInputFactory.newInstance().createXMLStreamReader(new StringReader(marshalled));
42  
43          String originalNS = null;
44          if (originalReader.next() == XMLStreamConstants.START_ELEMENT) {
45              originalNS = originalReader.getNamespaceURI();
46          }
47          String marshalledNS = null;
48          if (marshalledReader.next() == XMLStreamConstants.START_ELEMENT) {
49              marshalledNS = marshalledReader.getNamespaceURI();
50          }
51  
52          // only compare if namespace URIs are the same
53          if (originalNS.equals(marshalledNS)) {
54              compareXml(configId, original, marshalled, true);
55          }
56      }
57  
58  
59      @Override
60      protected AdditionalInitialization createAdditionalInitialization() {
61          return new AdditionalInitialization() {
62              @Override
63              protected RunningMode getRunningMode() {
64                  return RunningMode.ADMIN_ONLY;
65              }
66  
67              @Override
68              protected ValidationConfiguration getModelValidationConfiguration() {
69                  return ClusteringSubsystemTest.this.getModelValidationConfiguration();
70              }
71          };
72      }
73  
74      protected abstract ValidationConfiguration getModelValidationConfiguration();
75  }
