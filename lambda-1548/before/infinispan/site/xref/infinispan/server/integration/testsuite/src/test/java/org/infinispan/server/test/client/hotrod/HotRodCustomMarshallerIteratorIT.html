95     private static Archive<?> createFilterMarshallerArchive() throws IOException {
96        String protoFile = Util.read(HotRodCustomMarshallerIteratorIT.class.getResourceAsStream("/sample_bank_account/bank.proto"));
97  
98        return ShrinkWrap.create(JavaArchive.class, "filter-marshaller.jar")
99              // Add custom marshaller classes
100             .addClasses(CustomProtoStreamMarshaller.class, ProtoStreamMarshaller.class, BaseProtoStreamMarshaller.class)
101             .addClasses(HotRodClientException.class, UserMarshaller.class, GenderMarshaller.class, User.class, Address.class)
102                   // Add marshaller dependencies
103             .add(new StringAsset(protoFile), "/sample_bank_account/bank.proto")
104             .add(new StringAsset("Dependencies: org.infinispan.protostream"), "META-INF/MANIFEST.MF")
105                   // Register marshaller
106             .addAsServiceProvider(Marshaller.class, CustomProtoStreamMarshaller.class)
107                   // Add custom filterConverter classes
108             .addClasses(CustomFilterFactory.class, CustomFilterFactory.CustomFilter.class)
109                   // Register custom filterConverterFactory
110             .addAsServiceProvider(KeyValueFilterConverterFactory.class, CustomFilterFactory.class);
111    }

113    @Test
114    public void testIteration() {
115       remoteCache.clear();
116 
117       for (int i = 0; i < 10; i++) {
118          User user = new User();
119          user.setId(i);
120          user.setName("name" + i);
121          user.setSurname("surname" + i);
122          remoteCache.put(i, user);
123       }
124 
125       // Filter-less iteration
126       Map<Object, Object> entryMap = iteratorToMap(remoteCache.retrieveEntries(null, 10));
127       assertEquals(10, entryMap.size());
128       assertEquals(((User) entryMap.get(2)).getName(), "name2");
129 
130       // Iteration with filter
131       entryMap = iteratorToMap(remoteCache.retrieveEntries(TO_STRING_FILTER_CONVERTER_FACTORY_NAME, 10));
132       assertEquals(10, entryMap.size());
133       assertTrue(((String) entryMap.get(2)).startsWith("User{"));
134    }
135 
136    private Map<Object, Object> iteratorToMap(CloseableIterator<Entry<Object, Object>> iterator) {
137       Map<Object, Object> entryMap = new HashMap<>();
138       try {
139          while (iterator.hasNext()) {
140             Entry<Object, Object> next = iterator.next();
141             entryMap.put(next.getKey(), next.getValue());
142          }
143       } finally {
144          assert iterator != null;
145          iterator.close();
146       }
147       return entryMap;
148    }

152    public static class CustomProtoStreamMarshaller extends ProtoStreamMarshaller {
153       public CustomProtoStreamMarshaller() throws IOException {
154          SerializationContext serCtx = getSerializationContext();
155          serCtx.registerProtoFiles(FileDescriptorSource.fromResources(CustomProtoStreamMarshaller.class.getClassLoader(), "/sample_bank_account/bank.proto"));
156          serCtx.registerMarshaller(new UserMarshaller());
157          serCtx.registerMarshaller(new GenderMarshaller());
158       }
159    }

162    @NamedFactory(name = TO_STRING_FILTER_CONVERTER_FACTORY_NAME)
163    public static class CustomFilterFactory implements KeyValueFilterConverterFactory<Integer, User, String> {
164       @Override
165       public KeyValueFilterConverter<Integer, User, String> getFilterConverter() {
166          return new CustomFilter();
167       }

169       public static class CustomFilter extends AbstractKeyValueFilterConverter<Integer, User, String> implements Serializable {
170          @Override
171          public String filterAndConvert(Integer key, User value, Metadata metadata) {
172             return value.toString();
173          }
174       }
