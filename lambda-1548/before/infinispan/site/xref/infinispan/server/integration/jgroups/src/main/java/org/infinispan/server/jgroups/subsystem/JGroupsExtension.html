22  package org.infinispan.server.jgroups.subsystem;
23  
24  import java.security.AccessController;
25  import java.security.PrivilegedAction;
26  import java.util.EnumSet;
27  
28  import org.infinispan.server.jgroups.LogFactory;
29  import org.jboss.as.controller.Extension;
30  import org.jboss.as.controller.ExtensionContext;
31  import org.jboss.as.controller.ModelVersion;
32  import org.jboss.as.controller.SubsystemRegistration;
33  import org.jboss.as.controller.parsing.ExtensionParsingContext;
34  import org.jboss.as.controller.transform.description.TransformationDescription;
35  import org.jgroups.Global;
36  import org.kohsuke.MetaInfServices;
37  
38  /**
39   * Registers the JGroups subsystem.
40   *
41   * @author Paul Ferraro
42   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
43   */
44  @MetaInfServices
45  public class JGroupsExtension implements Extension {
46  
47      public static final String SUBSYSTEM_NAME = "datagrid-jgroups";
48  
49      // Workaround for JGRP-1475
50      // Configure JGroups to use jboss-logging.
51      static {
52          PrivilegedAction<Void> action = new PrivilegedAction<Void>() {
53              @Override
54              public Void run() {
55                  if (System.getProperty(Global.CUSTOM_LOG_FACTORY) == null) {
56                      System.setProperty(Global.CUSTOM_LOG_FACTORY, LogFactory.class.getName());
57                  }
58                  return null;
59              }
60          };
61          AccessController.doPrivileged(action);
62      }
63  
64      /**
65       * {@inheritDoc}
66       * @see org.jboss.as.controller.Extension#initialize(org.jboss.as.controller.ExtensionContext)
67       */
68      @Override
69      public void initialize(ExtensionContext context) {
70          SubsystemRegistration registration = context.registerSubsystem(SUBSYSTEM_NAME, JGroupsModel.CURRENT.getVersion());
71  
72          registration.registerSubsystemModel(new JGroupsSubsystemResourceDefinition(context.isRuntimeOnlyRegistrationValid()));
73          registration.registerXMLElementWriter(new JGroupsSubsystemXMLWriter());
74  
75          if (context.isRegisterTransformers()) {
76              // Register transformers for all but the current model
77              for (JGroupsModel model: EnumSet.complementOf(EnumSet.of(JGroupsModel.CURRENT))) {
78                  ModelVersion version = model.getVersion();
79                  TransformationDescription.Tools.register(JGroupsSubsystemResourceDefinition.buildTransformers(version), registration, version);
80              }
81          }
82      }
83  
84      /**
85       * {@inheritDoc}
86       * @see org.jboss.as.controller.Extension#initializeParsers(org.jboss.as.controller.parsing.ExtensionParsingContext)
87       */
88      @Override
89      public void initializeParsers(ExtensionParsingContext context) {
90          for (JGroupsSchema schema: JGroupsSchema.values()) {
91              context.setSubsystemXmlMapping(SUBSYSTEM_NAME, schema.getNamespaceUri(), new JGroupsSubsystemXMLReader(schema));
92          }
93      }
94  }
