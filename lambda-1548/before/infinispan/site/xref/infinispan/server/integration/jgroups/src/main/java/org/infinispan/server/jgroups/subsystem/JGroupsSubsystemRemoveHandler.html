22  package org.infinispan.server.jgroups.subsystem;
23  
24  import org.jboss.as.controller.AbstractRemoveStepHandler;
25  import org.jboss.as.controller.OperationContext;
26  import org.jboss.as.controller.OperationFailedException;
27  import org.jboss.as.controller.OperationStepHandler;
28  import org.jboss.as.controller.PathAddress;
29  import org.jboss.as.controller.operations.common.Util;
30  import org.jboss.dmr.ModelNode;
31  import org.jboss.dmr.Property;
32  
33  /**
34   * Handler for JGroups subsystem remove operations.
35   *
36   * @author Kabir Khan
37   */
38  public class JGroupsSubsystemRemoveHandler extends AbstractRemoveStepHandler {
39  
40      private final boolean allowRuntimeOnlyRegistration;
41  
42      JGroupsSubsystemRemoveHandler(boolean allowRuntimeOnlyRegistration) {
43          this.allowRuntimeOnlyRegistration = allowRuntimeOnlyRegistration;
44      }
45  
46      @Override
47      protected void performRemove(final OperationContext context, final ModelNode operation, final ModelNode model) throws OperationFailedException {
48          PathAddress address = context.getCurrentAddress();
49  
50          if (model.hasDefined(ChannelResourceDefinition.WILDCARD_PATH.getKey())) {
51              ModelNode channels = model.get(ChannelResourceDefinition.WILDCARD_PATH.getKey());
52              if (channels.isDefined()) {
53                  for (Property channel: channels.asPropertyList()) {
54                      PathAddress channelAddress = address.append(ChannelResourceDefinition.pathElement(channel.getName()));
55                      context.addStep(Util.createRemoveOperation(channelAddress), new ChannelRemoveHandler(this.allowRuntimeOnlyRegistration), OperationContext.Stage.MODEL);
56                  }
57              }
58          }
59  
60          if (model.hasDefined(StackResourceDefinition.WILDCARD_PATH.getKey())) {
61              ModelNode stacks = model.get(StackResourceDefinition.WILDCARD_PATH.getKey());
62              if (stacks.isDefined()) {
63                  for (Property stack: stacks.asPropertyList()) {
64                      PathAddress stackAddress = address.append(StackResourceDefinition.pathElement(stack.getName()));
65                      context.addStep(Util.createRemoveOperation(stackAddress), new StackRemoveHandler(), OperationContext.Stage.MODEL);
66                  }
67              }
68          }
69  
70          context.addStep(operation, new OperationStepHandler() {
71              @Override
72              public void execute(OperationContext context, ModelNode operation) throws OperationFailedException {
73                  JGroupsSubsystemRemoveHandler.super.performRemove(context, operation, model);
74              }
75          }, OperationContext.Stage.MODEL);
76      }
77  
78      @Override
79      protected void performRuntime(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
80          JGroupsSubsystemAddHandler.removeRuntimeServices(context, operation, model);
81      }
82  
83      @Override
84      protected void recoverServices(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
85          JGroupsSubsystemAddHandler.installRuntimeServices(context, operation, model);
86      }
87  }
