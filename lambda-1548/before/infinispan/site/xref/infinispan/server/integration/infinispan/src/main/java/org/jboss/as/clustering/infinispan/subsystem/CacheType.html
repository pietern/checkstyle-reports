23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import java.util.EnumSet;
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  import org.jboss.as.controller.PathElement;
30  import org.jboss.as.controller.descriptions.ResourceDescriptionResolver;
31  
32  /**
33   * Enumerates the supported cache types.
34   * @author Paul Ferraro
35   */
36  enum CacheType {
37      LOCAL(ModelKeys.LOCAL_CACHE, new LocalCacheAdd(), new CacheRemoveHandler()),
38      DISTRIBUTED(ModelKeys.DISTRIBUTED_CACHE, new DistributedCacheAdd(), LOCAL.getRemoveHandler()),
39      REPLICATED(ModelKeys.REPLICATED_CACHE, new ReplicatedCacheAdd(), LOCAL.getRemoveHandler()),
40      INVALIDATION(ModelKeys.INVALIDATION_CACHE, new InvalidationCacheAdd(), LOCAL.getRemoveHandler()),
41      ;
42  
43      private static final Map<String, CacheType> TYPES = new HashMap<>();
44      static {
45          for (CacheType type: values()) {
46              TYPES.put(type.key, type);
47          }
48      }
49  
50      static CacheType forName(String key) {
51          return TYPES.get(key);
52      }
53  
54      private final String key;
55      private final CacheAdd addHandler;
56      private final CacheRemoveHandler removeHandler;
57  
58      private CacheType(String key, CacheAdd addHandler, CacheRemoveHandler removeHandler) {
59          this.key = key;
60          this.addHandler = addHandler;
61          this.removeHandler = removeHandler;
62      }
63  
64      public String getKey() {
65          return key;
66      }
67  
68      public ResourceDescriptionResolver getResourceDescriptionResolver() {
69          return new InfinispanResourceDescriptionResolver(this.key);
70      }
71  
72      public PathElement pathElement() {
73          return pathElement(PathElement.WILDCARD_VALUE);
74      }
75  
76      public PathElement pathElement(String name) {
77          return PathElement.pathElement(this.key, name);
78      }
79  
80      public CacheAdd getAddHandler() {
81          return this.addHandler;
82      }
83  
84      public CacheRemoveHandler getRemoveHandler() {
85          return this.removeHandler;
86      }
87  
88      public boolean hasSharedState() {
89          return EnumSet.of(REPLICATED, DISTRIBUTED).contains(this);
90      }
91  }
