3   import org.infinispan.notifications.cachelistener.filter.CacheEventConverterFactory;
4   import org.infinispan.notifications.cachelistener.filter.CacheEventFilterConverterFactory;
5   import org.jboss.msc.service.ServiceName;

7   import org.jboss.msc.service.StopContext;
8   
9   import static org.infinispan.server.endpoint.EndpointLogger.ROOT_LOGGER;
10  
11  public final class FilterConverterFactoryExtensionProcessor extends AbstractNamedFactoryExtensionProcessor<CacheEventFilterConverterFactory> {
12  
13      public FilterConverterFactoryExtensionProcessor(ServiceName extensionManagerServiceName) {
14          super(extensionManagerServiceName);
15      }
16  
17      @Override
18      public AbstractExtensionManagerService<CacheEventFilterConverterFactory> createService(String name, CacheEventFilterConverterFactory instance) {
19          return new FilterConverterFactoryService(name, instance);
20      }
21  
22      @Override
23      public Class<CacheEventFilterConverterFactory> getServiceClass() {
24          return CacheEventFilterConverterFactory.class;
25      }
26  
27      private static class FilterConverterFactoryService extends AbstractExtensionManagerService<CacheEventFilterConverterFactory> {
28          private FilterConverterFactoryService(String name, CacheEventFilterConverterFactory converterFactory) {
29              super(name, converterFactory);
30          }
31  
32          @Override
33          public void start(StartContext context) {
34              ROOT_LOGGER.debugf("Started combined filter and converter service with name = %s", name);
35              extensionManager.getValue().addFilterConverterFactory(name, extension);
36          }
37  
38          @Override
39          public void stop(StopContext context) {
40              ROOT_LOGGER.debugf("Stopped combined filter and converter service with name = %s", name);
41              extensionManager.getValue().removeFilterConverterFactory(name);
42          }
43  
44          @Override
45          public CacheEventFilterConverterFactory getValue() {
46              return extension;
47          }
48  
49          @Override
50          public String getServiceTypeName() {
51              return "filter-converter-factory";
52          }
53      }
54  
55  }
