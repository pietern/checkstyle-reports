45  /**
46   * Handler for /subsystem=jgroups/channel=*:add() operations
47   * @author Paul Ferraro
48   */
49  public class ChannelAddHandler extends AbstractAddStepHandler {
50  
51      private final boolean allowRuntimeOnlyRegistration;
52  
53      ChannelAddHandler(boolean allowRuntimeOnlyRegistration) {
54          super(ChannelResourceDefinition.ATTRIBUTES);
55          this.allowRuntimeOnlyRegistration = allowRuntimeOnlyRegistration;
56      }
57  
58      @Override
59      protected void populateModel(OperationContext context, ModelNode operation, Resource resource) throws OperationFailedException {
60          super.populateModel(context, operation, resource);
61  
62          // Register runtime resource children for channel protocols
63          if (this.allowRuntimeOnlyRegistration && (context.getRunningMode() == RunningMode.NORMAL)) {
64              String name = context.getCurrentAddressValue();
65              String stack = ModelNodes.asString(ChannelResourceDefinition.STACK.resolveModelAttribute(context, resource.getModel()));
66  
67              PathAddress address = context.getCurrentAddress();
68              PathAddress subsystemAddress = address.subAddress(0, address.size() - 1);
69              // Lookup the name of the default stack if necessary
70              PathAddress stackAddress = subsystemAddress.append(StackResourceDefinition.pathElement((stack != null) ? stack : JGroupsSubsystemResourceDefinition.DEFAULT_STACK.resolveModelAttribute(context, context.readResourceFromRoot(subsystemAddress, false).getModel()).asString()));
71  
72              context.addStep(new ProtocolResourceRegistrationHandler(name, stackAddress), OperationContext.Stage.MODEL);
73          }
74      }
75  
76      @Override
77      protected void performRuntime(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
78          installRuntimeServices(context, operation, model);
79      }
80  
81      static void installRuntimeServices(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
82  
83          String name = context.getCurrentAddressValue();
84          String stack = ModelNodes.asString(ChannelResourceDefinition.STACK.resolveModelAttribute(context, model), ProtocolStackServiceNameFactory.DEFAULT_STACK);
85  
86          ModuleIdentifier module = ModelNodes.asModuleIdentifier(ChannelResourceDefinition.MODULE.resolveModelAttribute(context, model));
87  
88          ServiceTarget target = context.getServiceTarget();
89  
90          // Install channel factory alias
91          new AliasServiceBuilder<>(ChannelServiceName.FACTORY.getServiceName(name), ProtocolStackServiceName.CHANNEL_FACTORY.getServiceName(stack), ChannelFactory.class).build(target).install();
92  
93          // Install channel
94          new ChannelBuilder(name).build(target).install();
95  
96          // Install channel connector
97          new ChannelConnectorBuilder(name).build(target).install();
98  
99          // Install channel jndi binding
100         new BinderServiceBuilder<>(JGroupsBindingFactory.createChannelBinding(name), ChannelServiceName.CHANNEL.getServiceName(name), Channel.class).build(target).install();
101 
102         // Install fork channel factory
103         new ForkChannelFactoryBuilder(name).build(target).install();
104 
105         // Install fork channel factory jndi binding
106         new BinderServiceBuilder<>(JGroupsBindingFactory.createChannelFactoryBinding(name), ProtocolStackServiceName.CHANNEL_FACTORY.getServiceName(name), ChannelFactory.class).build(target).install();
107     }
108 
109     static void removeRuntimeServices(OperationContext context, ModelNode operation, ModelNode model) {
110         String name = context.getCurrentAddressValue();
111 
112         context.removeService(JGroupsBindingFactory.createChannelBinding(name).getBinderServiceName());
113 
114         for (ChannelServiceNameFactory factory : ChannelServiceName.values()) {
115             context.removeService(factory.getServiceName(name));
116         }
117 
118         context.removeService(JGroupsBindingFactory.createChannelFactoryBinding(name).getBinderServiceName());
119         context.removeService(ProtocolStackServiceName.CHANNEL_FACTORY.getServiceName(name));
120     }
121 }
