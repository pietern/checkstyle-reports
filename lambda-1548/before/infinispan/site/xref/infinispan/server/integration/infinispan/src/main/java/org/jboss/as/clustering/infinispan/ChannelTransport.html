23  package org.jboss.as.clustering.infinispan;
24  
25  import java.nio.ByteBuffer;
26  
27  import org.infinispan.remoting.responses.CacheNotFoundResponse;
28  import org.infinispan.remoting.transport.jgroups.CommandAwareRpcDispatcher;
29  import org.infinispan.remoting.transport.jgroups.JGroupsTransport;
30  import org.infinispan.remoting.transport.jgroups.MarshallerAdapter;
31  import org.infinispan.server.jgroups.spi.ChannelFactory;
32  import org.jgroups.Channel;
33  
34  /**
35   * Custom {@link JGroupsTransport} that uses a provided channel.
36   * @author Paul Ferraro
37   */
38  public class ChannelTransport extends JGroupsTransport {
39  
40      final ChannelFactory factory;
41  
42      public ChannelTransport(Channel channel, ChannelFactory factory) {
43          super(channel);
44          this.factory = factory;
45      }
46  
47      @Override
48      protected void initRPCDispatcher() {
49          this.dispatcher = new CommandAwareRpcDispatcher(channel, this, globalHandler, this.getTimeoutExecutor());
50          MarshallerAdapter adapter = new MarshallerAdapter(this.marshaller) {
51              @Override
52              public Object objectFromBuffer(byte[] buffer, int offset, int length) throws Exception {
53                  return ChannelTransport.this.factory.isUnknownForkResponse(ByteBuffer.wrap(buffer, offset, length)) ? CacheNotFoundResponse.INSTANCE : super.objectFromBuffer(buffer, offset, length);
54              }
55          };
56          this.dispatcher.setRequestMarshaller(adapter);
57          this.dispatcher.setResponseMarshaller(adapter);
58          this.dispatcher.start();
59      }
60  
61      @Override
62      protected synchronized void initChannel() {
63          this.channel.setDiscardOwnMessages(false);
64          this.connectChannel = true;
65          this.disconnectChannel = true;
66          this.closeChannel = false;
67      }
68  }
