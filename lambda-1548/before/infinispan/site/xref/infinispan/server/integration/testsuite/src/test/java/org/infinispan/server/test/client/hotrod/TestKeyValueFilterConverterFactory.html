16  @NamedFactory(name = "csv-key-value-filter-converter-factory")
17  public class TestKeyValueFilterConverterFactory implements KeyValueFilterConverterFactory {
18  
19     @Override
20     public KeyValueFilterConverter<String, SampleEntity, Summary> getFilterConverter() {
21        return new SampleKeyValueFilterConverter();
22     }
23  
24     static class SampleKeyValueFilterConverter extends AbstractKeyValueFilterConverter<String, SampleEntity, Summary> implements Serializable {
25        @Override
26        public Summary filterAndConvert(String key, SampleEntity entity, Metadata metadata) {
27           if ("ignore".equals(key)) {
28              return null;
29           }
30           return new Summary(Arrays.asList(entity.getCsvAttributes().split(",")));
31        }
32     }
