21  
22  public class ServerExtensionDependenciesProcessor implements DeploymentUnitProcessor {
23  
24      private static final ModuleIdentifier API = ModuleIdentifier.create("org.infinispan");
25  
26      @Override
27      public void deploy(DeploymentPhaseContext ctx) throws DeploymentUnitProcessingException {
28          if (hasInfinispanExtensions(ctx)) {
29              DeploymentUnit deploymentUnit = ctx.getDeploymentUnit();
30              ModuleSpecification moduleSpec = deploymentUnit.getAttachment(Attachments.MODULE_SPECIFICATION);
31              ModuleLoader moduleLoader = Module.getBootModuleLoader();
32              moduleSpec.addSystemDependency(new ModuleDependency(moduleLoader, API, false, false, false, false));
33          }
34      }
35  
36      private boolean hasInfinispanExtensions(DeploymentPhaseContext ctx) {
37          DeploymentUnit deploymentUnit = ctx.getDeploymentUnit();
38          ServicesAttachment servicesAttachment = deploymentUnit.getAttachment(Attachments.SERVICES);
39          if (servicesAttachment != null) {
40              List<String> filterFactories = servicesAttachment.getServiceImplementations(CacheEventFilterFactory.class.getName());
41              List<String> converterFactories = servicesAttachment.getServiceImplementations(CacheEventConverterFactory.class.getName());
42              List<String> filterConverterFactories = servicesAttachment.getServiceImplementations(CacheEventFilterConverterFactory.class.getName());
43              List<String> marshallers = servicesAttachment.getServiceImplementations(Marshaller.class.getName());
44              List<String> keyValueFilterConverterFactories = servicesAttachment.getServiceImplementations(KeyValueFilterConverterFactory.class.getName());
45              return !filterFactories.isEmpty() || !marshallers.isEmpty()
46                      || !converterFactories.isEmpty() || !filterConverterFactories.isEmpty() || !keyValueFilterConverterFactories.isEmpty();
47          }
48          return false;
49      }
50  
51      @Override
52      public void undeploy(DeploymentUnit context) {
53          // No-op
54      }
55  
56  }
