19  public class BooleShannonExpansionTest {
20  
21     private final EntityNamesResolver entityNamesResolver = new ReflectionEntityNamesResolver(null);
22     private final ReflectionPropertyHelper propertyHelper = new ReflectionPropertyHelper(entityNamesResolver);
23     private final QueryParser queryParser = new QueryParser();
24     private final BooleanFilterNormalizer booleanFilterNormalizer = new BooleanFilterNormalizer();
25  
26     /**
27      * @param jpaQuery        the input JPA query to parse and expand
28      * @param expectedExprStr the expected 'toString()' of the output AST
29      * @param expectedJpa     the expected equivalent JPA of the AST
30      */
31     private void assertExpectedTree(String jpaQuery, String expectedExprStr, String expectedJpa) {
32        FilterParsingResult<Class<?>> parsingResult = queryParser.parseQuery(jpaQuery, FilterProcessingChain.build(entityNamesResolver, propertyHelper, null));
33        BooleanExpr expr = booleanFilterNormalizer.normalize(parsingResult.getWhereClause());
34  
35        BooleShannonExpansion booleShannonExpansion = new BooleShannonExpansion(new BooleShannonExpansion.IndexedFieldProvider() {
36           @Override
37           public boolean isIndexed(List<String> propertyPath) {
38              String last = propertyPath.get(propertyPath.size() - 1);
39              return !last.equals("number") && !last.equals("license");
40           }
41        });
42  
43        expr = booleShannonExpansion.expand(expr);
44        String jpaOut = JPATreePrinter.printTree(parsingResult.getTargetEntityName(), expr, parsingResult.getSortFields());
45        assertEquals(expectedExprStr, expr.toString());
46        assertEquals(expectedJpa, jpaOut);
47     }
48  
49     @Test
50     public void testExpansionNotNeeded() throws Exception {
51        assertExpectedTree("from org.infinispan.objectfilter.test.model.Person p where " +
52                                 "p.surname = 'Adrian' or p.name = 'Nistor'",
53                           "OR(EQUAL(PROP(surname), CONST(Adrian)), EQUAL(PROP(name), CONST(Nistor)))",
54                           "FROM org.infinispan.objectfilter.test.model.Person WHERE (surname = \"Adrian\") OR (name = \"Nistor\")");
55     }
