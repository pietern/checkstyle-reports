9   public final class AndExpr extends BooleanOperatorExpr {
10  
11     public AndExpr(BooleanExpr... children) {
12        super(children);
13     }
14  
15     public AndExpr(List<BooleanExpr> children) {
16        super(children);
17     }
18  
19     @Override
20     public BooleanExpr acceptVisitor(Visitor visitor) {
21        return visitor.visit(this);
22     }
23  
24     @Override
25     public String toString() {
26        StringBuilder sb = new StringBuilder();
27        sb.append("AND(");
28        boolean isFirst = true;
29        for (BooleanExpr c : children) {
30           if (isFirst) {
31              isFirst = false;
32           } else {
33              sb.append(", ");
34           }
35           sb.append(c);
36        }
37        sb.append(")");
38        return sb.toString();
39     }
40  
41     @Override
42     public String toJpaString() {
43        StringBuilder sb = new StringBuilder();
44        for (int i = 0; i < children.size(); i++) {
45           if (i != 0) {
46              sb.append(" AND ");
47           }
48           sb.append("(").append(children.get(i).toJpaString()).append(")");
49        }
50        return sb.toString();
51     }
52  }
