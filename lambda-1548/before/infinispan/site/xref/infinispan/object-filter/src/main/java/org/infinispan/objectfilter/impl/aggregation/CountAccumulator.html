3   /**
4    * COUNT returns a Long greater or equal than 0. Null values are not counted.
5    *
6    * @author anistor@redhat.com
7    * @since 8.0
8    */
9   public final class CountAccumulator extends FieldAccumulator {
10  
11     public CountAccumulator(int pos) {
12        super(pos);
13     }
14  
15     @Override
16     public void init(Object[] row) {
17        row[pos] = row[pos] != null ? 1L : 0L;
18     }
19  
20     @Override
21     public void update(Object[] srcRow, Object[] destRow) {
22        if (srcRow[pos] != null) {
23           destRow[pos] = (Long) destRow[pos] + 1;
24        }
25     }
26  }
