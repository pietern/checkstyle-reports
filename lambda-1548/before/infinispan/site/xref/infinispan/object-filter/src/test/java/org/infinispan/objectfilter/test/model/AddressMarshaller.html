3   import org.infinispan.protostream.MessageMarshaller;
4   
5   import java.io.IOException;
6   
7   /**
8    * @author anistor@redhat.com

11  public class AddressMarshaller implements MessageMarshaller<Address> {
12  
13     @Override
14     public Address readFrom(ProtoStreamReader reader) throws IOException {
15        String street = reader.readString("street");
16        String postCode = reader.readString("postCode");
17  
18        Address address = new Address();
19        address.setStreet(street);
20        address.setPostCode(postCode);
21  
22        return address;
23     }
24  
25     @Override
26     public void writeTo(ProtoStreamWriter writer, Address address) throws IOException {
27        writer.writeString("street", address.getStreet());
28        writer.writeString("postCode", address.getPostCode());
29     }
30  
31     @Override
32     public Class<Address> getJavaClass() {
33        return Address.class;
34     }
35  
36     @Override
37     public String getTypeName() {
38        return "org.infinispan.objectfilter.test.model.Address";
39     }
40  }
