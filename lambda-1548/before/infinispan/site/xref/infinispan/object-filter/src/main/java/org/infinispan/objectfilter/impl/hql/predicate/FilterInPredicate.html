14  class FilterInPredicate extends InPredicate<BooleanExpr> {
15  
16     private final ValueExpr valueExpr;
17  
18     public FilterInPredicate(ValueExpr valueExpr, List<Object> values) {
19        super(valueExpr.toJpaString(), values);
20        this.valueExpr = valueExpr;
21     }
22  
23     @Override
24     public BooleanExpr getQuery() {
25        FilterDisjunctionPredicate predicate = new FilterDisjunctionPredicate();
26  
27        for (Object element : values) {
28           //todo need efficient implementation
29           FilterComparisonPredicate eq = new FilterComparisonPredicate(valueExpr, ComparisonPredicate.Type.EQUALS, element);
30           predicate.add(eq);
31        }
32  
33        return predicate.getQuery();
34     }
35  }
