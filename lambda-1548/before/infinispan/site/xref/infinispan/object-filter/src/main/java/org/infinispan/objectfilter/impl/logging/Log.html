3   import org.hibernate.hql.ParsingException;
4   import org.jboss.logging.BasicLogger;
5   import org.jboss.logging.annotations.Message;
6   import org.jboss.logging.annotations.MessageLogger;
7   
8   /**
9    * Log messages for the object filter parser backend..
10   *
11   * @author anistor@redhat.com
12   * @since 7.0
13   */
14  @MessageLogger(projectCode = "ISPN")
15  public interface Log extends BasicLogger {
16  
17     @Message(id = 402, value = "The type %s has no property named '%s'.")
18     ParsingException getNoSuchPropertyException(String typeName, String propertyName);
19  
20     @Message(id = 404, value = "Unknown alias: %s.")
21     ParsingException getUnknownAliasException(String unknownAlias);
22  
23     @Message(id = 405, value = "Property %2$s can not be selected from type %1$s since it is an embedded entity.")
24     ParsingException getProjectionOfCompleteEmbeddedEntitiesNotSupportedException(String typeName, String propertyPath);
25  
26     @Message(id = 406, value = "The property %s is an embedded entity and does not allow comparison predicates")
27     ParsingException getPredicatesOnCompleteEmbeddedEntitiesNotAllowedException(String propertyPath);
28  
29     @Message(id = 407, value = "Invalid numeric literal '%s'")
30     ParsingException getInvalidNumericLiteralException(String value);
31  
32     @Message(id = 408, value = "Invalid date literal '%s'")
33     ParsingException getInvalidDateLiteralException(String value);
34  
35     @Message(id = 409, value = "Invalid boolean literal '%s'")
36     ParsingException getInvalidBooleanLiteralException(String value);
37  
38     @Message(id = 410, value = "Invalid enum literal '%s' for enum type %s")
39     ParsingException getInvalidEnumLiteralException(String value, String enumType);
40  }
