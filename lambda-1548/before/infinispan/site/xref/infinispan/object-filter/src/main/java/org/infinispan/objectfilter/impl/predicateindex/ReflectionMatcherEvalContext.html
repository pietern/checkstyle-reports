3   import org.infinispan.objectfilter.impl.util.ReflectionHelper;
4   
5   import java.util.Iterator;
6   
7   /**
8    * @author anistor@redhat.com
9    * @since 7.0
10   */
11  public final class ReflectionMatcherEvalContext extends MatcherEvalContext<Class<?>, ReflectionHelper.PropertyAccessor, String> {
12  
13     private final Class<?> entityType;
14  
15     public ReflectionMatcherEvalContext(Object userContext, Object instance, Object eventType) {
16        super(userContext, instance, eventType);
17        entityType = instance.getClass();
18     }
19  
20     @Override
21     public Class<?> getEntityType() {
22        return entityType;
23     }
24  
25     @Override
26     protected void processAttributes(AttributeNode<ReflectionHelper.PropertyAccessor, String> node, Object instance) {
27        for (AttributeNode<ReflectionHelper.PropertyAccessor, String> childAttribute : node.getChildren()) {
28           if (instance == null) {
29              processAttribute(childAttribute, null);
30           } else {
31              ReflectionHelper.PropertyAccessor accessor = childAttribute.getMetadata();
32              if (accessor.isMultiple()) {
33                 Iterator valuesIt = accessor.getValueIterator(instance);
34                 if (valuesIt == null) {
35                    // try to evaluate eventual 'is null' predicates for this null collection
36                    processAttribute(childAttribute, null);
37                 } else {
38                    while (valuesIt.hasNext()) {
39                       Object attributeValue = valuesIt.next();
40                       processAttribute(childAttribute, attributeValue);
41                    }
42                 }
43              } else {
44                 Object attributeValue = accessor.getValue(instance);
45                 processAttribute(childAttribute, attributeValue);
46              }
47           }
48        }
49     }
50  
51     private void processAttribute(AttributeNode<ReflectionHelper.PropertyAccessor, String> attributeNode, Object attributeValue) {
52        attributeNode.processValue(attributeValue, this);
53        processAttributes(attributeNode, attributeValue);
54     }
55  }
