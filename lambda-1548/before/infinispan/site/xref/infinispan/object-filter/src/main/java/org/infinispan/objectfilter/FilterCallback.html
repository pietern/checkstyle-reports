3   /**
4    * A single-method callback that is specified when registering a filter with a Matcher. The {@link #onFilterResult}
5    * method is notified of all instances that were presented to {@link Matcher#match} and successfully matched the filter
6    * associated with this callback. The callback will receive the instance being matched, the projected fields (optional,
7    * if specified) and the 'order by' projections (optional, if specified). The 'order by' projection is an array of
8    * {@link java.lang.Comparable} that can be compared using the {@link java.util.Comparator} provided by {@link
9    * FilterSubscription#getComparator()}.
10   * <p/>
11   * Implementations of this interface are provided by the subscriber and must written is such a way that they can be
12   * invoked from multiple threads simultaneously.
13   *
14   * @author anistor@redhat.com
15   * @since 7.0
16   */
17  public interface FilterCallback {
18  
19     /**
20      * Receives notification that an instance matches the filter.
21      *
22      * @param userContext    the optional user context object passed to {@link Matcher#match}
23      * @param instance       the object being matched
24      * @param eventType      the optional event type discriminator object passed to {@link Matcher#match}
25      * @param projection     the projection, if a projection was requested or {@code null} otherwise
26      * @param sortProjection the projection of fields used for sorting, if sorting was requested or {@code null}
27      *                       otherwise
28      */
29     void onFilterResult(Object userContext, Object instance, Object eventType, Object[] projection, Comparable[] sortProjection);
30  }
