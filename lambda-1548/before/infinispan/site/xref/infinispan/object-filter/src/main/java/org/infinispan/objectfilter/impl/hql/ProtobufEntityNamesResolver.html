1   package org.infinispan.objectfilter.impl.hql;
2   
3   import org.hibernate.hql.ast.spi.EntityNamesResolver;
4   import org.infinispan.protostream.SerializationContext;
5   
6   /**
7    * @author anistor@redhat.com
8    * @since 7.0
9    */
10  public final class ProtobufEntityNamesResolver implements EntityNamesResolver {
11  
12     private final SerializationContext serializationContext;
13  
14     public ProtobufEntityNamesResolver(SerializationContext serializationContext) {
15        this.serializationContext = serializationContext;
16     }
17  
18     @Override
19     public Class<?> getClassFromName(String entityName) {
20        // The EntityNamesResolver of the HQL parser is not nicely designed to handle non-Class type metadata.
21        // Here we return a 'fake' class. It does not matter what we return as long as it is non-null.
22        return serializationContext.canMarshall(entityName) ? Object.class : null;
23     }
24  }
