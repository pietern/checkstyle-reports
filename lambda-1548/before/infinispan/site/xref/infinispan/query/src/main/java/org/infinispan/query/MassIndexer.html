1   package org.infinispan.query;

3   import org.infinispan.commons.util.concurrent.NotifyingFuture;
4   import org.infinispan.jmx.annotations.MBean;
5   import org.infinispan.jmx.annotations.ManagedOperation;
6   
7   /**
8    * Component to rebuild the indexes from the existing data.
9    * This process starts by removing all existing indexes, and then a distributed
10   * task is executed to rebuild the indexes. This task can take a long time to run,
11   * depending on data size, used stores, indexing complexity.
12   *
13   * While reindexing is being performed queries should not be executed as they
14   * will very likely miss many or all results.
15   *
16   * @author Sanne Grinovero <sanne@hibernate.org> (C) 2012 Red Hat Inc.
17   */
18  @MBean(objectName = "MassIndexer",
19        description = "Component that rebuilds the index from the cached data")
20  public interface MassIndexer {
21  
22     @ManagedOperation(description = "Starts rebuilding the index", displayName = "Rebuild index")
23     void start();
24  
25     /**
26      * @return {@link org.infinispan.commons.util.concurrent.NotifyingFuture}
27      */
28     NotifyingFuture<Void> startAsync();
29  
30  }
