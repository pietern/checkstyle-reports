39     public void testContinuousQuery() {
40        QueryFactory<?> qf = Search.getQueryFactory(cache());
41  
42        ContinuousQuery<Object, Object> cq = new ContinuousQuery<Object, Object>(cache());
43  
44        Query query = qf.from(Person.class)
45              .having("age").lte(30)
46              .toBuilder().build();
47  
48        CallCountingCQResultListener<Object, Object> listener = new CallCountingCQResultListener<>();
49        cq.addContinuousQueryListener(query, listener);
50  
51        final Map<Object, Integer> joined = listener.getJoined();
52        final Map<Object, Integer> left = listener.getLeft();
53        
54        assertEquals(0, joined.size());
55        assertEquals(0, left.size());
56  
57        for (int i = 0; i < 2; i++) {
58           Person value = new Person();
59           value.setName("John");
60           value.setAge(40);
61           cache().put(i, value);
62        }
63  
64        assertEquals(0, joined.size());
65        assertEquals(0, left.size());
66  
67        for (int i = 0; i < 10; i++) {
68           Person value = new Person();
69           value.setName("John");
70           value.setAge(i + 25);
71           cache().put(i, value);
72        }
73  
74        assertEquals(6, joined.size());
75        assertEquals(0, left.size());
76        joined.clear();
77  
78        for (int i = 0; i < 2; i++) {
79           Person value = new Person();
80           value.setName("John");
81           value.setAge(i + 40);
82           cache().put(i, value);
83        }
84  
85        assertEquals(0, joined.size());
86        assertEquals(2, left.size());
87        left.clear();
88  
89        for (int i = 4; i < 20; i++) {
90           cache().remove(i);
91        }
92  
93        assertEquals(0, joined.size());
94        assertEquals(2, left.size());
95        left.clear();
96  
97        cache().clear(); //todo [anistor] Does this generate MODIFY instead of REMOVE ???
98  
99        assertEquals(0, joined.size());
100       assertEquals(2, left.size());
101       left.clear();
102 
103       for (int i = 0; i < 2; i++) {
104          Person value = new Person();
105          value.setName("John");
106          value.setAge(i + 20);
107          cache().put(i, value, 5, TimeUnit.MILLISECONDS);
108       }
109 
110       assertEquals(2, joined.size());
111       assertEquals(0, left.size());
112       joined.clear();
113 
114       timeService.advance(6);
115       cache.getAdvancedCache().getExpirationManager().processExpiration();
116       assertEquals(0, cache().size());
117 
118       assertEquals(0, joined.size());
119       assertEquals(2, left.size());
120       left.clear();
121 
122       cq.removeContinuousQueryListener(listener);
123 
124       for (int i = 0; i < 3; i++) {
125          Person value = new Person();
126          value.setName("John");
127          value.setAge(i + 20);
128          cache().put(i, value);
129       }
130 
131       assertEquals(0, joined.size());
132       assertEquals(0, left.size());
133    }

135    public void testTwoSimilarCQ() {
136       QueryFactory<?> qf = Search.getQueryFactory(cache());
137       CallCountingCQResultListener<Object, Object> listener = new CallCountingCQResultListener<>();
138 
139       Query query1 = qf.from(Person.class)
140             .having("age").lte(30)
141             .and().having("name").eq("John").or().having("name").eq("Johny")
142             .toBuilder().build();
143       ContinuousQuery<Object, Object> cq1 = new ContinuousQuery<Object, Object>(cache()); 
144       cq1.addContinuousQueryListener(query1, listener);
145       
146       Query query2 = qf.from(Person.class)
147             .having("age").lte(30).or().having("name").eq("Joe")
148             .toBuilder().build();
149       ContinuousQuery<Object, Object> cq2 = new ContinuousQuery<Object, Object>(cache());
150       cq2.addContinuousQueryListener(query2, listener);
151       
152       final Map<Object, Integer> joined = listener.getJoined();
153       final Map<Object, Integer> left = listener.getLeft();
154 
155       assertEquals(0, joined.size());
156       assertEquals(0, left.size());
157       
158       Person value = new Person();
159       value.setName("John");
160       value.setAge(20);
161       cache().put(1, value);
162       
163       assertEquals(1, joined.size());
164       assertEquals(2, joined.get(1).intValue());
165       assertEquals(0, left.size());
166       joined.clear();
167       
168       value = new Person();
169       value.setName("Joe");
170       cache().replace(1, value);
171       assertEquals(0, joined.size());
172       assertEquals(1, left.size());
173       joined.clear();
174       left.clear();
175       
176       value = new Person();
177       value.setName("Joe");
178       value.setAge(31);
179       cache().replace(1, value);
180       assertEquals(0, joined.size());
181       assertEquals(0, left.size());
182       joined.clear();
183       left.clear();
184       
185       value = new Person();
186       value.setName("John");
187       value.setAge(29);
188       cache().put(1, value);
189       assertEquals(1, joined.size());
190       assertEquals(1, joined.get(1).intValue());
191       assertEquals(0, left.size());
192       joined.clear();
193       left.clear();
194       
195       value = new Person();
196       value.setName("Johny");
197       value.setAge(29);
198       cache().put(1, value);
199       assertEquals(0, joined.size());
200       assertEquals(0, left.size());
201       joined.clear();
202       left.clear();
203       
204       cache().clear();
205       assertEquals(0, joined.size());
206       assertEquals(1, left.size());
207       assertEquals(2, left.get(1).intValue());
208    }
