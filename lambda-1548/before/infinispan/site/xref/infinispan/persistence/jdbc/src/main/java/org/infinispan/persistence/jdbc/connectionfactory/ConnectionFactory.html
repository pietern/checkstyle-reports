1   package org.infinispan.persistence.jdbc.connectionfactory;

3   import org.infinispan.persistence.spi.PersistenceException;
4   import org.infinispan.commons.util.Util;
5   import org.infinispan.persistence.jdbc.configuration.ConnectionFactoryConfiguration;
6   
7   import java.sql.Connection;
8   
9   /**
10   * Defines the functionality a connection factory should implement.
11   *
12   * @author Mircea.Markus@jboss.com
13   */
14  public abstract class ConnectionFactory {
15  
16     /**
17      * Constructs a {@link ConnectionFactory} based on the supplied class
18      * name.
19      */
20     public static ConnectionFactory getConnectionFactory(String connectionFactoryClass, ClassLoader classLoader) throws PersistenceException {
21        return (ConnectionFactory) Util.getInstance(connectionFactoryClass, classLoader);
22     }
23  
24     /**
25      * Constructs a {@link ConnectionFactory} based on the supplied class
26      * name.
27      */
28     public static ConnectionFactory getConnectionFactory(Class<? extends ConnectionFactory> connectionFactoryClass) throws PersistenceException {
29        return Util.getInstance(connectionFactoryClass);
30     }
31  
32     /**
33      * Starts the connection factory. A pooled factory might be create connections here.
34      */
35     public abstract void start(ConnectionFactoryConfiguration factoryConfiguration, ClassLoader classLoader) throws
36                                                                                                              PersistenceException;
37  
38     /**
39      * Closes the connection factory, including all allocated connections etc.
40      */
41     public abstract void stop();
42  
43     /**
44      * Fetches a connection from the factory.
45      */
46     public abstract Connection getConnection() throws PersistenceException;
47  
48     /**
49      * Destroys a connection. Important: null might be passed in, as an valid argument.
50      */
51     public abstract void releaseConnection(Connection conn);
52  }
