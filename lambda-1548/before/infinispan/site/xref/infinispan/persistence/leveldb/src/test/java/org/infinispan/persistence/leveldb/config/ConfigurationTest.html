29  @Test(groups = "unit", testName = "persistence.leveldb.configuration.ConfigurationTest")
30  public class ConfigurationTest extends AbstractInfinispanTest {
31     private String tmpDirectory = TestingUtil.tmpDirectory(this.getClass());
32     private String tmpDataDirectory = tmpDirectory + "/data";
33     private String tmpExpiredDirectory = tmpDirectory + "/expired";
34  
35     @AfterClass(alwaysRun = true)
36     protected void clearTempDir() {
37        TestingUtil.recursiveFileRemove(tmpDirectory);
38     }
39  
40     public void testConfigBuilder() {
41        GlobalConfiguration globalConfig = new GlobalConfigurationBuilder().globalJmxStatistics().transport().defaultTransport().build();
42  
43        Configuration cacheConfig = new ConfigurationBuilder().persistence().addStore(LevelDBStoreConfigurationBuilder.class).location(tmpDataDirectory)
44              .expiredLocation(tmpExpiredDirectory).implementationType(LevelDBStoreConfiguration.ImplementationType.AUTO).build();
45  
46        StoreConfiguration cacheLoaderConfig = cacheConfig.persistence().stores().get(0);
47        assertTrue(cacheLoaderConfig instanceof LevelDBStoreConfiguration);
48        LevelDBStoreConfiguration leveldbConfig = (LevelDBStoreConfiguration) cacheLoaderConfig;
49        assertEquals(tmpDataDirectory, leveldbConfig.location());
50        assertEquals(tmpExpiredDirectory, leveldbConfig.expiredLocation());
51  
52        EmbeddedCacheManager cacheManager = new DefaultCacheManager(globalConfig);
53  
54        cacheManager.defineConfiguration("testCache", cacheConfig);
55  
56        cacheManager.start();
57        Cache<String, String> cache = cacheManager.getCache("testCache");
58  
59        cache.put("hello", "there");
60        cache.stop();
61        cacheManager.stop();
62     }
63  
64     public void testXmlConfig() throws IOException {
65        EmbeddedCacheManager cacheManager = new DefaultCacheManager("config/leveldb-config-" +
66              LevelDBStoreConfiguration.ImplementationType.AUTO.toString().toLowerCase() + ".xml");
67  
68        Cache<String, String> cache = cacheManager.getCache("testCache");
69  
70        cache.put("hello", "there 60 xml");
71        cache.stop();
72        cacheManager.stop();
73  
74        TestingUtil.recursiveFileRemove("/tmp/leveldb/60");
75     }
