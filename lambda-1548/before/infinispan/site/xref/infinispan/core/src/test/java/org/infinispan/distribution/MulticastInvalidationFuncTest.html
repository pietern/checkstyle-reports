12  import static org.testng.AssertJUnit.assertNull;

14  @Test(groups = "functional", testName = "distribution.MulticastInvalidationFuncTest")
15  public class MulticastInvalidationFuncTest extends BaseDistFunctionalTest<Object, String> {
16  	
17  	public static final String KEY1 = "k1";
18  
19     public MulticastInvalidationFuncTest() {
20        sync = true;
21        tx = false;
22        testRetVals = true;
23        l1Threshold = 0;
24     }
25  
26     public void testPut() {
27        initAndTest();
28        Cache<Object, String> nonOwner = getFirstNonOwner(KEY1);
29        Cache<Object, String> owner = getOwners(KEY1)[0];
30        Collection<ReplListener> listeners = new ArrayList<ReplListener>();
31        
32        // Put an object in from a non-owner, this will cause an L1 record to be created there
33        
34        nonOwner.put(KEY1, "foo");
35        assertNull(nonOwner.getAdvancedCache().getDataContainer().get(KEY1));
36        Assert.assertEquals(owner.getAdvancedCache().getDataContainer().get(KEY1).getValue(), "foo");
37        
38        // Check that all nodes (except the one we put to) are notified
39        // but only if the transport is multicast-capable
40        if (owner.getAdvancedCache().getRpcManager().getTransport().isMulticastCapable()) {
41           for (Cache<Object, String> c : getNonOwners(KEY1)) {
42              ReplListener rl = new ReplListener(c);
43              rl.expect(InvalidateL1Command.class);
44              listeners.add(rl);
45              log.debugf("Added nonowner %s", c);
46           }
47        } else {
48           ReplListener rl = new ReplListener(nonOwner);
49           rl.expect(InvalidateL1Command.class);
50           listeners.add(rl);
51        }
52        
53        // Put an object into an owner, this will cause the L1 records for this key to be invalidated
54        owner.put(KEY1, "bar");
55        
56        for (ReplListener rl : listeners) {
57        	rl.waitForRpc();
58        }
59        
60        Assert.assertNull(nonOwner.getAdvancedCache().getDataContainer().get(KEY1));
61     }
62     
63  }
