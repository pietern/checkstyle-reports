135    /**
136     * A default handler for all commands visited.  This is called for any visit method called, unless a visit command is
137     * appropriately overridden.
138     *
139     * @param ctx     invocation context
140     * @param command command to handle
141     * @return return value
142     * @throws Throwable in the case of a problem
143     */
144    protected Object handleDefault(InvocationContext ctx, VisitableCommand command) throws Throwable {
145       return null;
146    }
147 
148    /**
149     * Helper method to visit a collection of VisitableCommands.
150     *
151     * @param ctx     Invocation context
152     * @param toVisit collection of commands to visit
153     * @throws Throwable in the event of problems
154     */
155    public void visitCollection(InvocationContext ctx, Collection<? extends VisitableCommand> toVisit) throws Throwable {
156       for (VisitableCommand command : toVisit) {
157          command.acceptVisitor(ctx, this);
158       }
159    }
