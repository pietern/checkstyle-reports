9   /**
10   * It tests multiple scenarios where a split can happen during a transaction.
11   *
12   * @author Pedro Ruivo
13   * @since 8.0
14   */
15  @Test(groups = "functional", testName = "partitionhandling.PessimisticTxPartitionAndMergeDuringRuntimeTest")
16  public class PessimisticTxPartitionAndMergeDuringRuntimeTest extends BasePessimisticTxPartitionAndMergeTest {
17  
18     private static final Log log = LogFactory.getLog(PessimisticTxPartitionAndMergeDuringRuntimeTest.class);
19  
20     public void testDegradedPartitionWithDiscard() throws Exception {
21        doTest(SplitMode.BOTH_DEGRADED, true, true);
22     }
23  
24     public void testDegradedPartition() throws Exception {
25        doTest(SplitMode.BOTH_DEGRADED, true, false);
26     }
27  
28     public void testOriginatorIsolatedPartitionWithDiscard() throws Exception {
29        doTest(SplitMode.ORIGINATOR_ISOLATED, true, true);
30     }
31  
32     public void testOriginatorIsolatedPartition() throws Exception {
33        doTest(SplitMode.ORIGINATOR_ISOLATED, true, false);
34     }
35  
36     public void testPrimaryOwnerIsolatedPartitionWithDiscard() throws Exception {
37        doTest(SplitMode.PRIMARY_OWNER_ISOLATED, false, true);
38     }
39  
40     public void testPrimaryOwnerIsolatedPartition() throws Exception {
41        doTest(SplitMode.PRIMARY_OWNER_ISOLATED, false, false);
42     }
43  
44     @Override
45     protected void checkLocksDuringPartition(SplitMode splitMode, KeyInfo keyInfo, boolean discard) {
46        switch (splitMode) {
47           case ORIGINATOR_ISOLATED:
48              //lock command never received or ignored because the originator is in another partition
49              //(safe because we will rollback anyway)
50              assertEventuallyNotLocked(cache(1, PESSIMISTIC_TX_CACHE_NAME), keyInfo.getKey1());
51           case BOTH_DEGRADED:
52              //originator rollbacks which is received by cache1
53              assertEventuallyNotLocked(cache(1, PESSIMISTIC_TX_CACHE_NAME), keyInfo.getKey1());
54           case PRIMARY_OWNER_ISOLATED:
55              //originator can commit the transaction, key is eventually unlocked
56              assertEventuallyNotLocked(cache(1, PESSIMISTIC_TX_CACHE_NAME), keyInfo.getKey1());
57        }
58        //lock command is discarded since the transaction originator is missing
59        assertEventuallyNotLocked(cache(2, PESSIMISTIC_TX_CACHE_NAME), keyInfo.getKey2());
60     }
61  
62     @Override
63     protected boolean forceRollback() {
64        return false;
65     }
66  
67     @Override
68     protected Class<? extends TransactionBoundaryCommand> getCommandClass() {
69        return LockControlCommand.class;
70     }
71  
72     @Override
73     protected Log getLog() {
74        return log;
75     }
