11  import java.util.function.Supplier;
12  import java.util.stream.Stream;
13  
14  /**
15   * Local cache stream that returns a stream of cache entries
16   * @param <K> key type of the cache entries
17   * @param <V> value type of the cache entries
18   */
19  public class LocalEntryCacheStream<K, V> extends AbstractLocalCacheStream<CacheEntry<K, V>, K, V> {
20     private final Cache<K, V> cache;
21  
22     /**
23      * @param cache
24      * @param parallel
25      * @param supplier This must be a supplier that provides
26      */
27     public LocalEntryCacheStream(Cache<K, V> cache, boolean parallel, ConsistentHash hash,
28                                  Supplier<Stream<CacheEntry<K, V>>> supplier, ComponentRegistry registry) {
29        super(parallel, hash, supplier, registry);
30        this.cache = cache;
31     }
32  
33     @Override
34     protected Stream<CacheEntry<K, V>> getStream() {
35        Stream<CacheEntry<K, V>> stream;
36        if (keysToFilter != null) {
37           AdvancedCache<K, V> advancedCache = cache.getAdvancedCache();
38           log.tracef("Applying key filtering %s", keysToFilter);
39           stream = (parallel ? keysToFilter.parallelStream() : keysToFilter.stream()).map(k -> advancedCache.getCacheEntry(k)).filter(e -> e != null);
40        } else {
41           stream = supplier.get();
42        }
43        if (segmentsToFilter != null && hash != null) {
44           log.tracef("Applying segment filter %s", segmentsToFilter);
45           stream = stream.filter(k -> segmentsToFilter.contains(hash.getSegment(k.getKey())));
46        }
47        return stream;
48     }
49  
50     @Override
51     protected CloseableIterator<CacheEntry<K, V>> removableIterator(CloseableIterator<CacheEntry<K, V>> realIterator) {
52        return new RemovableCloseableIterator<>(realIterator, cache, CacheEntry::getKey);
53     }
54  
55     LocalValueCacheStream<K, V> toLocalValueCacheStream() {
56        return new LocalValueCacheStream<>(cache, parallel, hash, supplier, registry);
57     }
58  
59     LocalKeyCacheStream<K, V> toLocalKeyCacheStream() {
60        return new LocalKeyCacheStream<>(cache, parallel, hash, supplier, registry);
61     }
62  }
