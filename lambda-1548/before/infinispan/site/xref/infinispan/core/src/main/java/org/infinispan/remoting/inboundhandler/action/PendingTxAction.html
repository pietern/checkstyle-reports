1   package org.infinispan.remoting.inboundhandler.action;

3   import org.infinispan.context.impl.TxInvocationContext;
4   import org.infinispan.interceptors.locking.ClusteringDependentLogic;
5   import org.infinispan.util.concurrent.locks.PendingLockListener;
6   import org.infinispan.util.concurrent.locks.PendingLockManager;
7   import org.infinispan.util.concurrent.locks.PendingLockPromise;
8   import org.infinispan.util.concurrent.locks.RemoteLockCommand;
9   import org.infinispan.util.concurrent.locks.TransactionalRemoteLockCommand;
10  
11  import java.util.List;
12  import java.util.concurrent.CompletableFuture;
13  import java.util.concurrent.TimeUnit;
14  
15  /**
16   * An {@link Action} implementation that check for older topology transactions.
17   * <p/>
18   * This action is ready when no older topology transactions exists or is canceled when the timeout occurs.
19   *
20   * @author Pedro Ruivo
21   * @since 8.0
22   */
23  public class PendingTxAction extends BaseLockingAction implements PendingLockListener {
24  
25     private final PendingLockManager pendingLockManager;
26     private final CompletableFuture<Void> notifier;
27     private volatile PendingLockPromise pendingLockPromise;
28  
29     public PendingTxAction(PendingLockManager pendingLockManager, ClusteringDependentLogic clusteringDependentLogic) {
30        super(clusteringDependentLogic);
31        this.pendingLockManager = pendingLockManager;
32        notifier = new CompletableFuture<>();
33     }
34  
35     @Override
36     protected ActionStatus checking(ActionState state) {
37        PendingLockPromise promise = pendingLockPromise;
38        if (promise != null && promise.isReady() && cas(InternalState.CHECKING, InternalState.MAKE_READY)) {
39           if (promise.hasTimedOut()) {
40              cas(InternalState.MAKE_READY, InternalState.CANCELED);
41              return ActionStatus.CANCELED;
42           } else {
43              state.updateTimeout(promise.getRemainingTimeout());
44              cas(InternalState.MAKE_READY, InternalState.READY);
45              return ActionStatus.READY;
46           }
47        }
48        return ActionStatus.NOT_READY;
49     }
50  
51     @Override
52     protected ActionStatus init(ActionState state) {
53        if (!cas(InternalState.INIT, InternalState.CHECKING)) {
54           return ActionStatus.NOT_READY; //another thread is making progress
55        }
56        final TxInvocationContext<?> context = createContext(state);
57        if (context == null) {
58           //cancel everything. nobody else was able to update the state from checking, so no need to check the CAS
59           cas(InternalState.CHECKING, InternalState.CANCELED);
60           return ActionStatus.CANCELED;
61        }
62        final long timeout = state.getTimeout();
63        final List<Object> keysToLock = getAndUpdateFilteredKeys(state);
64  
65        if (keysToLock.isEmpty()) {
66           //nothing to do. nobody else was able to update the state from checking, so no need to check the CAS
67           cas(InternalState.CHECKING, InternalState.READY);
68           return ActionStatus.READY;
69        }
70  
71        keysToLock.forEach(context::addLockedKey);
72  
73        PendingLockPromise promise = keysToLock.size() == 1 ?
74              pendingLockManager.checkPendingTransactionsForKey(context, keysToLock.get(0), timeout, TimeUnit.MILLISECONDS) :
75              pendingLockManager.checkPendingTransactionsForKeys(context, keysToLock, timeout, TimeUnit.MILLISECONDS);
76  
77        if (promise == PendingLockPromise.NO_OP) {
78           //nothing to do. nobody else was able to update the state from checking, so no need to check the CAS
79           cas(InternalState.CHECKING, InternalState.READY);
80           return ActionStatus.READY;
81        }
82        pendingLockPromise = promise;
83        if (!promise.isReady()) {
84           promise.addListener(this);
85        }
86        return check(state);
87     }
88  
89     @Override
90     public void addListener(ActionListener listener) {
91        notifier.thenRun(listener::onComplete);
92     }
93  
94     @Override
95     public void onReady() {
96        notifier.complete(null);
97     }
98  
99     private TxInvocationContext<?> createContext(ActionState state) {
100       RemoteLockCommand command = state.getCommand();
101       if (command instanceof TransactionalRemoteLockCommand) {
102          return ((TransactionalRemoteLockCommand) command).createContext();
103       }
104       return null;
105    }
106 }
