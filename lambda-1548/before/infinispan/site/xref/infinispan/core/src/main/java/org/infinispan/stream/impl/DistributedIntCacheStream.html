15  /**
16   * Implementation of {@link IntStream} that utilizes a lazily evaluated distributed back end execution.  Note this
17   * class is only able to be created using {@link org.infinispan.CacheStream#mapToInt(ToIntFunction)} or similar
18   * methods from the {@link org.infinispan.CacheStream} interface.
19   */
20  public class DistributedIntCacheStream extends AbstractCacheStream<Integer, IntStream, IntConsumer>
21          implements IntStream {
22     /**
23      * This constructor is to be used only when a user calls a map or flat map method changing to an IntStream
24      * from a CacheStream, Stream, DoubleStream, LongStream etc.
25      * @param other other instance of {@link AbstractCacheStream} to copy details from
26      */
27     protected DistributedIntCacheStream(AbstractCacheStream other) {
28        super(other);
29     }

121    @Override
122    public void forEach(IntConsumer action) {
123       if (!rehashAware) {
124          performOperation(TerminalFunctions.forEachFunction(action), false, (v1, v2) -> null, null);
125       } else {
126          performRehashForEach(action);
127       }
128    }

130    @Override
131    KeyTrackingTerminalOperation<Object, Integer, Object> getForEach(IntConsumer consumer,
132            Supplier<Stream<CacheEntry>> supplier) {
133       if (iteratorOperation == IteratorOperation.FLAT_MAP) {
134          return new ForEachFlatMapIntOperation<>(intermediateOperations, supplier, distributedBatchSize, consumer);
135       } else {
136          return new ForEachIntOperation<>(intermediateOperations, supplier, distributedBatchSize, consumer);
137       }
138    }

140    @Override
141    public void forEachOrdered(IntConsumer action) {
142       if (intermediateType.shouldUseIntermediate(sorted, distinct)) {
143          performIntermediateRemoteOperation(s -> {
144             s.forEachOrdered(action);
145             return null;
146          });
147       } else {
148          forEach(action);
149       }
150    }

152    @Override
153    public int[] toArray() {
154       return performOperation(TerminalFunctions.toArrayIntFunction(), false,
155               (v1, v2) -> {
156                  int[] array = Arrays.copyOf(v1, v1.length + v2.length);
157                  System.arraycopy(v2, 0, array, v1.length, v2.length);
158                  return array;
159               }, null, false);
160    }

162    @Override
163    public int reduce(int identity, IntBinaryOperator op) {
164       return performOperation(TerminalFunctions.reduceFunction(identity, op), true, (i1, i2) -> op.applyAsInt(i1, i2),
165               null);

168    @Override
169    public OptionalInt reduce(IntBinaryOperator op) {
170       Integer result = performOperation(TerminalFunctions.reduceFunction(op), true,
171               (i1, i2) -> {
172                  if (i1 != null) {
173                     if (i2 != null) {
174                        return op.applyAsInt(i1, i2);
175                     }
176                     return i1;
177                  }
178                  return i2;
179               }, null);
180       if (result == null) {
181          return OptionalInt.empty();
182       } else {
183          return OptionalInt.of(result);
184       }
185    }

187    @Override
188    public <R> R collect(Supplier<R> supplier, ObjIntConsumer<R> accumulator, BiConsumer<R, R> combiner) {
189       return performOperation(TerminalFunctions.collectFunction(supplier, accumulator, combiner), true,

191                  combiner.accept(e1, e2);
192                  return e1;

201    @Override
202    public OptionalInt min() {
203       Integer value = performOperation(TerminalFunctions.minIntFunction(), false,
204               (i1, i2) -> {
205                  if (i1 != null) {
206                     if (i2 != null) {
207                        return i1 > i2 ? i2 : i1;
208                     }
209                     return i1;
210                  }
211                  return i2;
212               }, null);
213       if (value == null) {
214          return OptionalInt.empty();
215       } else {
216          return OptionalInt.of(value);
217       }
218    }

220    @Override
221    public OptionalInt max() {
222       Integer value = performOperation(TerminalFunctions.maxIntFunction(), false,
223               (i1, i2) -> {
224                  if (i1 != null) {
225                     if (i2 != null) {
226                        return i1 > i2 ? i1 : i2;

239    @Override
240    public OptionalDouble average() {
241       long[] results = performOperation(TerminalFunctions.averageIntFunction(), true,
242               (a1, a2) -> {
243                  a1[0] += a2[0];
244                  a1[1] += a2[1];
245                  return a1;
246               }, null);
247       if (results[1] > 0) {
248          return OptionalDouble.of((double) results[0] / results[1]);
249       } else {
250          return OptionalDouble.empty();
251       }
252    }

256       // TODO: maybe some day we can do this distributed way, currently IntSummaryStatistics is not serializable

288    @Override
289    public OptionalInt findAny() {
290       Integer result = performOperation(TerminalFunctions.findAnyIntFunction(), false,
291               (i1, i2) -> {
292                  if (i1 != null) {
293                     return i1;
294                  } else {
295                     return i2;
296                  }
297               }, a -> a != null);
298       if (result != null) {
299          return OptionalInt.of(result);
300       } else {
301          return OptionalInt.empty();
302       }
303    }

305    @Override
306    public PrimitiveIterator.OfInt iterator() {
307       if (intermediateType.shouldUseIntermediate(sorted, distinct)) {
308          IntStream stream = performIntermediateRemoteOperation(Function.identity());
309          return stream.iterator();
310       } else {
311          return remoteIterator();
312       }
313    }

315    PrimitiveIterator.OfInt remoteIterator() {
316       // TODO: need to add in way to not box these later
317       // Since this is a remote iterator we have to add it to the remote intermediate operations queue
318       intermediateOperations.add(BoxedIntOperation.getInstance());
319       DistributedCacheStream<Integer> stream = new DistributedCacheStream<>(this);
320       Iterator<Integer> iterator = stream.remoteIterator();
321       return new IntegerIteratorToPrimitiveInteger(iterator);
322    }
