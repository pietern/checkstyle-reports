3   /**
4    * Extends {@link Key2StringMapper} and allows a bidirectional transformation between keys and Strings.  Note that the
5    * object instance created by {@link #getKeyMapping(String)} is guaranteed to be <i>equal</i> to the original object
6    * used to generate the String, but not necessarily the same object reference.
7    * <p />
8    * The following condition should be satisfied by implementations of this interface:
9    * <code>
10   *   assert key.equals(mapper.getKeyMapping(mapper.getStringMapping(key)));
11   * </code>
12   *
13   * @author Mircea.Markus@jboss.com
14   * @author Manik Surtani
15   * @since 4.1
16   */
17  public interface TwoWayKey2StringMapper extends Key2StringMapper {
18     /**
19      * Maps a String back to its original key
20      * @param stringKey string representation of a key
21      * @return an object instance that is <i>equal</i> to the original object used to create the key mapping.
22      */
23     Object getKeyMapping(String stringKey);
24  }
