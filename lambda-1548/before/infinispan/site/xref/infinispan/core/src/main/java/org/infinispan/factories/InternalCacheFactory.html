27  /**
28   * An internal factory for constructing Caches.  Used by the {@link DefaultCacheManager}, this is not intended as public
29   * API.
30   * <p/>
31   * This is a special instance of a {@link AbstractComponentFactory} which contains bootstrap information for the {@link
32   * ComponentRegistry}.
33   * <p/>
34   *
35   * @author <a href="mailto:manik@jboss.org">Manik Surtani (manik@jboss.org)</a>
36   * @since 4.0
37   */
38  public class InternalCacheFactory<K, V> extends AbstractNamedCacheComponentFactory {
39     /**
40      * This implementation clones the configuration passed in before using it.
41      *
42      *
43      * @param configuration           to use
44      * @param globalComponentRegistry global component registry to attach the cache to
45      * @param cacheName               name of the cache
46      * @return a cache
47      * @throws CacheConfigurationException if there are problems with the cfg
48      */
49     public Cache<K, V> createCache(Configuration configuration,
50                                    GlobalComponentRegistry globalComponentRegistry,
51                                    String cacheName) throws CacheConfigurationException {
52        try {
53           if (configuration.simpleCache()) {
54              return createSimpleCache(configuration, globalComponentRegistry, cacheName);
55           } else {
56              return createAndWire(configuration, globalComponentRegistry, cacheName);
57           }
58        }
59        catch (CacheConfigurationException ce) {
60           throw ce;
61        }
62        catch (RuntimeException re) {
63           throw re;
64        }
65        catch (Exception e) {
66           throw new RuntimeException(e);
67        }
68     }
69  
70     protected AdvancedCache<K, V> createAndWire(Configuration configuration, GlobalComponentRegistry globalComponentRegistry,
71                                                 String cacheName) throws Exception {
72        AdvancedCache<K, V> cache = new CacheImpl<K, V>(cacheName);
73        bootstrap(cacheName, cache, configuration, globalComponentRegistry);
74        return cache;
75     }
76  
77     private AdvancedCache<K, V> createSimpleCache(Configuration configuration, GlobalComponentRegistry globalComponentRegistry,
78                                           String cacheName) {
79        AdvancedCache<K, V> cache;
80  
81        JMXStatisticsConfiguration jmxStatistics = configuration.jmxStatistics();
82        boolean statisticsAvailable = jmxStatistics != null && jmxStatistics.available();
83        if (statisticsAvailable) {
84           cache = new StatsCollectingCache<>(cacheName);
85        } else {
86           cache = new SimpleCacheImpl<>(cacheName);
87        }
88        this.configuration = configuration;
89        componentRegistry = new ComponentRegistry(cacheName, configuration, cache, globalComponentRegistry, globalComponentRegistry.getClassLoader()) {
90           @Override
91           public void cacheComponents() {
92              cacheMarshaler = getOrCreateComponent(StreamingMarshaller.class, KnownComponentNames.CACHE_MARSHALLER);
93              getOrCreateComponent(ExpirationManager.class);
94           }
95        };
96  
97        if (statisticsAvailable) {
98           componentRegistry.registerComponent(new StatsCollector.Factory(), StatsCollector.Factory.class);
99        }
100       componentRegistry.registerComponent(new CacheJmxRegistration(), CacheJmxRegistration.class.getName(), true);
101       componentRegistry.registerComponent(new ClusterEventManagerStub<K, V>(), ClusterEventManager.class);
102       componentRegistry.registerComponent(new PassivationManagerStub(), PassivationManager.class);
103       componentRegistry.registerComponent(new ActivationManagerStub(), ActivationManager.class);
104       componentRegistry.registerComponent(new PersistenceManagerStub(), PersistenceManager.class);
105       componentRegistry.registerComponent(new RollingUpgradeManager(), RollingUpgradeManager.class.getName(), true);
106       componentRegistry.registerComponent(cache, Cache.class.getName(), true);
107       return cache;
108    }
109 
110 
111    /**
112     * Bootstraps this factory with a Configuration and a ComponentRegistry.
113     */
114    private void bootstrap(String cacheName, AdvancedCache<?, ?> cache, Configuration configuration,
115                           GlobalComponentRegistry globalComponentRegistry) {
116       this.configuration = configuration;
117 
118       // injection bootstrap stuff
119       componentRegistry = new ComponentRegistry(cacheName, configuration, cache, globalComponentRegistry, globalComponentRegistry.getClassLoader());
120 
121       /*
122          --------------------------------------------------------------------------------------------------------------
123          This is where the bootstrap really happens.  Registering the cache in the component registry will cause
124          the component registry to look at the cache's @Inject methods, and construct various components and their
125          dependencies, in turn.
126          --------------------------------------------------------------------------------------------------------------
127        */
128       componentRegistry.registerComponent(cache, Cache.class.getName(), true);
129       componentRegistry.registerComponent(new CacheJmxRegistration(), CacheJmxRegistration.class.getName(), true);
130       if (configuration.transaction().transactionMode().isTransactional() && configuration.transaction().recovery().enabled()) {
131          componentRegistry.registerComponent(new RecoveryAdminOperations(), RecoveryAdminOperations.class.getName(), true);
132       }
133       if (configuration.sites().hasEnabledBackups()) {
134          componentRegistry.registerComponent(new XSiteAdminOperations(), XSiteAdminOperations.class.getName(), true);
135       }
136       // The RollingUpgradeManager should always be added so it is registered in JMX.
137       componentRegistry.registerComponent(new RollingUpgradeManager(), RollingUpgradeManager.class.getName(), true);
138    }
139 
140    @Override
141    public <T> T construct(Class<T> componentType) {
142       throw new UnsupportedOperationException("Should never be invoked - this is a bootstrap factory.");
143    }
