20  /**
21   * It tests multiple scenarios where a split can happen during a transaction.
22   *
23   * @author Pedro Ruivo
24   * @since 8.0
25   */
26  public abstract class BasePessimisticTxPartitionAndMergeTest extends BaseTxPartitionAndMergeTest {
27  
28     protected static final String PESSIMISTIC_TX_CACHE_NAME = "pes-cache";
29  
30     @Override
31     protected void createCacheManagers() throws Throwable {
32        super.createCacheManagers();
33        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC);
34        builder.clustering().partitionHandling().enabled(true);
35        builder.transaction().lockingMode(LockingMode.PESSIMISTIC).transactionMode(TransactionMode.TRANSACTIONAL).transactionManagerLookup(new DummyTransactionManagerLookup());
36        defineConfigurationOnAllManagers(PESSIMISTIC_TX_CACHE_NAME, builder);
37     }
38  
39     protected abstract void checkLocksDuringPartition(SplitMode splitMode, KeyInfo keyInfo, boolean discard);
40  
41     protected abstract boolean forceRollback();
42  
43     protected abstract Class<? extends TransactionBoundaryCommand> getCommandClass();
44  
45     protected void doTest(final SplitMode splitMode, boolean txFail, boolean discard) throws Exception {
46        waitForClusterToForm(PESSIMISTIC_TX_CACHE_NAME);
47  
48        final KeyInfo keyInfo = createKeys(PESSIMISTIC_TX_CACHE_NAME);
49        final Cache<Object, String> originator = cache(0, PESSIMISTIC_TX_CACHE_NAME);
50        final FilterCollection filterCollection = createFilters(PESSIMISTIC_TX_CACHE_NAME, discard, getCommandClass(), splitMode);
51  
52        Future<Void> put = fork(() -> {
53           final TransactionManager transactionManager = originator.getAdvancedCache().getTransactionManager();
54           transactionManager.begin();
55           final Transaction tx = transactionManager.getTransaction();
56           try {
57              keyInfo.putFinalValue(originator);
58              if (forceRollback()) {
59                 tx.setRollbackOnly();
60              }
61           } finally {
62              transactionManager.commit();
63           }
64           return null;
65        });
66  
67        filterCollection.await(30, TimeUnit.SECONDS);
68        splitMode.split(this);
69        filterCollection.unblock();
70  
71        try {
72           put.get();
73           assertFalse(txFail);
74        } catch (ExecutionException e) {
75           assertTrue(txFail);
76        }
77  
78        checkLocksDuringPartition(splitMode, keyInfo, discard);
79  
80        mergeCluster();
81        finalAsserts(PESSIMISTIC_TX_CACHE_NAME, keyInfo, txFail ? INITIAL_VALUE : FINAL_VALUE);
82     }
83  }
