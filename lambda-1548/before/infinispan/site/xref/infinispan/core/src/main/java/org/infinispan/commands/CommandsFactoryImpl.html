336     * @param isRemote true if the command is deserialized and is executed remote.
337     */
338    @Override
339    public void initializeReplicableCommand(ReplicableCommand c, boolean isRemote) {
340       if (c == null) return;
341       switch (c.getCommandId()) {
342          case PutKeyValueCommand.COMMAND_ID:
343             ((PutKeyValueCommand) c).init(notifier, configuration);
344             break;
345          case ReplaceCommand.COMMAND_ID:
346             ((ReplaceCommand) c).init(notifier, configuration);
347             break;
348          case PutMapCommand.COMMAND_ID:
349             ((PutMapCommand) c).init(notifier);
350             break;
351          case RemoveCommand.COMMAND_ID:
352             ((RemoveCommand) c).init(notifier, configuration);
353             break;
354          case MultipleRpcCommand.COMMAND_ID:
355             MultipleRpcCommand rc = (MultipleRpcCommand) c;
356             rc.init(interceptorChain, icf);
357             if (rc.getCommands() != null)
358                for (ReplicableCommand nested : rc.getCommands()) {
359                   initializeReplicableCommand(nested, false);
360                }
361             break;
362          case SingleRpcCommand.COMMAND_ID:
363             SingleRpcCommand src = (SingleRpcCommand) c;
364             src.init(interceptorChain, icf);
365             if (src.getCommand() != null)
366                initializeReplicableCommand(src.getCommand(), false);
367 
368             break;
369          case InvalidateCommand.COMMAND_ID:
370             InvalidateCommand ic = (InvalidateCommand) c;
371             ic.init(notifier, configuration);
372             break;
373          case InvalidateL1Command.COMMAND_ID:
374             InvalidateL1Command ilc = (InvalidateL1Command) c;
375             ilc.init(configuration, distributionManager, notifier, dataContainer);
376             break;
377          case PrepareCommand.COMMAND_ID:
378          case VersionedPrepareCommand.COMMAND_ID:
379          case TotalOrderNonVersionedPrepareCommand.COMMAND_ID:
380          case TotalOrderVersionedPrepareCommand.COMMAND_ID:
381             PrepareCommand pc = (PrepareCommand) c;
382             pc.init(interceptorChain, icf, txTable);
383             pc.initialize(notifier, recoveryManager);
384             if (pc.getModifications() != null)
385                for (ReplicableCommand nested : pc.getModifications())  {
386                   initializeReplicableCommand(nested, false);
387                }
388             pc.markTransactionAsRemote(isRemote);
389             if (configuration.deadlockDetection().enabled() && isRemote) {
390                DldGlobalTransaction transaction = (DldGlobalTransaction) pc.getGlobalTransaction();
391                transaction.setLocksHeldAtOrigin(pc.getAffectedKeys());
392             }
393             break;
394          case CommitCommand.COMMAND_ID:
395          case VersionedCommitCommand.COMMAND_ID:
396          case TotalOrderCommitCommand.COMMAND_ID:
397          case TotalOrderVersionedCommitCommand.COMMAND_ID:
398             CommitCommand commitCommand = (CommitCommand) c;
399             commitCommand.init(interceptorChain, icf, txTable);
400             commitCommand.markTransactionAsRemote(isRemote);
401             break;
402          case RollbackCommand.COMMAND_ID:
403          case TotalOrderRollbackCommand.COMMAND_ID:
404             RollbackCommand rollbackCommand = (RollbackCommand) c;
405             rollbackCommand.init(interceptorChain, icf, txTable);
406             rollbackCommand.markTransactionAsRemote(isRemote);
407             break;
408          case ClearCommand.COMMAND_ID:
409             ClearCommand cc = (ClearCommand) c;
410             cc.init(notifier, dataContainer);
411             break;
412          case ClusteredGetCommand.COMMAND_ID:
413             ClusteredGetCommand clusteredGetCommand = (ClusteredGetCommand) c;
414             clusteredGetCommand.initialize(icf, this, entryFactory,
415                   interceptorChain, distributionManager, txTable,
416                   configuration.dataContainer().keyEquivalence());
417             break;
418          case LockControlCommand.COMMAND_ID:
419             LockControlCommand lcc = (LockControlCommand) c;
420             lcc.init(interceptorChain, icf, txTable);
421             lcc.markTransactionAsRemote(isRemote);
422             if (configuration.deadlockDetection().enabled() && isRemote) {
423                DldGlobalTransaction gtx = (DldGlobalTransaction) lcc.getGlobalTransaction();
424                RemoteTransaction transaction = txTable.getRemoteTransaction(gtx);
425                if (transaction != null) {
426                   if (!configuration.clustering().cacheMode().isDistributed()) {
427                      Set<Object> keys = txTable.getLockedKeysForRemoteTransaction(gtx);
428                      GlobalTransaction gtx2 = transaction.getGlobalTransaction();
429                      ((DldGlobalTransaction) gtx2).setLocksHeldAtOrigin(keys);
430                      gtx.setLocksHeldAtOrigin(keys);
431                   } else {
432                      GlobalTransaction gtx2 = transaction.getGlobalTransaction();
433                      ((DldGlobalTransaction) gtx2).setLocksHeldAtOrigin(gtx.getLocksHeldAtOrigin());
434                   }
435                }
436             }
437             break;
438          case StateRequestCommand.COMMAND_ID:
439             ((StateRequestCommand) c).init(stateProvider);
440             break;
441          case StateResponseCommand.COMMAND_ID:
442             ((StateResponseCommand) c).init(stateConsumer);
443             break;
444          case GetInDoubtTransactionsCommand.COMMAND_ID:
445             GetInDoubtTransactionsCommand gptx = (GetInDoubtTransactionsCommand) c;
446             gptx.init(recoveryManager);
447             break;
448          case TxCompletionNotificationCommand.COMMAND_ID:
449             TxCompletionNotificationCommand ftx = (TxCompletionNotificationCommand) c;
450             ftx.init(txTable, lockManager, recoveryManager, stateTransferManager);
451             break;
452          case MapCombineCommand.COMMAND_ID:
453             MapCombineCommand mrc = (MapCombineCommand)c;
454             mrc.init(mapReduceManager);
455             break;
456          case ReduceCommand.COMMAND_ID:
457             ReduceCommand reduceCommand = (ReduceCommand)c;
458             reduceCommand.init(mapReduceManager);
459             break;
460          case DistributedExecuteCommand.COMMAND_ID:
461             DistributedExecuteCommand dec = (DistributedExecuteCommand)c;
462             dec.init(cache);
463             break;
464          case GetInDoubtTxInfoCommand.COMMAND_ID:
465             GetInDoubtTxInfoCommand gidTxInfoCommand = (GetInDoubtTxInfoCommand)c;
466             gidTxInfoCommand.init(recoveryManager);
467             break;
468          case CompleteTransactionCommand.COMMAND_ID:
469             CompleteTransactionCommand ccc = (CompleteTransactionCommand)c;
470             ccc.init(recoveryManager);
471             break;
472          case ApplyDeltaCommand.COMMAND_ID:
473             break;
474          case CreateCacheCommand.COMMAND_ID:
475             CreateCacheCommand createCacheCommand = (CreateCacheCommand)c;
476             createCacheCommand.init(cache.getCacheManager());
477             break;
478          case XSiteAdminCommand.COMMAND_ID:
479             XSiteAdminCommand xSiteAdminCommand = (XSiteAdminCommand)c;
480             xSiteAdminCommand.init(backupSender);
481             break;
482          case CancelCommand.COMMAND_ID:
483             CancelCommand cancelCommand = (CancelCommand)c;
484             cancelCommand.init(cancellationService);
485             break;
486          case XSiteStateTransferControlCommand.COMMAND_ID:
487             XSiteStateTransferControlCommand xSiteStateTransferControlCommand = (XSiteStateTransferControlCommand) c;
488             xSiteStateTransferControlCommand.initialize(xSiteStateProvider, xSiteStateConsumer, xSiteStateTransferManager);
489             break;
490          case XSiteStatePushCommand.COMMAND_ID:
491             XSiteStatePushCommand xSiteStatePushCommand = (XSiteStatePushCommand) c;
492             xSiteStatePushCommand.initialize(xSiteStateConsumer);
493             break;
494          case EntryRequestCommand.COMMAND_ID:
495             EntryRequestCommand entryRequestCommand = (EntryRequestCommand) c;
496             entryRequestCommand.init(entryRetriever);
497             break;
498          case EntryResponseCommand.COMMAND_ID:
499             EntryResponseCommand entryResponseCommand = (EntryResponseCommand) c;
500             entryResponseCommand.init(entryRetriever);
501             break;
502          case GetKeysInGroupCommand.COMMAND_ID:
503             GetKeysInGroupCommand getKeysInGroupCommand = (GetKeysInGroupCommand) c;
504             getKeysInGroupCommand.setGroupManager(groupManager);
505             break;
506          case ClusteredGetAllCommand.COMMAND_ID:
507             ClusteredGetAllCommand clusteredGetAllCommand = (ClusteredGetAllCommand) c;
508             clusteredGetAllCommand.init(icf, this, entryFactory, interceptorChain, txTable,
509                   configuration.dataContainer().keyEquivalence());
510             break;
511          case StreamRequestCommand.COMMAND_ID:
512             StreamRequestCommand streamRequestCommand = (StreamRequestCommand) c;
513             streamRequestCommand.inject(localStreamManager);
514             break;
515          case StreamResponseCommand.COMMAND_ID:
516             StreamResponseCommand streamResponseCommand = (StreamResponseCommand) c;
517             streamResponseCommand.inject(clusterStreamManager);
518             break;
519          case StreamSegmentResponseCommand.COMMAND_ID:
520             StreamSegmentResponseCommand streamSegmentResponseCommand = (StreamSegmentResponseCommand) c;
521             streamSegmentResponseCommand.inject(clusterStreamManager);
522             break;
523          case RemoveExpiredCommand.COMMAND_ID:
524             RemoveExpiredCommand removeExpiredCommand = (RemoveExpiredCommand) c;
525             removeExpiredCommand.init(notifier, configuration);
526             break;
527          default:
528             ModuleCommandInitializer mci = moduleCommandInitializers.get(c.getCommandId());
529             if (mci != null) {
530                mci.initializeReplicableCommand(c, isRemote);
531             } else {
532                if (trace) log.tracef("Nothing to initialize for command: %s", c);
533             }
534       }
535    }

766    private ValueMatcher getValueMatcher(Object o) {
767       SerializeFunctionWith ann = o.getClass().getAnnotation(SerializeFunctionWith.class);
768       if (ann != null)
769          return ValueMatcher.valueOf(ann.valueMatcher().toString());
770 
771       Externalizer ext = externalizerTable.getExternalizer(o);
772       if (ext != null && ext instanceof LambdaExternalizer)
773          return ValueMatcher.valueOf(((LambdaExternalizer) ext).valueMatcher(o).toString());
774 
775       return ValueMatcher.MATCH_ALWAYS;
776    }
