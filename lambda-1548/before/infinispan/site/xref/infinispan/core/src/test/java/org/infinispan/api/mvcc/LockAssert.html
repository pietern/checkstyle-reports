10  import static org.testng.AssertJUnit.assertEquals;
11  import static org.testng.AssertJUnit.assertFalse;
12  import static org.testng.AssertJUnit.assertTrue;
13  
14  /**
15   * Helper class to assert lock status in MVCC
16   *
17   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
18   */
19  public class LockAssert {
20     public static void assertLocked(Object key, LockManager lockManager, InvocationContextContainer icc) {
21        assertTrue("" + key + " not locked!", lockManager.isLocked(key));
22        InvocationContext context = icc.getInvocationContext(true);
23        if (context != null) {
24           assertTrue("" + key + " lock not recorded!", context.hasLockedKey(key));
25        }
26     }
27  
28     public static void assertNotLocked(Object key, InvocationContextContainer icc) {
29        // can't rely on the negative test since other entries may share the same lock with lock striping.
30        InvocationContext context = icc.getInvocationContext(true);
31        if (context != null) {
32           assertFalse("" + key + " lock recorded!", context.hasLockedKey(key));
33        }
34     }
35  
36     public static void assertNoLocks(LockManager lockManager) {
37        LockContainer lc = TestingUtil.extractField(lockManager, "lockContainer");
38        assertEquals("Stale locks exist! NumLocksHeld is " + lc.getNumLocksHeld() + " and lock info is " + lockManager.printLockInfo(),
39              0, lc.getNumLocksHeld());
40     }
41  
42     public static void assertNoLocks(Cache cache) {
43        LockManager lockManager = TestingUtil.extractComponentRegistry(cache).getComponent(LockManager.class);
44        assertNoLocks(lockManager);
45     }
46  }
