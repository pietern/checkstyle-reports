3   import org.testng.annotations.Test;

5   import static org.testng.AssertJUnit.assertEquals;
6   import static org.testng.AssertJUnit.assertTrue;

8   /**
9    * Tests if the conditional commands correctly fetch the value from cache loader even with the skip cache load/store
10   * flags.
11   * <p/>
12   * The configuration used is a tx distributed cache without passivation.
13   *
14   * @author Pedro Ruivo
15   * @since 7.0
16   */
17  @Test(groups = "functional", testName = "persistence.ClusteredTxConditionalCommandTest")
18  public class ClusteredTxConditionalCommandTest extends ClusteredConditionalCommandTest {
19  
20     public ClusteredTxConditionalCommandTest() {
21        super(true, false);
22     }
23  
24     @Override
25     protected <K, V> void assertLoadAfterOperation(CacheHelper<K, V> cacheHelper, ConditionalOperation operation, Ownership ownership, boolean skipLoad) {
26        switch (ownership) {
27           case PRIMARY_OWNER:
28              assertLoad(cacheHelper, skipLoad ? 0 : 1, 0, 0);
29              break;
30           case BACKUP_OWNER:
31              assertLoad(cacheHelper, 0, skipLoad ? 0 : 1, 0);
32              break;
33           case NON_OWNER:
34              if (!skipLoad) {
35                 assertTrue("any owner load", cacheHelper.loads(Ownership.PRIMARY_OWNER) +
36                       cacheHelper.loads(Ownership.BACKUP_OWNER) >= 1);
37                 assertEquals("non owner load", 0, cacheHelper.loads(Ownership.NON_OWNER));
38              } else {
39                 assertLoad(cacheHelper, 0, 0, 0);
40              }
41              break;
42        }
43     }
44  }
