3   import org.infinispan.commands.remote.CacheRpcCommand;
4   import org.infinispan.container.versioning.EntryVersionsMap;
5   import org.infinispan.util.logging.Log;

8   /**
9    * The default response generator for most cache modes
10   *
11   * @author Manik Surtani
12   * @author Dan Berindei
13   * @since 4.0
14   */
15  public class DefaultResponseGenerator implements ResponseGenerator {
16     private static final Log log = LogFactory.getLog(DefaultResponseGenerator.class);
17     private static final boolean trace = log.isTraceEnabled();
18  
19     public Response getResponse(CacheRpcCommand command, Object returnValue) {
20        if (returnValue instanceof Response)
21           return (Response) returnValue;
22  
23        if (command.isReturnValueExpected()) {
24           return SuccessfulResponse.create(returnValue);
25        } else {
26           if (returnValue != null) {
27              if (trace) log.tracef("Ignoring non-null response for command %s: %s", command, returnValue);
28           }
29           return null; // saves on serializing a response!
30        }
31     }
32  }
