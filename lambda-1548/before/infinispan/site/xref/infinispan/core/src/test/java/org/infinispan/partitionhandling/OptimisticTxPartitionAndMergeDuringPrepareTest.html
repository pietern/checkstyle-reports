10   * It tests multiple scenarios where a split can happen during a transaction.

12   * @author Pedro Ruivo

15  @Test(groups = "functional", testName = "partitionhandling.OptimisticTxPartitionAndMergeDuringPrepareTest")
16  public class OptimisticTxPartitionAndMergeDuringPrepareTest extends BaseOptimisticTxPartitionAndMergeTest {
17  
18     private static final Log log = LogFactory.getLog(OptimisticTxPartitionAndMergeDuringPrepareTest.class);
19  
20     public void testDegradedPartitionWithDiscard() throws Exception {
21        doTest(SplitMode.BOTH_DEGRADED, true, true);
22     }
23  
24     public void testDegradedPartition() throws Exception {
25        doTest(SplitMode.BOTH_DEGRADED, true, false);
26     }
27  
28     public void testOriginatorIsolatedPartitionWithDiscard() throws Exception {
29        doTest(SplitMode.ORIGINATOR_ISOLATED, true, true);
30     }
31  
32     public void testOriginatorIsolatedPartition() throws Exception {
33        doTest(SplitMode.ORIGINATOR_ISOLATED, true, false);
34     }
35  
36     public void testPrimaryOwnerIsolatedPartitionWithDiscard() throws Exception {
37        doTest(SplitMode.PRIMARY_OWNER_ISOLATED, false, true);
38     }
39  
40     public void testPrimaryOwnerIsolatedPartition() throws Exception {
41        doTest(SplitMode.PRIMARY_OWNER_ISOLATED, false, false);
42     }

44     @Override
45     protected void checkLocksDuringPartition(SplitMode splitMode, KeyInfo keyInfo, boolean discard) {
46        switch (splitMode) {
47           case ORIGINATOR_ISOLATED:
48              //they assume that the originator has crashed, so the prepare is never processed.
49              assertEventuallyNotLocked(cache(1, OPTIMISTIC_TX_CACHE_NAME), keyInfo.getKey1());
50              break;
51           case PRIMARY_OWNER_ISOLATED:
52              //the originator can recover and will retry the prepare command until it succeeds.
53              assertEventuallyNotLocked(cache(1, OPTIMISTIC_TX_CACHE_NAME), keyInfo.getKey1());
54              break;
55           case BOTH_DEGRADED:
56              //with the new changes, the rollback succeeds on the originator partition. Cache1 releases the lock.
57              assertEventuallyNotLocked(cache(1, OPTIMISTIC_TX_CACHE_NAME), keyInfo.getKey1());
58              break;
59        }
60        //or the prepare is never received, so key never locked, or it is received and it decides to rollback the transaction.
61        assertEventuallyNotLocked(cache(2, OPTIMISTIC_TX_CACHE_NAME), keyInfo.getKey2());
62     }
63  
64     @Override
65     protected boolean forceRollback() {
66        return false;
67     }
68  
69     @Override
70     protected Class<? extends TransactionBoundaryCommand> getCommandClass() {
71        return PrepareCommand.class;
72     }
73  
74     @Override
75     protected Log getLog() {
76        return log;
77     }
