1   package org.infinispan.util;
2   
3   import org.infinispan.commons.util.CloseableIterator;
4   
5   import java.util.NoSuchElementException;
6   import java.util.function.Consumer;
7   
8   public class CloseableSuppliedIterator<E> implements CloseableIterator<E> {
9      private final CloseableSupplier<? extends E> supplier;
10  
11     private E next;
12  
13     public CloseableSuppliedIterator(CloseableSupplier<? extends E> supplier) {
14        if (supplier == null) {
15           throw new NullPointerException();
16        }
17        this.supplier = supplier;
18     }
19  
20     @Override
21     public void close() {
22        supplier.close();
23     }
24  
25     private E getNext() {
26        return supplier.get();
27     }
28  
29     @Override
30     public boolean hasNext() {
31        if (next == null) {
32           next = getNext();
33        }
34        return next != null;
35     }
36  
37     @Override
38     public E next() {
39        E e = next == null ? getNext() : next;
40        if (e == null) {
41           throw new NoSuchElementException();
42        }
43        next = null;
44        return e;
45     }
46  
47     @Override
48     public void forEachRemaining(Consumer<? super E> action) {
49        E supplied;
50        if (next != null) {
51           action.accept(next);
52        }
53        while ((supplied = supplier.get()) != null) {
54           action.accept(supplied);
55        }
56     }
57  }
