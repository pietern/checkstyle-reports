3   import java.util.regex.Pattern;
4   
5   import org.apache.logging.log4j.Level;
6   import org.apache.logging.log4j.Marker;
7   import org.apache.logging.log4j.core.Filter;
8   import org.apache.logging.log4j.core.LogEvent;
9   import org.apache.logging.log4j.core.Logger;
10  import org.apache.logging.log4j.core.config.plugins.Plugin;
11  import org.apache.logging.log4j.core.config.plugins.PluginAttribute;
12  import org.apache.logging.log4j.core.config.plugins.PluginFactory;
13  import org.apache.logging.log4j.core.filter.AbstractFilter;
14  import org.apache.logging.log4j.message.Message;
15  
16  /**
17   * Log4j {@link Filter} that only allow events from threads matching a regular expression.
18   * Events with a level greater than {@code threshold} are always logged.
19   *
20   * @author Dan Berindei
21   * @since 5.2
22   */
23  @Plugin(name = "ThreadNameFilter", category = "Core", elementType = Filter.ELEMENT_TYPE, printObject = true)
24  public final class ThreadNameFilter extends AbstractFilter {
25     /** The serialVersionUID */
26     private static final long serialVersionUID = 1L;
27     private final Level level;
28     private final Pattern includePattern;
29  
30     public ThreadNameFilter(Level actualLevel, String includeRegex) {
31        this.level = actualLevel;
32        this.includePattern = Pattern.compile(includeRegex);
33     }
34  
35     @Override
36     public Result filter(final Logger logger, final Level level, final Marker marker, final String msg,
37                          final Object... params) {
38         return filter(level, Thread.currentThread().getName());
39     }
40  
41     @Override
42     public Result filter(final Logger logger, final Level level, final Marker marker, final Object msg,
43                          final Throwable t) {
44         return filter(level, Thread.currentThread().getName());
45     }
46  
47     @Override
48     public Result filter(final Logger logger, final Level level, final Marker marker, final Message msg,
49                          final Throwable t) {
50         return filter(level, Thread.currentThread().getName());
51     }
52  
53     @Override
54     public Result filter(final LogEvent event) {
55        return filter(event.getLevel(), event.getThreadName());
56     }
57  
58     private Result filter(final Level level, String threadName) {
59        if (level.isMoreSpecificThan(this.level)) {
60           return Result.NEUTRAL;
61        } else if (includePattern == null || includePattern.matcher(threadName).find()) {
62           return Result.NEUTRAL;
63        } else {
64           return Result.DENY;
65        }
66     }
67  
68     @Override
69     public String toString() {
70         return level.toString();
71     }
72  
73     /**
74      * Create a ThresholdFilter.
75      * @param level The log Level.
76      * @param match The action to take on a match.
77      * @param mismatch The action to take on a mismatch.
78      * @return The created ThresholdFilter.
79      */
80     @PluginFactory
81     public static ThreadNameFilter createFilter(
82             @PluginAttribute("level") final Level level,
83             @PluginAttribute("include") final String include) {
84         final Level actualLevel = level == null ? Level.DEBUG : level;
85         final String includeRegex = include == null ? ".*" : include;
86         return new ThreadNameFilter(actualLevel, includeRegex);
87     }
88  }
