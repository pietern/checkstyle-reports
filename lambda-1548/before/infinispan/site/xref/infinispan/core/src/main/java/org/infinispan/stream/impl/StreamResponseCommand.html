1   package org.infinispan.stream.impl;
2   
3   import org.infinispan.commands.remote.BaseRpcCommand;
4   import org.infinispan.context.InvocationContext;
5   import org.infinispan.factories.annotations.Inject;
6   import org.infinispan.remoting.transport.Address;
7   
8   import java.util.Collections;
9   import java.util.UUID;
10  
11  /**
12   * Stream response command used to handle returning intermediate or final responses from the remote node
13   * @param <R> the response type
14   */
15  public class StreamResponseCommand<R> extends BaseRpcCommand {
16     public static final byte COMMAND_ID = 48;
17  
18     protected ClusterStreamManager csm;
19  
20     protected UUID id;
21     protected boolean complete;
22     protected R response;
23  
24     // Only here for CommandIdUniquenessTest
25     protected StreamResponseCommand() { super(null); }
26  
27     public StreamResponseCommand(String cacheName) {
28        super(cacheName);
29     }
30  
31     public StreamResponseCommand(String cacheName, Address origin, UUID id, boolean complete, R response) {
32        super(cacheName);
33        setOrigin(origin);
34        this.id = id;
35        this.complete = complete;
36        this.response = response;
37     }
38  
39     @Inject
40     public void inject(ClusterStreamManager csm) {
41        this.csm = csm;
42     }
43  
44     @Override
45     public Object perform(InvocationContext ctx) throws Throwable {
46        csm.receiveResponse(id, getOrigin(), complete, Collections.emptySet(), response);
47        return null;
48     }
49  
50     @Override
51     public byte getCommandId() {
52        return COMMAND_ID;
53     }
54  
55     @Override
56     public Object[] getParameters() {
57        return new Object[]{getOrigin(), id, complete, response};
58     }
59  
60     @Override
61     public void setParameters(int commandId, Object[] parameters) {
62        int i = 0;
63        setOrigin((Address) parameters[i++]);
64        id = (UUID) parameters[i++];
65        complete = (Boolean) parameters[i++];
66        response = (R) parameters[i++];
67     }
68  
69     @Override
70     public boolean isReturnValueExpected() {
71        return false;
72     }
73  
74     @Override
75     public boolean canBlock() {
76        return true;
77     }
78  }
