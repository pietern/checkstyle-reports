3   import java.security.AccessController;
4   import java.security.PrivilegedAction;
5   
6   import org.infinispan.Cache;
7   import org.infinispan.distexec.DefaultExecutorService;
8   import org.infinispan.security.Security;
9   import org.infinispan.security.actions.GetDefaultExecutorServiceAction;
10  
11  /**
12   * SecurityActions for the org.infinispan.notifications.cachelistener.cluster package.
13   *
14   * Do not move. Do not change class and method visibility to avoid being called from other
15   * {@link java.security.CodeSource}s, thus granting privilege escalation to external code.
16   *
17   * @author Tristan Tarrant
18   * @since 7.1
19   */
20  final class SecurityActions {
21     private static <T> T doPrivileged(PrivilegedAction<T> action) {
22        if (System.getSecurityManager() != null) {
23           return AccessController.doPrivileged(action);
24        } else {
25           return Security.doPrivileged(action);
26        }
27     }
28  
29     static DefaultExecutorService getDefaultExecutorService(final Cache<?, ?> cache) {
30        GetDefaultExecutorServiceAction action = new GetDefaultExecutorServiceAction(cache);
31        return doPrivileged(action);
32     }
33  }
