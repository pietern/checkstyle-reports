1   package org.infinispan.configuration.parsing;
2   
3   import javax.xml.stream.XMLStreamException;
4   import javax.xml.stream.XMLStreamReader;
5   
6   /**
7    * An XML stream reader that can read nested {@code <xs:any>} content using a registered set of root
8    * elements.
9    *
10   * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
11   * @since 6.0
12   */
13  public interface XMLExtendedStreamReader extends XMLStreamReader {
14     /**
15      * Handle an {@code <xs:any>}-type nested element, passing in the given value, returning after
16      * the end of the element. Must be positioned on a {@code START_ELEMENT} or an exception will
17      * occur. On return the cursor will be positioned on the corresponding {@code END_ELEMENT}.
18      *
19      * @param value
20      *           the value to pass in
21      * @throws XMLStreamException
22      *            if an error occurs (e.g. the given value does not match the type of the handler for
23      *            the element, or the element is unknown)
24      */
25     void handleAny(ConfigurationBuilderHolder holder) throws XMLStreamException;
26  
27     /**
28      * Get the value of an attribute as a space-delimited string list.
29      *
30      * @param i the index of the attribute
31      */
32     String[] getListAttributeValue(int i);
33  
34     /**
35      * Returns the schema of currently being processed
36      *
37      * @return schema the current schema
38      */
39     Schema getSchema();
40  
41     /**
42      * Sets the current schema
43      *
44      * @param schema
45      */
46     void setSchema(Schema schema);
47  
48  }
