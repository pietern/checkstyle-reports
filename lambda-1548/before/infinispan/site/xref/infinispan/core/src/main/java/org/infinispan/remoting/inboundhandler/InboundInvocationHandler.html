1   package org.infinispan.remoting.inboundhandler;
2   
3   import org.infinispan.commands.ReplicableCommand;
4   import org.infinispan.remoting.transport.Address;
5   import org.infinispan.xsite.XSiteReplicateCommand;
6   
7   /**
8    * Interface to invoke when the {@link org.infinispan.remoting.transport.Transport} receives a command from other node
9    * or site.
10   *
11   * @author Pedro Ruivo
12   * @since 7.1
13   */
14  public interface InboundInvocationHandler {
15  
16     /**
17      * Handles the {@link org.infinispan.commands.ReplicableCommand} from other node belonging to local site.
18      *
19      * @param origin  the sender {@link org.infinispan.remoting.transport.Address}
20      * @param command the {@link org.infinispan.commands.ReplicableCommand} to handler
21      * @param reply   the return value is passed to this object in order to be sent back to the {@param origin}
22      * @param order   the {@link org.infinispan.remoting.inboundhandler.DeliverOrder} in which the command was sent
23      */
24     void handleFromCluster(Address origin, ReplicableCommand command, Reply reply, DeliverOrder order);
25  
26     /**
27      * Handles the {@link org.infinispan.commands.ReplicableCommand} from remote site.
28      * @param origin  the sender site
29      * @param command the {@link org.infinispan.commands.ReplicableCommand} to handle
30      * @param reply   the return value is passed to this object in order to be sent back to the {@param origin}
31      * @param order   the {@link DeliverOrder} in which the command was sent
32      */
33     void handleFromRemoteSite(String origin, XSiteReplicateCommand command, Reply reply, DeliverOrder order);
34  }
