1   package org.infinispan.util.concurrent.locks;

3   import java.util.Collection;
4   
5   /**
6    * Simple interface to extract all the keys that may need to be locked.
7    * <p>
8    * A {@link org.infinispan.commands.remote.CacheRpcCommand} that needs to acquire locks should implement this interface.
9    * This way, Infinispan tries to provide a better management to optimize the system resources usage.
10   *
11   * @author Pedro Ruivo
12   * @since 8.0
13   */
14  public interface RemoteLockCommand {
15  
16     /**
17      * It returns a {@link Collection} with the keys to be lock.
18      * <p>
19      * It may return an empty collection if no keys needs to be locked independently of the return value of {@link
20      * #hasSkipLocking()}. It may contains duplicated keys and {@code null} is not a valid return value.
21      *
22      * @return a {@link Collection} of keys to lock.
23      */
24     Collection<Object> getKeysToLock();
25  
26     /**
27      * It returns the lock owner of the key.
28      * <p>
29      * Usually, in transaction caches it is the {@link org.infinispan.transaction.xa.GlobalTransaction} and in
30      * non-transactional caches the {@link org.infinispan.commands.CommandInvocationId}.
31      *
32      * @return the lock owner of the key.
33      */
34     Object getKeyLockOwner();
35  
36     /**
37      * @return it the locks should be acquire with 0 (zero) acquisition timeout.
38      */
39     boolean hasZeroLockAcquisition();
40  
41     /**
42      * It checks if this command should acquire locks.
43      *
44      * @return {@code true} if locks should be acquired for the keys in {@link #getKeysToLock()}.
45      */
46     boolean hasSkipLocking();
47  }
