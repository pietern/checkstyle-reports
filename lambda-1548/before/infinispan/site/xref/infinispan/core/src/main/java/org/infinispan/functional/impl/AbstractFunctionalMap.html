6   /**

8    *
9    * @since 8.0
10   */
11  abstract class AbstractFunctionalMap<K, V> implements FunctionalMap<K, V> {
12  
13     protected final FunctionalMapImpl<K, V> fmap;
14  
15     protected AbstractFunctionalMap(FunctionalMapImpl<K, V> fmap) {
16        this.fmap = fmap;
17     }
18  
19     @Override
20     public String getName() {
21        return "";
22     }
23  
24     @Override
25     public Status getStatus() {
26        return fmap.getStatus();
27     }
28  
29     @Override
30     public void close() throws Exception {
31        fmap.close();
32     }
33  
34  }
