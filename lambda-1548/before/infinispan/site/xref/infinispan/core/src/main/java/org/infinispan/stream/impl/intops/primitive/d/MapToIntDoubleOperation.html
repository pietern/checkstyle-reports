3   import org.infinispan.stream.impl.intops.IntermediateOperation;
4   
5   import java.util.function.DoubleToIntFunction;
6   import java.util.stream.DoubleStream;
7   import java.util.stream.IntStream;
8   
9   /**
10   * Performs map to int operation on a {@link DoubleStream}
11   */
12  public class MapToIntDoubleOperation implements IntermediateOperation<Double, DoubleStream, Integer, IntStream> {
13     private final DoubleToIntFunction function;
14  
15     public MapToIntDoubleOperation(DoubleToIntFunction function) {
16        this.function = function;
17     }
18  
19     @Override
20     public IntStream perform(DoubleStream stream) {
21        return stream.mapToInt(function);
22     }
23  
24     public DoubleToIntFunction getFunction() {
25        return function;
26     }
27  }
