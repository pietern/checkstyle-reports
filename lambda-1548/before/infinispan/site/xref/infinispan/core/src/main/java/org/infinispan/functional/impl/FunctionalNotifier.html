1   package org.infinispan.functional.impl;
2   
3   import org.infinispan.commons.api.functional.EntryView.ReadEntryView;
4   import org.infinispan.commons.api.functional.Listeners.ReadWriteListeners;
5   import org.infinispan.commons.api.functional.Listeners.WriteListeners;
6   import org.infinispan.commons.util.Experimental;
7   import org.infinispan.factories.scopes.Scope;
8   import org.infinispan.factories.scopes.Scopes;
9   
10  import java.util.function.Supplier;
11  
12  /**
13   * Listener notifier.
14   *
15   * @since 8.0
16   */
17  @Scope(Scopes.NAMED_CACHE)
18  @Experimental
19  public interface FunctionalNotifier<K, V> extends ReadWriteListeners<K, V> {
20  
21     /**
22      * Notify registered {@link ReadWriteListener} instances of the created entry.
23      */
24     void notifyOnCreate(ReadEntryView<K, V> created);
25  
26     /**
27      * Notify registered {@link ReadWriteListener} instances of the modified
28      * entry passing the previous and new value.
29      */
30     void notifyOnModify(ReadEntryView<K, V> before, ReadEntryView<K, V> after);
31  
32     /**
33      * Notify registered {@link ReadWriteListener} instances of the removed
34      * entry passing in the removed entry.
35      */
36     void notifyOnRemove(ReadEntryView<K, V> removed);
37  
38     /**
39      * Notify registered {@link WriteListener} instances of the written entry.
40      *
41      * @apiNote By using a {@link Supplier} the entry view can be computed lazily
42      * only if any listeners has been registered.
43      */
44     void notifyOnWrite(Supplier<ReadEntryView<K, V>> write);
45  
46  }
