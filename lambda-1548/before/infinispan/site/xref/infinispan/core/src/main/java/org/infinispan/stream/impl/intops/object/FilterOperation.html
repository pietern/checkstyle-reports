1   package org.infinispan.stream.impl.intops.object;
2   
3   import org.infinispan.stream.impl.intops.IntermediateOperation;
4   
5   import java.util.function.Predicate;
6   import java.util.stream.Stream;
7   
8   /**
9    * Performs filter operation on a regular {@link Stream}
10   * @param <S> the type in the stream
11   */
12  public class FilterOperation<S> implements IntermediateOperation<S, Stream<S>, S, Stream<S>> {
13     private final Predicate<? super S> predicate;
14  
15     public FilterOperation(Predicate<? super S> predicate) {
16        this.predicate = predicate;
17     }
18  
19     @Override
20     public Stream<S> perform(Stream<S> stream) {
21        return stream.filter(predicate);
22     }
23  
24     public Predicate<? super S> getPredicate() {
25        return predicate;
26     }
27  }
