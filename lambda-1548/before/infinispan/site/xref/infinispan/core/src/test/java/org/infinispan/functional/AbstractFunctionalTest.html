25     @Override
26     protected void createCacheManagers() throws Throwable {
27        // Create local caches as default in a cluster of 2
28        createClusteredCaches(2, new ConfigurationBuilder());
29        // Create distributed caches
30        ConfigurationBuilder distBuilder = new ConfigurationBuilder();
31        distBuilder.clustering().cacheMode(CacheMode.DIST_SYNC).hash().numOwners(1);
32        cacheManagers.stream().forEach(cm -> cm.defineConfiguration(DIST, distBuilder.build()));
33        // Create replicated caches
34        ConfigurationBuilder replBuilder = new ConfigurationBuilder();
35        replBuilder.clustering().cacheMode(CacheMode.REPL_SYNC);
36        cacheManagers.stream().forEach(cm -> cm.defineConfiguration(REPL, replBuilder.build()));
37        // Wait for cluster to form
38        waitForClusterToForm(DIST, REPL);
39     }
