170    @Override
171    public void writeObject(ObjectOutput output, IntermediateOperation object) throws IOException {
172       int number = operations.get(object.getClass(), -1);
173       output.writeByte(number);
174       switch (number) {
175          case FILTER:
176             output.writeObject(((FilterOperation) object).getPredicate());
177             break;
178          case FLATMAP:
179             output.writeObject(((FlatMapOperation) object).getFunction());
180             break;
181          case FLATMAP_DOUBLE:
182             output.writeObject(((FlatMapToDoubleOperation) object).getFunction());
183             break;
184          case FLATMAP_INT:
185             output.writeObject(((FlatMapToIntOperation) object).getFunction());
186             break;
187          case FLATMAP_LONG:
188             output.writeObject(((FlatMapToLongOperation) object).getFunction());
189             break;
190          case LIMIT:
191             UnsignedNumeric.writeUnsignedLong(output, ((LimitOperation) object).getLimit());
192             break;
193          case MAP:
194             output.writeObject(((MapOperation) object).getFunction());
195             break;
196          case MAP_DOUBLE:
197             output.writeObject(((MapToDoubleOperation) object).getFunction());
198             break;
199          case MAP_INT:
200             output.writeObject(((MapToIntOperation) object).getFunction());
201             break;
202          case MAP_LONG:
203             output.writeObject(((MapToLongOperation) object).getFunction());
204             break;
205          case PEEK:

289    @Override
290    public IntermediateOperation readObject(ObjectInput input) throws IOException, ClassNotFoundException {
291       int number = input.readUnsignedByte();
292       switch (number) {
293          case DISTINCT:
294             return DistinctOperation.getInstance();
295          case FILTER:
296             return new FilterOperation<>((Predicate) input.readObject());
297          case FLATMAP:
298             return new FlatMapOperation<>((Function) input.readObject());
299          case FLATMAP_DOUBLE:
300             return new FlatMapToDoubleOperation((Function) input.readObject());
301          case FLATMAP_INT:
302             return new FlatMapToIntOperation((Function) input.readObject());
303          case FLATMAP_LONG:
304             return new FlatMapToLongOperation<>((Function) input.readObject());
305          case LIMIT:
306             return new LimitOperation<>(UnsignedNumeric.readUnsignedLong(input));
307          case MAP:
308             return new MapOperation<>((Function) input.readObject());
309          case MAP_DOUBLE:
310             return new MapToDoubleOperation<>((ToDoubleFunction) input.readObject());
311          case MAP_INT:
312             return new MapToIntOperation<>((ToIntFunction) input.readObject());
313          case MAP_LONG:
314             return new MapToLongOperation<>((ToLongFunction) input.readObject());
315          case PEEK:
316             return new PeekOperation<>((Consumer) input.readObject());
317          case SORTED_COMPARATOR:
318             return new SortedComparatorOperation<>((Comparator) input.readObject());
319          case SORTED:
320             return SortedOperation.getInstance();
321 
322          case DOUBLE_BOXED:
323             return BoxedDoubleOperation.getInstance();
324          case DOUBLE_DISTINCT:
325             return DistinctDoubleOperation.getInstance();
326          case DOUBLE_FILTER:
327             return new FilterDoubleOperation((DoublePredicate) input.readObject());
328          case DOUBLE_FLATMAP:
329             return new FlatMapDoubleOperation((DoubleFunction) input.readObject());
330          case DOUBLE_LIMIT:
331             return new LimitDoubleOperation(UnsignedNumeric.readUnsignedLong(input));
332          case DOUBLE_MAP:
333             return new MapDoubleOperation((DoubleUnaryOperator) input.readObject());
334          case DOUBLE_MAP_INT:
335             return new MapToIntDoubleOperation((DoubleToIntFunction) input.readObject());
336          case DOUBLE_MAP_LONG:
337             return new MapToLongDoubleOperation((DoubleToLongFunction) input.readObject());
338          case DOUBLE_MAP_OBJ:
339             return new MapToObjDoubleOperation<>((DoubleFunction) input.readObject());
340          case DOUBLE_PEEK:
341             return new PeekDoubleOperation((DoubleConsumer) input.readObject());
342          case DOUBLE_SORTED:
343             return SortedDoubleOperation.getInstance();
344 
345          case INT_AS_DOUBLE:
346             return AsDoubleIntOperation.getInstance();
347          case INT_AS_LONG:
348             return AsLongIntOperation.getInstance();
349          case INT_BOXED:
350             return BoxedIntOperation.getInstance();
351          case INT_DISTINCT:
352             return DistinctIntOperation.getInstance();
353          case INT_FILTER:
354             return new FilterIntOperation((IntPredicate) input.readObject());
355          case INT_FLATMAP:
356             return new FlatMapIntOperation((IntFunction) input.readObject());
357          case INT_LIMIT:
358             return new LimitIntOperation(UnsignedNumeric.readUnsignedLong(input));
359          case INT_MAP:
360             return new MapIntOperation((IntUnaryOperator) input.readObject());
361          case INT_MAP_DOUBLE:
362             return new MapToDoubleIntOperation((IntToDoubleFunction) input.readObject());
363          case INT_MAP_LONG:
364             return new MapToLongIntOperation((IntToLongFunction) input.readObject());
365          case INT_MAP_OBJ:
366             return new MapToObjIntOperation<>((IntFunction) input.readObject());
367          case INT_PEEK:
368             return new PeekIntOperation((IntConsumer) input.readObject());
369          case INT_SORTED:
370             return SortedIntOperation.getInstance();
371 
372          case LONG_AS_DOUBLE:
373             return AsDoubleLongOperation.getInstance();
374          case LONG_BOXED:
375             return BoxedLongOperation.getInstance();
376          case LONG_DISTINCT:
377             return DistinctLongOperation.getInstance();
378          case LONG_FILTER:
379             return new FilterLongOperation((LongPredicate) input.readObject());
380          case LONG_FLATMAP:
381             return new FlatMapLongOperation((LongFunction) input.readObject());
382          case LONG_LIMIT:
383             return new LimitLongOperation(UnsignedNumeric.readUnsignedLong(input));
384          case LONG_MAP:
385             return new MapLongOperation((LongUnaryOperator) input.readObject());
386          case LONG_MAP_DOUBLE:
387             return new MapToDoubleLongOperation((LongToDoubleFunction) input.readObject());
388          case LONG_MAP_INT:
389             return new MapToIntLongOperation((LongToIntFunction) input.readObject());
390          case LONG_MAP_OBJ:
391             return new MapToObjLongOperation<>((LongFunction) input.readObject());
392          case LONG_PEEK:
393             return new PeekLongOperation((LongConsumer) input.readObject());
394          case LONG_SORTED:
395             return SortedLongOperation.getInstance();
396 
397          default:
398             throw new IllegalArgumentException("Found invalid number " + number);
399       }
