20     @Override
21     protected void createCacheManagers() throws Throwable {
22        ConfigurationBuilder dcc = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
23        dcc.transaction().lockingMode(LockingMode.PESSIMISTIC);
24        createCluster(dcc, 2);
25        waitForClusterToForm();
26        cache(0).put("k", "v");

28     }

33           @Override
34           protected Object handleDefault(InvocationContext ctx, VisitableCommand command) throws Throwable {
35              throw new RuntimeException("Induced!");
36           }
37        }, PessimisticLockingInterceptor.class);
