1   package org.infinispan.util.concurrent.locks;
2   
3   /**
4    * An extended {@link LockPromise} interface that allows a better control over it.
5    *
6    * @author Pedro Ruivo
7    * @since 8.0
8    */
9   public interface ExtendedLockPromise extends LockPromise {
10  
11     /**
12      * It cancels the {@link LockPromise} if possible.
13      *
14      * @param cause the cancellation cause. The possible values are {@link LockState#DEADLOCKED} and {@link
15      *              LockState#TIMED_OUT}.
16      * @throws IllegalArgumentException if the argument {@code cause} is not valid.
17      */
18     void cancel(LockState cause);
19  
20     /**
21      * @return the lock owner associated to this {@link LockPromise}.
22      */
23     Object getRequestor();
24  
25     /**
26      * @return the current lock owner.
27      */
28     Object getOwner();
29  
30  }
