18  /**
19   * Available flags, which may be set on a per-invocation basis.  These are
20   * provided using the {@link AdvancedCache} interface, using some of the
21   * overloaded methods that allow passing in of a variable number of Flags.
22   *
23   * When making modifications to these enum, do not change the order of
24   * enumerations, so always append any new enumerations after the last one.
25   * Finally, enumerations should not be removed.
26   *
27   * @author Manik Surtani
28   * @author Galder Zamarre√±o
29   * @since 4.0
30   */
31  public enum Flag {
32     /**
33      * Overrides the {@link Configuration#setLockAcquisitionTimeout(long)} configuration setting by ensuring lock
34      * managers use a 0-millisecond lock acquisition timeout.  Useful if you only want to acquire a lock on an entry
35      * <i>if and only if</i> the lock is uncontended.
36      */
37     ZERO_LOCK_ACQUISITION_TIMEOUT,
38     /**
39      * Forces LOCAL mode operation, even if the cache is configured to use a clustered mode like replication,
40      * invalidation or distribution. Applying this flag will suppress any RPC messages otherwise associated with this
41      * invocation. Write operations mat not acquire the entry locks. In distributed mode, the modifications performed
42      * during an operation in a non-owner node are going to L1, if it is enabled, otherwise the operation is a no-op in
43      * that node.
44      */
45     CACHE_MODE_LOCAL,
46     /**
47      * Bypasses lock acquisition for this invocation altogether. A potentially dangerous flag, as it can lead to
48      * inconsistent data: a Lock is needed to make sure the same value is written to each node replica; a lock
49      * is also needed to guarantee that several writes on the same key are not applied out of order to an async CacheLoader
50      * storage engine.
51      * So this flag is useful only as an optimization when the same key is written once and never again, or as
52      * an unsafe optimisation if the period between writes on the same key is large enough to make a race condition
53      * never happen in practice. If this is unclear, avoid it.
54      */
55     SKIP_LOCKING,
56     /**
57      * Forces a write lock, even if the invocation is a read operation.  Useful when reading an entry to later update it
58      * within the same transaction, and is analogous in behavior and use case to a <tt>select ... for update ... </tt>
59      * SQL statement.
60      */
61     FORCE_WRITE_LOCK,
62     /**
63      * Skips checking whether a cache is in a receptive state, i.e. is {@link ComponentStatus#RUNNING}.  May break
64      * operation in weird ways!
65      *
66      * @deprecated This flag is no longer in use.
67      */
68     @Deprecated
69     SKIP_CACHE_STATUS_CHECK,
70     /**
71      * Forces asynchronous network calls where possible, even if otherwise configured to use synchronous network calls.
72      * Only applicable to non-local, clustered caches.
73      */
74     FORCE_ASYNCHRONOUS,
75     /**
76      * Forces synchronous network calls where possible, even if otherwise configured to use asynchronous network calls.
77      * Only applicable to non-local, clustered caches.
78      */
79     FORCE_SYNCHRONOUS,
80     /**
81      * Skips storing an entry to any configured {@link org.infinispan.persistence.spi.CacheLoader}s.
82      */
83     SKIP_CACHE_STORE,
84     /**
85      * Skips loading an entry from any configured {@link org.infinispan.persistence.spi.CacheLoader}s.
86      * Useful for example to perform a {@link Cache#put(Object, Object)} operation while not interested
87      * in the return value (i.e. the previous value of the key).
88      * <br>
89      * Note that the loader will be skipped even if that changes the meaning of the operation, e.g. for
90      * conditional or {@link org.infinispan.atomic.DeltaAware} write operations. If that is not intended,
91      * you should use {@link #IGNORE_RETURN_VALUES} instead.
92      */
93     SKIP_CACHE_LOAD,
94     /**
95      * <p>Swallows any exceptions, logging them instead at a low log level.  Will prevent a failing operation from
96      * affecting any ongoing JTA transactions as well.</p>
97      * <p>This Flag will not be replicated to remote nodes, but it will still protect the invoker from remote exceptions.</p>
98      * <p>When using this flag with Optimistic caches, lock acquisition happen in the prepare phase at which
99      * point this flag will be ignored in order to ensure that Infinispan reports the correct exception
100     * back to the transaction manager. This is done for safety reasons to avoid inconsistent cache contents.</p>
101     */
102    FAIL_SILENTLY,
103    /**
104     * When used with <b>distributed</b> cache mode, will prevent retrieving a remote value either when
105     * executing a {@link Cache#get(Object)} or {@link Cache#containsKey(Object)},
106     * or to return the overwritten value for {@link Cache#put(Object, Object)} or {@link Cache#remove(Object)}.
107     * This would render return values for most operations unusable, in exchange for the performance gains of
108     * reducing remote calls.
109     * <br>
110     * Note that the remote lookup will be skipped even if that changes the meaning of the operation, e.g. for
111     * conditional or {@link org.infinispan.atomic.DeltaAware} write operations. If that is not intended,
112     * you should use {@link #IGNORE_RETURN_VALUES} instead.
113     */
114    SKIP_REMOTE_LOOKUP,
115 
116    /**
117     * Used by the Query module only, it will prevent the indexes to be updated as a result of the current operations.
118     */
119    SKIP_INDEXING,
120 
121    /**
122     * Flags the invocation as a {@link Cache#putForExternalRead(Object, Object)}
123     * call, as opposed to a regular {@link Cache#put(Object, Object)}. This
124     * flag was created purely for internal Infinispan usage, and should not be
125     * used by clients calling into Infinispan.
126     */
127    PUT_FOR_EXTERNAL_READ,
128 
129    /**
130     * Flags the invocation as a put operation done internally by the state transfer.
131     * This flag was created purely for internal Infinispan usage, and should not be
132     * used by clients calling into Infinispan.
133     *
134     * Note for internal users: PUT_FOR_STATE_TRANSFER only applies to state transfer-specific actions,
135     * in order to avoid loading the previous value one should add the IGNORE_RETURN_VALUES flag explicitly.
136     */
137    PUT_FOR_STATE_TRANSFER,
138 
139    /**
140     * Flags the invocation as a put operation done internally by the cross-site state transfer. This flag was created
141     * purely for internal Infinispan usage, and should not be used by clients calling into Infinispan.
142     */
143    PUT_FOR_X_SITE_STATE_TRANSFER,
144 
145    /**
146     * If this flag is enabled, if a cache store is shared, then storage to the store is skipped.
147     */
148    SKIP_SHARED_CACHE_STORE,
149    /**
150     * This flag has only effect when it's used before calling {@link
151     * org.infinispan.Cache#stop()} and its effect is that apart from stopping
152     * the cache, it removes all of its content from both memory and any backing
153     * cache store.
154     *
155     * @deprecated No longer in use.
156     */
157    @Deprecated
158    REMOVE_DATA_ON_STOP,
159    /**
160     * Used by the DistLockingInterceptor to commit the change no matter what (if the flag is set). This is used when
161     * a node A pushes state to another node B and A doesn't want B to check if the state really belongs to it
162     */
163    SKIP_OWNERSHIP_CHECK,
164    /**
165     * Signals when a particular cache write operation is writing a delta of
166     * the object, rather than the full object. This can be useful in order to
167     * make decisions such as whether the cache store needs checking to see if
168     * the previous value needs to be loaded and merged.
169     */
170    DELTA_WRITE,
171 
172    /**
173     * Signals that a write operation's return value will be ignored, so reading
174     * the existing value from a store or from a remote node is not necessary.
175     *
176     * Typical operations whose return value might be ignored include
177     * {@link java.util.Map#put(Object, Object)} whose return value indicates
178     * previous value. So, a user might decide to the put something in the
179     * cache but might not be interested in the return value.
180     *
181     * This flag is ignored for operations that need the existing value to execute
182     * correctly, e.g. {@link Cache#get(Object)},
183     * conditional remove ({@link Cache#remove(Object, Object)}),
184     * and replace with an expected value ({@link Cache#replace(Object, Object, Object)}).
185     *
186     * That means it is safe to use {@code IGNORE_RETURN_VALUES} for all the operations on a cache,
187     * unlike {@link Flag#SKIP_REMOTE_LOOKUP} and {@link Flag#SKIP_CACHE_LOAD}.
188     */
189    IGNORE_RETURN_VALUES,
190 
191    /**
192     * If cross-site replication is enabled, this would skip the replication to any remote site.
193     */
194    SKIP_XSITE_BACKUP,
195 
196    /**
197     * Using a synchronous cache (whether replicated or distributed) provides
198     * the cache caller guarantees that data has been sent to other cluster
199     * nodes correctly and has been applied successfully. At the network
200     * level, message delivery acknowledgement protocols are used to provide
201     * these guarantees.
202     *
203     * In order to increase performance and achieve better throughput, it's
204     * common to use negative acknowledgment protocols to confirm the
205     * delivery of messages. The problem with these protocols is that if the
206     * last message is lost, it can be difficult to recover it because a new
207     * message would need to be sent to find the gap.
208     *
209     * Some cache use cases might involve storing an entry in a synchronously
210     * replicated or distributed cache, and if that store operation fails, the
211     * application fails to start. One such example is the Infinispan Hot Rod
212     * server. When it starts, the first thing it does is add its endpoint
213     * information to a cache which is used to notify clients of topology
214     * changes. If this operation fails, the server cannot start because
215     * topologies won't include this node, and no more cache operations
216     * are attempted.
217     *
218     * So, in exceptional use cases such as this, a cluster wide cache update
219     * should be positively acknowledged by the other nodes, so that if the
220     * data is lost, it can be retransmitted immediately without the need
221     * to wait for an extra cluster wide operation to detect the lost message.
222     * The way to force a particular cache operation to be positively
223     * acknowledged is to send this {@link #GUARANTEED_DELIVERY} flag.
224     *
225     * Note that this is flag is <b>EXPERIMENTAL</b> and so there is a high
226     * probability that it will be removed in future Infinispan versions.
227     */
228    GUARANTEED_DELIVERY,
229 
230    /**
231     * This flag skips listener notifications as a result of a cache operation.
232     * For example, if this flag is passed as a result of a {@link Cache#get(Object)}
233     * call, no callbacks will be made on listeners annotated with
234     * {@link org.infinispan.notifications.cachelistener.annotation.CacheEntryVisited}.
235     */
236    SKIP_LISTENER_NOTIFICATION,
237 
238    /**
239     * This flag skips statistics updates as a result of a cache operation.
240     * For example, if this flag is passed as a result of a {@link Cache#get(Object)}
241     * call, no cache hits or cache miss counters will be updated.
242     */
243    SKIP_STATISTICS,

262     */
263    SKIP_INDEX_CLEANUP,

274    /**
275     * Creates a copy of a Flag Set removing instances of FAIL_SILENTLY.
276     * The copy might be the same instance if no change is required,
277     * and should be considered immutable.
278     * @param flags
279     * @return might return the same instance
280     */
281    public static Set<Flag> copyWithoutRemotableFlags(Set<Flag> flags) {
282       //FAIL_SILENTLY should not be sent to remote nodes
283       if (flags != null && flags.contains(Flag.FAIL_SILENTLY)) {
284          EnumSet<Flag> copy = EnumSet.copyOf(flags);
285          copy.remove(Flag.FAIL_SILENTLY);
286          if (copy.isEmpty()) {
287             return InfinispanCollections.emptySet();
288          }
289          else {
290             return copy;
291          }
292       } else {
293          return flags;
294       }
295    }

297    public static Set<Flag> addFlag(Set<Flag> flags, Flag newFlag) {
298       if (flags == null || flags.isEmpty()) {
299          return EnumSet.of(newFlag);
300       }
301       if (flags.contains(newFlag)) {
302          return flags;
303       }
304       EnumSet<Flag> copy = EnumSet.copyOf(flags);
305       copy.add(newFlag);
306       return copy;
307    }
308 
309    public static Set<Flag> addFlags(Set<Flag> flags, Flag... newFlags) {
310       if (newFlags == null || newFlags.length == 0) {
311          return flags;
312       }
313       EnumSet<Flag> copy;
314       if (flags == null) {
315          copy = EnumSet.noneOf(Flag.class);
316       } else {
317          copy = EnumSet.copyOf(flags);
318       }
319       Collections.addAll(copy, newFlags);
320       return copy;
321    }
322 
323    public static Set<Flag> addFlags(Set<Flag> flags, Set<Flag> newFlags) {
324       if (newFlags == null || newFlags.isEmpty()) {
325          return flags;
326       }
327       if (flags == null || flags.isEmpty()) {
328          return newFlags;
329       }
330       EnumSet<Flag> copy = EnumSet.copyOf(flags);
331       copy.addAll(newFlags);
332       return copy;
333    }

335    public static class Externalizer extends AbstractExternalizer<Flag> {

337       @Override
338       public Integer getId() {
339          return Ids.FLAG;

344          return Util.<Class<? extends Flag>>asSet(Flag.class);

347       @Override
348       public void writeObject(ObjectOutput output, Flag flag) throws IOException {
349          output.writeByte(flag.ordinal());
350       }
351 
352       @Override
353       public Flag readObject(ObjectInput input) throws IOException, ClassNotFoundException {
354          byte flagByte = input.readByte();
355          try {
356             // values() cached by Class.getEnumConstants()
357             return Flag.values()[flagByte];
358          } catch (ArrayIndexOutOfBoundsException e) {
359             throw new IllegalStateException("Unknown flag index: " + flagByte);
360          }
361       }
