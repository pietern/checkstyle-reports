3   /**
4    * This event subtype is passed in to any method annotated with
5    * {@link org.infinispan.notifications.cachelistener.annotation.CacheEntryExpired}.
6    * <p />
7    * The {@link #getValue()} method returns the value of the entry before it expired.  Note this value may be null if
8    * the entry expired from a cache store
9    * <p>
10   * This is a post only event
11   * <p>
12   * This event can be raised multiple times in sequence for a single expiration event if concurrent reads for the same
13   * key occur on different nodes.  This should rarely happen though since this window is narrowed internally by the
14   * cache.
15   * @author William Burns
16   * @since 8.0
17   */
18  public interface CacheEntryExpiredEvent<K, V> extends CacheEntryEvent<K, V> {
19  
20     /**
21      * Retrieves the value of the entry being expired.  Note this event is raised after the value has been expired.
22      * <p />
23      * @return the value of the entry expired
24      */
25     @Override
26     V getValue();
27  }
