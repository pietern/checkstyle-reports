1   package org.infinispan.persistence.spi;

3   import org.infinispan.commons.util.Experimental;
4   import org.infinispan.marshall.core.MarshalledEntry;
5   
6   import java.util.concurrent.Executor;
7   
8   /**
9    * Defines functionality for advanced expiration techniques.  Note this interface allows for providing not just the key
10   * when an entry is expired.  This is important so that proper cluster wide expiration can be performed.
11   * @param <K>
12   * @param <V>
13   * @since 8.0
14   */
15  @Experimental
16  public interface AdvancedCacheExpirationWriter<K, V> extends AdvancedCacheWriter<K, V> {
17     /**
18      * Using the thread in the pool, removed all the expired data from the persistence storage. For each removed entry,
19      * the supplied listener is invoked.  This should be preferred to
20      * {@link AdvancedCacheWriter#purge(Executor, PurgeListener)} since it allows for value and metadata to be provided
21      * which provides more accurate expiration when coordination is required.
22      *
23      * @param executor the executor to invoke the given command on
24      * @param listener the listener that is notified for each expired entry
25      * @throws PersistenceException in case of an error, e.g. communicating with the external storage
26      */
27     void purge(Executor executor, ExpirationPurgeListener<K, V> listener);
28  
29     /**
30      * Callback to be notified when an entry is removed by the {@link #purge(Executor, ExpirationPurgeListener)} method.
31      * Note this interface adds a new method to the purge listener.  It is possible that a cache store may want to
32      * have a key only expiration and a key/metadata for various performance reasons.
33      */
34     interface ExpirationPurgeListener<K, V> extends PurgeListener<K> {
35  
36        /**
37         * Optional. If possible, {@link AdvancedCacheExpirationWriter} implementors should invoke this method for every
38         * entry that is purged from the store. One of the side effects of not implementing this method is that listeners
39         * do not receive {@link org.infinispan.notifications.cachelistener.annotation.CacheEntryExpired} for the
40         * entries that are removed from the persistent store directly.
41         */
42        void marshalledEntryPurged(MarshalledEntry<K,V> entry);
43     }
44  }
