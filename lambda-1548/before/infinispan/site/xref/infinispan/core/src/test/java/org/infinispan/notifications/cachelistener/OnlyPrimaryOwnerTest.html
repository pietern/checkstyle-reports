111    public void testOwnership() {
112       // Is not owner nor primary owner
113       cdl.isOwner = false;
114       cdl.isPrimaryOwner = false;
115       n.notifyCacheEntryCreated("reject", "v1", null, true, ctx, null);
116       n.notifyCacheEntryCreated("reject", "v1", null, false, ctx, null);
117 
118       assert !cl.isReceivedPost();
119       assert !cl.isReceivedPre();
120       assert cl.getInvocationCount() == 0;
121 
122       // Is an owner but not primary owner
123       cdl.isOwner = true;
124       cdl.isPrimaryOwner = false;
125       n.notifyCacheEntryCreated("reject", "v1", null, true, ctx, null);
126       n.notifyCacheEntryCreated("reject", "v1", null, false, ctx, null);
127 
128       assert !cl.isReceivedPost();
129       assert !cl.isReceivedPre();
130       assert cl.getInvocationCount() == 0;
131 
132       // Is primary owner
133       cdl.isOwner = true;
134       cdl.isPrimaryOwner = true;
135       n.notifyCacheEntryCreated("accept", "v1", null, true, ctx, null);
136       n.notifyCacheEntryCreated("accept", "v1", null, false, ctx, null);
137 
138       assert cl.isReceivedPost();
139       assert cl.isReceivedPre();
140       assert cl.getInvocationCount() == 2;
141       assert cl.getEvents().get(0).getCache() == mockCache;
142       assert cl.getEvents().get(0).getType() == Event.Type.CACHE_ENTRY_CREATED;
143       assert ((CacheEntryCreatedEvent) cl.getEvents().get(0)).getKey().equals("accept");
144       assert ((CacheEntryCreatedEvent) cl.getEvents().get(0)).getValue() == null;
145       assert cl.getEvents().get(1).getCache() == mockCache;
146       assert cl.getEvents().get(1).getType() == Event.Type.CACHE_ENTRY_CREATED;
147       assert ((CacheEntryCreatedEvent) cl.getEvents().get(1)).getKey().equals("accept");
148       assert ((CacheEntryCreatedEvent) cl.getEvents().get(1)).getValue().equals("v1");
149 
150    }
