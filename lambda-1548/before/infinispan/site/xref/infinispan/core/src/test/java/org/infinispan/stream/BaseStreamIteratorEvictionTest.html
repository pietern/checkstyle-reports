18  /**
19   * Base class to test if evicted entries are not returned via stream
20   *
21   * @author wburns
22   * @since 8.0
23   */
24  @Test(groups = "functional", testName = "stream.BaseStreamIteratorEvictionTest")
25  public abstract class BaseStreamIteratorEvictionTest extends BaseSetupStreamIteratorTest {
26     public BaseStreamIteratorEvictionTest(boolean tx, CacheMode mode) {
27        super(tx, mode);
28     }
29  
30     public void testExpiredEntryNotReturned() throws InterruptedException {
31        Cache<Object, String> cache = cache(0, CACHE_NAME);
32        // First put some values in there
33        Map<Object, String> valuesInserted = new LinkedHashMap<Object, String>();
34        for (int i = 0; i < 5; ++i) {
35           Object key = i;
36           String value = key + " stay in cache";
37           cache.put(key, value);
38           valuesInserted.put(key, value);
39        }
40  
41        int expectedTime = 2;
42        // Now we insert a value that will expire in 2 seconds
43        cache.put("expired", "this shouldn't be returned", expectedTime, TimeUnit.SECONDS);
44  
45        // We have to wait the time limit to make sure it is evicted before proceeding
46        Thread.sleep(TimeUnit.SECONDS.toMillis(expectedTime) + 50);
47  
48        cache.getAdvancedCache().filterEntries(new KeyFilterAsKeyValueFilter<Object, String>(
49              new CollectionKeyFilter<>(Collections.emptySet())));
50  
51        Map<Object, String> results;
52        try (Stream<CacheEntry<Object, String>> stream = cache.getAdvancedCache().cacheEntrySet().stream()) {
53           results = mapFromStream(stream);
54        }
55  
56        assertEquals(valuesInserted, results);
57     }
58  }
