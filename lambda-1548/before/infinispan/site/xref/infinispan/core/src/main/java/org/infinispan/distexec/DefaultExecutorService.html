49  /**
50   * Infinispan's implementation of an {@link ExecutorService} and {@link DistributedExecutorService}.
51   * This ExecutorService provides methods to submit tasks for an execution on a cluster of Infinispan
52   * nodes.
53   * <p>
54   *
55   *
56   * Note that due to potential task migration to another nodes every {@link Callable},
57   * {@link Runnable} and/or {@link DistributedCallable} submitted must be either {@link Serializable}
58   * or {@link Externalizable}. Also the value returned from a callable must be {@link Serializable}
59   * or {@link Externalizable}. Unfortunately if the value returned is not serializable then a
60   * {@link NotSerializableException} will be thrown.
61   *
62   * @author Vladimir Blagojevic
63   * @since 5.0
64   *
65   */
66  public class DefaultExecutorService extends AbstractExecutorService implements DistributedExecutorService {

148     * parallel execution of task ran on this node. All distributed task executions will be initiated
149     * from this Infinispan cache node.
150     *
151     * @param masterCacheNode
152     *           Cache node initiating distributed task
153     * @param localExecutorService
154     *           ExecutorService to run local tasks
155     * @param takeExecutorOwnership
156     *           if true {@link DistributedExecutorService#shutdown()} and
157     *           {@link DistributedExecutorService#shutdownNow()} method will shutdown
158     *           localExecutorService as well
159     *

161    public DefaultExecutorService(Cache<?, ?> masterCacheNode, ExecutorService localExecutorService,
162             boolean takeExecutorOwnership) {
163       super();
164       if (masterCacheNode == null)
165          throw new IllegalArgumentException("Can not use null cache for DefaultExecutorService");
166       else if (localExecutorService == null)
167          throw new IllegalArgumentException("Can not use null instance of ExecutorService");
168       else {
169          try {
170             if (localExecutorService.isShutdown())
171                throw new IllegalArgumentException("Can not use an instance of ExecutorService which is shutdown");
172          } catch (IllegalStateException e) {
173             if (takeExecutorOwnership) {
174                throw new IllegalArgumentException("Can not take ownership of a ManagedExecutorService");
175             }
176          }
177       }
178 
179       ensureAccessPermissions(masterCacheNode.getAdvancedCache());
180       ensureProperCacheState(masterCacheNode.getAdvancedCache());
181       ensureFullCache(masterCacheNode.getAdvancedCache());
182 
183       this.cache = masterCacheNode.getAdvancedCache();
184       ComponentRegistry registry = SecurityActions.getCacheComponentRegistry(cache);
185 
186       this.rpc = SecurityActions.getCacheRpcManager(cache);
187       this.invoker = registry.getComponent(InterceptorChain.class);
188       this.factory = registry.getComponent(CommandsFactory.class);
189       this.marshaller = registry.getComponent(StreamingMarshaller.class, CACHE_MARSHALLER);
190       this.cancellationService = registry.getComponent(CancellationService.class);
191       this.localExecutorService = localExecutorService;
192       this.takeExecutorOwnership = takeExecutorOwnership;
193       this.timeService = registry.getTimeService();
194       this.clusterDependentLogic = registry.getComponent(ClusteringDependentLogic.class);
195    }

292    /**
293     * the main mechanics of invokeAny. This was essentially copied from
294     * {@link AbstractExecutorService} doInvokeAny except that we replaced the
295     * {@link ExecutorCompletionService} with our {@link DistributedExecutionCompletionService}.
296     */
297    private <T> T doInvokeAny(Collection<? extends Callable<T>> tasks, boolean timed, long nanos)
298             throws InterruptedException, ExecutionException, TimeoutException {
299       if (tasks == null)
300          throw new NullPointerException();
301       int ntasks = tasks.size();
302       if (ntasks == 0)
303          throw new IllegalArgumentException();
304       List<Future<T>> futures = new ArrayList<Future<T>>(ntasks);
305       CompletionService<T> ecs = new DistributedExecutionCompletionService<T>(this);
306 
307       // For efficiency, especially in executors with limited
308       // parallelism, check to see if previously submitted tasks are
309       // done before submitting more of them. This interleaving
310       // plus the exception mechanics account for messiness of main
311       // loop.
312 
313       try {
314          // Record exceptions so that if we fail to obtain any
315          // result, we can throw the last exception we got.
316          ExecutionException ee = null;
317          long lastTime = (timed) ? timeService.time() : 0;
318          Iterator<? extends Callable<T>> it = tasks.iterator();
319 
320          // Start one task for sure; the rest incrementally
321          futures.add(ecs.submit(it.next()));
322          --ntasks;
323          int active = 1;
324 
325          for (;;) {
326             Future<T> f = ecs.poll();
327             if (f == null) {
328                if (ntasks > 0) {
329                   --ntasks;
330                   futures.add(ecs.submit(it.next()));
331                   ++active;
332                } else if (active == 0)
333                   break;
334                else if (timed) {
335                   f = ecs.poll(nanos, TimeUnit.NANOSECONDS);
336                   if (f == null)
337                      throw new TimeoutException();
338                   long now = timeService.time();
339                   nanos -= timeService.timeDuration(lastTime, now, TimeUnit.NANOSECONDS);
340                   lastTime = now;
341                } else
342                   f = ecs.take();
343             }
344             if (f != null) {
345                --active;
346                try {
347                   return f.get();
348                } catch (InterruptedException ie) {
349                   throw ie;
350                } catch (ExecutionException eex) {
351                   ee = eex;
352                } catch (RuntimeException rex) {
353                   ee = new ExecutionException(rex);
354                }
355             }
356          }
357 
358          if (ee == null)
359             ee = new ExecutionException() {
360                private static final long serialVersionUID = 200818694545553992L;
361             };
362          throw ee;
363 
364       } finally {
365          for (Future<T> f : futures)
366             f.cancel(true);
367       }
368    }

370    @Override
371    public void execute(Runnable command) {
372       if (!isShutdown.get()) {
373          DistributedTaskPart<Object> cmd;
374          if (command instanceof DistributedTaskPart<?>) {
375             cmd = (DistributedTaskPart<Object>) command;
376          } else if (command instanceof Serializable) {
377             cmd = (DistributedTaskPart<Object>) newTaskFor(command, null);
378          } else {
379             throw new IllegalArgumentException("Runnable command is not Serializable  " + command);
380          }
381          cmd.execute();
382       } else {
383          throw new RejectedExecutionException();
384       }
385    }

412    public <T> NotifyingFuture<T> submit(Address target, DistributedTask<T> task) {
413       if (task == null)
414          throw new NullPointerException();
415       if (target == null)
416          throw new NullPointerException();
417       List<Address> members = getMembers();
418       if (!members.contains(target)) {
419          return new NoOpFuture<>(new SuspectException("Target node " + target
420                   + " is not a cluster member, members are " + members));
421       }
422       Address me = getAddress();
423       DistributedExecuteCommand<T> c = null;
424       if (target.equals(me)) {
425          c = factory.buildDistributedExecuteCommand(clone(task.getCallable()), me, null);
426       } else {
427          c = factory.buildDistributedExecuteCommand(task.getCallable(), me, null);
428       }
429       DistributedTaskPart<T> part = createDistributedTaskPart(task, c, target, 0);
430       part.execute();
431       return part;
432    }

442    public <T, K> NotifyingFuture<T> submit(DistributedTask<T> task, K... input) {
443       if (task == null) throw new NullPointerException();
444 
445       if(inputKeysSpecified(input)){
446          Map<Address, List<K>> nodesKeysMap = keysToExecutionNodes(task.getTaskExecutionPolicy(), input);
447          checkExecutionPolicy(task, nodesKeysMap, input);
448          Address me = getAddress();
449          DistributedExecuteCommand<T> c = factory.buildDistributedExecuteCommand(task.getCallable(), me, Arrays.asList(input));
450          ArrayList<Address> nodes = new ArrayList<Address>(nodesKeysMap.keySet());
451          DistributedTaskPart<T> part = createDistributedTaskPart(task, c, Arrays.asList(input), selectExecutionNode(nodes), 0);
452          part.execute();
453          return part;
454       } else {
455          return submit(task.getCallable());
456       }
457    }

494    @Override
495    public <T, K> List<Future<T>> submitEverywhere(DistributedTask<T> task, K... input) {
496       if (task == null) throw new NullPointerException();
497       if(inputKeysSpecified(input)) {
498          List<Future<T>> futures = new ArrayList<Future<T>>(input.length * 2);
499          Address me = getAddress();
500          Map<Address, List<K>> nodesKeysMap = keysToExecutionNodes(task.getTaskExecutionPolicy(), input);
501          checkExecutionPolicy(task, nodesKeysMap, input);
502          for (Entry<Address, List<K>> e : nodesKeysMap.entrySet()) {
503             Address target = e.getKey();
504             DistributedExecuteCommand<T> c = null;
505             if (target.equals(me)) {
506                c = factory.buildDistributedExecuteCommand(clone(task.getCallable()), me, e.getValue());
507             } else {
508                c = factory.buildDistributedExecuteCommand(task.getCallable(), me, e.getValue());
509             }
510             DistributedTaskPart<T> part = createDistributedTaskPart(task, c, e.getValue(), target, 0);
511             futures.add(part);
512             part.execute();
513          }
514          return futures;
515       } else {
516          return submitEverywhere(task);
517       }
518    }

561    protected List<Address> randomClusterMembers(final List<Address> members, int numNeeded) {
562       if(members == null || members.isEmpty())
563          throw new IllegalArgumentException("Invalid member list " + members);
564 
565       if (members.size() < numNeeded) {
566          log.cannotSelectRandomMembers(numNeeded, members);
567          numNeeded = members.size();
568       }
569       List<Address> membersCopy = new ArrayList<Address>(members);
570       List<Address> chosen = new ArrayList<Address>(numNeeded);
571       Random r = new Random();
572       while (!membersCopy.isEmpty() && numNeeded >= chosen.size()) {
573          int count = membersCopy.size();
574          Address address = membersCopy.remove(r.nextInt(count));
575          chosen.add(address);
576       }
577       return chosen;
578    }

580    protected <K> Map<Address, List<K>> keysToExecutionNodes(DistributedTaskExecutionPolicy policy, K... input) {
581       DistributionManager dm = cache.getDistributionManager();
582       Map<Address, List<K>> addressToKey = new HashMap<Address, List<K>>(input.length * 2);
583       boolean usingREPLMode = dm == null;
584       for (K key : input) {
585          Address ownerOfKey = null;
586          if (usingREPLMode) {
587             List<Address> members = new ArrayList<Address>(getMembers());
588             members =  filterMembers(policy, members);
589             // using REPL mode https://issues.jboss.org/browse/ISPN-1886
590             // since keys and values are on all nodes, lets just pick randomly
591             Collections.shuffle(members);
592             ownerOfKey = members.get(0);
593          } else {
594             // DIST mode
595             List<Address> owners = dm.locate(key);
596             List<Address> filtered = filterMembers(policy, owners);
597             if(!filtered.isEmpty()){
598                ownerOfKey = filtered.get(0);
599             } else {
600                ownerOfKey = owners.get(0);
601             }
602          }
603          List<K> keysAtNode = addressToKey.get(ownerOfKey);
604          if (keysAtNode == null) {
605             keysAtNode = new LinkedList<K>();
606             addressToKey.put(ownerOfKey, keysAtNode);
607          }
608          keysAtNode.add(key);
609       }
610       return addressToKey;
611    }

613    private List<Address> filterMembers(DistributedTaskExecutionPolicy policy, List<Address> members) {
614       NodeFilter filter = null;
615       switch (policy) {
616          case SAME_MACHINE:
617             filter = SAME_MACHINE_FILTER;
618             break;
619          case SAME_SITE:
620             filter = SAME_SITE_FILTER;
621             break;
622          case SAME_RACK:
623             filter = SAME_RACK_FILTER;
624             break;
625          case ALL:
626             filter = ALL_FILTER;
627             break;
628          default:
629             filter = ALL_FILTER;
630             break;
631       }
632       List<Address> result = new ArrayList<Address>();
633       for (Address address : members) {
634          if(address instanceof TopologyAwareAddress){
635             TopologyAwareAddress taa = (TopologyAwareAddress)address;
636             if(filter.include(taa, (TopologyAwareAddress)getAddress())){
637                result.add(address);
638             }
639          } else {
640             result.add(address);
641          }
642       }
643       return result;
644    }

647       AuthorizationManager authorizationManager = SecurityActions.getCacheAuthorizationManager(cache);

669    private static class RandomNodeTaskFailoverPolicy implements DistributedTaskFailoverPolicy {
670 
671       public RandomNodeTaskFailoverPolicy() {
672          super();
673       }
674 
675       @Override
676       public Address failover(FailoverContext fc) {
677          return randomNode(fc.executionCandidates(),fc.executionFailureLocation());
678       }
679 
680       protected Address randomNode(List<Address> candidates, Address failedExecutionLocation){
681          Random r = new Random();
682          candidates.remove(failedExecutionLocation);
683          if (candidates.isEmpty())
684             throw new IllegalStateException("There are no candidates for failover: " + candidates);
685          int tIndex = r.nextInt(candidates.size());
686          return candidates.get(tIndex);
687       }
688 
689       @Override
690       public int maxFailoverAttempts() {
691          return 1;
692       }
693    }

794    /**
795     * DistributedTaskPart represents a unit of work sent to remote VM and executed there
796     *
797     *
798     * @author Mircea Markus
799     * @author Vladimir Blagojevic
800     */
801    private abstract class DistributedTaskPart<V> implements NotifyingFuture<V>, RunnableFuture<V> {
802 
803       protected final DistributedExecuteCommand<V> distCommand;
804       private final List<Object> inputKeys;
805       private final DistributedTask<V> owningTask;
806       private int failedOverCount;
807       private volatile boolean cancelled;
808 
809       protected DistributedTaskPart(List<Object> inputKeys, DistributedExecuteCommand<V> command, DistributedTask<V> task, int failedOverCount) {
810          this.inputKeys = inputKeys;
811          this.distCommand = command;
812          this.owningTask = task;
813          this.failedOverCount = failedOverCount;
814       }
815 
816       public List<Object> getInputKeys() {
817          return inputKeys;
818       }
819 
820       public DistributedExecuteCommand<V> getCommand() {
821          return distCommand;
822       }
823 
824       public DistributedTask<V> getOwningTask() {
825          return owningTask;
826       }
827 
828       public abstract Address getExecutionTarget();
829 
830       private DefaultExecutorService getOuterType() {
831          return DefaultExecutorService.this;
832       }
833 
834       public abstract void execute();
835 
836       @Override
837       public void run() {
838          //intentionally empty
839       }
840 
841       @Override
842       public boolean isCancelled() {
843          return cancelled;
844       }
845 
846       @Override
847       public V get() throws InterruptedException, ExecutionException {
848          try {
849             return innerGet(0, TimeUnit.MILLISECONDS);
850          } catch (TimeoutException e) {
851             throw new ExecutionException(e);
852          }
853       }
854 
855       @Override
856       public V get(long timeout, TimeUnit unit)
857             throws InterruptedException, ExecutionException, TimeoutException {
858          return innerGet(timeout, unit);
859       }
860 
861       protected V innerGet(long timeout, TimeUnit unit)
862             throws ExecutionException, TimeoutException, InterruptedException {
863          if (isCancelled())
864             throw new CancellationException("Task already cancelled");
865 
866          long timeoutNanos = computeTimeoutNanos(timeout, unit);
867          long endNanos = timeService.expectedEndTime(timeoutNanos, TimeUnit.NANOSECONDS);
868          try {
869             return getResult(timeoutNanos);
870          } catch (TimeoutException te) {
871             throw te;
872          } catch (Exception e) {
873             // The RPC could have finished with a org.infinispan.util.concurrent.TimeoutException right before
874             // the Future.get timeout expired. If that's the case, we want to throw a TimeoutException.
875             if (e instanceof ExecutionException && e.getCause() instanceof org.infinispan.util.concurrent.TimeoutException) {
876                if (trace) log.tracef(
877                      "Distributed task timed out, throwing a TimeoutException and ignoring exception", e);
878                throw new TimeoutException();
879             }
880             boolean canFailover = failedOverCount++ < getOwningTask().getTaskFailoverPolicy().maxFailoverAttempts();
881             if (canFailover) {
882                try {
883                   return failoverExecution(e, timeoutNanos, TimeUnit.NANOSECONDS);
884                } catch (Exception failedOver) {
885                   throw wrapIntoExecutionException(failedOver);
886                }
887             } else {
888                throw wrapIntoExecutionException(e);
889             }
890          }
891       }
892 
893       protected abstract V getResult(long timeoutNanos) throws Exception;
894 
895       protected long computeTimeoutNanos(long timeout, TimeUnit unit) {
896          long taskTimeout = TimeUnit.MILLISECONDS.toNanos(getOwningTask().timeout());
897          long futureTimeout = TimeUnit.NANOSECONDS.convert(timeout, unit);
898          long actualTimeout;
899          if (taskTimeout > 0 && futureTimeout > 0) {
900             actualTimeout = Math.min(taskTimeout, futureTimeout);
901          } else {
902             actualTimeout = Math.max(taskTimeout, futureTimeout);
903          }
904          return actualTimeout;
905       }
906 
907       protected ExecutionException wrapIntoExecutionException(Exception e){
908          if (e instanceof ExecutionException) {
909             return (ExecutionException) e;
910          } else {
911             return new ExecutionException(e);
912          }
913       }
914 
915       protected V failoverExecution(final Exception cause, long timeout, TimeUnit unit)
916             throws Exception {
917          final List<Address> executionCandidates = executionCandidates(getOwningTask());
918          FailoverContext fc = new FailoverContext() {
919             @Override
920             public <K> List<K> inputKeys() {
921                return (List<K>) getInputKeys();
922             }
923 
924             @Override
925             public Address executionFailureLocation() {
926                return getExecutionTarget();
927             }
928 
929             @Override
930             public List<Address> executionCandidates() {
931                return executionCandidates;
932             }
933 
934             @Override
935             public Throwable cause() {
936                return cause;
937             }
938          };
939 
940          Address failoverTarget = getOwningTask().getTaskFailoverPolicy().failover(fc);
941          log.distributedTaskFailover(fc.executionFailureLocation(), failoverTarget, cause);
942          DistributedTaskPart<V> part = createDistributedTaskPart(owningTask, distCommand,
943                getInputKeys(), failoverTarget, failedOverCount);
944          part.execute();
945          return part.get(timeout, unit);
946       }
947 
948       protected void setCancelled() {
949          cancelled = true;
950       }
951    }

953    private class RemoteDistributedTaskPart<V> extends DistributedTaskPart<V> {
954       private final Address executionTarget;
955       private CompletableFuture<Map<Address, Response>> future;
956 
957       public RemoteDistributedTaskPart(DistributedTask<V> task, DistributedExecuteCommand<V> command,
958                                  List<Object> inputKeys, Address executionTarget, int failoverCount) {
959          super(inputKeys, command, task, failoverCount);
960          if (getAddress().equals(executionTarget)) {
961             throw new IllegalArgumentException("This task should be executed as local.");
962          }
963          this.executionTarget = executionTarget;
964       }
965 
966       @Override
967       public Address getExecutionTarget() {
968          return executionTarget;
969       }
970 
971       @Override
972       public void execute() {
973          if (trace) log.tracef("Sending %s to remote execution at node %s", this, getExecutionTarget());
974          try {
975             CompletableFuture<Map<Address, Response>> remoteFuture = rpc.invokeRemotelyAsync(
976                   Collections.singletonList(getExecutionTarget()), getCommand(), rpc.getRpcOptionsBuilder(
977                         ResponseMode.SYNCHRONOUS).timeout(getOwningTask().timeout(), TimeUnit.MILLISECONDS)
978                         .build());
979             future = remoteFuture;
980          } catch (Throwable e) {
981             log.remoteExecutionFailed(getExecutionTarget(), e);
982          }
983       }
984 
985       @Override
986       public boolean cancel(boolean mayInterruptIfRunning) {
987          if (!isCancelled()) {
988             CancelCommand ccc = factory.buildCancelCommandCommand(distCommand.getUUID());
989             rpc.invokeRemotely(Collections.singletonList(getExecutionTarget()), ccc, rpc.getDefaultRpcOptions(true));
990             setCancelled();
991             return future.cancel(true);
992          } else {
993             //already cancelled
994             return false;
995          }
996       }
997 
998       @Override
999       public boolean isDone() {
1000          return future.isDone();
1001       }
1002 
1003       @Override
1004       protected V getResult(long timeoutNanos) throws Exception {
1005          if (timeoutNanos > 0) {
1006             return retrieveResult(future.get(timeoutNanos, TimeUnit.NANOSECONDS));
1007          } else {
1008             return retrieveResult(future.get());
1009          }
1010       }
1011 
1012       private V retrieveResult(Object response) throws Exception {
1013          V result = null;
1014          //this is application level Exception that was raised in execution
1015          //simply rethrow it (might be good candidate for failover)
1016          if (response instanceof Exception) {
1017             throw ((Exception) response);
1018          }
1019          //these two should never happen, mark them with IllegalStateException
1020          if (response == null || !(response instanceof Map<?, ?>)) {
1021             throw new IllegalStateException("Invalid response received " + response);
1022          }
1023          Map<Address, Response> mapResult = (Map<Address, Response>) response;
1024          if (mapResult.size() == 1) {
1025             for (Entry<Address, Response> e : mapResult.entrySet()) {
1026                Response value = e.getValue();
1027                if (value instanceof SuccessfulResponse) {
1028                   result = (V) ((SuccessfulResponse) value).getResponseValue();
1029                } else {
1030                   throw new ExecutionException(new Exception(value != null ? value.toString() : "Unknown cause"));
1031                }
1032             }
1033          } else {
1034             //should never happen as we send DistributedTaskPart to one node for
1035             //execution only, therefore we should get only one response
1036             throw new IllegalStateException("Invalid response " + response);
1037          }
1038          return result;
1039       }
1040 
1041       @Override
1042       public NotifyingFuture<V> attachListener(final FutureListener<V> listener) {
1043          future.whenComplete((value, throwable) -> {
1044             listener.futureDone(this);
1045          });
1046          return this;
1047       }
1048    }

1050    private class LocalDistributedTaskPart<V> extends DistributedTaskPart<V> {
1051       private final NotifyingFutureImpl<V> future = new NotifyingFutureImpl<V>();
1052 
1053       public LocalDistributedTaskPart(DistributedTask<V> task, DistributedExecuteCommand<V> command,
1054                List<Object> inputKeys, int failoverCount) {
1055          super(inputKeys, command, task, failoverCount);
1056       }
1057 
1058       @Override
1059       public boolean isDone() {
1060          return future.isDone();
1061       }
1062 
1063       @Override
1064       public synchronized boolean cancel(boolean mayInterruptIfRunning) {
1065          if (!isCancelled()) {
1066             CancelCommand ccc = factory.buildCancelCommandCommand(distCommand.getUUID());
1067             ccc.init(cancellationService);
1068             try {
1069                ccc.perform(null);
1070             } catch (Throwable e) {
1071                log.couldNotExecuteCancellationLocally(e.getLocalizedMessage());
1072             }
1073             setCancelled();
1074             return future.cancel(true);
1075          } else {
1076             //already cancelled
1077             return false;
1078          }
1079       }
1080 
1081       @Override
1082       protected V getResult(long timeoutNanos) throws Exception {
1083          if (timeoutNanos > 0) {
1084             return future.get(timeoutNanos, TimeUnit.NANOSECONDS);
1085          } else {
1086             return future.get();
1087          }
1088       }
1089 
1090       @Override
1091       public Address getExecutionTarget() {
1092          return getAddress();
1093       }
1094 
1095       @Override
1096       public void execute() {
1097          log.debugf("Sending %s to self", this);
1098          try {
1099             Callable<V> call = new Callable<V>() {
1100 
1101                @Override
1102                public V call() throws Exception {
1103                   return doLocalInvoke();
1104                }
1105 
1106                private V doLocalInvoke() throws Exception {
1107                   getCommand().init(cache);
1108                   DistributedTaskLifecycleService lifecycle = DistributedTaskLifecycleService.getInstance();
1109                   try {
1110                      // hook into lifecycle
1111                      lifecycle.onPreExecute(getCommand().getCallable(), cache);
1112                      cancellationService.register(Thread.currentThread(), getCommand().getUUID());
1113                      V result = getCommand().perform(null);
1114                      future.notifyDone(result);
1115                      return result;
1116                   } catch (Exception e) {
1117                      future.notifyException(e);
1118                      throw e;
1119                   } finally {
1120                      // hook into lifecycle
1121                      lifecycle.onPostExecute(getCommand().getCallable());
1122                      cancellationService.unregister(getCommand().getUUID());
1123                   }
1124                }
1125             };
1126             future.setFuture(localExecutorService.submit(call));
1127          } catch (Throwable e1) {
1128             log.localExecutionFailed(e1);
1129          }
1130       }
1131 
1132       @Override
1133       public NotifyingFuture<V> attachListener(final FutureListener<V> listener) {
1134          future.attachListener(listener);
1135          return this;
1136       }
1137    }
