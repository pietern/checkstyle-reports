22     public static <T> Traversable<T> of(Stream<T> stream) {
23        return new StreamTraversable<>(stream);
24     }

26     public static <T> CloseableIterator<T> asIterator(Traversable<T> traversable) {
27        if (traversable instanceof StreamTraversable)
28           return Closeables.iterator(((StreamTraversable<T>) traversable).stream);
29        else {
30           List<T> collected = traversable.collect(ArrayList::new, ArrayList::add, ArrayList::addAll);
31           return Closeables.iterator(collected.iterator());
32        }
33     }

40     private static final class StreamTraversable<T> implements Traversable<T> {
41        final Stream<T> stream;
42  
43        private StreamTraversable(Stream<T> stream) {
44           this.stream = stream;
45        }
46  
47        @Override
48        public Traversable<T> filter(Predicate<? super T> p) {
49           return new StreamTraversable<>(stream.filter(p));
50        }
51  
52        @Override
53        public <R> Traversable<R> map(Function<? super T, ? extends R> f) {
54           return new StreamTraversable<>(stream.map(f));
55        }
56  
57        @Override
58        public <R> Traversable<R> flatMap(Function<? super T, ? extends Traversable<? extends R>> f) {
59           Function<? super T, ? extends Stream<? extends R>> mapper = new Function<T, Stream<? extends R>>() {
60              @Override
61              public Stream<? extends R> apply(T t) {
62                 StreamTraversable<? extends R> applied = (StreamTraversable<? extends R>) f.apply(t);
63                 return applied.stream;
64              }
65           };
66  
67           return new StreamTraversable<>(stream.flatMap(mapper));
68        }
69  
70        @Override
71        public void forEach(Consumer<? super T> c) {
72           stream.forEach(c);
73        }
74  
75        @Override
76        public T reduce(T z, BinaryOperator<T> folder) {
77           return stream.reduce(z, folder);
78        }
79  
80        @Override
81        public Optional<T> reduce(BinaryOperator<T> folder) {
82           return stream.reduce(folder);
83        }
84  
85        @Override
86        public <U> U reduce(U z, BiFunction<U, ? super T, U> mapper, BinaryOperator<U> folder) {
87           return stream.reduce(z, mapper, folder);
88        }
89  
90        @Override
91        public <R> R collect(Supplier<R> s, BiConsumer<R, ? super T> accumulator, BiConsumer<R, R> combiner) {
92           return stream.collect(s, accumulator, combiner);
93        }
94  
95        @Override
96        public <R, A> R collect(Collector<? super T, A, R> collector) {
97           return stream.collect(collector);
98        }
