16  /**
17   * This test is useful to test how much memory is in use by the data container.  Since the Java GC may not clean up
18   * everything on 1 pass, we have to run multiple passes through until we get a number that is relatively stable.
19   * @author William Burns
20   */
21  @Test(groups = "profiling", testName = "eviction.MemoryEvictionTest")
22  public class MemoryEvictionTest extends SingleCacheManagerTest {
23  
24     private final long MAX_MEMORY = 400 * 1000 * 1000;
25     private final int MATCH_COUNT = 5;
26  
27     private final static Log log = LogFactory.getLog(MemoryEvictionTest.class);
28  
29     @Override
30     protected EmbeddedCacheManager createCacheManager() throws Exception {
31        ConfigurationBuilder cfg = new ConfigurationBuilder();
32        cfg
33           .eviction().strategy(EvictionStrategy.LRU).type(EvictionType.MEMORY).size(MAX_MEMORY)
34           .build();
35        EmbeddedCacheManager cm = TestCacheManagerFactory.createCacheManager(cfg);
36        cache = cm.getCache();
37        return cm;
38     }
39  
40     public void testSimpleSizeEviction() {
41        log.debugf("Max memory: %d", MAX_MEMORY);
42        DataContainer dc = cache.getAdvancedCache().getDataContainer();
43        printMemoryUsage(dc.size());
44        Random random = new Random();
45        int matchCount = 0;
46        int byteKeySize = 10;
47        int byteValueSize = 100;
48        long previousMemorySize = 0;
49        for (int j = 0; j < 200; ++j) {
50           while (matchCount < this.MATCH_COUNT) {
51              for (int i = 0; i < 20000; ++i) {
52                 byte[] keyBytes = new byte[byteKeySize];
53                 byte[] valueBytes = new byte[byteValueSize];
54                 random.nextBytes(keyBytes);
55                 random.nextBytes(valueBytes);
56                 cache.getAdvancedCache().put(keyBytes, valueBytes);
57              }
58              long memorySize = printMemoryUsage(dc.size());
59              if (memorySize == previousMemorySize) {
60                 matchCount++;
61              }
62              previousMemorySize = memorySize;
63           }
64        }
65     }
66  
67     // Also returns size
68     private long printMemoryUsage(int cacheSize) {
69        System.gc(); System.gc();
70        Runtime runtime = Runtime.getRuntime();
71        long usedMemory = runtime.totalMemory() - runtime.freeMemory();
72        log.debugf("Used memory = %d, cache size = %d", usedMemory, cacheSize);
73        return usedMemory;
74     }
75  }
