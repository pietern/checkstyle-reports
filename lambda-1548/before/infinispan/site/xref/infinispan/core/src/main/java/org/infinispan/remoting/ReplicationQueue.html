4   import org.infinispan.commons.api.Lifecycle;
5   
6   /**
7    * Periodically (or when certain size is exceeded) takes elements and replicates them.
8    *
9    * @author <a href="mailto:bela@jboss.org">Bela Ban</a>
10   * @author Mircea.Markus@jboss.com
11   * @since 4.0
12   */
13  public interface ReplicationQueue extends Lifecycle {
14  
15  
16     /**
17      * @return true if this replication queue is enabled, false otherwise.
18      */
19     boolean isEnabled();
20  
21     /**
22      * Adds a new command to the replication queue.
23      *
24      * @param job command to add to the queue
25      */
26     void add(ReplicableCommand job);
27  
28     /**
29      * Flushes existing jobs in the replication queue, and returns the number of jobs flushed.
30      * @return the number of jobs flushed
31      */
32     int flush();
33  
34     /**
35      * @return the number of elements in the replication queue.
36      */
37     int getElementsCount();
38  
39     /**
40      * Resets the replication queue, typically used when a cache is restarted.
41      */
42     void reset();
43  }
