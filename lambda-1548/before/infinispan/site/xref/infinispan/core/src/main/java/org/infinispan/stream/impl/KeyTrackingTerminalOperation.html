11  /**
12   * A terminal operation for a {@link org.infinispan.CacheStream} that allows tracking keys during a rehash event.
13   * @param <K> key type for the entry returned
14   * @param <R> return type when not utilizing rehash aware
15   * @param <R2> value type for the entry returned
16   */
17  public interface KeyTrackingTerminalOperation<K, R, R2> extends SegmentAwareOperation {
18     /**
19      * Collector used to collect items from intermediate responses of operations
20      * @param <C> type of collected item
21      */
22     interface IntermediateCollector<C> {
23        /**
24         * Called back when a response is sent back to the invoker
25         * @param response the returned data
26         */
27        void sendDataResonse(C response);
28     }
29  
30     /**
31      * Invoked when a key aware operation is desired without rehash being enabled.
32      * @param response the collector that will be called back for any intermediate results
33      * @return the final response from the remote node
34      */
35     Collection<R> performOperation(IntermediateCollector<Collection<R>> response);
36  
37     /**
38      * Invoked when a key and rehash aware operation is desired.
39      * @param response the collector that will be called back for any intermediate results
40      * @return the final response from the remote node
41      */
42     Collection<CacheEntry<K, R2>> performOperationRehashAware(
43             IntermediateCollector<Collection<CacheEntry<K, R2>>> response);
44  
45     /**
46      * This method is to be invoked only locally after a key tracking operation has been serialized to a new node
47      * @param supplier the supplier to use
48      */
49     void setSupplier(Supplier<? extends Stream<?>> supplier);
50  
51     /**
52      * Handles injection of components for various intermediate and this operation.
53      * @param registry component registry to use
54      */
55     void handleInjection(ComponentRegistry registry);
56  }
