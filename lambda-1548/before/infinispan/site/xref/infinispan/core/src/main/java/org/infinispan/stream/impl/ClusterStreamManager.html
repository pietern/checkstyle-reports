15  /**
16   * Manages distribution of various stream operations that are sent to remote nodes.  Note usage of any operations
17   * should <b>always</b> be accompanied with a subsequent call to {@link ClusterStreamManager#forgetOperation(UUID)}
18   * so that the operation is fully released.  This is important especially for early terminating operations.
19   * @param <K> The key type for the underlying cache
20   */
21  public interface ClusterStreamManager<K> {
22     /**
23      * A callback that is used for result processing from the remote nodes.
24      * @param <R> the type of results returned
25      */
26     interface ResultsCallback<R> {
27        /**
28         * Called back for intermediate data returned from an operation.  This is useful for operations that utilized
29         * batch fetching such as {@link CacheStream#iterator()}, {@link CacheStream#spliterator()},
30         * {@link CacheStream#forEach(Consumer)} and {@link CacheStream#toArray()}.
31         * @param address Which node this data came from
32         * @param results The results obtained so far.
33         * @return the segments that completed with some value
34         */
35        Set<Integer> onIntermediateResult(Address address, R results);
36  
37        /**
38         * Essentially the same as {@link ClusterStreamManager.ResultsCallback#onIntermediateResult(Address address, Object)}
39         * except that this is the last time this callback will be invoked and it tells which segments were completed
40         * @param address Which node this data came from
41         * @param results The last batch of results for this operator
42         */
43        void onCompletion(Address address, Set<Integer> completedSegments, R results);
44  
45        /**
46         * Called back when a segment is found to have been lost that is no longer remote
47         * This method should return as soon as possible and not block in any fashion.
48         * This method may be invoked concurrently with any of the other methods
49         * @param segments The segments that were requested but are now local
50         */
51        void onSegmentsLost(Set<Integer> segments);
52     }

54     /**
55      * Performs the remote stream operation without rehash awareness.
56      * @param parallelDistribution whether or not parallel distribution is enabled
57      * @param parallelStream whether or not the stream is paralllel
58      * @param ch the consistent hash to use when determining segment ownership
59      * @param segments the segments that this request should utilize
60      * @param keysToInclude which keys to include in the request
61      * @param keysToExclude which keys to exclude in the request
62      * @param includeLoader whether or not to use a loader
63      * @param operation the actual operation to perform
64      * @param callback the callback to collect individual node results
65      * @param earlyTerminatePredicate a predicate to determine if this operation should stop based on intermediate results
66      * @param <R> the type of response
67      * @return the operation id to be used for further calls
68      */
69     <R> UUID remoteStreamOperation(boolean parallelDistribution, boolean parallelStream, ConsistentHash ch,
70             Set<Integer> segments, Set<K> keysToInclude, Map<Integer, Set<K>> keysToExclude, boolean includeLoader,
71             TerminalOperation<R> operation, ResultsCallback<R> callback, Predicate<? super R> earlyTerminatePredicate);
72  
73     /**
74      * Performs the remote stream operation with rehash awareness.
75      * @param parallelDistribution whether or not parallel distribution is enabled
76      * @param parallelStream whether or not the stream is paralllel
77      * @param ch the consistent hash to use when determining segment ownership
78      * @param segments the segments that this request should utilize
79      * @param keysToInclude which keys to include in the request
80      * @param keysToExclude which keys to exclude in the request
81      * @param includeLoader whether or not to use a loader
82      * @param operation the actual operation to perform
83      * @param callback the callback to collect individual node results
84      * @param earlyTerminatePredicate a predicate to determine if this operation should stop based on intermediate results
85      * @param <R> the type of response
86      * @return the operation id to be used for further calls
87      */
88     <R> UUID remoteStreamOperationRehashAware(boolean parallelDistribution, boolean parallelStream, ConsistentHash ch,
89             Set<Integer> segments, Set<K> keysToInclude, Map<Integer, Set<K>> keysToExclude, boolean includeLoader,
90             TerminalOperation<R> operation, ResultsCallback<R> callback, Predicate<? super R> earlyTerminatePredicate);

92     /**
93      * Key tracking remote operation that doesn't have rehash enabled.
94      * @param parallelDistribution whether or not parallel distribution is enabled
95      * @param parallelStream whether or not the stream is paralllel
96      * @param ch the consistent hash to use when determining segment ownership
97      * @param segments the segments that this request should utilize
98      * @param keysToInclude which keys to include in the request
99      * @param keysToExclude which keys to exclude in the request
100     * @param includeLoader whether or not to use a loader
101     * @param operation the actual operation to perform
102     * @param callback the callback to collect individual node results
103     * @param <R> the type of response
104     * @return the operation id to be used for further calls
105     */
106    <R> UUID remoteStreamOperation(boolean parallelDistribution, boolean parallelStream, ConsistentHash ch,
107            Set<Integer> segments, Set<K> keysToInclude, Map<Integer, Set<K>> keysToExclude, boolean includeLoader,
108            KeyTrackingTerminalOperation<K, R, ?> operation, ResultsCallback<Collection<R>> callback);

115     * @param segments the segments that this request should utilize

117     * @param keysToExclude which keys to exclude in the request

125                                                  Set<Integer> segments, Set<K> keysToInclude,

130    /**
131     * Tests whether this operation is still pending or not.
132     * @param id the id of the operation that was returned from the invocation
133     * @return whether or not it is completed
134     */
135    boolean isComplete(UUID id);
136 
137    /**
138     * Awaits completion of the given request.  Returns true when the request completes otherwise returns false after
139     * the time elapses
140     * @param id the ide of the operation that was returned from the invocation
141     * @param time how long to wait before returning false
142     * @param unit controls how long the time wait is
143     * @return whether or not the request is complete
144     */
145    boolean awaitCompletion(UUID id, long time, TimeUnit unit) throws InterruptedException;
146 
147    /**
148     * Frees any resources related to this operation and signals to any ongoing remote operations to no longer continue
149     * processing
150     * @param id the ide of the operation that was returned from the invocation
151     */
152    void forgetOperation(UUID id);
153 
154    /**
155     * Receives a response for a given request
156     * @param id The request id
157     * @param origin The origin of the response
158     * @param complete Whether or not this is a completed response
159     * @param segments The segments that were suspected
160     * @param response The actual response value
161     * @param <R1> The type of the response
162     * @return Whether or not the operation should continue operating, only valid if complete was false
163     */
164    <R1> boolean receiveResponse(UUID id, Address origin, boolean complete, Set<Integer> segments, R1 response);
