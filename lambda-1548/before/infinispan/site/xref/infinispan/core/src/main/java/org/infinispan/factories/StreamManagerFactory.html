1   package org.infinispan.factories;
2   
3   import org.infinispan.factories.annotations.DefaultFactoryFor;
4   import org.infinispan.stream.impl.ClusterStreamManager;
5   import org.infinispan.stream.impl.LocalStreamManager;
6   import org.infinispan.stream.impl.ClusterStreamManagerImpl;
7   import org.infinispan.stream.impl.LocalStreamManagerImpl;
8   import org.infinispan.stream.impl.PartitionAwareClusterStreamManager;
9   
10  /**
11   * Factory that allows creation of a {@link LocalStreamManager} or {@link ClusterStreamManager} based on the provided
12   * configuration.
13   *
14   * @author wburns
15   * @since 8.0
16   */
17  @DefaultFactoryFor(classes = {LocalStreamManager.class, ClusterStreamManager.class})
18  public class StreamManagerFactory extends AbstractNamedCacheComponentFactory implements AutoInstantiableFactory {
19     @Override
20     public <T> T construct(Class<T> componentType) {
21        if (configuration.clustering().cacheMode().isDistributed()) {
22           if (componentType.equals(LocalStreamManager.class)) {
23              return componentType.cast(new LocalStreamManagerImpl<>());
24           }
25           if (componentType.equals(ClusterStreamManager.class)) {
26              if (configuration.clustering().partitionHandling().enabled()) {
27                 return componentType.cast(new PartitionAwareClusterStreamManager<>());
28              } else {
29                 return componentType.cast(new ClusterStreamManagerImpl<>());
30              }
31           }
32        }
33        return null;
34     }
35  }
