8   /**
9    * Entry Size calculator that returns an approximation of how much various primitives, primitive wrappers, Strings,
10   * and arrays
11   * @author wburns
12   * @since 8.0
13   */
14  public class PrimitiveEntrySizeCalculator extends AbstractEntrySizeCalculatorHelper<Object, Object> {
15     public long calculateSize(Object key, Object value) {
16        return handleObject(key) + handleObject(value);
17     }
18  
19     protected long handleObject(Object object) {
20        Class<?> objClass = object.getClass();
21        if (objClass == String.class) {
22           String realString = (String) object;
23           // The string is an object and has a reference to its class, int for the hash code and a pointer to the char[]
24           long objectSize = roundUpToNearest8(OBJECT_SIZE + POINTER_SIZE + 4 + POINTER_SIZE);
25           // We then include the char[] offset and size
26           return objectSize + roundUpToNearest8(Unsafe.ARRAY_CHAR_BASE_OFFSET + realString.length() *
27                   Unsafe.ARRAY_CHAR_INDEX_SCALE);
28        } else if (objClass == Long.class) {
29           long longValue = ((Long) object).longValue();
30           if (longValue >= LongCacheConstraints.MIN_CACHE_VALUE &&
31                   longValue <= LongCacheConstraints.MAX_CACHE_VALUE) {
32              return 0;
33           }
34           // We add in the size for a long, plus the object reference and the class ref
35           return roundUpToNearest8(Unsafe.ARRAY_LONG_INDEX_SCALE + OBJECT_SIZE + POINTER_SIZE);
36        } else if (objClass == Integer.class) {
37           int intValue = ((Integer) object).intValue();
38           if (intValue >= IntegerCacheConstraints.MIN_CACHE_VALUE &&
39                   intValue <= IntegerCacheConstraints.MAX_CACHE_VALUE) {
40              return 0;
41           }
42           // We add in the size for a long, plus the object reference and the class ref
43           return roundUpToNearest8(Unsafe.ARRAY_INT_INDEX_SCALE + OBJECT_SIZE + POINTER_SIZE);
44        } else if (objClass == Short.class) {
45           short shortValue = ((Short) object).shortValue();
46           if (shortValue >= ShortCacheConstraints.MIN_CACHE_VALUE &&
47                   shortValue <= ShortCacheConstraints.MAX_CACHE_VALUE) {
48              return 0;
49           }
50           return roundUpToNearest8(Unsafe.ARRAY_SHORT_INDEX_SCALE + OBJECT_SIZE + POINTER_SIZE);
51        } else if (objClass == Double.class) {
52           return roundUpToNearest8(Unsafe.ARRAY_DOUBLE_INDEX_SCALE + OBJECT_SIZE + POINTER_SIZE);
53        } else if (objClass == Float.class) {
54           return roundUpToNearest8(Unsafe.ARRAY_FLOAT_INDEX_SCALE + OBJECT_SIZE + POINTER_SIZE);
55        } else if (objClass == Boolean.class) {
56           // We assume all provided booleans are cached
57           return 0;
58        } else if (objClass == Character.class) {
59           char charValue = ((Character) object).charValue();
60           if (charValue >= CharacterCacheConstraints.MIN_CACHE_VALUE &&
61                   charValue <= CharacterCacheConstraints.MAX_CACHE_VALUE) {
62              return 0;
63           }
64           return roundUpToNearest8(Unsafe.ARRAY_CHAR_INDEX_SCALE + OBJECT_SIZE + POINTER_SIZE);
65        } else if (objClass == Byte.class) {
66           // All byte values are cached
67           return 0;
68        } else if (objClass.isArray()) {
69           // We assume the array is of a type that supports shallow copy, such as the ones above
70           // We don't verify cached values if the array contains Booleans for example
71           Unsafe unsafe = getUnsafe();
72           Class<?> compClass = objClass.getComponentType();
73           int arrayLength = Array.getLength(object);
74           // Every array has a base offset which defines how large the array is and other overhead.
75           // Then each element in the array is indexed contiguously in memory thus we can simply multiply how
76           // many elements are in the array by how much of an offset each element requires.  A normal object for example
77           // takes up the standard Object pointer worth of size but primitives take up space equivalent to how many byte
78           // they occupy.
79           long arraySize = roundUpToNearest8(unsafe.arrayBaseOffset(objClass) + unsafe.arrayIndexScale(objClass) *
80                   arrayLength);
81           // If the component type isn't primitive we have to add in each of the instances
82           if (!compClass.isPrimitive()) {
83              // TODO: we could assume some values for given primitive wrappers.
84              for (int i = 0; i < arrayLength; ++i) {
85                 arraySize += handleObject(Array.get(object, i));
86              }
87           }
88           return arraySize;
89        } else {
90           throw new IllegalArgumentException("Size of Class " + objClass +
91                   " cannot be determined using given entry size calculator :" + getClass());
92        }
93     }

95     static class CharacterCacheConstraints {
96        static final short MAX_CACHE_VALUE = 127;
97        static final short MIN_CACHE_VALUE = 0;
98     }
99  
100    static class ShortCacheConstraints {
101       static final short MAX_CACHE_VALUE = 127;
102       static final short MIN_CACHE_VALUE = -128;
103    }
104 
105    static class LongCacheConstraints {
106       static final long MAX_CACHE_VALUE = 127;
107       static final long MIN_CACHE_VALUE = -128;
108    }
109 
110    static class IntegerCacheConstraints {
111       static final int MAX_CACHE_VALUE = calculateMaxIntCache();
112       static final int MIN_CACHE_VALUE = -128;
113 
114       static int calculateMaxIntCache() {
115          int h = 127;
116          String integerCacheHighPropValue =
117                  sun.misc.VM.getSavedProperty("java.lang.Integer.IntegerCache.high");
118          if (integerCacheHighPropValue != null) {
119             try {
120                int i = Integer.parseInt(integerCacheHighPropValue);
121                i = Math.max(i, 127);
122                // Maximum array size is Integer.MAX_VALUE
123                h = Math.min(i, Integer.MAX_VALUE - (-128) -1);
124             } catch( NumberFormatException nfe) {
125                // If the property cannot be parsed into an int, ignore it.
126             }
127          }
128          return h;
129       }
130    }

132    /**
133     * Returns a sun.misc.Unsafe.  Suitable for use in a 3rd party package.
134     * Replace with a simple call to Unsafe.getUnsafe when integrating
135     * into a jdk.
136     *
137     * @return a sun.misc.Unsafe
138     */
139    static Unsafe getUnsafe() {
140       try {
141          return Unsafe.getUnsafe();
142       } catch (SecurityException tryReflectionInstead) {}
143       try {
144          return java.security.AccessController.doPrivileged
145                  (new java.security.PrivilegedExceptionAction<Unsafe>() {
146                     public Unsafe run() throws Exception {
147                        Class<Unsafe> k = Unsafe.class;
148                        for (java.lang.reflect.Field f : k.getDeclaredFields()) {
149                           f.setAccessible(true);
150                           Object x = f.get(null);
151                           if (k.isInstance(x))
152                              return k.cast(x);
153                        }
154                        throw new NoSuchFieldError("the Unsafe");
155                     }});
156       } catch (java.security.PrivilegedActionException e) {
157          throw new RuntimeException("Could not initialize intrinsics",
158                  e.getCause());
159       }
160    }
