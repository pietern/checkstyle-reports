1   package org.infinispan.stream.impl;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.commons.util.CloseableIterator;
5   
6   import java.util.function.Function;
7   
8   /**
9    * A CloseableIterator implementation that allows for a CloseableIterator that doesn't allow remove operations to
10   * implement remove by delegating the call to the provided cache to remove the previously read value.  The key used
11   * to remove from the cache is determined by first applying the removeFunction to the value retrieved from the
12   * iterator.
13   *
14   * @author wburns
15   * @since 8.0
16   */
17  public class RemovableCloseableIterator<K, C> extends RemovableIterator<K, C> implements CloseableIterator<C> {
18     protected final CloseableIterator<C> realIterator;
19  
20     public RemovableCloseableIterator(CloseableIterator<C> realIterator, Cache<K, ?> cache,
21             Function<? super C, K> removeFunction) {
22        super(realIterator, cache, removeFunction);
23        this.realIterator = realIterator;
24     }
25  
26     @Override
27     public void close() {
28        currentValue = null;
29        previousValue = null;
30        realIterator.close();
31     }
32  }
