3   /**
4    * Versions can be compared to each other to result in one version being before, after or at the same time as another
5    * version.  This is different from the JDK's {@link Comparable} interface, which is much more simplistic in that it
6    * doesn't differentiate between something that is the same versus equal-but-different.
7    *
8    * @author Manik Surtani
9    * @since 5.1
10   */
11  public enum InequalVersionComparisonResult {
12     /**
13      * Denotes a version that was created temporally <i>before</i> another version.
14      */
15     BEFORE,
16     /**
17      * Denotes a version that was created temporally <i>after</i> another version.
18      */
19     AFTER,
20     /**
21      * Denotes that the two versions being compared are equal.
22      */
23     EQUAL,
24     /**
25      * Denotes a version that was created at the same time as another version, but is not equal.  This is only really
26      * useful when using a partition-aware versioning scheme, such as vector or Lamport clocks.
27      */
28     CONFLICTING
29  }
