9   import java.util.Map;

11  /**
12   * Modules which wish to implement their own commands and visitors must also provide an implementation of this interface
13   * and declare it in their <tt>infinispan-module.properties</tt> file under key <tt>infinispan.module.command.factory</tt>.
14   * <p />
15   * Implementations <b>must</b> be public classes with a public, no-arg constructor for instantiation.
16   * <p />
17   * Note that this is a {@link Scopes#GLOBAL} component and as such cannot have {@link Inject} methods referring to
18   * {@link Scopes#NAMED_CACHE} scoped components.  For such components, use a corresponding {@link Scopes#NAMED_CACHE}-scoped
19   * {@link ModuleCommandInitializer}.
20   * <p />
21   * @author Manik Surtani
22   * @since 5.0
23   */
24  @Scope(Scopes.GLOBAL)
25  public interface ModuleCommandFactory {
26     /**
27      * Provides a map of command IDs to command types of all the commands handled by the command factory instance.
28      * Unmarshalling requests for these command IDs will be dispatched to this implementation.
29      *
30      * @return map of command IDs to command types handled by this implementation.
31      */
32     Map<Byte, Class<? extends ReplicableCommand>> getModuleCommands();
33  
34     /**
35      * Construct and initialize a {@link ReplicableCommand} based on the command
36      * id and argument array passed in.
37      *
38      * @param commandId command id to construct
39      * @param args array of arguments with which to initialize the ReplicableCommand
40      * @return a ReplicableCommand
41      */
42     ReplicableCommand fromStream(byte commandId, Object[] args);
43  
44     /**
45      * Construct and initialize a {@link CacheRpcCommand} based on the command
46      * id and argument array passed in.
47      *
48      * @param commandId  command id to construct
49      * @param args       array of arguments with which to initialize the {@link CacheRpcCommand}
50      * @param cacheName  cache name at which command to be created is directed
51      * @return           a {@link CacheRpcCommand}
52      */
53     CacheRpcCommand fromStream(byte commandId, Object[] args, String cacheName);
54  }
