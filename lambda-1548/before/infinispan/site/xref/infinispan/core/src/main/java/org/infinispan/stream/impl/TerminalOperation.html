5   import java.util.function.Supplier;
6   import java.util.stream.Stream;

8   /**
9    * Interface describing an operation that is a terminal one that doesn't track keys.
10   * @param <R> the returing result
11   */
12  public interface TerminalOperation<R> extends SegmentAwareOperation {
13     /**
14      * Actually runs the terminal operation returning the result from the operation
15      * @return the value retrieved for the operation
16      */
17     R performOperation();
18  
19     /**
20      * Sets the local supplier for the stream.  This is to be invoked on a remote node after the object is unmarshalled
21      * to set the supplier to use
22      * @param supplier the supplier that will return the stream that the operations are performed on
23      */
24     void setSupplier(Supplier<? extends Stream<?>> supplier);
25  
26     /**
27      * Handles injection of components for various intermediate and this operation.
28      * @param registry component registry to use
29      */
30     void handleInjection(ComponentRegistry registry);
31  }
