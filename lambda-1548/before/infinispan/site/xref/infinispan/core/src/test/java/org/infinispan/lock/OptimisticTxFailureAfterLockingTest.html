17  /**
18   * Test the failures after lock acquired for Optimistic transactional caches.
19   *
20   * @author Pedro Ruivo
21   * @since 6.0
22   */
23  @Test(groups = "functional", testName = "lock.OptimisticTxFailureAfterLockingTest")
24  @CleanupAfterMethod
25  public class OptimisticTxFailureAfterLockingTest extends MultipleCacheManagersTest {
26  
27     /**
28      * ISPN-3556
29      */
30     public void testInOwner() throws Exception {
31        //primary owner is cache(0) and the failure is executed in cache(0)
32        doTest(0, 0);
33     }
34  
35     /**
36      * ISPN-3556
37      */
38     public void testInNonOwner() throws Exception {
39        //primary owner is cache(1) and the failure is executed in cache(0)
40        doTest(1, 0);
41     }
42  
43     @Override
44     protected void createCacheManagers() throws Throwable {
45        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
46        builder.locking()
47              .isolationLevel(IsolationLevel.REPEATABLE_READ)
48              .writeSkewCheck(true);
49        builder.transaction()
50              .lockingMode(LockingMode.OPTIMISTIC);
51        builder.clustering().hash()
52              .numOwners(2);
53        builder.versioning()
54              .enable()
55              .scheme(VersioningScheme.SIMPLE);
56        createClusteredCaches(3, builder);
57     }
58  
59     private void doTest(int primaryOwnerIndex, int execIndex) throws Exception {
60        final Object key = new MagicKey(cache(primaryOwnerIndex), cache(2));
61  
62        cache(primaryOwnerIndex).put(key, "v1");
63  
64        tm(execIndex).begin();
65        AssertJUnit.assertEquals("v1", cache(execIndex).get(key));
66        final Transaction transaction = tm(execIndex).suspend();
67  
68        cache(primaryOwnerIndex).put(key, "v2");
69  
70        tm(execIndex).resume(transaction);
71        AssertJUnit.assertEquals("v1", cache(execIndex).put(key, "v3"));
72        try {
73           tm(execIndex).commit();
74           AssertJUnit.fail("Exception expected!");
75        } catch (RollbackException e) {
76           //expected
77        }
78  
79        assertNoTransactions();
80        assertEventuallyNotLocked(cache(primaryOwnerIndex), key);
81     }
82  }
