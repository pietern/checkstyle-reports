19  @Test(groups = "functional", testName = "replication.InvokeRemotelyInFutureTest")
20  public class InvokeRemotelyInFutureTest extends MultipleCacheManagersTest {
21  
22     @Override
23     protected void createCacheManagers() throws Throwable {
24        ConfigurationBuilder c = getDefaultClusteredCacheConfig(CacheMode.REPL_SYNC, true);
25        createClusteredCaches(2, "futureRepl", c);
26     }
27  
28     public void testInvokeRemotelyInFutureWithListener() throws Exception {
29  
30        Cache cache1 = cache(0,"futureRepl");
31  
32        //never mind use of KeyValueCommand, the point is that callback is called and completes without Exceptions thrown
33        final AtomicBoolean futureDoneOk = new AtomicBoolean();
34        NotifyingFutureImpl<Map<Address, Response>> f = new NotifyingFutureImpl<>();
35        f.attachListener(future -> {
36           try {
37              future.get();
38              futureDoneOk.set(true);
39           } catch (Exception e) {
40              log.errorf(e, "Error reading future result");
41           }
42        });
43        CommandsFactory cf = cache1.getAdvancedCache().getComponentRegistry().getComponent(CommandsFactory.class);
44        PutKeyValueCommand put = cf.buildPutKeyValueCommand("k", "v",
45              new EmbeddedMetadata.Builder().build(), null);
46        cache1.getAdvancedCache().getRpcManager().invokeRemotelyInFuture(f, null,
47              put, cache1.getAdvancedCache().getRpcManager().getDefaultRpcOptions(true));
48        TestingUtil.sleepThread(2000);
49        assert futureDoneOk.get();
50     }
51  }
