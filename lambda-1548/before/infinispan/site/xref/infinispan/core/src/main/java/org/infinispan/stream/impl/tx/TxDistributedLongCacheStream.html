14  import java.util.function.Supplier;
15  import java.util.stream.Collectors;
16  import java.util.stream.Stream;

18  /**
19   * Long variant of tx cache stream
20   * @see TxDistributedCacheStream
21   */
22  public class TxDistributedLongCacheStream extends DistributedLongCacheStream {
23     private final Address localAddress;
24     private final LocalTxInvocationContext ctx;
25     private final ConsistentHash hash;
26  
27     TxDistributedLongCacheStream(AbstractCacheStream stream, Address localAddress, ConsistentHash hash,
28             LocalTxInvocationContext ctx) {
29        super(stream);
30        this.localAddress = localAddress;
31        this.ctx = ctx;
32        this.hash = hash;
33     }
34  
35     @Override
36     protected Supplier<Stream<CacheEntry>> supplierForSegments(ConsistentHash ch, Set<Integer> targetSegments,
37             Set<Object> excludedKeys, boolean primaryOnly) {
38        return () -> {
39           Supplier<Stream<CacheEntry>> supplier = super.supplierForSegments(ch, targetSegments, excludedKeys, primaryOnly);
40           Set<CacheEntry> set = ctx.getLookedUpEntries().values().stream().filter(
41                   e -> !localAddress.equals(ch.locatePrimaryOwner(e.getKey()))).collect(Collectors.toSet());
42           Stream<CacheEntry> suppliedStream = supplier.get();
43           if (!set.isEmpty()) {
44              return Stream.concat(set.stream(), suppliedStream);
45           }
46           return suppliedStream;
47        };
48     }
49  
50     @Override
51     protected <R> DistributedCacheStream<R> cacheStream() {
52        return new TxDistributedCacheStream<R>(this, localAddress, hash, ctx);
53     }
54  
55     @Override
56     protected DistributedIntCacheStream intCacheStream() {
57        return new TxDistributedIntCacheStream(this, localAddress, hash, ctx);
58     }
59  
60     @Override
61     protected DistributedDoubleCacheStream doubleCacheStream() {
62        return new TxDistributedDoubleCacheStream(this, localAddress, hash, ctx);
63     }
64  }
