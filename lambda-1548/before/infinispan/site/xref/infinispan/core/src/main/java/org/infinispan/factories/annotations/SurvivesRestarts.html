1   package org.infinispan.factories.annotations;
2   
3   import java.lang.annotation.ElementType;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   /**
9    * This annotation is used for components that will be registered in the {@link org.infinispan.factories.ComponentRegistry},
10   * that are meant to be retained in the component registry even after the component registry is stopped.  Components
11   * annotated as such would not have to be recreated and rewired between stopping and starting a component registry.
12   * <br />
13   * As a rule of thumb though, use of this annotation on components should be avoided, since resilient components are
14   * retained even after a component registry is stopped and consume resources.  Only components necessary for and critical
15   * to bootstrapping and restarting the component registry should be annotated as such.
16   *
17   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
18   * @since 4.0
19   */
20  // ensure this annotation is available at runtime.
21  @Retention(RetentionPolicy.RUNTIME)
22  
23  // only applies to classes.
24  @Target(ElementType.TYPE)
25  public @interface SurvivesRestarts {
26  }
