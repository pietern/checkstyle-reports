57     @Override
58     protected Object perform(TestWriteOperation op, AdvancedCache<Object, Object> cache0, String key) {
59        try {
60           return super.perform(op, cache0, key);
61        } catch (IllegalArgumentException e) {
62           switch (op) {
63              case REPLACE_META_FUNCTIONAL:
64                 BiFunction<Object, ReadWriteEntryView<Object, Object>, Boolean> f =
65                    (BiFunction<Object, ReadWriteEntryView<Object, Object>, Boolean> & Serializable) (v, rw) -> {
66                       Class<MetaEntryVersion<Long>> type = MetaEntryVersion.type();
67                       return rw.findMetaParam(type)
68                          .filter(ver -> ver.get().compareTo(new NumericEntryVersion(1)) == EQUAL)
69                          .map(ver -> {
70                             rw.set(v, new MetaEntryVersion<>(new NumericEntryVersion(2)));
71                             return true;
72                          }).orElse(false);
73                    };
74                 return FunctionalTestUtils.await(rw(cache0).eval(key, "v1", f));
75              default:
76                 throw new AssertionError("Unknown operation: " + op);
77           }
78        }
79     }
