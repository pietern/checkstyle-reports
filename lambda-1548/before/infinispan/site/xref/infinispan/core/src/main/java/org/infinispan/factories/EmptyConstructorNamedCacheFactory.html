55  /**
56   * Simple factory that just uses reflection and an empty constructor of the component type.
57   *
58   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
59   * @author Pedro Ruivo
60   * @since 4.0
61   */
62  @DefaultFactoryFor(classes = {CacheNotifier.class, ClusterCacheNotifier.class, CommandsFactory.class,
63                                PersistenceManager.class, InvocationContextContainer.class,
64                                PassivationManager.class, ActivationManager.class,
65                                BatchContainer.class, EvictionManager.class,
66                                TransactionCoordinator.class, RecoveryAdminOperations.class, StateTransferLock.class,
67                                ClusteringDependentLogic.class, L1Manager.class, TransactionFactory.class, BackupSender.class,
68                                TotalOrderManager.class, ByteBufferFactory.class, MarshalledEntryFactory.class,
69                                RemoteValueRetrievedListener.class, InvocationContextFactory.class, CommitManager.class,
70                                XSiteStateTransferManager.class, XSiteStateConsumer.class, XSiteStateProvider.class,
71                                FunctionalNotifier.class})
72  public class EmptyConstructorNamedCacheFactory extends AbstractNamedCacheComponentFactory implements AutoInstantiableFactory {
73  
74     @Override
75     @SuppressWarnings("unchecked")
76     public <T> T construct(Class<T> componentType) {
77        Class<?> componentImpl;
78        if (componentType.equals(ClusteringDependentLogic.class)) {
79           CacheMode cacheMode = configuration.clustering().cacheMode();
80           if (!cacheMode.isClustered()) {
81              return componentType.cast(new ClusteringDependentLogic.LocalLogic());
82           } else if (cacheMode.isInvalidation()) {
83              return componentType.cast(new ClusteringDependentLogic.InvalidationLogic());
84           } else if (cacheMode.isReplicated()) {
85              return componentType.cast(new ClusteringDependentLogic.ReplicationLogic());
86           } else {
87              return componentType.cast(new ClusteringDependentLogic.DistributionLogic());
88           }
89        } else {
90           boolean isTransactional = configuration.transaction().transactionMode().isTransactional();
91           if (componentType.equals(InvocationContextFactory.class)) {
92              componentImpl = isTransactional ? TransactionalInvocationContextFactory.class
93                    : NonTransactionalInvocationContextFactory.class;
94              return componentType.cast(getInstance(componentImpl));
95           } else if (componentType.equals(InvocationContextContainer.class)) {
96              return (T) new InvocationContextContainerImpl();
97           } else if (componentType.equals(CacheNotifier.class)) {
98              return (T) new CacheNotifierImpl();
99           } else if (componentType.equals(CommandsFactory.class)) {
100             return (T) new CommandsFactoryImpl();
101          } else if (componentType.equals(PersistenceManager.class)) {
102             return (T) new PersistenceManagerImpl();
103          } else if (componentType.equals(PassivationManager.class)) {
104             return (T) new PassivationManagerImpl();
105          } else if (componentType.equals(ActivationManager.class)) {
106             return (T) new ActivationManagerImpl();
107          } else if (componentType.equals(BatchContainer.class)) {
108             return (T) new BatchContainer();
109          } else if (componentType.equals(TransactionCoordinator.class)) {
110             return (T) new TransactionCoordinator();
111          } else if (componentType.equals(RecoveryAdminOperations.class)) {
112             return (T) new RecoveryAdminOperations();
113          } else if (componentType.equals(StateTransferLock.class)) {
114             return (T) new StateTransferLockImpl();
115          } else if (componentType.equals(EvictionManager.class)) {
116             return (T) new EvictionManagerImpl();
117          } else if (componentType.equals(L1Manager.class)) {
118             return (T) new L1ManagerImpl();
119          } else if (componentType.equals(TransactionFactory.class)) {
120             return (T) new TransactionFactory();
121          } else if (componentType.equals(BackupSender.class)) {
122             return (T) new BackupSenderImpl(globalConfiguration.sites().localSite());
123          } else if (componentType.equals(TotalOrderManager.class)) {
124             return isTransactional && configuration.transaction().transactionProtocol().isTotalOrder() ?
125                   (T) new TotalOrderManager() : null;
126          } else if (componentType.equals(ByteBufferFactory.class)) {
127             return (T) new ByteBufferFactoryImpl();
128          } else if (componentType.equals(MarshalledEntryFactory.class)) {
129             return (T) new MarshalledEntryFactoryImpl();
130          } else if (componentType.equals(RemoteValueRetrievedListener.class)) {
131             // L1Manager is currently only listener for remotely retrieved values
132             return (T) componentRegistry.getComponent(L1Manager.class);
133          } else if (componentType.equals(ClusterCacheNotifier.class)) {
134             return (T) componentRegistry.getComponent(CacheNotifier.class);
135          } else if (componentType.equals(CommitManager.class)) {
136             return (T) new CommitManager(configuration.dataContainer().keyEquivalence());
137          } else if (componentType.equals(XSiteStateTransferManager.class)) {
138             return (T) (configuration.sites().allBackups().isEmpty() ? null : new XSiteStateTransferManagerImpl());
139          } else if (componentType.equals(XSiteStateConsumer.class)) {
140             return (T) new XSiteStateConsumerImpl();
141          } else if (componentType.equals(XSiteStateProvider.class)) {
142             return (T) new XSiteStateProviderImpl();
143          } else if (componentType.equals(FunctionalNotifier.class)) {
144             return (T) new FunctionalNotifierImpl<>();
145          }
146       }
147 
148       throw new CacheConfigurationException("Don't know how to create a " + componentType.getName());
149 
150    }
