9   import org.infinispan.stream.impl.AbstractDelegatingCacheSet;

11  
12  import java.util.stream.StreamSupport;
13  
14  /**
15   * Abstract cache key set that delegates to the underlying cache for stream usage
16   * @param <K> key type of the cache
17   * @param <V> value type of the cache
18   */
19  public abstract class AbstractDelegatingKeyCacheSet<K, V> extends AbstractDelegatingCacheSet<K> {
20     private final Cache<K, V> cache;
21     private final CacheSet<K> set;
22  
23     protected AbstractDelegatingKeyCacheSet(Cache<K, V> cache, CacheSet<K> set) {
24        this.cache = cache;
25        this.set = set;
26     }
27  
28     @Override
29     protected final CacheSet<K> delegate() {
30        return set;
31     }
32  
33     @Override
34     public CacheStream<K> stream() {
35        return getStream(false);
36     }
37  
38     @Override
39     public CacheStream<K> parallelStream() {
40        return getStream(true);
41     }
42  
43     private CacheStream<K> getStream(boolean parallel) {
44        DistributionManager dm = cache.getAdvancedCache().getDistributionManager();
45        CloseableSpliterator<CacheEntry<K, V>> closeableSpliterator = cache.getAdvancedCache().cacheEntrySet().spliterator();
46        // TODO: add custom local key cache stream that doesn't use entries - this way it doesn't need to use entry set
47        CacheStream<K> stream = new LocalKeyCacheStream<>(cache, parallel, dm != null ? dm.getConsistentHash() : null,
48                () -> StreamSupport.stream(closeableSpliterator, parallel), cache.getAdvancedCache().getComponentRegistry());
49        // We rely on the fact that on close returns the same instance
50        stream.onClose(() -> closeableSpliterator.close());
51        return stream;
52     }
53  }
