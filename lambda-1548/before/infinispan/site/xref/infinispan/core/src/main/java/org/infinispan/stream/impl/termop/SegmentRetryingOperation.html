1   package org.infinispan.stream.impl.termop;
2   
3   import org.infinispan.stream.impl.TerminalOperation;
4   import org.infinispan.stream.impl.intops.IntermediateOperation;
5   import org.infinispan.util.logging.Log;
6   import org.infinispan.util.logging.LogFactory;
7   
8   import java.util.concurrent.atomic.AtomicBoolean;
9   import java.util.concurrent.atomic.AtomicReference;
10  import java.util.function.Function;
11  import java.util.function.Supplier;
12  import java.util.stream.BaseStream;
13  import java.util.stream.Stream;
14  
15  /**
16   * A terminal based operation that runs the provided function to evaluate the operation.  If a segment is lost during
17   * the evaluation of the function the function results will be ignored and subsequently retried with the new stable
18   * segments.  This is repeated until either a full stable run is completed of the function or if the lost segment
19   * states that there are no more segments left.
20   * @param <E> output type of the function
21   * @param <T> type of the stream entries
22   * @param <S> type of the stream itself
23   */
24  public class SegmentRetryingOperation<E, T, S extends BaseStream<T, S>> extends BaseTerminalOperation
25          implements TerminalOperation<E> {
26     private static final Log log = LogFactory.getLog(SegmentRetryingOperation.class);
27  
28     private static final BaseStream<?, ?> EMPTY = Stream.empty();
29  
30     private final Function<S, ? extends E> function;
31     private transient AtomicReference<BaseStream<?, ?>> streamRef = new AtomicReference<>(EMPTY);
32     private transient AtomicBoolean continueTrying = new AtomicBoolean(true);
33  
34     public SegmentRetryingOperation(Iterable<IntermediateOperation> intermediateOperations,
35             Supplier<? extends Stream<?>> supplier, Function<S, ? extends E> function) {
36        super(intermediateOperations, supplier);
37        this.function = function;
38     }
39  
40     @Override
41     public boolean lostSegment(boolean stopIfLost) {
42        BaseStream<?, ?> oldStream = streamRef.get();
43        continueTrying.set(!stopIfLost);
44        boolean affected;
45        if (oldStream != null) {
46           // If the stream was non null and wasn't empty that means we were processing it at the time of the segment
47           // being lost - so we tell that one to close
48           if (oldStream != EMPTY) {
49              // This can only fail if the operation completes concurrently
50              if ((affected = streamRef.compareAndSet(oldStream, EMPTY))) {
51                 // This can short circuit some things like sending a response or waiting for retrieval from a
52                 // cache loader
53                 oldStream.close();
54              }
55           } else {
56              affected = true;
57           }
58        } else {
59           affected = false;
60        }
61        return affected;
62     }
63  
64     private E innerPerformOperation(BaseStream<?, ?> stream) {
65        for (IntermediateOperation intOp : intermediateOperations) {
66           stream = intOp.perform(stream);
67        }
68        return function.apply((S) stream);
69     }
70  
71     @Override
72     public E performOperation() {
73        boolean keepTrying = true;
74        BaseStream<?, ?> stream;
75        E value;
76        do {
77           stream = supplier.get();
78           streamRef.set(stream);
79           value = innerPerformOperation(stream);
80           log.trace("Completed an operation, trying to see if we are done.");
81        } while (!streamRef.compareAndSet(stream, null) && (keepTrying = continueTrying.get()));
82        log.tracef("Operation now done, due to try denial: " + !keepTrying);
83        return keepTrying ? value : null;
84     }
85  
86     public Function<S, ? extends E> getFunction() {
87        return function;
88     }
89  }
