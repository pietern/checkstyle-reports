1   package org.infinispan.expiration;

3   import net.jcip.annotations.ThreadSafe;
4   
5   import org.infinispan.configuration.cache.ExpirationConfigurationBuilder;
6   import org.infinispan.configuration.global.GlobalConfigurationBuilder;
7   import org.infinispan.container.entries.InternalCacheEntry;
8   import org.infinispan.factories.scopes.Scope;
9   import org.infinispan.factories.scopes.Scopes;
10  import org.infinispan.marshall.core.MarshalledEntry;
11  
12  /**
13   * Central component that deals with expiration of cache entries.
14   * <p />
15   * Typically, {@link #processExpiration()} is called periodically by the expiration thread (which can be configured using
16   * {@link ExpirationConfigurationBuilder#wakeUpInterval(long)} and {@link GlobalConfigurationBuilder#expirationThreadPool()}).
17   * <p />
18   * If the expiration thread is disabled - by setting {@link ExpirationConfigurationBuilder#wakeUpInterval(long)} to <tt>0</tt> -
19   * then this method could be called directly, perhaps by any other maintenance thread that runs periodically in the application.
20   * <p />
21   * @author William Burns
22   * @since 7.2
23   */
24  @ThreadSafe
25  @Scope(Scopes.NAMED_CACHE)
26  public interface ExpirationManager<K, V> {
27  
28     /**
29      * Processes the expiration event queue.
30      */
31     void processExpiration();
32  
33     /**
34      * @return true if expiration reaper thread is enabled, false otherwise
35      */
36     boolean isEnabled();
37  
38     /**
39      * This should be invoked passing in an entry that is now expired.  This method may attempt to lock this key to
40      * preserve atomicity.
41      * @param entry entry that is now expired
42      * @param currentTime the current time in milliseconds
43      */
44     void handleInMemoryExpiration(InternalCacheEntry<K, V> entry, long currentTime);
45  
46     /**
47      * This is to be invoked when a store entry expires.  This method may attempt to lock this key to preserve atomicity.
48      * <p>
49      * Note this method doesn't currently take a {@link InternalCacheEntry} and this is due to a limitation in the
50      * cache store API.  This may cause some values to be removed if they were updated at the same time.
51      * @param key the key of the expired entry
52      */
53     void handleInStoreExpiration(K key);
54  
55     /**
56      * This is to be invoked when a store entry expires and the value and/or metadata is available to be used.  This
57      * method is preferred over {@link ExpirationManager#handleInStoreExpiration(Object)} as it allows for more
58      * specific expiration to possibly occur.
59      * @param marshalledEntry the entry that can be unmarshalled as needed
60      */
61     void handleInStoreExpiration(MarshalledEntry<K, V> marshalledEntry);
62  
63     /**
64      * This is to be invoked with a when a write is known to occur to prevent expiration from happening.  This way we
65      * won't have a swarm of remote calls required.
66      * @param key the key to use
67      */
68     void registerWriteIncoming(K key);
69  
70     /**
71      * This should always be invoked after registering write but after performing any operations required.
72      * @param key the key to use
73      */
74     void unregisterWrite(K key);
75  }
