31     public void testConfigurationOverride() throws Exception {
32        ConfigurationBuilder defaultCfgBuilder = new ConfigurationBuilder();
33        defaultCfgBuilder.eviction().maxEntries(200).strategy(LIRS);
34  
35        cm = TestCacheManagerFactory.createCacheManager(defaultCfgBuilder);
36        final ConfigurationBuilder cacheCfgBuilder =
37              new ConfigurationBuilder().read(defaultCfgBuilder.build());
38        cm.defineConfiguration("my-cache", cacheCfgBuilder.build());
39        Cache<?, ?> cache = cm.getCache("my-cache");
40        assertEquals(200,
41              cache.getCacheConfiguration().eviction().maxEntries());
42        assertEquals(LIRS,
43              cache.getCacheConfiguration().eviction().strategy());
44     }
45  
46     public void testSimpleDistributedClusterModeDefault() throws Exception {
47        ConfigurationBuilder builder = new ConfigurationBuilder();
48        builder.clustering().cacheMode(DIST_SYNC)
49              .hash().numOwners(3).numSegments(51);
50  
51        cm = TestCacheManagerFactory.createClusteredCacheManager(builder);
52  
53        Cache<?, ?> cache = cm.getCache("my-cache");
54        // These are all overridden values
55        ClusteringConfiguration clusteringCfg =
56              cache.getCacheConfiguration().clustering();
57        assertEquals(DIST_SYNC, clusteringCfg.cacheMode());
58        assertEquals(3, clusteringCfg.hash().numOwners());
59        assertEquals(51, clusteringCfg.hash().numSegments());
60     }

62     public void testSimpleDistributedClusterModeNamedCache() throws Exception {
63        final String cacheName = "my-cache";
64        final Configuration config = new ConfigurationBuilder()
65              .clustering().cacheMode(DIST_SYNC)
66              .hash().numOwners(3).numSegments(51).build();
67  
68        cm = TestCacheManagerFactory.createClusteredCacheManager();
69        cm.defineConfiguration(cacheName, config);
70        Cache<?, ?> cache = cm.getCache(cacheName);
71        ClusteringConfiguration clusteringCfg =
72              cache.getCacheConfiguration().clustering();
73        assertEquals(DIST_SYNC, clusteringCfg.cacheMode());
74        assertEquals(3, clusteringCfg.hash().numOwners());
75        assertEquals(51, clusteringCfg.hash().numSegments());
76     }

78     public void testOverrideWithStore() {
79        final ConfigurationBuilder builder1 = new ConfigurationBuilder();
80        builder1.persistence().addStore(DummyInMemoryStoreConfigurationBuilder.class);
81        cm = new DefaultCacheManager(new GlobalConfigurationBuilder().build(), builder1.build());
82        ConfigurationBuilder builder2 = new ConfigurationBuilder();
83        builder2.read(cm.getDefaultCacheConfiguration());
84        builder2.eviction().maxEntries(1000);
85        Configuration configuration = cm.defineConfiguration("named", builder2.build());
86        assertEquals(1, configuration.persistence().stores().size());
87     }

89     public void testPartialOverride() {
90        ConfigurationBuilder baseBuilder = new ConfigurationBuilder();
91        baseBuilder.eviction().maxEntries(200).strategy(LIRS);
92        Configuration base = baseBuilder.build();
93        ConfigurationBuilder overrideBuilder = new ConfigurationBuilder();
94        overrideBuilder.read(base).locking().concurrencyLevel(31);
95        Configuration override = overrideBuilder.build();
96        assertEquals(200, base.eviction().maxEntries());
97        assertEquals(200, override.eviction().maxEntries());
98        assertEquals(LIRS, base.eviction().strategy());
99        assertEquals(LIRS, override.eviction().strategy());
100       assertEquals(32, base.locking().concurrencyLevel());
101       assertEquals(31, override.locking().concurrencyLevel());
102    }
