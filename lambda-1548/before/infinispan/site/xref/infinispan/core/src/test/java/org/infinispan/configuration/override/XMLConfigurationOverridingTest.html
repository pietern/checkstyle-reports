311             //Changing the data in referenced variables

313                indexedClass.description = "other value";
314             }
315 
316             //Verifying that the entry in the cache is not changed.
317             for (int i = 0; i < 10; i++) {
318                Assert.assertEquals(new NonIndexedClass("value" + i), cm.getCache(simpleCacheName).get("key" + i));

464    class NonIndexedClass implements Serializable {
465       public String description;
466 
467       NonIndexedClass(String description) {
468          this.description = description;
469       }
470 
471       @Override
472       public boolean equals(Object o) {
473          if (this == o) return true;
474          if (o == null || getClass() != o.getClass()) return false;
475 
476          NonIndexedClass that = (NonIndexedClass) o;
477 
478          if (!description.equals(that.description)) return false;
479 
480          return true;
481       }
482 
483       @Override
484       public int hashCode() {
485          return description.hashCode();
486       }
487    }

489    class SimpleInterceptor extends CommandInterceptor {
490       private boolean putOkay;
491 
492       @Override
493       public Object visitPutKeyValueCommand(InvocationContext ctx, PutKeyValueCommand command) throws Throwable {
494          if (isRightType(ctx)) putOkay = true;
495          return super.visitPutKeyValueCommand(ctx, command);
496       }
497 
498       private boolean isRightType(InvocationContext ctx) {
499          return ctx instanceof SingleKeyNonTxInvocationContext;
500       }
