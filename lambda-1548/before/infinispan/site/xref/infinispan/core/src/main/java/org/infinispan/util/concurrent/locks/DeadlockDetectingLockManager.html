21  /**
22   * Lock manager in charge with processing deadlock detections.
23   * </p>
24   * Implementation notes: if a deadlock is detected, then one of the transactions has to rollback. The transaction that
25   * rollbacks is determined by comparing the coin toss from {@link org.infinispan.transaction.xa.DldGlobalTransaction}.
26   * </p>
27   * A thread calling {@link #lock(Object, Object, long, TimeUnit)} or {@link #lockAll(Collection, Object, long, TimeUnit)}
28   * would run the deadlock detection algorithm only if all of the following take place:
29   * - the call is made in the scope of a transaction (either locally originated or remotely originated)
30   * - it cannot acquire lock on the given key and the lock owner is another transaction
31   * - when comparing coin toss, this thread would loose against the other one - so it's always the potential loser that runs DLD.
32   * If deadlock is detected then a {@link org.infinispan.util.concurrent.locks.DeadlockDetectedException} is thrown.
33   * This is subsequently handled in the interceptor chain - locks owned by this tx are released.
34   *
35   * @author Mircea.Markus@jboss.com
36   */
37  @MBean(objectName = "DeadlockDetectingLockManager", description = "Information about the number of deadlocks that were detected")
38  public class DeadlockDetectingLockManager extends DefaultLockManager implements DeadlockChecker, Runnable {
39  
40     private static final Log log = LogFactory.getLog(DeadlockDetectingLockManager.class);
41     private static final boolean trace = log.isTraceEnabled();
42  
43     private ScheduledFuture<?> scheduledFuture;
44  
45     protected volatile boolean exposeJmxStats;
46  
47     private AtomicLong localTxStopped = new AtomicLong(0);
48  
49     private AtomicLong remoteTxStopped = new AtomicLong(0);
50  
51     private AtomicLong cannotRunDld = new AtomicLong(0);
52  
53     @Start
54     public void init() {
55        long spinDuration = configuration.deadlockDetection().spinDuration();
56        exposeJmxStats = configuration.jmxStatistics().enabled();
57        scheduledFuture = scheduler.scheduleWithFixedDelay(this, spinDuration, spinDuration, TimeUnit.MILLISECONDS);
58     }
59  
60     @Stop
61     public void stopScheduler() {
62        if (scheduledFuture != null) {
63           scheduledFuture.cancel(false);
64           scheduledFuture = null;
65        }
66     }

68     @Override
69     public KeyAwareLockPromise lock(Object key, Object lockOwner, long time, TimeUnit unit) {
70        if (lockOwner instanceof DldGlobalTransaction) {
71           ((DldGlobalTransaction) lockOwner).setLockIntention(Collections.singleton(key));
72           return super.lock(key, lockOwner, time, unit);
73        }
74        return super.lock(key, lockOwner, time, unit);
75     }
76  
77     @Override
78     public KeyAwareLockPromise lockAll(Collection<?> keys, Object lockOwner, long time, TimeUnit unit) {
79        if (lockOwner instanceof DldGlobalTransaction) {
80           ((DldGlobalTransaction) lockOwner).setLockIntention(new HashSet<>(keys));
81           return super.lockAll(keys, lockOwner, time, unit);
82        }
83        return super.lockAll(keys, lockOwner, time, unit);
84     }
85  
86     @Override
87     public void run() {
88        lockContainer.deadlockCheck(this);
89     }

91     private boolean isDeadlockAndIAmLoosing(DldGlobalTransaction lockOwnerTx, DldGlobalTransaction thisTx) {
92        //run the lose check first as it is cheaper
93        boolean wouldWeLoose = thisTx.wouldLose(lockOwnerTx);
94        if (!wouldWeLoose) {
95           if (trace)
96              log.tracef("We (%s) win against the other (%s) transaction, so no point running rest of DLD", thisTx, lockOwnerTx);
97           return false;
98        }
99        //do we have lock on what other tx intends to acquire?
100       return ownsAnyLocalIntention(thisTx, lockOwnerTx) || ownsRemoteIntention(lockOwnerTx, thisTx) || isSameKeyDeadlock(thisTx, lockOwnerTx);
101    }
102 
103    private boolean isSameKeyDeadlock(DldGlobalTransaction thisTx, DldGlobalTransaction lockOwnerTx) {
104       boolean iHaveRemoteLock = !thisTx.isRemote(); //this relies on the fact that when DLD is enabled a lock is first acquired remotely and then locally
105       boolean otherHasLocalLock = lockOwnerTx.isRemote();
106 
107       //if we are here then 1) the other tx has a lock on this local key AND 2) I have a lock on the same key remotely
108       if (iHaveRemoteLock && otherHasLocalLock) {
109          if (trace) log.tracef("Same key deadlock between %s and %s.", thisTx, lockOwnerTx);
110          return true;
111       }
112       return false;
113    }
114 
115    /**
116     * This happens with two nodes replicating same tx at the same time.
117     */
118    private boolean ownsRemoteIntention(DldGlobalTransaction lockOwnerTx, DldGlobalTransaction thisTx) {
119       boolean localLockOwner = !lockOwnerTx.isRemote();
120       if (localLockOwner) {
121          // I've already acquired lock on this key before replicating here, so this mean we are in deadlock. This assumes the fact that
122          // if trying to acquire a remote lock, a tx first acquires a local lock.
123          if (thisTx.hasAnyLockAtOrigin(lockOwnerTx)) {
124             if (trace)
125                log.trace("Same key deadlock detected: lock owner tries to acquire a lock remotely but we have it!");
126             return true;
127          }
128       } else {
129          if (trace) log.tracef("Lock owner is remote: %s", lockOwnerTx);
130       }
131       return false;
132    }
133 
134    private boolean ownsAnyLocalIntention(DldGlobalTransaction thisTx, DldGlobalTransaction otherTx) {
135       for (Object key : otherTx.getLockIntention()) {
136          if (ownsLock(key, thisTx)) {
137             if (trace) log.tracef("Local intention is '%s' and we (%s) own the lock.", key, thisTx);
138             return true;
139          }
140       }
141       return false;
142    }
143 
144    @Override
145    public boolean deadlockDetected(Object pendingOwner, Object currentOwner) {
146       if (!(pendingOwner instanceof DldGlobalTransaction) || !(currentOwner instanceof DldGlobalTransaction)) {
147          if (trace) {
148             log.tracef("Unable to run DLD with %s and %s. One of them are not a DldGlobalTransaction.",
149                        pendingOwner, currentOwner);
150          }
151          cannotRunDld.incrementAndGet();
152          return false;
153       }
154       if (trace) {
155          log.tracef("Could not acquire lock. It is locked by %s (%s)", currentOwner, System.identityHashCode(currentOwner));
156       }
157 
158       final DldGlobalTransaction ownerTx = (DldGlobalTransaction) currentOwner;
159       final DldGlobalTransaction pendingTx = (DldGlobalTransaction) pendingOwner;
160 
161       if (isDeadlockAndIAmLoosing(ownerTx, pendingTx)) {
162          updateStats(pendingTx);
163          log.tracef("Deadlock found and we (%s) shall not continue. Other tx is %s", pendingOwner, currentOwner);
164          return true;
165       }
166       return false;
167    }

170       this.exposeJmxStats = exposeJmxStats;

173    @ManagedAttribute(description = "Total number of local detected deadlocks", displayName = "Number of total detected deadlocks", measurementType = MeasurementType.TRENDSUP)
174    public long getTotalNumberOfDetectedDeadlocks() {
175       return localTxStopped.get() + remoteTxStopped.get();
176    }
177 
178    @ManagedOperation(description = "Resets statistics gathered by this component", displayName = "Reset statistics")
179    public void resetStatistics() {
180       localTxStopped.set(0);
181       remoteTxStopped.set(0);
182       cannotRunDld.set(0);
183    }

185    @ManagedAttribute(description = "Number of remote transactions that were rolled-back due to deadlocks", displayName = "Number of remote transaction that were roll backed due to deadlocks", measurementType = MeasurementType.TRENDSUP)
186    public long getDetectedRemoteDeadlocks() {
187       return remoteTxStopped.get();
188    }

190    @ManagedAttribute(description = "Number of local transactions that were rolled-back due to deadlocks", displayName = "Number of local transaction that were roll backed due to deadlocks", measurementType = MeasurementType.TRENDSUP)
191    public long getDetectedLocalDeadlocks() {
192       return localTxStopped.get();
193    }

195    @ManagedAttribute(description = "Number of situations when we try to determine a deadlock and the other lock owner is NOT a transaction. In this scenario we cannot run the deadlock detection mechanism", displayName = "Number of unsolvable deadlock situations", measurementType = MeasurementType.TRENDSUP)
196    public long getOverlapWithNotDeadlockAwareLockOwners() {
197       return cannotRunDld.get();
198    }

202    private void updateStats(DldGlobalTransaction tx) {
203       if (exposeJmxStats) {
204          if (tx.isRemote())
205             remoteTxStopped.incrementAndGet();
206          else
207             localTxStopped.incrementAndGet();
208       }
209    }
