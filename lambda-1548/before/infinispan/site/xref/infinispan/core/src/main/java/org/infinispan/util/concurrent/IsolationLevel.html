3   /**
4    * Various transaction isolation levels as an enumerated class.  Note that <a href="http://wiki.jboss.org/wiki/JBossCacheMVCC">MVCC</a>
5    * only supports {@link #READ_COMMITTED} and {@link #REPEATABLE_READ}, upgrading where possible.
6    * <p/>
7    * Also note that Infinispan defaults to {@link #READ_COMMITTED}.
8    * <p/>
9    *
10   * @author (various)
11   * @see <a href="http://en.wikipedia.org/wiki/Isolation_%28computer_science%29">Isolation levels</a>
12   * @since 4.0
13   */
14  public enum IsolationLevel {
15     /**
16      * No isolation.
17      */
18     NONE,
19     SERIALIZABLE,
20     REPEATABLE_READ,
21     READ_COMMITTED,
22     READ_UNCOMMITTED
23  }
