8   import java.util.Arrays;
9   import java.util.List;
10  import java.util.concurrent.CompletableFuture;
11  import java.util.concurrent.ExecutorService;
12  import java.util.function.Supplier;
13  
14  /**
15   * Internal class that encapsulates collection of parameters used to tweak
16   * functional map operations.
17   *
18   * <p>Internally, parameters are stored in an array which is indexed by
19   * a parameter's {@link Param#id()}
20   *
21   * <p>All parameters have default values which are stored in a static
22   * array field in {@link Params} class, which are used to as base collection
23   * when adding or overriding parameters.
24   *
25   * @since 8.0
26   */
27  @Experimental
28  public final class Params {
29  
30     private static final Param<?>[] DEFAULTS = new Param<?>[]{
31        FutureMode.defaultValue(),
32        PersistenceMode.defaultValue(),
33     };
34  
35     final Param<?>[] params;
36  
37     private Params(Param<?>[] params) {
38        this.params = params;
39     }
40  
41     /**
42      * Checks whether all the parameters passed in are already present in the
43      * current parameters. This method can be used to optimise the decision on
44      * whether the parameters collection needs updating at all.
45      */
46     public boolean containsAll(Param<?>... ps) {
47        List<Param<?>> paramsToCheck = Arrays.asList(ps);
48        List<Param<?>> paramsCurrent = Arrays.asList(params);
49        return paramsCurrent.containsAll(paramsToCheck);
50     }
51  
52     /**
53      * Adds all parameters and returns a new parameter collection.
54      */
55     public Params addAll(Param<?>... ps) {
56        List<Param<?>> paramsToAdd = Arrays.asList(ps);
57        Param<?>[] paramsAll = Arrays.copyOf(params, params.length);
58        paramsToAdd.forEach(p -> paramsAll[p.id()] = p);
59        return new Params(paramsAll);
60     }
61  
62     /**
63      * Retrieve a param given its identifier. Callers are expected to know the
64      * exact type of parameter that will be returned. Such assumption is
65      * possible because as indicated in {@link Param} implementations will
66      * only come from Infinispan itself.
67      */
68     @SuppressWarnings("unchecked")
69     public <T> Param<T> get(int index) {
70        // TODO: Provide a more type safe API. E.g. make index a pojo typed on T which is aligned with the Param<T>
71        return (Param<T>) params[index];
72     }
73  
74     @Override
75     public String toString() {
76        return "Params=" + Arrays.toString(params);
77     }
78  
79     public static Params create() {
80        return new Params(DEFAULTS);
81     }
82  
83     public static Params from(Param<?>... ps) {
84        List<Param<?>> paramsToAdd = Arrays.asList(ps);
85        List<Param<?>> paramsDefaults = Arrays.asList(DEFAULTS);
86        if (paramsDefaults.containsAll(paramsToAdd))
87           return create(); // All parameters are defaults, don't do more work
88  
89        Param<?>[] paramsAll = Arrays.copyOf(DEFAULTS, DEFAULTS.length);
90        paramsToAdd.forEach(p -> paramsAll[p.id()] = p);
91        return new Params(paramsAll);
92     }
93  
94     static <T> CompletableFuture<T> withFuture(Param<FutureMode> futureParam,
95           ExecutorService asyncExec, Supplier<T> s) {
96        switch (futureParam.get()) {
97           case COMPLETED:
98              // If completed, complete the future directly with the result.
99              // No separate thread or executor is instantiated.
100             return CompletableFuture.completedFuture(s.get());
101          case ASYNC:
102             // If async, execute the supply function asynchronously,
103             // and return a future that's completed when the supply
104             // function returns.
105             return CompletableFuture.supplyAsync(s, asyncExec);
106          default:
107             throw new IllegalStateException();
108       }
109    }
