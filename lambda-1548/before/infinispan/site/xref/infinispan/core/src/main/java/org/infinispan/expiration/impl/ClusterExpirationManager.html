28  /**
29   * Allows for cluster based expirations to occur.  This provides guarantees that when an entry is expired that it will
30   * expire that entry across the entire cluster at once.  This requires obtaining the lock for said entry before
31   * expiration is performed.  Since expiration can occur without holding onto the lock it is possible for an expiration
32   * to occur immediately after a value has been updated.  This can cause a premature expiration to occur.  Attempts
33   * are made to prevent this by using the expired entry's value and lifespan to limit this expiration so it only happens
34   * in a smaller amount of cases.
35   * <p>
36   * Cache stores however do not supply the value or metadata information which means if an entry is purged from the cache
37   * store that it will forcibly remove the value even if a concurrent write updated it just before.  This will be
38   * addressed by future SPI changes to the cache store.
39   * @param <K>
40   * @param <V>
41   */
42  @ThreadSafe
43  public class ClusterExpirationManager<K, V> extends ExpirationManagerImpl<K, V> {
44     protected static final Log log = LogFactory.getLog(ClusterExpirationManager.class);
45     protected static final boolean trace = log.isTraceEnabled();
46  
47     private ExecutorService asyncExecutor;
48     private AdvancedCache<K, V> cache;
49  
50     @Inject
51     public void inject(AdvancedCache<K, V> cache,
52             @ComponentName(KnownComponentNames.ASYNC_OPERATIONS_EXECUTOR) ExecutorService asyncExecutor) {
53        this.cache = cache;
54        this.asyncExecutor = asyncExecutor;
55     }
56  
57     @Override
58     public void processExpiration() {
59        long start = 0;
60        if (!Thread.currentThread().isInterrupted()) {
61           try {
62              if (trace) {
63                 log.trace("Purging data container of expired entries");
64                 start = timeService.time();
65              }
66              long currentTimeMillis = timeService.wallClockTime();
67              for (Iterator<InternalCacheEntry<K, V>> purgeCandidates = dataContainer.iteratorIncludingExpired();
68                   purgeCandidates.hasNext();) {
69                 InternalCacheEntry<K, V> e = purgeCandidates.next();
70                 if (e.canExpire()) {
71                    if (ExpiryHelper.isExpiredMortal(e.getLifespan(), e.getCreated(), currentTimeMillis)) {
72                       handleLifespanExpireEntry(e);
73                    } else if (ExpiryHelper.isExpiredTransient(e.getMaxIdle(), e.getLastUsed(), currentTimeMillis)) {
74                       super.handleInMemoryExpiration(e, currentTimeMillis);
75                    }
76                 }
77              }
78              if (trace) {
79                 log.tracef("Purging data container completed in %s",
80                         Util.prettyPrintTime(timeService.timeDuration(start, TimeUnit.MILLISECONDS)));
81              }
82           } catch (Exception e) {
83              log.exceptionPurgingDataContainer(e);
84           }
85        }
86  
87        if (!Thread.currentThread().isInterrupted()) {
88           persistenceManager.purgeExpired();
89        }
90     }
91  
92     void handleLifespanExpireEntry(InternalCacheEntry<K, V> entry) {
93        K key = entry.getKey();
94        // The most used case will be a miss so no extra read before
95        if (expiring.putIfAbsent(key, key) == null) {
96           long lifespan = entry.getLifespan();
97           if (trace) {
98              log.tracef("Submitting expiration removal for key %s which had lifespan of %s", key, lifespan);
99           }
100          asyncExecutor.submit(() -> {
101             try {
102                removeExpired(key, entry.getValue(), lifespan);
103             } finally {
104                expiring.remove(key);
105             }
106          });
107       }
108    }
109 
110    private void removeExpired(K key, V value, Long lifespan) {
111       cache.removeExpired(key, value, lifespan);
112    }
113 
114    @Override
115    public void handleInMemoryExpiration(InternalCacheEntry<K, V> entry, long currentTime) {
116       if (ExpiryHelper.isExpiredMortal(entry.getLifespan(), entry.getCreated(), currentTime)) {
117          handleLifespanExpireEntry(entry);
118       } else {
119          super.handleInMemoryExpiration(entry, currentTime);
120       }
121    }
122 
123    @Override
124    public void handleInStoreExpiration(K key) {
125       expiring.put(key, key);
126       // Unfortunately stores don't pull the entry so we can't tell exactly why it expired and thus we have to remove
127       // the entire value.  Unfortunately this could cause a concurrent write to be undone
128       try {
129          removeExpired(key, null, null);
130       } finally {
131          expiring.remove(key);
132       }
133    }
134 
135    @Override
136    public void handleInStoreExpiration(MarshalledEntry<K, V> marshalledEntry) {
137       K key = marshalledEntry.getKey();
138       expiring.put(key, key);
139       try {
140          InternalMetadata metadata = marshalledEntry.getMetadata();
141          removeExpired(key, marshalledEntry.getValue(), metadata.lifespan() == -1 ? null : metadata.lifespan());
142       } finally {
143          expiring.remove(key);
144       }
145    }
146 }
