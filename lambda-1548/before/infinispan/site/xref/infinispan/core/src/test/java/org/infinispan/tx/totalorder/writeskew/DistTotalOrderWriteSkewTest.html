15  /**
16   * A simple write skew check test for total order based protocol in distributed mode
17   *
18   * @author Pedro Ruivo
19   * @since 5.3
20   */
21  @Test(groups = "functional", testName = "tx.totalorder.DistTotalOrderWriteSkewTest")
22  @CleanupAfterMethod
23  public class DistTotalOrderWriteSkewTest extends DistWriteSkewTest {
24     public void transactionCleanupWithWriteSkew() throws Exception {
25        cache(0).put("k", "v");
26        tm(0).begin();
27        assertEquals("v", cache(0).get("k"));
28        cache(0).put("k", "v2");
29        Transaction suspend = tm(0).suspend();
30  
31        cache(0).put("k", "v3");
32        tm(0).resume(suspend);
33        try {
34           tm(0).commit();
35           assert false;
36        } catch (Throwable e) {
37           log.debug("Ignoring expected write skew check exception", e);
38        }
39        assertEquals("v3", cache(0).get("k"));
40        assertEventuallyEquals(1, "k", "v3");
41        assertNoTransactions();
42     }
43  
44     @Override
45     protected void decorate(ConfigurationBuilder builder) {
46        builder.transaction().transactionProtocol(TransactionProtocol.TOTAL_ORDER)
47              .recovery().disable();
48        builder.locking().isolationLevel(IsolationLevel.REPEATABLE_READ).writeSkewCheck(true)
49              .versioning().enable().scheme(VersioningScheme.SIMPLE);
50     }
51  }
