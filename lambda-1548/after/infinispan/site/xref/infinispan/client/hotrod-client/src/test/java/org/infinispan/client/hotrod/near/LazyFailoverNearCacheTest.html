27     public void testNearCacheClearedUponFailover() {
28        AssertsNearCache<Integer, String> stickyClient = createStickyAssertClient();
29        try {
30           stickyClient.put(1, "v1").expectNearRemove(1, headClient(), tailClient());
31           stickyClient.get(1, "v1").expectNearGetNull(1).expectNearPutIfAbsent(1, "v1");
32           stickyClient.put(2, "v1").expectNearRemove(2, headClient(), tailClient());
33           stickyClient.get(2, "v1").expectNearGetNull(2).expectNearPutIfAbsent(2, "v1");
34           stickyClient.put(3, "v1").expectNearRemove(3, headClient(), tailClient());
35           stickyClient.get(3, "v1").expectNearGetNull(3).expectNearPutIfAbsent(3, "v1");
36           findServerAndKill(stickyClient.manager, servers, cacheManagers);
37           // Since each client is separate remote cache manager, you need to get
38           // each client to do an operation to receive the near cache clear.
39           // These gets should return non-null, but the get should be resolved remotely!
40           stickyClient.get(1, "v1").expectNearClear().expectNearPutIfAbsent(1, "v1");
41           stickyClient.expectNoNearEvents();
42           headClient().get(2, "v1").expectNearClear().expectNearPutIfAbsent(2, "v1");
43           headClient().expectNoNearEvents();
44           tailClient().get(3, "v1").expectNearClear().expectNearPutIfAbsent(3, "v1");
45           tailClient().expectNoNearEvents();
46        } finally {
47           stickyClient.stop();
48        }
49     }
