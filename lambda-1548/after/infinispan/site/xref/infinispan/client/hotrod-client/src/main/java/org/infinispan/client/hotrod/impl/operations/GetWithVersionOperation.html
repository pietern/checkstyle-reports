16  /**
17   * Corresponds to getWithVersion operation as described by
18   * <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod protocol specification</a>.
19   *
20   * @author Mircea.Markus@jboss.com
21   * @since 4.1
22   */
23  @Immutable
24  @Deprecated
25  public class GetWithVersionOperation extends AbstractKeyOperation<VersionedValue<byte[]>> {
26  
27     private static final Log log = LogFactory.getLog(GetWithVersionOperation.class);
28  
29     public GetWithVersionOperation(Codec codec, TransportFactory transportFactory,
30              byte[] key, byte[] cacheName, AtomicInteger topologyId, Flag[] flags) {
31        super(codec, transportFactory, key, cacheName, topologyId, flags);
32     }
33  
34     @Override
35     protected VersionedValue<byte[]> executeOperation(Transport transport) {
36        short status = sendKeyOperation(key, transport, GET_WITH_VERSION, GET_WITH_VERSION_RESPONSE);
37        VersionedValue<byte[]> result = null;
38        if (status == KEY_DOES_NOT_EXIST_STATUS) {
39           result = null;
40        } else if (status == NO_ERROR_STATUS) {
41           long version = transport.readLong();
42           if (log.isTraceEnabled()) {
43              log.tracef("Received version: %d", version);
44           }
45           byte[] value = transport.readArray();
46           result = new VersionedValueImpl<byte[]>(version, value);
47        }
48        return result;
49     }
50  }
