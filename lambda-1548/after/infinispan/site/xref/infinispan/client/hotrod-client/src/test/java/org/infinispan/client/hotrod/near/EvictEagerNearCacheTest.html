20     protected <K, V> AssertsNearCache<K, V> createClient() {
21        ConfigurationBuilder builder = new ConfigurationBuilder();
22        builder.addServer().host("127.0.0.1").port(hotrodServer.getPort());
23        builder.nearCache().mode(getNearCacheMode()).maxEntries(2);
24        return AssertsNearCache.create(this.<byte[], Object>cache(), builder);
25     }
26  
27     protected NearCacheMode getNearCacheMode() {
28        return NearCacheMode.EAGER;
29     }
30  
31     public void testEvictAfterReachingMax() {
32        assertClient.expectNoNearEvents();
33        assertClient.put(1, "v1").expectNearPut(1, "v1");
34        assertClient.put(2, "v1").expectNearPut(2, "v1");
35        assertClient.get(1, "v1").expectNearGetValue(1, "v1");
36        assertClient.get(2, "v1").expectNearGetValue(2, "v1");
37        assertClient.put(3, "v1").expectNearPut(3, "v1");
38        assertClient.get(3, "v1").expectNearGetValue(3, "v1");
39        assertClient.get(1, "v1").expectNearGetNull(1).expectNearPutIfAbsent(1, "v1");
40     }
