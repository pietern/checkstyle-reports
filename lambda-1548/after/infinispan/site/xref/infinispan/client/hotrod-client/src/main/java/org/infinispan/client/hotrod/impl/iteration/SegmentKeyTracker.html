1   package org.infinispan.client.hotrod.impl.iteration;
2   
3   import java.util.BitSet;
4   import java.util.HashSet;
5   import java.util.Set;
6   import java.util.concurrent.atomic.AtomicReferenceArray;
7   import java.util.stream.IntStream;
8   
9   import org.infinispan.client.hotrod.impl.consistenthash.SegmentConsistentHash;
10  import org.infinispan.client.hotrod.logging.Log;
11  import org.infinispan.client.hotrod.logging.LogFactory;
12  import org.infinispan.commons.equivalence.ByteArrayEquivalence;
13  import org.infinispan.commons.util.CollectionFactory;
14  import org.infinispan.commons.util.Util;
15  
16  /**
17   * @author gustavonalle
18   * @since 8.0
19   */
20  public class SegmentKeyTracker implements KeyTracker {
21  
22     private static final Log log = LogFactory.getLog(SegmentKeyTracker.class);
23  
24     private final AtomicReferenceArray<Set<byte[]>> keysPerSegment;
25     private final SegmentConsistentHash segmentConsistentHash;
26  
27     public SegmentKeyTracker(SegmentConsistentHash segmentConsistentHash) {
28        int numSegments = segmentConsistentHash.getNumSegments();
29        keysPerSegment = new AtomicReferenceArray<>(numSegments);
30        if (log.isDebugEnabled()) log.debugf("Created SegmentKeyTracker with %d segments", numSegments);
31        this.segmentConsistentHash = segmentConsistentHash;
32        IntStream.range(0, segmentConsistentHash.getNumSegments())
33                .forEach(i -> keysPerSegment.set(i, CollectionFactory.makeSet(ByteArrayEquivalence.INSTANCE)));
34     }
35  
36     public boolean track(byte[] key) {
37        int segment = segmentConsistentHash.getSegment(key);
38        boolean result = keysPerSegment.get(segment).add(key);
39        if (log.isDebugEnabled())
40           log.debugf("Tracking key %s belonging to segment %d, seenBefore? = %s", Util.printArray(key), segment, !result);
41        return result;
42     }
43  
44     public Set<Integer> missedSegments() {
45        int length = keysPerSegment.length();
46        if (length == 0) return null;
47        Set<Integer> missed = new HashSet<>(length);
48        for (int i = 0; i < keysPerSegment.length(); i++) {
49           if (keysPerSegment.get(i) != null) {
50              missed.add(i);
51           }
52        }
53        return missed;
54     }
55  
56     public void segmentsFinished(byte[] finishedSegments) {
57        if (finishedSegments != null) {
58           BitSet bitSet = BitSet.valueOf(finishedSegments);
59           if (log.isDebugEnabled()) log.debugf("Removing completed segments %s", bitSet);
60           bitSet.stream().forEach(seg -> keysPerSegment.set(seg, null));
61        }
62     }
63  }
