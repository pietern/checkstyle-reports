3   /**
4    * The events generated by default contain just enough information to make the
5    * event relevant but they avoid cramming too much information in order to reduce
6    * the cost of sending them. Optionally, the information shipped in the events
7    * can be customised in order to contain more information, such as values, or to
8    * contain even less information. This customization is done with {@link org.infinispan.filter.Converter}
9    * instances generated by a {@link org.infinispan.filter.ConverterFactory}.
10   *
11   * As a result of this conversion, custom events are reprenseted by this class,
12   * and are expected in methods annotation with either
13   * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryCreated},
14   * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryModified} or,
15   * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryRemoved}.
16   * The event parameter for any of these callbacks is always a {@link ClientCacheEntryCustomEvent},
17   * and if needed, the event's {@link org.infinispan.client.hotrod.event.ClientCacheEntryCustomEvent#getType()}
18   * can be queried to find out whether the originating event was the result of create,
19   * modified or removed.
20   *
21   * @param <T> Type of customized event data. It needs to be marshallable.
22   */
23  public interface ClientCacheEntryCustomEvent<T> extends ClientEvent {
24  
25     /**
26      * Customized event data. It can be any type as long as it can be converted
27      * to binary format for shipping between the server and client.
28      *
29      * @return an instance of the customised event data.
30      */
31     T getEventData();
32  
33     /**
34      * This will be true if the write command that caused this had to be retried
35      * again due to a topology change.  This could be a sign that this event
36      * has been duplicated or another event was dropped and replaced
37      * (eg: ModifiedEvent replaced CreateEvent)
38      *
39      * @return Whether the command that caused this event was retried
40      */
41     boolean isCommandRetried();
42  
43  }
