73     protected EmbeddedCacheManager createCacheManager() throws Exception {
74        org.infinispan.configuration.cache.ConfigurationBuilder builder = createConfigBuilder();
75        // the default key equivalence works only for byte[] so we need to override it with one that works for Object
76        builder.dataContainer().keyEquivalence(AnyEquivalence.getInstance());
77  
78        cacheManager = TestCacheManagerFactory.createCacheManager(builder);
79        cache = cacheManager.getCache();
80  
81        hotRodServer = HotRodClientTestingUtil.startHotRodServer(cacheManager);
82  
83        ConfigurationBuilder clientBuilder = new ConfigurationBuilder();
84        clientBuilder.addServer().host("127.0.0.1").port(hotRodServer.getPort());
85        clientBuilder.marshaller(new ProtoStreamMarshaller());
86        remoteCacheManager = new RemoteCacheManager(clientBuilder.build());
87  
88        remoteCache = remoteCacheManager.getCache();
89  
90        //initialize client-side serialization context
91        SerializationContext clientSerCtx = ProtoStreamMarshaller.getSerializationContext(remoteCacheManager);
92        MarshallerRegistration.registerMarshallers(clientSerCtx);
93        clientSerCtx.registerProtoFiles(FileDescriptorSource.fromString("not_indexed.proto", NOT_INDEXED_PROTO_SCHEMA));
94        clientSerCtx.registerMarshaller(new NotIndexedMarshaller());
95  
96        //initialize server-side serialization context
97        ProtobufMetadataManager protobufMetadataManager = cacheManager.getGlobalComponentRegistry().getComponent(ProtobufMetadataManager.class);
98        protobufMetadataManager.registerProtofile("sample_bank_account/bank.proto", Util.read(Util.getResourceAsStream("/sample_bank_account/bank.proto", getClass().getClassLoader())));
99        protobufMetadataManager.registerProtofile("not_indexed.proto", NOT_INDEXED_PROTO_SCHEMA);
100       assertNull(protobufMetadataManager.getFileErrors("sample_bank_account/bank.proto"));
101       assertNull(protobufMetadataManager.getFileErrors("not_indexed.proto"));
102       assertNull(protobufMetadataManager.getFilesWithErrors());
103 
104       protobufMetadataManager.registerMarshaller(new EmbeddedAccountMarshaller());
105       protobufMetadataManager.registerMarshaller(new EmbeddedUserMarshaller());
106       protobufMetadataManager.registerMarshaller(new NotIndexedMarshaller());
107 
108       return cacheManager;
109    }

114       builder.indexing().index(Index.ALL)

117       return builder;

129 
130       // try to get the object through the local cache interface and check it's the same object we put

132       Object key = cache.keySet().iterator().next();
133       Object localObject = cache.get(key);
134       assertAccount((Account) localObject, AccountHS.class);
135 
136       // get the object through the remote cache interface and check it's the same object we put
137       Account fromRemoteCache = remoteCache.get(1);
138       assertAccount(fromRemoteCache, AccountPB.class);

364    public void testIterationForRemote() throws Exception {
365       IntStream.range(0, 10).forEach(id -> remoteCache.put(id, createAccountPB(id)));
366 
367       // Remote unfiltered iteration
368       CloseableIterator<Map.Entry<Object, Object>> remoteUnfilteredIterator = remoteCache.retrieveEntries(null, null, 10);
369       remoteUnfilteredIterator.forEachRemaining(e -> {
370          Integer key = (Integer) e.getKey();
371          AccountPB value = (AccountPB) e.getValue();
372          assertTrue(key < 10);
373          assertTrue(value.getId() == key);
374       });
375 
376       // Remote filtered iteration
377       KeyValueFilterConverterFactory<Integer, Account, String> filterConverterFactory = () ->
378             new AbstractKeyValueFilterConverter<Integer, Account, String>() {
379                @Override
380                public String filterAndConvert(Integer key, Account value, Metadata metadata) {
381                   if (key % 2 == 0) {
382                      return value.toString();
383                   }
384                   return null;
385                }
386             };
387 
388       hotRodServer.addKeyValueFilterConverterFactory("filterConverterFactory", filterConverterFactory);
389 
390       CloseableIterator<Map.Entry<Object, Object>> remoteFilteredIterator = remoteCache.retrieveEntries("filterConverterFactory", null, 10);
391       remoteFilteredIterator.forEachRemaining(e -> {
392          Integer key = (Integer) e.getKey();
393          String value = (String) e.getValue();
394          assertTrue(key < 10);
395          assertTrue(value.equals(createAccountHS(key).toString()));
396       });
397 
398       // Embedded  iteration
399       Cache<Integer, AccountHS> ourCache = cache();
400       Iterator<Map.Entry<Integer, AccountHS>> localUnfilteredIterator = ourCache.entrySet().stream().iterator();
401       localUnfilteredIterator.forEachRemaining(e -> {
402          Integer key = e.getKey();
403          AccountHS value = e.getValue();
404          assertTrue(key < 10);
405          assertTrue(value.getId() == key);
406       });
407    }
