19  /**
20   * @author gfernandes

23  @SuppressWarnings("unchecked")
24  public interface AbstractRemoteIteratorTest {
25  
26     default <T> Set<T> setOf(T... values) {
27        return Stream.of(values).collect(Collectors.toSet());
28     }
29  
30     default <T> void populateCache(int numElements, Function<Integer, T> supplier, RemoteCache<Integer, T> remoteCache) {
31        IntStream.range(0, numElements).forEach(i -> remoteCache.put(i, supplier.apply(i)));
32     }
33  
34     default <T> void assertForAll(Set<T> elements, Predicate<? super T> condition) {
35        assertTrue(elements.stream().allMatch(condition));
36     }
37  
38     default AccountHS newAccount(int id) {
39        AccountHS account = new AccountHS();
40        account.setId(id);
41        account.setDescription("description for " + id);
42        account.setCreationDate(new Date());
43        return account;
44     }
45  
46     default AccountPB newAccountPB(int id) {
47        AccountPB account = new AccountPB();
48        account.setId(id);
49        account.setDescription("description for " + id);
50        account.setCreationDate(new Date());
51        return account;
52     }
53  
54     default Set<Integer> rangeAsSet(int minimum, int maximum) {
55        return IntStream.range(minimum, maximum).boxed().collect(Collectors.toSet());
56     }
57  
58     default <K> Set<K> extractKeys(Collection<Map.Entry<Object, Object>> entries) {
59        return entries.stream().map(e -> (K) e.getKey()).collect(Collectors.<K>toSet());
60     }
61  
62     default <V> Set<V> extractValues(Collection<Map.Entry<Object, Object>> entries) {
63        return entries.stream().map(e -> (V) e.getValue()).collect(Collectors.<V>toSet());
64     }
65  
66     default byte[] toByteBuffer(Object key, Marshaller marshaller) {
67        try {
68           return marshaller.objectToByteBuffer(key);
69        } catch (Exception ignored) { }
70        return null;
71     }
72  
73     default void assertKeysInSegment(Set<Map.Entry<Object, Object>> entries, Set<Integer> segments,
74                                      Marshaller marshaller, Function<byte[], Integer> segmentCalculator) {
75        entries.forEach(e -> {
76           Integer key = (Integer) e.getKey();
77           int segment = segmentCalculator.apply(toByteBuffer(key, marshaller));
78           assertTrue(segments.contains(segment));
79        });
80     }
81  
82  }
