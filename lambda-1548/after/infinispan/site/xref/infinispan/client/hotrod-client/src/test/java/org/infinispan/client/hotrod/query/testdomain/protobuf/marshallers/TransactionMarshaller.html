1   package org.infinispan.client.hotrod.query.testdomain.protobuf.marshallers;
2   
3   import org.infinispan.client.hotrod.query.testdomain.protobuf.TransactionPB;
4   import org.infinispan.protostream.MessageMarshaller;
5   
6   import java.io.IOException;
7   import java.util.Date;
8   
9   /**
10   * @author anistor@redhat.com
11   * @since 7.0
12   */
13  public class TransactionMarshaller implements MessageMarshaller<TransactionPB> {
14  
15     @Override
16     public String getTypeName() {
17        return "sample_bank_account.Transaction";
18     }
19  
20     @Override
21     public Class<TransactionPB> getJavaClass() {
22        return TransactionPB.class;
23     }
24  
25     @Override
26     public TransactionPB readFrom(ProtoStreamReader reader) throws IOException {
27        int id = reader.readInt("id");
28        String description = reader.readString("description");
29        int accountId = reader.readInt("accountId");
30        long date = reader.readLong("date");
31        double amount = reader.readDouble("amount");
32        boolean isDebit = reader.readBoolean("isDebit");
33  
34        TransactionPB transaction = new TransactionPB();
35        transaction.setId(id);
36        transaction.setDescription(description);
37        transaction.setAccountId(accountId);
38        transaction.setDate(new Date(date));
39        transaction.setAmount(amount);
40        transaction.setDebit(isDebit);
41        return transaction;
42     }
43  
44     @Override
45     public void writeTo(ProtoStreamWriter writer, TransactionPB transaction) throws IOException {
46        writer.writeInt("id", transaction.getId());
47        writer.writeString("description", transaction.getDescription());
48        writer.writeInt("accountId", transaction.getAccountId());
49        writer.writeLong("date", transaction.getDate().getTime());
50        writer.writeDouble("amount", transaction.getAmount());
51        writer.writeBoolean("isDebit", transaction.isDebit());
52     }
53  }
