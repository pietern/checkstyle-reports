20  /**
21   * @author gustavonalle
22   * @since 8.0
23   */
24  public class IterationStartOperation extends RetryOnFailureOperation<IterationStartResponse> {
25  
26     private static final Log log = LogFactory.getLog(IterationStartOperation.class);
27  
28  
29     private final String filterConverterFactory;
30     private final Set<Integer> segments;
31     private final int batchSize;
32     private final TransportFactory transportFactory;
33  
34     protected IterationStartOperation(Codec codec, Flag[] flags, byte[] cacheName, AtomicInteger topologyId,
35                                       String filterConverterFactory, Set<Integer> segments, int batchSize, TransportFactory transportFactory) {
36        super(codec, transportFactory, cacheName, topologyId, flags);
37        this.filterConverterFactory = filterConverterFactory;
38        this.segments = segments;
39        this.batchSize = batchSize;
40        this.transportFactory = transportFactory;
41     }
42  
43     @Override
44     protected Transport getTransport(int retryCount, Set<SocketAddress> failedServers) {
45        return transportFactory.getTransport(failedServers, cacheName);
46     }
47  
48     @Override
49     protected IterationStartResponse executeOperation(Transport transport) {
50        HeaderParams params = writeHeader(transport, ITERATION_START_REQUEST);
51        if (segments == null) {
52           transport.writeSignedVInt(-1);
53        } else {
54           // TODO use a more compact BitSet implementation, like http://roaringbitmap.org/
55           BitSet bitSet = new BitSet();
56           segments.stream().forEach(bitSet::set);
57           transport.writeOptionalArray(bitSet.toByteArray());
58        }
59        transport.writeOptionalString(filterConverterFactory);
60        transport.writeVInt(batchSize);
61        transport.flush();
62  
63        readHeaderAndValidate(transport, params);
64  
65        return new IterationStartResponse(transport.readString(), (SegmentConsistentHash) transportFactory.getConsistentHash(cacheName), topologyId.get(), transport);
66     }
67  
68     @Override
69     protected void releaseTransport(Transport transport) {
70     }
71  }
