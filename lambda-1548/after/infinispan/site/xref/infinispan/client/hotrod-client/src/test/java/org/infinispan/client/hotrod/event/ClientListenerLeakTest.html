14  @Test(groups = "functional", testName = "client.hotrod.event.ClientListenerLifecycleTest")
15  public class ClientListenerLeakTest extends SingleHotRodServerTest {
16  
17     private String cacheName;
18  
19     @Override
20     protected EmbeddedCacheManager createCacheManager() throws Exception {
21        EmbeddedCacheManager cm = super.createCacheManager();
22        ConfigurationBuilder cfg = hotRodCacheConfiguration();
23        cacheName = this.getClass().getSimpleName();
24        cm.defineConfiguration(cacheName, cfg.build());
25        return cm;
26     }
27  
28     public void testNoLeaksAfterShutdown() {
29        withRemoteCacheManager(new RemoteCacheManagerCallable(getRemoteCacheManager()) {
30           @Override
31           public void call() {
32              RemoteCache<Integer, String> remote = rcm.getCache(cacheName);
33              EventLogListener<Integer> eventListener = new EventLogListener<>();
34              remote.addClientListener(eventListener);
35              eventListener.expectNoEvents();
36              remote.put(1, "one");
37              eventListener.expectOnlyCreatedEvent(1, cache(cacheName));
38           }
39        });
40        detectThreadLeaks(".*Client-Listener-ClientListenerLeakTest-.*");
41     }
