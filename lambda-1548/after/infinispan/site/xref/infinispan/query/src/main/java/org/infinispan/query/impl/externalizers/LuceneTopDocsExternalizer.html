1   package org.infinispan.query.impl.externalizers;
2   
3   import java.io.IOException;
4   import java.io.ObjectInput;
5   import java.io.ObjectOutput;
6   import java.util.Set;
7   
8   import org.apache.lucene.search.ScoreDoc;
9   import org.apache.lucene.search.TopDocs;
10  import org.infinispan.commons.io.UnsignedNumeric;
11  import org.infinispan.commons.marshall.AbstractExternalizer;
12  import org.infinispan.commons.util.Util;
13  
14  public class LuceneTopDocsExternalizer extends AbstractExternalizer<TopDocs> {
15  
16     @Override
17     public Set<Class<? extends TopDocs>> getTypeClasses() {
18        return Util.<Class<? extends TopDocs>>asSet(TopDocs.class);
19     }
20  
21     @Override
22     public TopDocs readObject(final ObjectInput input) throws IOException, ClassNotFoundException {
23        final int totalHits = UnsignedNumeric.readUnsignedInt(input);
24        final float maxScore = input.readFloat();
25        final int scoreCount = UnsignedNumeric.readUnsignedInt(input);
26        final ScoreDoc[] scoreDocs = new ScoreDoc[scoreCount];
27        for (int i=0; i<scoreCount; i++) {
28           scoreDocs[i] = (ScoreDoc) input.readObject();
29        }
30        return new TopDocs(totalHits, scoreDocs, maxScore);
31     }
32  
33     @Override
34     public void writeObject(final ObjectOutput output, final TopDocs topDocs) throws IOException {
35        UnsignedNumeric.writeUnsignedInt(output, topDocs.totalHits);
36        output.writeFloat(topDocs.getMaxScore());
37        final ScoreDoc[] scoreDocs = topDocs.scoreDocs;
38        final int count = scoreDocs.length;
39        UnsignedNumeric.writeUnsignedInt(output, count);
40        for (ScoreDoc scoreDoc : scoreDocs) {
41           output.writeObject(scoreDoc);
42        }
43     }
44  
45     @Override
46     public Integer getId() {
47        return ExternalizerIds.LUCENE_TOPDOCS;
48     }
49  
50  }
