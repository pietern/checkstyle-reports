13  /**
14   * A query that does not return any results because the query filter is a boolean contradiction.
15   *
16   * @author anistor@redhat.com
17   * @since 8.0
18   */
19  final class EmptyResultQuery extends BaseEmbeddedQuery {
20  
21     EmptyResultQuery(QueryFactory queryFactory, AdvancedCache<?, ?> cache, String jpaQuery, Map<String, Object> namedParameters,
22                      long startOffset, int maxResults) {
23        super(queryFactory, cache, jpaQuery, namedParameters, null, startOffset, maxResults);
24     }
25  
26     @Override
27     protected Comparator<Comparable[]> getComparator() {
28        return null;
29     }
30  
31     @Override
32     protected CloseableIterator<ObjectFilter.FilterResult> getIterator() {
33        return new CloseableIterator<ObjectFilter.FilterResult>() {
34  
35           @Override
36           public void close() {
37           }
38  
39           @Override
40           public boolean hasNext() {
41              return false;
42           }
43  
44           @Override
45           public ObjectFilter.FilterResult next() {
46              throw new NoSuchElementException();
47           }
48        };
49     }
50  
51     @Override
52     public String toString() {
53        return "EmptyResultQuery{" +
54              "jpaQuery=" + jpaQuery +
55              ", namedParameters=" + namedParameters +
56              ", projection=" + Arrays.toString(projection) +
57              ", startOffset=" + startOffset +
58              ", maxResults=" + maxResults +
59              '}';
60     }
61  }
