5   import java.util.StringTokenizer;

7   public class CustomTransformer implements Transformer {
8      @Override
9      public Object fromString(String s) {
10        StringTokenizer strtok = new StringTokenizer(s, ",");
11        int[] ints = new int[3];
12        int i = 0;
13        while (strtok.hasMoreTokens()) {
14           String token = strtok.nextToken();
15           String[] contents = token.split("=");
16           ints[i++] = Integer.parseInt(contents[1]);
17        }
18        return new CustomKey(ints[0], ints[1], ints[2]);
19     }
20  
21     @Override
22     public String toString(Object customType) {
23        CustomKey ck = (CustomKey) customType;
24        return "i=" + ck.i + ",j=" + ck.j + ",k=" + ck.k;
25     }
