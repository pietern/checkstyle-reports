19  /**
20   * Tests that the default embedded cache configuration can be overridden.
21   *
22   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
23   */
24  @Test(groups = "functional", testName = "cdi.test.cachemanager.embedded.DefaultConfigurationTest")
25  public class DefaultConfigurationTest extends Arquillian {
26  
27     @Deployment
28     public static Archive<?> deployment() {
29        return baseDeployment()
30              .addClass(DefaultConfigurationTest.class)
31              .addClass(DefaultTestEmbeddedCacheManagerProducer.class);
32     }
33  
34     @Inject
35     private Cache<?, ?> cache;
36  
37     public void testDefaultConfiguration() {
38        assertEquals(cache.getCacheConfiguration().eviction().maxEntries(), 16);
39        assertEquals(cache.getName(), DEFAULT_CACHE_NAME);
40     }
41  
42     /**
43      * Overrides the default embedded cache configuration used for the initialization of the default embedded cache
44      * manager.
45      */
46     public static class Config {
47        @Produces
48        public Configuration customDefaultConfiguration() {
49           return new ConfigurationBuilder()
50                 .eviction().maxEntries(16)
51                 .build();
52        }
53     }
54  }
