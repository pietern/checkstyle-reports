1   package org.infinispan.cdi;
2   
3   import org.infinispan.cdi.util.defaultbean.DefaultBean;
4   import org.infinispan.client.hotrod.RemoteCacheManager;
5   
6   import javax.enterprise.context.ApplicationScoped;
7   import javax.enterprise.inject.Disposes;
8   import javax.enterprise.inject.Produces;
9   
10  /**
11   * <p>The default {@link RemoteCacheManager} producer.</p>
12   *
13   * <p>The remote cache manager used by default can be overridden by creating a producer which produces the new default
14   * remote cache manager. The remote cache manager produced must have the scope {@link ApplicationScoped} and the
15   * {@linkplain javax.enterprise.inject.Default Default} qualifier.</p>
16   *
17   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
18   */
19  public class DefaultRemoteCacheManagerProducer {
20     /**
21      * Produces the default remote cache manager with the default settings.
22      *
23      * @return the default remote cache manager.
24      * @see org.infinispan.client.hotrod.RemoteCacheManager#RemoteCacheManager()
25      */
26     @Produces
27     @ApplicationScoped
28     @DefaultBean(RemoteCacheManager.class)
29     public RemoteCacheManager getDefaultRemoteCacheManager() {
30        return new RemoteCacheManager();
31     }
32  
33     /**
34      * Stops the default remote cache manager when the corresponding instance is released.
35      *
36      * @param defaultRemoteCacheManager the default remote cache manager.
37      */
38     @SuppressWarnings("unused")
39     private void stopRemoteCacheManager(@Disposes RemoteCacheManager defaultRemoteCacheManager) {
40        defaultRemoteCacheManager.stop();
41     }
42  
43  }
