20  @Test(groups = "functional", testName = "api.MetadataAPIDefaultExpiryTest")
21  public class MetadataAPIDefaultExpiryTest extends SingleCacheManagerTest {
22  
23     public static final int EXPIRATION_TIMEOUT = 1000;
24  
25     private final ControlledTimeService controlledTimeService = new ControlledTimeService(0);
26  
27     @Override
28     protected EmbeddedCacheManager createCacheManager() throws Exception {
29        ConfigurationBuilder builder = new ConfigurationBuilder();
30        builder.expiration().lifespan(EXPIRATION_TIMEOUT);
31        EmbeddedCacheManager manager = TestCacheManagerFactory.createCacheManager(builder);
32        TestingUtil.replaceComponent(manager, TimeService.class, controlledTimeService, true);
33        return manager;
34     }
35  
36     public void testDefaultLifespanPut() {
37        cache().put(1, "v1");
38        expectCachedThenExpired(1, "v1");
39        cache().getAdvancedCache().put(2, "v2", new EmbeddedMetadata.Builder().build());
40        expectCachedThenExpired(2, "v2");
41     }
42  
43     public void testDefaultLifespanReplace() {
44        cache().put(1, "v1");
45        cache().replace(1, "v11");
46        expectCachedThenExpired(1, "v11");
47        cache().getAdvancedCache().put(2, "v2", new EmbeddedMetadata.Builder().build());
48        cache().getAdvancedCache().replace(2, "v22", new EmbeddedMetadata.Builder().build());
49        expectCachedThenExpired(2, "v22");
50     }
51  
52     public void testDefaultLifespanReplaceWithOldValue() {
53        cache().put(1, "v1");
54        cache().replace(1, "v1", "v11");
55        expectCachedThenExpired(1, "v11");
56        cache().getAdvancedCache().put(2, "v2", new EmbeddedMetadata.Builder().build());
57        cache().getAdvancedCache().replace(2, "v2", "v22", new EmbeddedMetadata.Builder().build());
58        expectCachedThenExpired(2, "v22");
59     }
60  
61     public void testDefaultLifespanPutIfAbsent() {
62        cache().putIfAbsent(1, "v1");
63        expectCachedThenExpired(1, "v1");
64        cache().getAdvancedCache().putIfAbsent(2, "v2", new EmbeddedMetadata.Builder().build());
65        expectCachedThenExpired(2, "v2");
66     }
67  
68     public void testDefaultLifespanPutForExternalRead() {
69        cache().putForExternalRead(1, "v1");
70        expectCachedThenExpired(1, "v1");
71        cache().getAdvancedCache().putForExternalRead(2, "v2", new EmbeddedMetadata.Builder().build());
72        expectCachedThenExpired(2, "v2");
73     }
74  
75     public void testDefaultLifespanPutAsync() throws Exception {
76        NotifyingFuture<Object> f = cache().putAsync(1, "v1");
77        f.get(10, TimeUnit.SECONDS);
78        expectCachedThenExpired(1, "v1");
79        f = cache().getAdvancedCache().putAsync(2, "v2", new EmbeddedMetadata.Builder().build());
80        f.get(10, TimeUnit.SECONDS);
81        expectCachedThenExpired(2, "v2");
82     }
83  
84     private void expectCachedThenExpired(Integer key, String value) {
85        // Check that it doesn't expire too early
86        controlledTimeService.advance(EXPIRATION_TIMEOUT - 1);
87        String v = this.<Integer, String>cache().get(key);
88        assertEquals(value, v);
89  
90        // But not too late either
91        controlledTimeService.advance(2);
92        assertNull(cache.get(key));
93     }
94  }
