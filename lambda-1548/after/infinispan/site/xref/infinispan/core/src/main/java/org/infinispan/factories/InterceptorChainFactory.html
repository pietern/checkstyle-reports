79     private static final Log log = LogFactory.getLog(InterceptorChainFactory.class);

81     private CommandInterceptor createInterceptor(CommandInterceptor interceptor, Class<? extends CommandInterceptor> interceptorType) {
82        CommandInterceptor chainedInterceptor = componentRegistry.getComponent(interceptorType);
83        if (chainedInterceptor == null) {
84           chainedInterceptor = interceptor;
85           register(interceptorType, chainedInterceptor);
86        } else {
87           // wipe next/last chaining!!
88           chainedInterceptor.setNext(null);
89        }
90        return chainedInterceptor;
91     }
92  
93  
94     private void register(Class<? extends CommandInterceptor> clazz, CommandInterceptor chainedInterceptor) {
95        try {
96           componentRegistry.registerComponent(chainedInterceptor, clazz);
97        } catch (RuntimeException e) {
98           log.unableToCreateInterceptor(clazz, e);
99           throw e;
100       }
101    }

103    private boolean isUsingMarshalledValues(Configuration c) {
104       return c.storeAsBinary().enabled() && (c.storeAsBinary().storeKeysAsBinary() || c.storeAsBinary().storeValuesAsBinary());
105    }

107    public InterceptorChain buildInterceptorChain() {
108       TransactionMode transactionMode = configuration.transaction().transactionMode();
109       boolean needsVersionAwareComponents = transactionMode.isTransactional() &&
110               Configurations.isVersioningEnabled(configuration);
111 
112       InterceptorChain interceptorChain = new InterceptorChain(componentRegistry.getComponentMetadataRepo());
113       // add the interceptor chain to the registry first, since some interceptors may ask for it.
114       componentRegistry.registerComponent(interceptorChain, InterceptorChain.class);
115 
116       boolean invocationBatching = configuration.invocationBatching().enabled();
117       boolean isTotalOrder = configuration.transaction().transactionProtocol().isTotalOrder();
118       CacheMode cacheMode = configuration.clustering().cacheMode();
119 
120       if (configuration.clustering().cacheMode().isDistributed()) {
121          interceptorChain.appendInterceptor(createInterceptor(new DistributionBulkInterceptor<>(),
122                  DistributionBulkInterceptor.class), false);
123       }
124 
125       // load the icInterceptor first
126       if (invocationBatching) {
127          interceptorChain.appendInterceptor(createInterceptor(new BatchingInterceptor(), BatchingInterceptor.class), false);
128       }
129       interceptorChain.appendInterceptor(createInterceptor(new InvocationContextInterceptor(), InvocationContextInterceptor.class), false);
130 
131       CompatibilityModeConfiguration compatibility = configuration.compatibility();
132       if (compatibility.enabled()) {
133          Marshaller compatibilityMarshaller = compatibility.marshaller();
134          if (compatibilityMarshaller != null) {
135             componentRegistry.wireDependencies(compatibilityMarshaller);
136          }
137          interceptorChain.appendInterceptor(createInterceptor(
138                new TypeConverterInterceptor(compatibilityMarshaller), TypeConverterInterceptor.class), false);
139       }
140 
141       // add marshallable check interceptor for situations where we want to figure out before marshalling
142       // Store as binary marshalls keys/values eagerly now, so avoid extra serialization
143       if (configuration.clustering().async().useReplQueue() || hasAsyncStore())
144          interceptorChain.appendInterceptor(createInterceptor(new IsMarshallableInterceptor(), IsMarshallableInterceptor.class), false);
145 
146       // load the cache management interceptor next
147       if (configuration.jmxStatistics().available()) {
148          interceptorChain.appendInterceptor(createInterceptor(new CacheMgmtInterceptor(), CacheMgmtInterceptor.class), false);
149       }
150 
151       // load the state transfer lock interceptor
152       // the state transfer lock ensures that the cache member list is up-to-date
153       // so it's necessary even if state transfer is disabled
154       if (cacheMode.isDistributed() || cacheMode.isReplicated()) {
155          if (isTotalOrder) {
156             interceptorChain.appendInterceptor(createInterceptor(new TotalOrderStateTransferInterceptor(),
157                                                                  TotalOrderStateTransferInterceptor.class), false);
158          } else {
159             interceptorChain.appendInterceptor(createInterceptor(new StateTransferInterceptor(), StateTransferInterceptor.class), false);
160          }
161          if (transactionMode.isTransactional()) {
162             interceptorChain.appendInterceptor(createInterceptor(new TransactionSynchronizerInterceptor(), TransactionSynchronizerInterceptor.class), false);
163          }
164       }
165 
166       // The partition handling must run every time the state transfer interceptor retries a command (in non-transactional caches)
167       if (configuration.clustering().partitionHandling().enabled()
168             && (cacheMode.isDistributed() || cacheMode.isReplicated())) {
169          interceptorChain.appendInterceptor(createInterceptor(new PartitionHandlingInterceptor(), PartitionHandlingInterceptor.class), false);
170       }
171 
172 
173       //load total order interceptor
174       if (isTotalOrder) {
175          interceptorChain.appendInterceptor(createInterceptor(new TotalOrderInterceptor(), TotalOrderInterceptor.class), false);
176       }
177 
178       // load the tx interceptor
179       if (transactionMode.isTransactional())
180          interceptorChain.appendInterceptor(createInterceptor(new TxInterceptor(), TxInterceptor.class), false);
181 
182       if (isUsingMarshalledValues(configuration)) {
183          CommandInterceptor interceptor;
184             interceptor = createInterceptor(
185                   new MarshalledValueInterceptor(), MarshalledValueInterceptor.class);
186 
187          interceptorChain.appendInterceptor(interceptor, false);
188       }
189 
190       // NotificationInterceptor is used only for Prepare/Commit/Rollback notifications
191       if (transactionMode.isTransactional() && configuration.transaction().notifications()) {
192          interceptorChain.appendInterceptor(createInterceptor(new NotificationInterceptor(), NotificationInterceptor.class), false);
193       }
194 
195       if (configuration.transaction().useEagerLocking()) {
196          configuration.transaction().lockingMode(LockingMode.PESSIMISTIC);
197       }
198 
199       //the total order protocol doesn't need locks
200       if (!isTotalOrder) {
201          if (transactionMode.isTransactional()) {
202             if (configuration.transaction().lockingMode() == LockingMode.PESSIMISTIC) {
203                interceptorChain.appendInterceptor(createInterceptor(new PessimisticLockingInterceptor(), PessimisticLockingInterceptor.class), false);
204             } else {
205                interceptorChain.appendInterceptor(createInterceptor(new OptimisticLockingInterceptor(), OptimisticLockingInterceptor.class), false);
206             }
207          } else {
208             interceptorChain.appendInterceptor(createInterceptor(new NonTransactionalLockingInterceptor(), NonTransactionalLockingInterceptor.class), false);
209          }
210       }
211 
212       if (configuration.sites().hasEnabledBackups() && !configuration.sites().disableBackups()) {
213          if (transactionMode == TransactionMode.TRANSACTIONAL) {
214             if (configuration.transaction().lockingMode() == LockingMode.OPTIMISTIC) {
215                interceptorChain.appendInterceptor(createInterceptor(new OptimisticBackupInterceptor(), OptimisticBackupInterceptor.class), false);
216             } else {
217                interceptorChain.appendInterceptor(createInterceptor(new PessimisticBackupInterceptor(), PessimisticBackupInterceptor.class), false);
218             }
219          } else {
220             interceptorChain.appendInterceptor(createInterceptor(new NonTransactionalBackupInterceptor(), NonTransactionalBackupInterceptor.class), false);
221          }
222       }
223 
224       // This needs to be added after the locking interceptor (for tx caches) but before the wrapping interceptor.
225       if (configuration.clustering().l1().enabled()) {
226          interceptorChain.appendInterceptor(createInterceptor(new L1LastChanceInterceptor(), L1LastChanceInterceptor.class), false);
227       }
228 
229       if (configuration.clustering().hash().groups().enabled()) {
230          interceptorChain.appendInterceptor(createInterceptor(new GroupingInterceptor(), GroupingInterceptor.class), false);
231       }
232 
233       // The expiration interceptor must come before the entry wrapping interceptor as it can stop data container
234       // access from expiring things.
235       if (transactionMode.isTransactional() && configuration.transaction().lockingMode() == LockingMode.OPTIMISTIC) {
236          interceptorChain.appendInterceptor(createInterceptor(new OptimisticTxExpirationInterceptor<>(),
237                  ExpirationInterceptor.class), false);
238       } else {
239          interceptorChain.appendInterceptor(createInterceptor(new ExpirationInterceptor<>(),
240                  ExpirationInterceptor.class), false);
241       }
242 
243       if (needsVersionAwareComponents && cacheMode.isClustered()) {
244          if (isTotalOrder) {
245             interceptorChain.appendInterceptor(createInterceptor(new TotalOrderVersionedEntryWrappingInterceptor(),
246                                                                  TotalOrderVersionedEntryWrappingInterceptor.class), false);
247          } else {
248             interceptorChain.appendInterceptor(createInterceptor(new VersionedEntryWrappingInterceptor(), VersionedEntryWrappingInterceptor.class), false);
249          }
250       } else
251          interceptorChain.appendInterceptor(createInterceptor(new EntryWrappingInterceptor(), EntryWrappingInterceptor.class), false);
252 
253       if (configuration.persistence().usingStores()) {
254          if (configuration.persistence().passivation()) {
255             if (cacheMode.isClustered())
256                interceptorChain.appendInterceptor(createInterceptor(new ClusteredActivationInterceptor(), ClusteredActivationInterceptor.class), false);
257             else
258                interceptorChain.appendInterceptor(createInterceptor(new ActivationInterceptor(), ActivationInterceptor.class), false);
259          } else {
260             if (cacheMode.isClustered())
261                interceptorChain.appendInterceptor(createInterceptor(new ClusteredCacheLoaderInterceptor(), ClusteredCacheLoaderInterceptor.class), false);
262             else
263                interceptorChain.appendInterceptor(createInterceptor(new CacheLoaderInterceptor(), CacheLoaderInterceptor.class), false);
264             switch (cacheMode) {
265                case DIST_SYNC:
266                case DIST_ASYNC:
267                case REPL_SYNC:
268                case REPL_ASYNC:
269                   interceptorChain.appendInterceptor(createInterceptor(new DistCacheWriterInterceptor(), DistCacheWriterInterceptor.class), false);
270                   break;
271                default:
272                   interceptorChain.appendInterceptor(createInterceptor(new CacheWriterInterceptor(), CacheWriterInterceptor.class), false);
273                   break;
274             }
275          }
276       }
277 
278       if (configuration.deadlockDetection().enabled() && !isTotalOrder) {
279          interceptorChain.appendInterceptor(createInterceptor(new DeadlockDetectingInterceptor(), DeadlockDetectingInterceptor.class), false);
280       }
281 
282       if (configuration.clustering().l1().enabled()) {
283          if (transactionMode.isTransactional()) {
284             interceptorChain.appendInterceptor(createInterceptor(new L1TxInterceptor(), L1TxInterceptor.class), false);
285          }
286          else {
287             interceptorChain.appendInterceptor(createInterceptor(new L1NonTxInterceptor(), L1NonTxInterceptor.class), false);
288          }
289       }
290 
291       switch (cacheMode) {
292          case INVALIDATION_SYNC:
293          case INVALIDATION_ASYNC:
294             interceptorChain.appendInterceptor(createInterceptor(new InvalidationInterceptor(), InvalidationInterceptor.class), false);
295             break;
296          case DIST_SYNC:
297          case REPL_SYNC:
298             if (needsVersionAwareComponents) {
299                if (isTotalOrder) {
300                   interceptorChain.appendInterceptor(createInterceptor(new TotalOrderVersionedDistributionInterceptor(),
301                                                                        TotalOrderVersionedDistributionInterceptor.class), false);
302                } else {
303                   interceptorChain.appendInterceptor(createInterceptor(new VersionedDistributionInterceptor(), VersionedDistributionInterceptor.class), false);
304                }
305                break;
306             }
307          case DIST_ASYNC:
308          case REPL_ASYNC:
309             if (transactionMode.isTransactional()) {
310                if (isTotalOrder) {
311                   interceptorChain.appendInterceptor(createInterceptor(new TotalOrderDistributionInterceptor(), TotalOrderDistributionInterceptor.class), false);
312                } else {
313                   interceptorChain.appendInterceptor(createInterceptor(new TxDistributionInterceptor(), TxDistributionInterceptor.class), false);
314                }
315             } else {
316                interceptorChain.appendInterceptor(createInterceptor(new NonTxDistributionInterceptor(), NonTxDistributionInterceptor.class), false);
317             }
318             break;
319          case LOCAL:
320             //Nothing...
321       }
322 
323       CommandInterceptor callInterceptor = createInterceptor(new CallInterceptor(), CallInterceptor.class);
324       interceptorChain.appendInterceptor(callInterceptor, false);
325       log.trace("Finished building default interceptor chain.");
326       buildCustomInterceptors(interceptorChain, configuration.customInterceptors());
327       return interceptorChain;
328    }

330    private void buildCustomInterceptors(InterceptorChain interceptorChain, CustomInterceptorsConfiguration customInterceptors) {
331       for (InterceptorConfiguration config : customInterceptors.interceptors()) {
332          if (interceptorChain.containsInterceptorType(config.interceptor().getClass())) continue;
333 
334          CommandInterceptor customInterceptor = config.interceptor();
335          SecurityActions.applyProperties(customInterceptor, config.properties());
336          register(customInterceptor.getClass(), customInterceptor);
337          if (config.first())
338             interceptorChain.addInterceptor(customInterceptor, 0);
339          else if (config.last())
340             interceptorChain.addInterceptorBefore(customInterceptor, CallInterceptor.class);
341          else if (config.index() >= 0)
342             interceptorChain.addInterceptor(customInterceptor, config.index());
343          else if (config.after() != null) {
344             List<CommandInterceptor> withClassName = interceptorChain.getInterceptorsWithClass(config.after());
345             if (withClassName.isEmpty()) {
346                throw new CacheConfigurationException("Cannot add after class: " + config.after()
347                                                       + " as no such interceptor exists in the default chain");
348             }
349             interceptorChain.addInterceptorAfter(customInterceptor, withClassName.get(0).getClass());
350          } else if (config.before() != null) {
351             List<CommandInterceptor> withClassName = interceptorChain.getInterceptorsWithClass(config.before());
352             if (withClassName.isEmpty()) {
353                throw new CacheConfigurationException("Cannot add before class: " + config.after()
354                                                       + " as no such interceptor exists in the default chain");
355             }
356             interceptorChain.addInterceptorBefore(customInterceptor, withClassName.get(0).getClass());
357          } else if (config.position() == InterceptorConfiguration.Position.OTHER_THAN_FIRST_OR_LAST) {
358             interceptorChain.addInterceptor(customInterceptor, 1);
359          }
360       }
361 
362    }

364    private boolean hasAsyncStore() {
365       List<StoreConfiguration> loaderConfigs = configuration.persistence().stores();
366       for (StoreConfiguration loaderConfig : loaderConfigs) {
367          if (loaderConfig.async().enabled())
368             return true;
369       }
370       return false;
371    }

373    @Override
374    public <T> T construct(Class<T> componentType) {
375       try {
376          return componentType.cast(buildInterceptorChain());
377       } catch (CacheException ce) {
378          throw ce;
379       } catch (Exception e) {
380          throw new CacheConfigurationException("Unable to build interceptor chain", e);
381       }
382    }

384    public static InterceptorChainFactory getInstance(ComponentRegistry componentRegistry, Configuration configuration) {
385       InterceptorChainFactory icf = new InterceptorChainFactory();
386       icf.componentRegistry = componentRegistry;
387       icf.configuration = configuration;
388       return icf;
389    }
