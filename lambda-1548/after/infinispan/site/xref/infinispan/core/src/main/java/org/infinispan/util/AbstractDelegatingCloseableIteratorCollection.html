15  /**
16   * Delegating collection that produces closeable iterators and spliterators from the collection returned from
17   * {@link AbstractDelegatingCloseableIteratorCollection#delegate()} method.
18   * @param <E> The type in the collection
19   */
20  public abstract class AbstractDelegatingCloseableIteratorCollection<E> extends AbstractDelegatingCollection<E>
21          implements CloseableIteratorCollection<E> {
22  
23     protected abstract CloseableIteratorCollection<E> delegate();
24  
25     @Override
26     public CloseableIterator<E> iterator() {
27        return delegate().iterator();
28     }
29  
30     @Override
31     public CloseableSpliterator<E> spliterator() {
32        return delegate().spliterator();
33     }
34  }
