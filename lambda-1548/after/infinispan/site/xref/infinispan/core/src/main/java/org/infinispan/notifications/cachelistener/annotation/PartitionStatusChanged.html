10  /**
11   * This annotation should be used on methods that need to be notified when the
12   * {@link org.infinispan.partitionhandling.AvailabilityMode} in use by the
13   * {@link org.infinispan.partitionhandling.impl.PartitionHandlingManager} changes due to a change in cluster topology.
14   * This is only fired in a {@link CacheMode#DIST_SYNC}, {@link CacheMode#DIST_ASYNC}, {@link CacheMode#REPL_SYNC} or
15   * {@link CacheMode#REPL_ASYNC} configured cache.
16   * <p/>
17   * Methods annotated with this annotation should accept a single parameter, a
18   * {@link org.infinispan.notifications.cachelistener.event.PartitionStatusChangedEvent} otherwise a
19   * {@link org.infinispan.notifications.IncorrectListenerException} will be thrown when registering your listener.
20   * <p/>
21   * Note that methods marked with this annotation will be fired <i>before</i> and <i>after</i> the updated
22   * {@link org.infinispan.partitionhandling.AvailabilityMode}
23   * is updated, i.e., your method will be called twice, with
24   * {@link org.infinispan.notifications.cachelistener.event.Event#isPre()} being set to <tt>true</tt> as well
25   * as <tt>false</tt>.
26   * <p/>
27   *
28   * @author William Burns
29   * @see org.infinispan.notifications.Listener
30   * @since 7.0
31   */
32  // ensure this annotation is available at runtime.
33  @Retention(RetentionPolicy.RUNTIME)
34  // ensure that this annotation is applied to classes.
35  @Target(ElementType.METHOD)
36  public @interface PartitionStatusChanged {
37  }
