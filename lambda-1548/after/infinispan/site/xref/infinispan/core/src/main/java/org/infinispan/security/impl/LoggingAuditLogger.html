1   package org.infinispan.security.impl;

3   import javax.security.auth.Subject;
4   
5   import org.infinispan.security.AuditContext;
6   import org.infinispan.security.AuditLogger;
7   import org.infinispan.security.AuditResponse;
8   import org.infinispan.security.AuthorizationPermission;
9   import org.infinispan.security.Security;
10  import org.jboss.logging.Logger;
11  
12  /**
13   * LoggingAuditLogger. A simple {@link AuditLogger} which send audit messages to a named
14   * logger "org.infinispan.AUDIT"
15   *
16   * @author Tristan Tarrant
17   * @since 7.0
18   * @public
19   */
20  public class LoggingAuditLogger implements AuditLogger {
21     static final AuditMessages auditLog = Logger.getMessageLogger(AuditMessages.class, "org.infinispan.AUDIT");
22  
23     @Override
24     public void audit(Subject subject, AuditContext context, String contextName, AuthorizationPermission permission,
25           AuditResponse response) {
26        auditLog.auditMessage(response, Security.getSubjectUserPrincipal(subject), permission, context, contextName);
27     }
28  }
