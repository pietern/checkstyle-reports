5   /**
6    * An {@link Action} implementation that checks if the command topology id is valid.
7    * <p/>
8    * The command topology id is valid when it is higher or equal thant the first topology as member for this node.
9    *
10   * @author Pedro Ruivo
11   * @since 8.0
12   */
13  public class CheckTopologyAction implements Action {
14  
15     private final BasePerCacheInboundInvocationHandler handler;
16  
17     public CheckTopologyAction(BasePerCacheInboundInvocationHandler handler) {
18        this.handler = handler;
19     }
20  
21     @Override
22     public ActionStatus check(ActionState state) {
23        return handler.isCommandSentBeforeFirstTopology(state.getCommandTopologyId()) ?
24              ActionStatus.CANCELED :
25              ActionStatus.READY;
26     }
27  
28     @Override
29     public void addListener(ActionListener listener) {
30        //no-op
31     }
32  }
