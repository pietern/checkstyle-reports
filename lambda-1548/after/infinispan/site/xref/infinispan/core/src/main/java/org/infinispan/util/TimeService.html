1   package org.infinispan.util;
2   
3   import org.infinispan.factories.scopes.Scope;
4   import org.infinispan.factories.scopes.Scopes;
5   
6   import java.util.concurrent.TimeUnit;
7   
8   /**
9    * Encapsulates all the time related logic in this interface.
10   *
11   * @author Pedro Ruivo
12   * @since 5.3
13   */
14  @Scope(Scopes.GLOBAL)
15  public interface TimeService {
16  
17     /**
18      * @return the current clock time in milliseconds. Note that it depends of the system time.
19      */
20     long wallClockTime();
21  
22     /**
23      * @return the current cpu time in nanoseconds. Note that some platforms do not provide nanosecond precision.
24      */
25     long time();
26  
27     /**
28      * It is equivalent to {@code timeDuration(startTime, time(), outputTimeUnit)}.
29      *
30      * @param startTime      start cpu time in nanoseconds, usually returned by {@link #time()}.
31      * @param outputTimeUnit the {@link TimeUnit} of the returned value.
32      * @return the duration between the current cpu time and startTime. It returns zero if startTime is less than zero or
33      *         if startTime is greater than the current cpu time.
34      */
35     long timeDuration(long startTime, TimeUnit outputTimeUnit);
36  
37     /**
38      * @param startTime      start cpu time in nanoseconds, usually returned by {@link #time()}.
39      * @param endTime        end cpu time in nanoseconds, usually returned by {@link #time()}.
40      * @param outputTimeUnit the {@link TimeUnit} of the returned value.
41      * @return the duration between the endTime and startTime. It returns zero if startTime or endTime are less than zero
42      *         or if startTime is greater than the endTime.
43      */
44     long timeDuration(long startTime, long endTime, TimeUnit outputTimeUnit);
45  
46     /**
47      * @param endTime a cpu time in nanoseconds, usually returned by {@link #time()}
48      * @return {@code true} if the endTime is less or equals than the current cpu time.
49      */
50     boolean isTimeExpired(long endTime);
51  
52     /**
53      * @param endTime        the end cpu time in nanoseconds.
54      * @param outputTimeUnit the {@link TimeUnit} of the returned value.
55      * @return the remaining cpu time until the endTime is reached.
56      */
57     long remainingTime(long endTime, TimeUnit outputTimeUnit);
58  
59     /**
60      * @param duration      the duration.
61      * @param inputTimeUnit the {@link TimeUnit} of the duration.
62      * @return the expected end time in nano seconds. If duration is less or equals to zero, the current cpu time is returned ({@link
63      *         #time()}).
64      */
65     long expectedEndTime(long duration, TimeUnit inputTimeUnit);
66  
67  }
