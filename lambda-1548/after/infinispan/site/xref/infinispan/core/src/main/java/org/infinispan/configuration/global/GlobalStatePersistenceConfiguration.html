1   package org.infinispan.configuration.global;
2   
3   import org.infinispan.commons.configuration.attributes.Attribute;
4   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
5   import org.infinispan.commons.configuration.attributes.AttributeSet;
6   
7   /**
8    *
9    * GlobalStatePersistenceConfiguration.
10   *
11   * @author Tristan Tarrant
12   * @since 8.1
13   */
14  public class GlobalStatePersistenceConfiguration {
15     public static final AttributeDefinition<Boolean> ENABLED = AttributeDefinition.builder("enabled", false).immutable().build();
16     public static final AttributeDefinition<String> LOCATION = AttributeDefinition.builder("location", null, String.class).immutable().build();
17  
18     public static AttributeSet attributeDefinitionSet() {
19        return new AttributeSet(GlobalStatePersistenceConfiguration.class, ENABLED, LOCATION);
20     }
21  
22     private final AttributeSet attributes;
23     private final Attribute<Boolean> enabled;
24     private Attribute<String> location;
25  
26     public GlobalStatePersistenceConfiguration(AttributeSet attributes) {
27        this.attributes = attributes.checkProtection();
28        this.enabled = attributes.attribute(ENABLED);
29        this.location = attributes.attribute(LOCATION);
30     }
31  
32     public boolean enabled() {
33        return enabled.get();
34     }
35  
36     public String location() {
37        return location.get();
38     }
39  
40     @Override
41     public String toString() {
42        return "GlobalStatePersistenceConfiguration [attributes=" + attributes + "]";
43     }
44  }
