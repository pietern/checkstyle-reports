1   package org.infinispan.stream.impl.termop;
2   
3   import org.infinispan.stream.impl.TerminalOperation;
4   import org.infinispan.stream.impl.intops.IntermediateOperation;
5   
6   import java.util.concurrent.atomic.AtomicBoolean;
7   import java.util.function.Function;
8   import java.util.function.Supplier;
9   import java.util.stream.BaseStream;
10  
11  /**
12   * Runs the provided function once only and returns the result.  This is useful for operations that can be performed
13   * and its result is still valid even when a segment is lost.
14   * @param <E> type of the output of the function
15   * @param <R> type of the stream entries
16   * @param <S> type of the stream itself
17   */
18  public class SingleRunOperation<E, R, S extends BaseStream<R, S>> extends BaseTerminalOperation
19          implements TerminalOperation<E> {
20     private final Function<S, ? extends E> function;
21     private transient AtomicBoolean complete;
22  
23     public SingleRunOperation(Iterable<IntermediateOperation> intermediateOperations,
24                               Supplier<? extends BaseStream<?, ?>> supplier, Function<S, ? extends E> function) {
25        super(intermediateOperations, supplier);
26        this.function = function;
27        this.complete = new AtomicBoolean();
28     }
29  
30     @Override
31     public boolean lostSegment(boolean stopIfLost) {
32        return !complete.get();
33     }
34  
35     @Override
36     public E performOperation() {
37        BaseStream<?, ?> stream = supplier.get();
38        for (IntermediateOperation intOp : intermediateOperations) {
39           stream = intOp.perform(stream);
40        }
41        E value = function.apply((S) stream);
42        complete.set(true);
43        return value;
44     }
45  
46     public Function<S, ? extends E> getFunction() {
47        return function;
48     }
49  }
