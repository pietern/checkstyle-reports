3   /**
4    * It checks or waits until the required topology is installed.
5    *
6    * @author Pedro Ruivo
7    * @since 8.0
8    */
9   public enum TopologyMode {
10     WAIT_TOPOLOGY,
11     READY_TOPOLOGY,
12     WAIT_TX_DATA,
13     READY_TX_DATA;
14  
15     public static TopologyMode create(boolean onExecutor, boolean txData) {
16        if (onExecutor && txData) {
17           return TopologyMode.READY_TX_DATA;
18        } else if (onExecutor) {
19           return TopologyMode.READY_TOPOLOGY;
20        } else if (txData) {
21           return TopologyMode.WAIT_TX_DATA;
22        } else {
23           return TopologyMode.WAIT_TOPOLOGY;
24        }
25     }
