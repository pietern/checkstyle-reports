43        Cache<MagicKey, String> cache0 = cache(0);
44        cache0.put(new MagicKey(cache(1), cache(2)), "not-local");
45        cache0.put(new MagicKey(cache(0), cache(1)), "local");
46  
47        splitCluster(new int[]{0, 1}, new int[]{2, 3});
48        partition(0).assertDegradedMode();
49  
50        try (CloseableIterator<Map.Entry<MagicKey, String>> iterator = Closeables.iterator(cache0.getAdvancedCache()
51                .withFlags(Flag.CACHE_MODE_LOCAL).entrySet().stream())) {
52           assertNotNull(iterator.next());
53           assertNotNull(iterator.next());
54           assertFalse(iterator.hasNext());
