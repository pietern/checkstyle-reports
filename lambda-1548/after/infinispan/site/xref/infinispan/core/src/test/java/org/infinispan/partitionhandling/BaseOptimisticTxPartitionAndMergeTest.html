20  /**
21   * It tests multiple scenarios where a split can happen during a transaction.
22   *
23   * @author Pedro Ruivo
24   * @since 8.0
25   */
26  public abstract class BaseOptimisticTxPartitionAndMergeTest extends BaseTxPartitionAndMergeTest {
27  
28     protected static final String OPTIMISTIC_TX_CACHE_NAME = "opt-cache";
29  
30     @Override
31     protected void createCacheManagers() throws Throwable {
32        super.createCacheManagers();
33        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC);
34        builder.clustering().partitionHandling().enabled(true);
35        builder.transaction().lockingMode(LockingMode.OPTIMISTIC).transactionMode(TransactionMode.TRANSACTIONAL).transactionManagerLookup(new DummyTransactionManagerLookup());
36        defineConfigurationOnAllManagers(OPTIMISTIC_TX_CACHE_NAME, builder);
37     }
38  
39     protected abstract void checkLocksDuringPartition(SplitMode splitMode, KeyInfo keyInfo, boolean discard);
40  
41     protected abstract boolean forceRollback();
42  
43     protected abstract Class<? extends TransactionBoundaryCommand> getCommandClass();
44  
45     protected void doTest(final SplitMode splitMode, boolean txFail, boolean discard) throws Exception {
46        waitForClusterToForm(OPTIMISTIC_TX_CACHE_NAME);
47  
48        final KeyInfo keyInfo = createKeys(OPTIMISTIC_TX_CACHE_NAME);
49        final Cache<Object, String> originator = cache(0, OPTIMISTIC_TX_CACHE_NAME);
50        final FilterCollection filterCollection = createFilters(OPTIMISTIC_TX_CACHE_NAME, discard, getCommandClass(), splitMode);
51  
52        Future<Void> put = fork(() -> {
53           final DummyTransactionManager transactionManager = (DummyTransactionManager) originator.getAdvancedCache().getTransactionManager();
54           transactionManager.begin();
55           keyInfo.putFinalValue(originator);
56           final DummyTransaction transaction = transactionManager.getTransaction();
57           transaction.runPrepare();
58           transaction.runCommit(forceRollback());
59           transaction.throwRollbackExceptionIfAny();
60           return null;
61        });
62  
63        filterCollection.await(30, TimeUnit.SECONDS);
64        splitMode.split(this);
65        filterCollection.unblock();
66  
67        try {
68           put.get();
69           assertFalse(txFail);
70        } catch (ExecutionException e) {
71           assertTrue(txFail);
72        }
73  
74        checkLocksDuringPartition(splitMode, keyInfo, discard);
75  
76        mergeCluster();
77        finalAsserts(OPTIMISTIC_TX_CACHE_NAME, keyInfo, txFail ? INITIAL_VALUE : FINAL_VALUE);
78     }
79  }
