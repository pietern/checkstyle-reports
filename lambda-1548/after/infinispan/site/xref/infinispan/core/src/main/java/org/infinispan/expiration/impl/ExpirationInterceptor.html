1   package org.infinispan.expiration.impl;
2   
3   import org.infinispan.commands.write.ClearCommand;
4   import org.infinispan.commands.write.DataWriteCommand;
5   import org.infinispan.commands.write.PutKeyValueCommand;
6   import org.infinispan.commands.write.PutMapCommand;
7   import org.infinispan.commands.write.RemoveCommand;
8   import org.infinispan.context.InvocationContext;
9   import org.infinispan.expiration.ExpirationManager;
10  import org.infinispan.factories.annotations.Inject;
11  import org.infinispan.interceptors.base.CommandInterceptor;
12  
13  import java.util.Set;
14  
15  /**
16   * Interceptor used to ensure that when a writie occurs we don't expire the entry at the same time.  This can be both
17   * costly by going remote an extra time or even removing from the cache store when not required as it will be
18   * overwritten.
19   */
20  public class ExpirationInterceptor<K, V> extends CommandInterceptor {
21     protected ExpirationManager<K, V> expirationManager;
22  
23     @Inject
24     public void inject(ExpirationManager<K, V> expirationManager) {
25        this.expirationManager = expirationManager;
26     }
27  
28     @Override
29     public Object visitRemoveCommand(InvocationContext ctx, RemoveCommand command) throws Throwable {
30        return handleWriteCommand(ctx, command);
31     }
32  
33     @Override
34     public Object visitPutKeyValueCommand(InvocationContext ctx, PutKeyValueCommand command) throws Throwable {
35        return handleWriteCommand(ctx, command);
36     }
37  
38     @Override
39     public Object visitClearCommand(InvocationContext ctx, ClearCommand command) throws Throwable {
40        // TODO: do we want to stop all expirations?
41        return super.visitClearCommand(ctx, command);
42     }
43  
44     @Override
45     public Object visitPutMapCommand(InvocationContext ctx, PutMapCommand command) throws Throwable {
46        Set<K> keys = (Set<K>) command.getAffectedKeys();
47        keys.forEach(expirationManager::registerWriteIncoming);
48        try {
49           return super.visitPutMapCommand(ctx, command);
50        } finally {
51           // TODO: what about for transactions?
52           keys.forEach(expirationManager::unregisterWrite);
53        }
54     }
55  
56     private Object handleWriteCommand(InvocationContext ctx, DataWriteCommand command) throws Throwable {
57        K key = (K) command.getKey();
58        expirationManager.registerWriteIncoming(key);
59        try {
60           return handleDefault(ctx, command);
61        } finally {
62           // TODO: what about for transactions?
63           expirationManager.unregisterWrite(key);
64        }
65     }
66  }
