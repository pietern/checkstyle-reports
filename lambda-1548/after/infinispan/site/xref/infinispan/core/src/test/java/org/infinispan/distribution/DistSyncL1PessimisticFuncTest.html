44     public void testWriteLockBlockingForceWriteL1Update() throws Exception {
45        final String key = "some-key";
46        String value = "some-value";
47        final String otherValue = "some-new-value";
48  
49        final Cache<Object, String> nonOwner = getFirstNonOwner(key);
50        final Cache<Object, String> owner = getFirstOwner(key);
51  
52        owner.put(key, value);
53        // Get put in L1
54        nonOwner.get(key);
55  
56        assertIsInL1(nonOwner, key);
57  
58        try {
59           // Owner now does a write
60           TransactionManager ownerManger = TestingUtil.getTransactionManager(owner);
61           ownerManger.begin();
62           // This should lock the key
63           owner.put(key, otherValue);
64  
65           // Now non owner tries to lock the key, but should get blocked
66           Future<String> futureGet = fork(new Callable<String>() {
67  
68              @Override
69              public String call() throws Exception {
70                 TransactionManager mgr = TestingUtil.getTransactionManager(nonOwner);
71                 mgr.begin();
72                 try {
73                    return nonOwner.getAdvancedCache().withFlags(Flag.FORCE_WRITE_LOCK).get(key);
74                 } finally {
75                    mgr.commit();
76                 }
77              }
78           });
79  
80           // Get should not be able to complete
81           try {
82              futureGet.get(1, TimeUnit.SECONDS);
83              fail("Get command should have blocked waiting");
84           } catch (TimeoutException e) {
85  
86           }
87  
88           ownerManger.commit();
89  
90           assertEquals(otherValue, futureGet.get(1, TimeUnit.SECONDS));
91  
92           assertIsInL1(nonOwner, key);
93        } finally {
94           nonOwner.getAdvancedCache().removeInterceptor(BlockingInterceptor.class);
95        }
96     }

98     public void testForceWriteLockWithL1Invalidation() throws Exception {
99        final String key = "some-key";
100       String value = "some-value";
101       final String otherValue = "some-new-value";
102 
103       final Cache<Object, String> nonOwner = getFirstNonOwner(key);
104       final Cache<Object, String> owner = getFirstOwner(key);
105 
106       owner.put(key, value);
107       // Get put in L1
108       nonOwner.get(key);
109 
110       assertIsInL1(nonOwner, key);
111 
112       try {
113          // Owner now does a write
114          TransactionManager ownerManger = TestingUtil.getTransactionManager(owner);
115          ownerManger.begin();
116          // This should lock the key
117          assertEquals(value, nonOwner.getAdvancedCache().withFlags(Flag.FORCE_WRITE_LOCK).get(key));
118 
119          // Now non owner tries to lock the key, but should get blocked
120          Future<String> futurePut = fork(new Callable<String>() {
121 
122             @Override
123             public String call() throws Exception {
124                TransactionManager mgr = TestingUtil.getTransactionManager(nonOwner);
125                mgr.begin();
126                try {
127                   return owner.put(key, otherValue);
128                } finally {
129                   mgr.commit();
130                }
131             }
132          });
133 
134          // Put should not be able to complete
135          try {
136             futurePut.get(1, TimeUnit.SECONDS);
137             fail("Get command should have blocked waiting");
138          } catch (TimeoutException e) {
139 
140          }
141 
142          ownerManger.commit();
143 
144          assertEquals(value, futurePut.get(1, TimeUnit.SECONDS));
145 
146          eventually(new Condition() {
147             @Override
148             public boolean isSatisfied() throws Exception {
149                // Value should be removed from L1 eventually
150                return !isInL1(nonOwner, key);
151             }
152          });
153 
154          assertIsNotInL1(nonOwner, key);
155       } finally {
156          nonOwner.getAdvancedCache().removeInterceptor(BlockingInterceptor.class);
157       }
158    }
