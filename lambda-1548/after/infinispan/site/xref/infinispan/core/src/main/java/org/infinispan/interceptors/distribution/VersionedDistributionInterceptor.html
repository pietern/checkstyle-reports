17  /**
18   * A version of the {@link TxDistributionInterceptor} that adds logic to handling prepares when entries are versioned.
19   *
20   * @author Manik Surtani
21   * @since 5.1
22   */
23  public class VersionedDistributionInterceptor extends TxDistributionInterceptor {
24  
25     private static final Log log = LogFactory.getLog(VersionedDistributionInterceptor.class);
26  
27     @Override
28     protected Log getLog() {
29        return log;
30     }
31  
32     @Override
33     protected void prepareOnAffectedNodes(TxInvocationContext<?> ctx, PrepareCommand command, Collection<Address> recipients) {
34        // Perform the RPC
35        try {
36           Map<Address, Response> resps = rpcManager.invokeRemotely(recipients, command, createPrepareRpcOptions());
37           checkTxCommandResponses(resps, command, (LocalTxInvocationContext) ctx, recipients);
38  
39           // Now store newly generated versions from lock owners for use during the commit phase.
40           CacheTransaction ct = ctx.getCacheTransaction();
41           for (Response r : resps.values()) readVersionsFromResponse(r, ct);
42        } finally {
43           transactionRemotelyPrepared(ctx);
44        }
45     }
46  }
