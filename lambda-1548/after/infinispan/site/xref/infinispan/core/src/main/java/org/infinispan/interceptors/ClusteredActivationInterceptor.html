1   package org.infinispan.interceptors;
2   
3   import org.infinispan.commands.write.WriteCommand;
4   import org.infinispan.context.Flag;
5   import org.infinispan.context.InvocationContext;
6   import org.infinispan.factories.annotations.Inject;
7   import org.infinispan.factories.annotations.Start;
8   import org.infinispan.interceptors.locking.ClusteringDependentLogic;
9   import org.infinispan.statetransfer.StateTransferManager;
10  import org.infinispan.util.logging.Log;
11  import org.infinispan.util.logging.LogFactory;
12  
13  /**
14   * The same as a regular cache loader interceptor, except that it contains additional logic to force loading from the
15   * cache loader if needed on a remote node, in certain conditions.
16   *
17   * @author Manik Surtani
18   * @since 5.1
19   */
20  public class ClusteredActivationInterceptor extends ActivationInterceptor {
21  
22     private static final Log log = LogFactory.getLog(ClusteredActivationInterceptor.class);
23     private static final boolean trace = log.isTraceEnabled();
24  
25     private boolean transactional;
26     private ClusteringDependentLogic cdl;
27     private StateTransferManager stateTransferManager;
28     private boolean distributed;
29  
30     @Inject
31     private void injectDependencies(ClusteringDependentLogic cdl, StateTransferManager stateTransferManager) {
32        this.cdl = cdl;
33        this.stateTransferManager = stateTransferManager;
34     }
35  
36     @Start(priority = 15)
37     private void startClusteredActivationInterceptor() {
38        transactional = cacheConfiguration.transaction().transactionMode().isTransactional();
39        distributed = cacheConfiguration.clustering().cacheMode().isDistributed();
40     }
41  
42     @Override
43     protected boolean skipLoadForWriteCommand(WriteCommand cmd, Object key, InvocationContext ctx) {
44        if (!cmd.alwaysReadsExistingValues()) {
45           if (transactional) {
46              if (!ctx.isOriginLocal()) {
47                 if (trace) log.tracef("Skip load for remote tx write command %s.", cmd);
48                 return true;
49              }
50           } else {
51              if (!cdl.localNodeIsPrimaryOwner(key) && !cmd.hasFlag(Flag.CACHE_MODE_LOCAL)) {
52                 if (trace) {
53                    log.tracef("Skip load for command %s. This node is not the primary owner of %s", cmd, key);
54                 }
55                 return true;
56              }
57           }
58        }
59        return super.skipLoadForWriteCommand(cmd, key, ctx);
60     }
61  
62     @Override
63     protected boolean canLoad(Object key) {
64        // Don't load the value if we are using distributed mode and aren't in the read CH
65        return stateTransferManager.isJoinComplete() && (!distributed || isKeyLocal(key));
66     }
67  
68     private boolean isKeyLocal(Object key) {
69        return stateTransferManager.getCacheTopology().getReadConsistentHash().isKeyLocalToNode(cdl.getAddress(), key);
70     }
71  }
