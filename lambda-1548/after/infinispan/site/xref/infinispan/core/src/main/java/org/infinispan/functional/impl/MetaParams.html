21  /**
22   * Represents a {@link MetaParam} collection.
23   *
24   * <p>In {@link Params}, the internal array where each parameter was
25   * stored is indexed by an integer. This worked fine because the available
26   * parameters are exclusively controlled by the Infinispan. This is not the
27   * case with {@link MetaParam} instances where users are expected to add their
28   * own types. So, for {@link MetaParams}, an array is still used but the lookup
29   * is done sequentially comparing the type of the {@link MetaParam} looked for
30   * against the each individual {@link MetaParam} instance stored in
31   * {@link MetaParams}.
32   *
33   * <p>Having sequential {@link MetaParam} lookups over an array is O(n),
34   * but this is not problematic since the number of {@link MetaParam} to be
35   * stored with each cached entry is expected to be small, less than 10 per
36   * {@link MetaParams} collection. So, the performance impact is quite small.
37   *
38   * <p>Storing {@link MetaParam} instances in an array adds the least
39   * amount of overhead to keeping a collection of {@link MetaParam} in memory
40   * along with each cached entry, while retaining flexibility to add or remove
41   * {@link MetaParam} instances.
42   *
43   * <p>This {@link MetaParams} collection is not thread safe because
44   * it is expected that any updates will be done having acquired write locks
45   * on the entire {@link org.infinispan.container.entries.CacheEntry} which
46   * references the {@link MetaParams} collection. Hence, any updates could be
47   * done without the need to keep {@link MetaParams} concurrently safe.
48   * Also, although users can retrieve or update individual {@link MetaParam}
49   * instances, they cannot act on the globally at the {@link MetaParams} level,
50   * and hence there is no risk of users misusing {@link MetaParams}.
51   *
52   * @since 8.0
53   */
54  @NotThreadSafe
55  @Experimental
56  public final class MetaParams {
57  
58     private MetaParam<?>[] metas;
59  
60     private MetaParams(MetaParam<?>[] metas) {
61        this.metas = metas;
62     }
63  
64     public boolean isEmpty() {
65        return metas.length == 0;
66     }
67  
68     public int size() {
69        return metas.length;
70     }
71  
72     public <T> Optional<T> find(Class<T> type) {
73        return Optional.ofNullable(findNullable(type));
74     }
75  
76     @SuppressWarnings("unchecked")
77     private <T> T findNullable(Class<T> type) {
78        for (MetaParam<?> meta : metas) {
79           if (meta.getClass().isAssignableFrom(type))
80              return (T) meta;
81        }
82  
83        return null;
84     }
85  
86     public void add(MetaParam.Writable meta) {
87        if (metas.length == 0)
88           metas = new MetaParam[]{meta};
89        else {
90           boolean found = false;
91           for (int i = 0; i < metas.length; i++) {
92              if (metas[i].getClass().isAssignableFrom(meta.getClass())) {
93                 metas[i] = meta;
94                 found = true;
95              }
96           }
97  
98           if (!found) {
99              MetaParam<?>[] newMetas = Arrays.copyOf(metas, metas.length + 1);
100             newMetas[newMetas.length - 1] = meta;
101             metas = newMetas;
102          }
103       }
104    }
105 
106    public void addMany(MetaParam.Writable... metaParams) {
107       if (metas.length == 0) metas = metaParams;
108       else {
109          List<MetaParam<?>> notFound = new ArrayList<>(metaParams.length);
110          for (MetaParam.Writable newMeta : metaParams) {
111             boolean found = false;
112             for (int i = 0; i < metas.length; i++) {
113                if (metas[i].getClass().isAssignableFrom(newMeta.getClass())) {
114                   metas[i] = newMeta;
115                   found = true;
116                }
117             }
118             if (!found)
119                notFound.add(newMeta);
120          }
121 
122          if (!notFound.isEmpty()) {
123             List<MetaParam<?>> allMetasList = new ArrayList<>(Arrays.asList(metas));
124             allMetasList.addAll(notFound);
125             metas = allMetasList.toArray(new MetaParam[metas.length + notFound.size()]);
126          }
127       }
128    }
129 
130    @Override
131    public String toString() {
132       return "MetaParams{" +
133          "metas=" + Arrays.toString(metas) +
134          '}';
135    }
136 
137    /**
138     * Construct a collection of {@link MetaParam} instances. If multiple
139     * instances of the same {@link MetaParam} are present, the last value is
140     * only considered since there can only be one instance per type in the
141     * {@link MetaParams} collection.
142     *
143     * @param metas Meta parameters to create the collection with
144     * @return a collection of meta parameters without type duplicates
145     */
146    static MetaParams of(MetaParam... metas) {
147       return new MetaParams(filterDuplicates(metas));
148    }
149 
150    static MetaParams of(MetaParam meta) {
151       return new MetaParams(new MetaParam[]{meta});
152    }
153 
154    static MetaParams empty() {
155       return new MetaParams(new MetaParam[]{});
156    }
157 
158    private static MetaParam[] filterDuplicates(MetaParam... metas) {
159       Map<Class<?>, MetaParam<?>> all = new HashMap<>();
160       for (MetaParam meta : metas)
161          all.put(meta.getClass(), meta);
162 
163       return all.values().toArray(new MetaParam[all.size()]);
164    }
165 
166    public static final class Externalizer extends AbstractExternalizer<MetaParams> {
167       @Override
168       public void writeObject(ObjectOutput oo, MetaParams o) throws IOException {
169          oo.writeInt(o.metas.length);
170          for (Object meta : o.metas) oo.writeObject(meta);
171       }
172 
173       @Override
174       public MetaParams readObject(ObjectInput input) throws IOException, ClassNotFoundException {
175          int length = input.readInt();
176          MetaParam[] metas = new MetaParam[length];
177          for (int i = 0; i < length; i++)
178             metas[i] = (MetaParam) input.readObject();
179 
180          return MetaParams.of(metas);
181       }
182 
183       @Override
184       public Set<Class<? extends MetaParams>> getTypeClasses() {
185          return Util.<Class<? extends MetaParams>>asSet(MetaParams.class);
186       }
187 
188       @Override
189       public Integer getId() {
190          return Ids.META_PARAMS;
191       }
192    }
