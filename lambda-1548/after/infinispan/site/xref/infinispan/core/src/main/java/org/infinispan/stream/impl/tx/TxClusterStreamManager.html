20  /**
21   * This is a delegating cluster stream manager that sends all calls to the underlying cluster stream manager.  However
22   * in the case of performing an operation it adds all entries that are in the provided tx context to the map of
23   * keys to exclude so those values are not processed in the remote nodes.
24   * @param <K> the key type
25   */
26  public class TxClusterStreamManager<K> implements ClusterStreamManager<K> {
27     private final ClusterStreamManager<K> manager;
28     private final LocalTxInvocationContext ctx;
29     private final ConsistentHash hash;
30  
31     public TxClusterStreamManager(ClusterStreamManager<K> manager, LocalTxInvocationContext ctx, ConsistentHash hash) {
32        this.manager = manager;
33        this.ctx = ctx;
34        this.hash = hash;
35     }
36  
37     @Override
38     public <R> UUID remoteStreamOperation(boolean parallelDistribution, boolean parallelStream, ConsistentHash ch,
39             Set<Integer> segments, Set<K> keysToInclude, Map<Integer, Set<K>> keysToExclude, boolean includeLoader,
40             TerminalOperation<R> operation, ResultsCallback<R> callback, Predicate<? super R> earlyTerminatePredicate) {
41        TxExcludedKeys<K> txExcludedKeys = new TxExcludedKeys<>(keysToExclude, ctx, hash);
42        return manager.remoteStreamOperation(parallelDistribution, parallelStream, ch, segments, keysToInclude,
43                txExcludedKeys, includeLoader, operation, callback, earlyTerminatePredicate);
44     }
45  
46     @Override
47     public <R> UUID remoteStreamOperationRehashAware(boolean parallelDistribution, boolean parallelStream,
48             ConsistentHash ch, Set<Integer> segments, Set<K> keysToInclude, Map<Integer, Set<K>> keysToExclude,
49             boolean includeLoader, TerminalOperation<R> operation, ResultsCallback<R> callback,
50             Predicate<? super R> earlyTerminatePredicate) {

52        return manager.remoteStreamOperationRehashAware(parallelDistribution, parallelStream, ch, segments, keysToInclude,
53                txExcludedKeys, includeLoader, operation, callback, earlyTerminatePredicate);
54     }

94     private static class TxExcludedKeys<K> extends AbstractDelegatingMap<Integer, Set<K>> {
95        private final Map<Integer, Set<K>> map;
96        private final Map<Integer, Set<K>> ctxMap;
97  
98        private TxExcludedKeys(Map<Integer, Set<K>> map, LocalTxInvocationContext ctx, ConsistentHash hash) {
99           this.map = map;
100          this.ctxMap = contextToMap(ctx, hash);
101       }
102 
103       Map<Integer, Set<K>> contextToMap(LocalTxInvocationContext ctx, ConsistentHash hash) {
104          Map<Integer, Set<K>> contextMap = new HashMap<>();
105          ctx.getLookedUpEntries().forEach((k, v) -> {
106             Integer segment = hash.getSegment(k);
107             Set<K> innerSet = contextMap.get(segment);
108             if (innerSet == null) {
109                innerSet = new HashSet<K>();
110                contextMap.put(segment, innerSet);
111             }
112             innerSet.add((K) k);
113          });
114          return contextMap;
115       }
116 
117       @Override
118       protected Map<Integer, Set<K>> delegate() {
119          return map;
120       }
121 
122       @Override
123       public Set<K> get(Object key) {
124          if (!(key instanceof Integer)) {
125             return null;
126          }
127          Set<K> ctxSet = ctxMap.get(key);
128          Set<K> excludedSet = super.get(key);
129          if (ctxSet != null) {
130             if (excludedSet != null) {
131                ctxSet.addAll(excludedSet);
132             }
133             return ctxSet;
134          }
135          return excludedSet;
136       }
137 
138       @Override
139       public boolean isEmpty() {
140          return ctxMap.isEmpty() && super.isEmpty();
141       }
142    }
