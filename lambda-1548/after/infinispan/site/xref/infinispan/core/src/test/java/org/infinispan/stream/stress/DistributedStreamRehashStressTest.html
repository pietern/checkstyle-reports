64        builderUsed.clustering().stateTransfer().timeout(240, TimeUnit.SECONDS);
65        createClusteredCaches(CACHE_COUNT, CACHE_NAME, builderUsed);

68     protected EmbeddedCacheManager addClusterEnabledCacheManager(TransportFlags flags) {
69        GlobalConfigurationBuilder gcb = GlobalConfigurationBuilder.defaultClusteredBuilder();
70        // Amend first so we can increase the transport thread pool
71        TestCacheManagerFactory.amendGlobalConfiguration(gcb, flags);
72        // we need to increase the transport and remote thread pools to default values
73        BlockingThreadPoolExecutorFactory executorFactory = new BlockingThreadPoolExecutorFactory(
74              25, 25, 10000, 30000);
75        gcb.transport().transportThreadPool().threadPoolFactory(executorFactory);
76  
77        gcb.transport().remoteCommandThreadPool().threadPoolFactory(executorFactory);
78  
79        EmbeddedCacheManager cm = TestCacheManagerFactory.newDefaultCacheManager(true, gcb, new ConfigurationBuilder(),
80                false);
81        cacheManagers.add(cm);
82        return cm;
83     }

85     public void testStressNodesLeavingWhileMultipleCollectors() throws InterruptedException, ExecutionException,
86             TimeoutException {
87        testStressNodesLeavingWhilePerformingCallable((masterValues, cache, iteration) -> {
88           Map<Integer, Integer> results = cache.entrySet().stream().filter(
89                   (Serializable & Predicate<Map.Entry<Integer, Integer>>)
90                           e -> (e.getKey().intValue() & 1) == 1).collect(
91                   CacheCollectors.serializableCollector(() -> Collectors.toMap(e -> e.getKey(), e -> e.getValue())));
92           assertEquals(CACHE_ENTRY_COUNT / 2, results.size());
93           for (Map.Entry<Integer, Integer> entry : results.entrySet()) {
94              assertEquals(entry.getKey(), entry.getValue());
95              assertTrue((entry.getKey() & 1) == 1, "Mismatched value was " + entry.getKey());
96           }
97        });

109    public void testStressNodesLeavingWhileMultipleIterators() throws InterruptedException, ExecutionException,
110            TimeoutException {
111       testStressNodesLeavingWhilePerformingCallable((masterValues, cache, iteration) -> {
112          Map<Integer, Integer> seenValues = new HashMap<>();
113          Iterator<Map.Entry<Integer, Integer>> iterator = cache.entrySet().stream()
114                  .distributedBatchSize(50000)
115                  .iterator();
116          while (iterator.hasNext()) {
117             Map.Entry<Integer, Integer> entry = iterator.next();
118             if (seenValues.containsKey(entry.getKey())) {
119                log.tracef("Seen values were: %s", seenValues);
120                throw new IllegalArgumentException(Thread.currentThread() + "-Found duplicate value: " + entry.getKey() + " on iteration " + iteration);
121             } else if (!masterValues.get(entry.getKey()).equals(entry.getValue())) {
122                log.tracef("Seen values were: %s", seenValues);
123                throw new IllegalArgumentException(Thread.currentThread() + "-Found incorrect value: " + entry.getKey() + " with value " + entry.getValue() + " on iteration " + iteration);
124             }
125             seenValues.put(entry.getKey(), entry.getValue());
126          }
127          if (seenValues.size() != masterValues.size()) {
128             Map<Integer, Set<Map.Entry<Integer, Integer>>> target = generateEntriesPerSegment(cache.getAdvancedCache().getDistributionManager().getConsistentHash(), masterValues.entrySet());
129             Map<Integer, Set<Map.Entry<Integer, Integer>>> actual = generateEntriesPerSegment(cache.getAdvancedCache().getDistributionManager().getConsistentHash(), seenValues.entrySet());
130             for (Map.Entry<Integer, Set<Map.Entry<Integer, Integer>>> entry : target.entrySet()) {
131                Set<Map.Entry<Integer, Integer>> entrySet = entry.getValue();
132                Set<Map.Entry<Integer, Integer>> actualEntries = actual.get(entry.getKey());
133                if (actualEntries != null) {
134                   entrySet.removeAll(actualEntries);
135                }
136                if (!entrySet.isEmpty()) {
137                   throw new IllegalArgumentException(Thread.currentThread() + "-Found incorrect amount " +
138                           (actualEntries != null ? actualEntries.size() : 0) + " of entries, expected " +
139                           entrySet.size() + " for segment " + entry.getKey() + " missing entries " + entrySet
140                           + " on iteration " + iteration);
141                }
142             }
143          }
144       });
145    }

147    void testStressNodesLeavingWhilePerformingCallable(final PerformOperation operation)
148            throws InterruptedException, ExecutionException, TimeoutException {
149       final Map<Integer, Integer> masterValues = new HashMap<Integer, Integer>();
150       // First populate our caches
151       for (int i = 0; i < CACHE_ENTRY_COUNT; ++i) {
152          masterValues.put(i, i);
153       }
154 
155       cache(0, CACHE_NAME).putAll(masterValues);
156 
157       System.out.println("Done with inserts!");
158 
159       final AtomicBoolean complete = new AtomicBoolean(false);
160       final Exchanger<Throwable> exchanger = new Exchanger<>();
161       // Now we spawn off CACHE_COUNT of threads.  All but one will constantly calling to iterator while another
162       // will constantly be killing and adding new caches
163       Future<Void>[] futures = new Future[(CACHE_COUNT - 1) * THREAD_MULTIPLIER + 1];
164       for (int j = 0; j < THREAD_MULTIPLIER; ++j) {
165       // We iterate over all but the last cache since we kill it constantly
166       for (int i = 0; i < CACHE_COUNT - 1; ++i) {
167          final Cache<Integer, Integer> cache = cache(i, CACHE_NAME);
168          futures[i + j * (CACHE_COUNT -1)] = fork(() -> {
169             try {
170                int iteration = 0;
171                while (!complete.get()) {
172                   log.warnf("Starting iteration %s", iteration++);
173                   operation.perform(masterValues, cache, iteration);
174                }
175                return null;
176             } catch (Throwable e) {
177                log.fatal("Exception encountered:", e);
178                // Stop all the others as well
179                complete.set(true);
180                exchanger.exchange(e);
181                throw e;
182             }
183          });
184       }
185       }
186 
187       // Then spawn a thread that just constantly kills the last cache and recreates over and over again
188       futures[futures.length - 1] = fork(() -> {
189          TestResourceTracker.testThreadStarted(DistributedStreamRehashStressTest.this);
190          try {
191             Cache<?, ?> cacheToKill = cache(CACHE_COUNT - 1);
192             while (!complete.get()) {
193                Thread.sleep(1000);
194                if (cacheManagers.remove(cacheToKill.getCacheManager())) {
195                   log.fatal("Killing cache to force rehash");
196                   cacheToKill.getCacheManager().stop();
197                   List<Cache<Object, Object>> caches = caches(CACHE_NAME);
198                   if (caches.size() > 0) {
199                      TestingUtil.blockUntilViewsReceived(60000, false, caches);
200                      TestingUtil.waitForRehashToComplete(caches);
201                   }
202                } else {
203                   throw new IllegalStateException("Cache Manager " + cacheToKill.getCacheManager() +
204                                                         " wasn't found for some reason!");
205                }
206 
207                log.trace("Adding new cache again to force rehash");
208                // We should only create one so just make it the next cache manager to kill
209                cacheToKill = createClusteredCaches(1, CACHE_NAME, builderUsed).get(0);
210                log.trace("Added new cache again to force rehash");
211             }
212             return null;
213          } catch (Exception e) {
214             // Stop all the others as well
215             complete.set(true);
216             exchanger.exchange(e);
217             throw e;
218          }
219       });
220 
221       try {
222          // If this returns means we had an issue
223          Throwable e = exchanger.exchange(null, 5, TimeUnit.MINUTES);
224          fail("Found an exception in at least 1 thread", e);
225       } catch (TimeoutException e) {
226 
227       }
228 
229       complete.set(true);
230 
231       // Make sure they all finish properly
232       for (int i = 0; i < futures.length; ++i) {
233          try {
234             futures[i].get(2, TimeUnit.MINUTES);
235          } catch (TimeoutException e) {
236             log.errorf("Future %s did not complete in time allotted.", i);
237             throw e;
238          }
239       }
240    }

246    private <K, V> Map<Integer, Set<Map.Entry<K, V>>> generateEntriesPerSegment(ConsistentHash hash, Iterable<Map.Entry<K, V>> entries) {
247       Map<Integer, Set<Map.Entry<K, V>>> returnMap = new HashMap<Integer, Set<Map.Entry<K, V>>>();
248 
249       for (Map.Entry<K, V> value : entries) {
250          int segment = hash.getSegment(value.getKey());
251          Set<Map.Entry<K, V>> set = returnMap.get(segment);
252          if (set == null) {
253             set = new HashSet<Map.Entry<K, V>>();
254             returnMap.put(segment, set);
255          }
256          set.add(new ImmortalCacheEntry(value.getKey(), value.getValue()));
257       }
258       return returnMap;
259    }
