32  /**
33   * A {@link org.infinispan.remoting.inboundhandler.PerCacheInboundInvocationHandler} implementation for non-total order
34   * caches.
35   *
36   * @author Pedro Ruivo
37   * @since 7.1
38   */
39  public class NonTotalOrderPerCacheInboundInvocationHandler extends BasePerCacheInboundInvocationHandler implements LockListener {
40  
41     private static final Log log = LogFactory.getLog(NonTotalOrderPerCacheInboundInvocationHandler.class);
42     private static final boolean trace = log.isTraceEnabled();
43  
44     private final CheckTopologyAction checkTopologyAction;
45  
46     private LockManager lockManager;
47     private ClusteringDependentLogic clusteringDependentLogic;
48     private Configuration configuration;
49  
50     public NonTotalOrderPerCacheInboundInvocationHandler() {
51        checkTopologyAction = new CheckTopologyAction(this);
52     }
53  
54     @Inject
55     public void inject(LockManager lockManager, ClusteringDependentLogic clusteringDependentLogic, Configuration configuration) {
56        this.lockManager = lockManager;
57        this.clusteringDependentLogic = clusteringDependentLogic;
58        this.configuration = configuration;
59     }
60  
61     @Override
62     public void handle(CacheRpcCommand command, Reply reply, DeliverOrder order) {
63        if (order == DeliverOrder.TOTAL) {
64           unexpectedDeliverMode(command, order);
65        }
66        try {
67           final int commandTopologyId = extractCommandTopologyId(command);
68           final boolean onExecutorService = executeOnExecutorService(order, command);
69           final BlockingRunnable runnable;
70  
71           switch (command.getCommandId()) {
72              case SingleRpcCommand.COMMAND_ID:
73                 runnable = createReadyActionRunnable(command, reply, commandTopologyId, true, onExecutorService,
74                                                      createReadyActionForSingleRpcCommand(commandTopologyId, (SingleRpcCommand) command));
75                 break;
76              case MultipleRpcCommand.COMMAND_ID:
77                 runnable = createReadyActionRunnable(command, reply, commandTopologyId, true, onExecutorService,
78                                                    createReadyActionForMultipleRpcCommand(commandTopologyId, (MultipleRpcCommand) command));
79                 break;
80              default:
81                 runnable = createDefaultRunnable(command, reply, commandTopologyId, command.getCommandId() != StateRequestCommand.COMMAND_ID, onExecutorService);
82                 break;
83           }
84           handleRunnable(runnable, onExecutorService);
85        } catch (Throwable throwable) {
86           reply.reply(exceptionHandlingCommand(command, throwable));
87        }
88     }
89  
90     @Override
91     public void onEvent(LockState state) {
92        remoteCommandsExecutor.checkForReadyTasks();
93     }

96     protected Log getLog() {
97        return log;

105    private BlockingRunnable createReadyActionRunnable(CacheRpcCommand command, Reply reply, int commandTopologyId,
106                                                               boolean waitTransactionalData, boolean onExecutorService,
107                                                               ReadyAction readyAction) {
108       final TopologyMode topologyMode = TopologyMode.create(onExecutorService, waitTransactionalData);
109       if (onExecutorService && readyAction != null) {
110          readyAction.addListener(remoteCommandsExecutor::checkForReadyTasks);
111          return new DefaultTopologyRunnable(this, command, reply, topologyMode, commandTopologyId) {
112             @Override
113             public boolean isReady() {
114                return super.isReady() && readyAction.isReady();
115             }
116          };
117       } else {
118          return new DefaultTopologyRunnable(this, command, reply, topologyMode, commandTopologyId);
119       }
120    }
121 
122    private ReadyAction createReadyAction(int topologyId, RemoteLockCommand command) {
123       if (command.hasSkipLocking()) {
124          return null;
125       }
126       Collection<Object> keys = command.getKeysToLock();
127       if (keys.isEmpty()) {
128          return null;
129       }
130       final long timeoutMillis = command.hasZeroLockAcquisition() ? 0 : configuration.locking().lockAcquisitionTimeout();
131 
132       DefaultReadyAction action = new DefaultReadyAction(new ActionState(command, topologyId, timeoutMillis),
133                                                          checkTopologyAction,
134                                                          new LockAction(lockManager, clusteringDependentLogic));
135       action.registerListener();
136       return action;
137    }
138 
139    private ReadyAction createReadyActionForSingleRpcCommand(int topologyId, SingleRpcCommand singleRpcCommand) {
140       ReplicableCommand command = singleRpcCommand.getCommand();
141       return command instanceof RemoteLockCommand ? createReadyAction(topologyId, (RemoteLockCommand) command) : null;
142    }
143 
144    private ReadyAction createReadyActionForMultipleRpcCommand(int topologyId, MultipleRpcCommand command) {
145       ReplicableCommand[] commands = command.getCommands();
146       List<ReadyAction> list = new ArrayList<>(commands.length);
147       forEach(commands, cmd -> {
148          if (cmd instanceof RemoteLockCommand) {
149             ReadyAction action = createReadyAction(topologyId, (RemoteLockCommand) cmd);
150             if (action != null) {
151                list.add(action);
152             }
153          }
154       });
155 
156       if (list.isEmpty()) {
157          return null;
158       } else if (list.size() == 1) {
159          return list.get(0);
160       }
161 
162       CompositeAction action = new CompositeAction(list);
163       action.registerListener();
164       return action;
165    }
166 
167    private static class CompositeAction implements ReadyAction, ActionListener {
168 
169       private final Collection<ReadyAction> actions;
170       private final AtomicBoolean notify;
171       private volatile ActionListener listener;
172 
173       private CompositeAction(Collection<ReadyAction> actions) {
174          this.actions = actions;
175          notify = new AtomicBoolean(false);
176       }
177 
178       public void registerListener() {
179          actions.forEach(readyAction -> readyAction.addListener(this));
180       }
181 
182       @Override
183       public boolean isReady() {
184          for (ReadyAction action : actions) {
185             if (!action.isReady()) {
186                return false;
187             }
188          }
189          return true;
190       }
191 
192       @Override
193       public void addListener(ActionListener listener) {
194          this.listener = listener;
195       }
196 
197       @Override
198       public void onComplete() {
199          ActionListener actionListener = listener;
200          if (isReady() && actionListener != null && notify.compareAndSet(false, true)) {
201             actionListener.onComplete();
202          }
203       }
204    }
