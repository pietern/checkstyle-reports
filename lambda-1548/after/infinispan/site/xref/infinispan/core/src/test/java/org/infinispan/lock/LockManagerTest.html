86     private void doSingleCounterTest(LockManager lockManager) throws ExecutionException, InterruptedException {
87        final NotThreadSafeCounter counter = new NotThreadSafeCounter();
88        final String key = "key";
89        final int numThreads = 8;
90        final int maxCounterValue = 100;
91        final ExecutorService executorService = Executors.newFixedThreadPool(numThreads);
92        final CyclicBarrier barrier = new CyclicBarrier(numThreads);
93        List<Future<Collection<Integer>>> callableResults = new ArrayList<>(numThreads);
94  
95        for (int i = 0; i < numThreads; ++i) {
96           callableResults.add(executorService.submit(() -> {
97              final Thread lockOwner = Thread.currentThread();
98              AssertJUnit.assertEquals(0, counter.getCount());
99              List<Integer> seenValues = new LinkedList<>();
100             barrier.await();
101             while (true) {
102                lockManager.lock(key, lockOwner, 1, TimeUnit.MINUTES).lock();
103                AssertJUnit.assertEquals(lockOwner, lockManager.getOwner(key));
104                AssertJUnit.assertTrue(lockManager.isLocked(key));
105                AssertJUnit.assertTrue(lockManager.ownsLock(key, lockOwner));
106                try {
107                   int value = counter.getCount();
108                   if (value == maxCounterValue) {
109                      return seenValues;
110                   }
111                   seenValues.add(value);
112                   counter.setCount(value + 1);
113                } finally {
114                   lockManager.unlock(key, lockOwner);
115                }
116             }
117          }));
118       }
119 
120       Set<Integer> seenResults = new HashSet<>();
121       try {
122          for (Future<Collection<Integer>> future : callableResults) {
123             for (Integer integer : future.get()) {
124                AssertJUnit.assertTrue(seenResults.add(integer));
125             }
126          }
127       } finally {
128          executorService.shutdown();
129          executorService.awaitTermination(30, TimeUnit.SECONDS);
130       }
131       AssertJUnit.assertEquals(maxCounterValue, seenResults.size());
132       for (int i = 0; i < maxCounterValue; ++i) {
133          AssertJUnit.assertTrue(seenResults.contains(i));
134       }
135 
136       AssertJUnit.assertEquals(0, lockManager.getNumberOfLocksHeld());
137    }

139    private void doMultipleCounterTest(LockManager lockManager) throws ExecutionException, InterruptedException {
140       final int numCounters = 8;
141       final NotThreadSafeCounter[] counters = new NotThreadSafeCounter[numCounters];
142       final String[] keys = new String[numCounters];
143       final int numThreads = 8;
144       final int maxCounterValue = 100;
145       final ExecutorService executorService = Executors.newFixedThreadPool(numThreads);
146       final CyclicBarrier barrier = new CyclicBarrier(numThreads);
147 
148       for (int i = 0; i < numCounters; ++i) {
149          counters[i] = new NotThreadSafeCounter();
150          keys[i] = "key-" + i;
151       }
152 
153       List<Future<Collection<Integer>>> callableResults = new ArrayList<>(numThreads);
154 
155       for (int i = 0; i < numThreads; ++i) {
156          final List<String> threadKeys = new ArrayList<>(Arrays.asList(keys));
157          Collections.shuffle(threadKeys);
158          callableResults.add(executorService.submit(() -> {
159             final Thread lockOwner = Thread.currentThread();
160             List<Integer> seenValues = new LinkedList<>();
161             barrier.await();
162             while (true) {
163                lockManager.lockAll(threadKeys, lockOwner, 1, TimeUnit.MINUTES).lock();
164                for (String key : threadKeys) {
165                   AssertJUnit.assertEquals(lockOwner, lockManager.getOwner(key));
166                   AssertJUnit.assertTrue(lockManager.isLocked(key));
167                   AssertJUnit.assertTrue(lockManager.ownsLock(key, lockOwner));
168                }
169                try {
170                   int value = -1;
171                   for (NotThreadSafeCounter counter : counters) {
172                      if (value == -1) {
173                         value = counter.getCount();
174                         if (value == maxCounterValue) {
175                            return seenValues;
176                         }
177                         seenValues.add(value);
178                      } else {
179                         AssertJUnit.assertEquals(value, counter.getCount());
180                      }
181                      counter.setCount(value + 1);
182                   }
183                } finally {
184                   lockManager.unlockAll(threadKeys, lockOwner);
185                }
186             }
187          }));
188       }
189 
190       Set<Integer> seenResults = new HashSet<>();
191       try {
192          for (Future<Collection<Integer>> future : callableResults) {
193             for (Integer integer : future.get()) {
194                AssertJUnit.assertTrue(seenResults.add(integer));
195             }
196          }
197       } finally {
198          executorService.shutdown();
199          executorService.awaitTermination(30, TimeUnit.SECONDS);
200       }
201       AssertJUnit.assertEquals(maxCounterValue, seenResults.size());
202       for (int i = 0; i < maxCounterValue; ++i) {
203          AssertJUnit.assertTrue(seenResults.contains(i));
204       }
205 
206       AssertJUnit.assertEquals(0, lockManager.getNumberOfLocksHeld());
207    }

209    private void doTestWithFailAcquisition(LockManager lockManager) throws InterruptedException {
210       final String lockOwner1 = "LO1";
211       final String lockOwner2 = "LO2";
212       final String key = "key";
213       final String key2 = "key2";
214       final String key3 = "key2";
215 
216       lockManager.lock(key, lockOwner1, 0, TimeUnit.MILLISECONDS).lock();
217       AssertJUnit.assertEquals(lockOwner1, lockManager.getOwner(key));
218       AssertJUnit.assertTrue(lockManager.isLocked(key));
219       AssertJUnit.assertTrue(lockManager.ownsLock(key, lockOwner1));
220 
221       try {
222          lockManager.lockAll(Arrays.asList(key, key2, key3), lockOwner2, 0, TimeUnit.MILLISECONDS).lock();
223          AssertJUnit.assertEquals(1, lockManager.getNumberOfLocksHeld());
224          AssertJUnit.fail();
225       } catch (TimeoutException t) {
226          //expected
227       }
228 
229       AssertJUnit.assertEquals(lockOwner1, lockManager.getOwner(key));
230       AssertJUnit.assertTrue(lockManager.isLocked(key));
231       AssertJUnit.assertTrue(lockManager.ownsLock(key, lockOwner1));
232 
233       LockPromise lockPromise = lockManager.lockAll(Arrays.asList(key, key2, key3), lockOwner2, 1, TimeUnit.MINUTES);
234 
235       AssertJUnit.assertFalse(lockPromise.isAvailable());
236 
237       lockManager.unlock(key, lockOwner1);
238 
239       AssertJUnit.assertTrue(lockPromise.isAvailable());
240       lockPromise.lock();
241 
242       AssertJUnit.assertEquals(lockOwner2, lockManager.getOwner(key));
243       AssertJUnit.assertTrue(lockManager.isLocked(key));
244       AssertJUnit.assertTrue(lockManager.ownsLock(key, lockOwner2));
245 
246       lockManager.unlockAll(Arrays.asList(key, key2, key3), lockOwner2);
247       AssertJUnit.assertEquals(0, lockManager.getNumberOfLocksHeld());
248    }
