14  @Test(groups = "functional", testName = "tx.locking.SizeDistTxReadCommittedTest")

16  
17     public SizeDistTxReadCommittedTest() {
18        isolation = IsolationLevel.READ_COMMITTED;
19     }
20  
21     /**
22      * Manifestation of inconsistent behaviour of READ_COMMITTED mode. More information
23      * in https://issues.jboss.org/browse/ISPN-4910 and in comments below.
24      */
25     @Override
26     public void testSizeWithPreviousRead() throws Exception {
27        preloadCacheAndCheckSize();
28  
29        tm(0).begin();
30        //make sure we read k1 in this transaction
31        assertEquals("v1", cache(0).get(k1));
32        final Transaction tx1 = tm(0).suspend();
33  
34        //another tx working on the same keys
35        tm(0).begin();
36        //remove the key that was previously read in another tx
37        cache(0).remove(k1);
38        cache(0).put(k0, "v2");
39        tm(0).commit();
40  
41        tm(0).resume(tx1);
42  
43        //We read k1 earlier and in READ_COMMITTED mode, the size() method should return the most
44        //up-to-date size. However, the current transaction does not see the parallel changes
45        //and still returns incorrect size == 2. The expected size would be 1.
46        //assertEquals(1, cache(0).size()); //This assertion would fail
47        assertEquals(2, cache(0).size());
48        assertEquals("v2", cache(0).get(k0));
49  
50        //The current transaction does not see changes made to k1 in the parallel transaction.
51        //As a result, the get() operation returns k1 instead of expected null.
52        //assertNull(cache(0).get(k1)); //This assertion would fail
53        assertEquals("v1", cache(0).get(k1));
54        tm(0).commit();
55  
56        assertNull(cache(1).get(k1));
57     }
58  
59     @Override
60     public void testSizeWithReadFromRemoteNode() throws Exception {
61        preloadCacheAndCheckSize();
62  
63        tm(0).begin();
64        assertEquals("v1", cache(0).get(k1));
65        assertEquals(2, cache(0).size());
66        tm(0).rollback();
67     }
