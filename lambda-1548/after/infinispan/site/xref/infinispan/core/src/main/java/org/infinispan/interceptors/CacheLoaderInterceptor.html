174    private Object visitDataCommand(InvocationContext ctx, AbstractDataCommand command) throws Throwable {
175       if (enabled) {
176          Object key;
177          if ((key = command.getKey()) != null) {
178             loadIfNeeded(ctx, key, command);
179          }
180       }
181       return invokeNextInterceptor(ctx, command);
182    }

184    @Override
185    public Object visitGetKeysInGroupCommand(final InvocationContext ctx, GetKeysInGroupCommand command) throws Throwable {
186       final String groupName = command.getGroupName();
187       if (!command.isGroupOwner() || !enabled || hasSkipLoadFlag(command)) {
188          return invokeNextInterceptor(ctx, command);
189       }
190 
191       final KeyFilter<Object> keyFilter = new CompositeKeyFilter<>(new GroupFilter<>(groupName, groupManager),
192                                                                    new CollectionKeyFilter<>(ctx.getLookedUpEntries().keySet()));
193       persistenceManager.processOnAllStores(keyFilter, new AdvancedCacheLoader.CacheLoaderTask() {
194          @Override
195          public void processEntry(MarshalledEntry marshalledEntry, AdvancedCacheLoader.TaskContext taskContext) throws InterruptedException {
196             synchronized (ctx) {
197                //the process can be made in multiple threads, so we need to synchronize in the context.
198                entryFactory.wrapEntryForReading(ctx, marshalledEntry.getKey(), convert(marshalledEntry, iceFactory)).setSkipLookup(true);
199             }
200          }
201       }, true, true);
202       return invokeNextInterceptor(ctx, command);
203    }

206    public CacheSet<CacheEntry<K, V>> visitEntrySetCommand(InvocationContext ctx, EntrySetCommand command) throws Throwable {
207       CacheSet<CacheEntry<K, V>> entrySet = (CacheSet<CacheEntry<K, V>>) invokeNextInterceptor(ctx, command);
208       if (!enabled || hasSkipLoadFlag(command)) {
209          return entrySet;
210       }
211       return new AbstractDelegatingEntryCacheSet<K, V>(getCacheWithFlags(cache, command), entrySet) {
212 
213          @Override
214          public CloseableIterator<CacheEntry<K, V>> iterator() {
215             CloseableIterator<CacheEntry<K, V>> iterator = Closeables.iterator(entrySet.stream());
216             // TODO: can we use data container equivalence?
217             Set<K> seenKeys = new EquivalentHashSet<K>(cache.getAdvancedCache().getDataContainer().size(),
218                     new AnyServerEquivalence());
219             // TODO: how to handle concurrent activation....
220             return new DistinctKeyDoubleEntryCloseableIterator<>(iterator, new CloseableSuppliedIterator<>(
221                     // TODO: how to pass in key filter...
222                     new PersistenceManagerCloseableSupplier<>(executorService, persistenceManager, iceFactory,
223                             new CollectionKeyFilter<>(seenKeys), 10, TimeUnit.SECONDS, 2048)), e -> e.getKey(),
224                     seenKeys);
225          }
226 
227          @Override
228          public CloseableSpliterator<CacheEntry<K, V>> spliterator() {
229             return spliteratorFromIterator(iterator());
230          }
231 
232          private <E> CloseableSpliterator<E> spliteratorFromIterator(CloseableIterator<E> iterator) {
233             return new IteratorAsSpliterator.Builder<>(iterator)
234                     .setCharacteristics(Spliterator.CONCURRENT | Spliterator.DISTINCT | Spliterator.NONNULL)
235                     .get();
236          }
237 
238          @Override
239          public int size() {
240             long size = stream().count();
241             if (size > Integer.MAX_VALUE) {
242                return Integer.MAX_VALUE;
243             }
244             return (int) size;
245          }
246       };

342    /**
343     * Loads from the cache loader the entry for the given key.  A found value is loaded into the current context.  The
344     * method returns whether the value was found or not, or even if the cache loader was checked.
345     * @param ctx The current invocation's context
346     * @param key The key for the entry to look up
347     * @param cmd The command that was called that now wants to query the cache loader
348     * @return Whether or not the entry was found in the cache loader.  A value of null means the cache loader was never
349     *         queried for the value, so it was neither a hit or a miss.
350     * @throws Throwable
351     */
352    protected final Boolean loadIfNeeded(final InvocationContext ctx, Object key, final FlagAffectedCommand cmd) {
353       if (skipLoad(cmd, key, ctx)) {
354          return null;
355       }
356 
357       return loadInContext(ctx, key, cmd);
358    }
359 
360    private Boolean loadInContext(InvocationContext ctx, Object key, FlagAffectedCommand cmd) {
361       final AtomicReference<Boolean> isLoaded = new AtomicReference<>();
362       InternalCacheEntry<K, V> entry = PersistenceUtil.loadAndStoreInDataContainer(dataContainer, persistenceManager, (K) key,
363                                                                              ctx, timeService, isLoaded);
364       Boolean isLoadedValue = isLoaded.get();
365       if (trace) {
366          log.tracef("Entry was loaded? %s", isLoadedValue);
367       }
368       if (getStatisticsEnabled()) {
369          if (isLoadedValue == null) {
370             // the entry was in data container, we haven't touched cache store
371          } else if (isLoadedValue) {
372             cacheLoads.incrementAndGet();
373          } else {
374             cacheMisses.incrementAndGet();
375          }
376       }
377 
378       if (entry != null) {
379          CacheEntry wrappedEntry = wrapInternalCacheEntry(ctx, key, cmd, entry);
380          if (isLoadedValue != null && isLoadedValue.booleanValue() && wrappedEntry != null) {
381             recordLoadedEntry(ctx, key, wrappedEntry, entry, cmd);
382          }
383       }
384       return isLoadedValue;
385    }

387    private boolean skipLoad(FlagAffectedCommand cmd, Object key, InvocationContext ctx) {
388       if (!shouldAttemptLookup(ctx.lookupEntry(key))) {
389          if (trace) {
390             log.tracef("Skip load for command %s. Entry already exists in context.", cmd);
391          }
392          return true;
393       }
394 
395       if (!canLoad(key)) {
396          if (trace) {
397             log.tracef("Skip load for command %s. Cannot load the key.", cmd);
398          }
399          return true;
400       }
401 
402       boolean skip;
403       if (cmd instanceof WriteCommand) {
404          skip = skipLoadForWriteCommand((WriteCommand) cmd, key, ctx);
405          if (trace) {
406             log.tracef("Skip load for write command %s? %s", cmd, skip);
407          }
408       } else {
409          //read command
410          skip = hasSkipLoadFlag(cmd);
411          if (trace) {
412             log.tracef("Skip load for command %s?. %s", cmd, skip);
413          }
414       }
415       return skip;
416    }
417 
418    protected boolean skipLoadForWriteCommand(WriteCommand cmd, Object key, InvocationContext ctx) {
419       if (cmd.readsExistingValues()) {
420          // TODO Could make DELTA_WRITE/ApplyDeltaCommand override SKIP_CACHE_LOAD by changing the next line to
421          // if (hasSkipLoadFlag(cmd) && !cmd.alwaysReadsExistingValues)
422          if (hasSkipLoadFlag(cmd)) {
423             log.tracef("Skipping load for command that reads existing values %s", cmd);
424             return true;
425          } else {
426             return false;
427          }
428       }
429       return true;
430    }

432    private CacheEntry wrapInternalCacheEntry(InvocationContext ctx, Object key, FlagAffectedCommand cmd,
433                                              InternalCacheEntry ice) {
434       if (cmd instanceof ApplyDeltaCommand) {
435          ctx.putLookedUpEntry(key, ice);
436          return entryFactory.wrapEntryForDelta(ctx, key, ((ApplyDeltaCommand) cmd).getDelta());
437       } else {
438          return entryFactory.wrapEntryForPut(ctx, key, ice, false, cmd, false);
439       }
440    }

442    /**
443     * Only perform if context doesn't have a value found (Read Committed) or if we can do a remote
444     * get only if the value is null (Repeatable Read)
445     */
446    private boolean shouldAttemptLookup(CacheEntry e) {
447       return e == null || (e.isNull() || e.getValue() == null) && !e.skipLookup();
448    }

450    /**
451     * This method records a loaded entry, performing the following steps: <ol>
452     * <li>updates the 'entry' reference (an entry in the current thread's InvocationContext) with the contents
453     * of 'loadedEntry' (freshly loaded from the CacheStore) so that the loaded details will be flushed to the
454     * DataContainer when the call returns (in the LockingInterceptor, when locks are released)</li>
455     * <li>notifies listeners</li> </ol>
456     */
457    private void recordLoadedEntry(InvocationContext ctx, Object key,
458                                   CacheEntry entry, InternalCacheEntry loadedEntry, FlagAffectedCommand cmd) {
459       final Object value = loadedEntry.getValue();
460       // FIXME: There's no point to trigger the entryLoaded/Activated event twice.
461       sendNotification(key, value, true, ctx, cmd);
462       entry.setValue(value);
463 
464       Metadata metadata = cmd.getMetadata();
465       Metadata loadedMetadata = loadedEntry.getMetadata();
466       if (metadata != null && loadedMetadata != null)
467          metadata = Metadatas.applyVersion(loadedMetadata, metadata);
468       else if (metadata == null)
469          metadata = loadedMetadata;
470 
471       entry.setMetadata(metadata);
472       // TODO shouldn't we also be setting last used and created timestamps?
473       entry.setValid(true);
474 
475       sendNotification(key, value, false, ctx, cmd);
476    }
477 
478    protected void sendNotification(Object key, Object value, boolean pre,
479          InvocationContext ctx, FlagAffectedCommand cmd) {
480       notifier.notifyCacheEntryLoaded(key, value, pre, ctx, cmd);
481    }
