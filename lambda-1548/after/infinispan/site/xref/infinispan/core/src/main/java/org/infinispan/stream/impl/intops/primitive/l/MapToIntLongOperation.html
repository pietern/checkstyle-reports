3   import org.infinispan.stream.impl.intops.IntermediateOperation;
4   
5   import java.util.function.LongToIntFunction;
6   import java.util.stream.LongStream;
7   import java.util.stream.IntStream;
8   
9   /**
10   * Performs map to int operation on a {@link LongStream}
11   */
12  public class MapToIntLongOperation implements IntermediateOperation<Long, LongStream, Integer, IntStream> {
13     private final LongToIntFunction function;
14  
15     public MapToIntLongOperation(LongToIntFunction function) {
16        this.function = function;
17     }
18  
19     @Override
20     public IntStream perform(LongStream stream) {
21        return stream.mapToInt(function);
22     }
23  
24     public LongToIntFunction getFunction() {
25        return function;
26     }
27  }
