5   /**
6    * Factory interface, allows {@link org.infinispan.Cache} to use different transactional systems. Names of implementors of
7    * this class can be configured using {@link Configuration#setTransactionManagerLookupClass}.
8    * Thread safety: it is possible for the same instance of this class to be used by multiple caches at the same time e.g.
9    * when the same instance is passed to multiple configurations:
10   * {@link org.infinispan.configuration.cache.TransactionConfigurationBuilder#transactionManagerLookup(TransactionManagerLookup)}.
11   * As infinispan supports parallel test startup, it might be possible for multiple threads to invoke the
12   * getTransactionManager() method concurrently, so it is highly recommended for instances of this class to be thread safe.
13   *
14   * @author Bela Ban, Aug 26 2003
15   * @since 4.0
16   */
17  public interface TransactionManagerLookup {
18  
19     /**
20      * Returns a new TransactionManager.
21      *
22      * @throws Exception if lookup failed
23      */
24     TransactionManager getTransactionManager() throws Exception;
