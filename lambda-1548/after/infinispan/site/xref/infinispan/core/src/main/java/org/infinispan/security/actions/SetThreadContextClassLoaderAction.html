3   import java.security.PrivilegedAction;

5   /**
6    * SetThreadContextClassLoaderAction.
7    *
8    * @author Tristan Tarrant
9    * @since 7.0

11  public class SetThreadContextClassLoaderAction implements PrivilegedAction<ClassLoader> {
12  
13     private final ClassLoader classLoader;
14  
15     public SetThreadContextClassLoaderAction(ClassLoader classLoader) {
16        this.classLoader = classLoader;
17     }
18  
19     public SetThreadContextClassLoaderAction(Class<?> klass) {
20        this.classLoader = klass.getClassLoader();
21     }
22  
23     @Override
24     public ClassLoader run() {
25        ClassLoader old = Thread.currentThread().getContextClassLoader();
26        Thread.currentThread().setContextClassLoader(classLoader);
27        return old;
28     }
