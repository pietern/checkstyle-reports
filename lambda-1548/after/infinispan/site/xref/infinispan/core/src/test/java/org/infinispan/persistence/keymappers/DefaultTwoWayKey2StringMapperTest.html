5   @Test(groups = "unit", testName = "persistence.keymappers.DefaultTwoWayKey2StringMapperTest")
6   public class DefaultTwoWayKey2StringMapperTest {

8      DefaultTwoWayKey2StringMapper mapper = new DefaultTwoWayKey2StringMapper();
9   
10     public void testKeyMapper() {
11        String skey = mapper.getStringMapping("k1");
12        assert skey.equals("k1");
13  
14        skey = mapper.getStringMapping(Integer.valueOf(100));
15  
16        assert !skey.equals("100");
17  
18        Integer i = (Integer) mapper.getKeyMapping(skey);
19        assert i == 100;
20  
21        skey = mapper.getStringMapping(Boolean.TRUE);
22  
23        assert !skey.equalsIgnoreCase("true");
24  
25        Boolean b = (Boolean) mapper.getKeyMapping(skey);
26  
27        assert b;
28  
29        skey = mapper.getStringMapping(Double.valueOf(3.141592d));
30  
31        assert !skey.equals("3.141592");
32  
33        Double d = (Double) mapper.getKeyMapping(skey);
34  
35        assert d == 3.141592d;
36     }
37  
38     public void testPrimitivesAreSupported() {
39        assert mapper.isSupportedType(Integer.class);
40        assert mapper.isSupportedType(Byte.class);
41        assert mapper.isSupportedType(Short.class);
42        assert mapper.isSupportedType(Long.class);
43        assert mapper.isSupportedType(Double.class);
44        assert mapper.isSupportedType(Float.class);
45        assert mapper.isSupportedType(Boolean.class);
46        assert mapper.isSupportedType(String.class);
47     }
48  
49     public void testTwoWayContract() {
50        Object[] toTest = { 0, new Byte("1"), new Short("2"), (long) 3, new Double("3.4"), new Float("3.5"), Boolean.FALSE, "some string" };
51        for (Object o : toTest) {
52           Class<?> type = o.getClass();
53           String rep = mapper.getStringMapping(o);
54           assert o.equals(mapper.getKeyMapping(rep)) : String.format("Failed on type %s and value %s", type, rep);
55        }
56     }
57  
58     public void testAssumption() {
59        // even if they have the same value, they have a different type
60        assert !new Float(3.0f).equals(new Integer(3));
61     }
62  
63     public void testString() {
64        assert mapper.isSupportedType(String.class);
65        assert assertWorks("") : "Expected empty string, was " + mapper.getStringMapping("");
66        assert assertWorks("mircea") : "Expected 'mircea', was " + mapper.getStringMapping("mircea");
67     }
