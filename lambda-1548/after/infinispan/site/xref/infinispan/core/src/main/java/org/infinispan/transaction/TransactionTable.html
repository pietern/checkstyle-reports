1   package org.infinispan.transaction;
2   
3   import org.infinispan.transaction.xa.GlobalTransaction;
4   
5   import javax.transaction.Transaction;
6   import java.util.Collection;
7   
8   /**
9    * Interface that allows to fetch the {@link org.infinispan.transaction.xa.GlobalTransaction} associated to local or
10   * remote transactions.
11   *
12   * @author Pedro Ruivo
13   * @since 7.1
14   */
15  public interface TransactionTable {
16  
17     /**
18      * @param transaction the local transaction. Must be non-null.
19      * @return the {@link org.infinispan.transaction.xa.GlobalTransaction} associated with the transaction or {@code
20      * null} if doesn't exists.
21      */
22     GlobalTransaction getGlobalTransaction(Transaction transaction);
23  
24     /**
25      * @return an unmodified collection of {@link org.infinispan.transaction.xa.GlobalTransaction} associated with local
26      * running transactions.
27      */
28     Collection<GlobalTransaction> getLocalGlobalTransaction();
29  
30     /**
31      * @return an unmodified collection of {@link org.infinispan.transaction.xa.GlobalTransaction} associated with remote
32      * transactions.
33      */
34     Collection<GlobalTransaction> getRemoteGlobalTransaction();
35  
36  }
