7   /**
8    * Intermediate operation that can be applied to a stream to change its processing.
9    * @param <InputType> the type of the input stream
10   * @param <InputStream> the input stream type
11   * @param <OutputType> the type of the output stream
12   * @param <OutputStream> the output stream type
13   */
14  public interface IntermediateOperation<InputType, InputStream extends BaseStream<InputType, InputStream>,
15          OutputType, OutputStream extends BaseStream<OutputType, OutputStream>> {
16     /**
17      * Performs the actualy intermediate operation returning the resulting stream
18      * @param stream the stream to have the operation performed on
19      * @return the resulting stream after the operation was applied
20      */
21     OutputStream perform(InputStream stream);
22  
23     /**
24      * Handles injection of components for various dependencies that the intermediate operation has
25      * @param registry the registry to use
26      */
27     default void handleInjection(ComponentRegistry registry) {
28        // Default is nothing is done
29     }
30  }
