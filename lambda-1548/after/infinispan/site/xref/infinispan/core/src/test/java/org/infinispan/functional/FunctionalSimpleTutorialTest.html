32     public void testSimpleTutorial() throws Exception {
33        AdvancedCache<String, String> cache = cacheManager.<String, String>getCache().getAdvancedCache();
34        FunctionalMapImpl<String, String> functionalMap = FunctionalMapImpl.create(cache);
35        WriteOnlyMap<String, String> writeOnlyMap = WriteOnlyMapImpl.create(functionalMap);
36        ReadOnlyMap<String, String> readOnlyMap = ReadOnlyMapImpl.create(functionalMap);
37  
38        // Execute two parallel write-only operation to store key/value pairs
39        CompletableFuture<Void> writeFuture1 = writeOnlyMap.eval("key1", "value1",
40           (v, writeView) -> writeView.set(v));
41        CompletableFuture<Void> writeFuture2 = writeOnlyMap.eval("key2", "value2",
42           (v, writeView) -> writeView.set(v));
43  
44        // When each write-only operation completes, execute a read-only operation to retrieve the value
45        CompletableFuture<String> readFuture1 =
46           writeFuture1.thenCompose(r -> readOnlyMap.eval("key1", ReadEntryView::get));
47        CompletableFuture<String> readFuture2 =
48           writeFuture2.thenCompose(r -> readOnlyMap.eval("key2", ReadEntryView::get));
49  
50        // When the read-only operation completes, print it out
51        printf("Created entries: %n");
52        CompletableFuture<Void> end = readFuture1.thenAcceptBoth(readFuture2, (v1, v2) ->
53           printf("key1 = %s%nkey2 = %s%n", v1, v2));
54  
55        // Wait for this read/write combination to finish
56        end.get();
57  
58        // Create a read-write map
59        ReadWriteMap<String, String> readWriteMap = ReadWriteMapImpl.create(functionalMap);
60  
61        // Use read-write multi-key based operation to write new values
62        // together with lifespan and return previous values
63        Map<String, String> data = new HashMap<>();
64        data.put("key1", "newValue1");
65        data.put("key2", "newValue2");
66        Traversable<String> previousValues = readWriteMap.evalMany(data, (v, readWriteView) -> {
67           String prev = readWriteView.find().orElse(null);
68           readWriteView.set(v, new MetaLifespan(Duration.ofHours(1).toMillis()));
69           return prev;
70        });
71  
72        // Use read-only multi-key operation to read current values for multiple keys
73        Traversable<ReadEntryView<String, String>> entryViews =
74              readOnlyMap.evalMany(data.keySet(), readOnlyView -> readOnlyView);
75        printf("Updated entries: %n");
76        entryViews.forEach(view -> printf("%s%n", view));
77  
78        // Finally, print out the previous entry values
79        printf("Previous entry values: %n");
80        previousValues.forEach(prev -> printf("%s%n", prev));
81     }
82  
83     private void printf(String format, Object ... args) {
84        log.infof(format, args);
85     }
