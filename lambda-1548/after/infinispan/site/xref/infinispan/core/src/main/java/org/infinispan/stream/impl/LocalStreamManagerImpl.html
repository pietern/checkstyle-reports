39  /**
40   * Local stream manager implementation that handles injection of the stream supplier, invoking the operation and
41   * subsequently notifying the operation if a rehash has changed one of its segments.
42   * @param <K> key type of underlying cache
43   * @param <V> value type of underlying cache
44   */
45  @Listener(observation = Listener.Observation.POST)
46  public class LocalStreamManagerImpl<K, V> implements LocalStreamManager<K> {
47     private final static Log log = LogFactory.getLog(LocalStreamManagerImpl.class);
48  
49     private AdvancedCache<K, V> cache;
50     private ComponentRegistry registry;
51     private StateTransferManager stm;
52     private RpcManager rpc;
53     private CommandsFactory factory;
54     private boolean hasLoader;
55  
56     private Address localAddress;

58     private final ConcurrentMap<UUID, SegmentListener> changeListener = CollectionFactory.makeConcurrentMap();
59  
60     class SegmentListener {
61        private final Set<Integer> segments;
62        private final SegmentAwareOperation op;
63        private final Set<Integer> segmentsLost;
64  
65        SegmentListener(Set<Integer> segments, SegmentAwareOperation op) {
66           this.segments = new HashSet<>(segments);
67           this.op = op;
68           this.segmentsLost = new HashSet<>();
69        }
70  
71        public void localSegments(Set<Integer> localSegments) {
72           segments.forEach(s -> {
73              if (!localSegments.contains(s)) {
74                 log.tracef("Could not process segment %s", s);
75                 segmentsLost.add(s);
76              }
77           });
78        }
79  
80        public void lostSegments(Set<Integer> lostSegments) {
81           for (Integer segment : lostSegments) {
82              if (segments.contains(segment)) {
83                 log.tracef("Lost segment %s", segment);
84                 if (op.lostSegment(false)) {
85                    if (segmentsLost.add(segment) && segmentsLost.size() == segments.size()) {
86                       log.tracef("All segments %s are now lost", segments);
87                       op.lostSegment(true);
88                    }
89                 }
90              }
91           }
92        }
93     }

112    /**
113     * We need to listen to data rehash events in case if data moves while we are iterating over it.
114     * If a rehash occurs causing this node to lose a segment and there is something iterating over the stream looking
115     * for values of that segment, we can't guarantee that the data has all been seen correctly, so we must therefore
116     * suspect that node by sending it back to the owner.
117     * @param event The data rehash event
118     */
119    @DataRehashed
120    public void dataRehashed(DataRehashedEvent<K, V> event) {
121       ConsistentHash startHash = event.getConsistentHashAtStart();
122       ConsistentHash endHash = event.getConsistentHashAtEnd();
123       boolean trace = log.isTraceEnabled();
124       if (startHash != null && endHash != null) {
125          log.tracef("Data rehash occurred startHash: %s and endHash: %s with new topology %s and was pre %s", startHash, endHash,
126                  event.getNewTopologyId(), event.isPre());
127 
128          if (!changeListener.isEmpty()) {
129             if (trace) {
130                log.tracef("Previous segments %s ", startHash.getSegmentsForOwner(localAddress));
131                log.tracef("After segments %s ", endHash.getSegmentsForOwner(localAddress));
132             }
133             // we don't care about newly added segments, since that means our run wouldn't include them anyways
134             Set<Integer> beforeSegments = new HashSet<>(startHash.getSegmentsForOwner(localAddress));
135             // Now any that were there before but aren't there now should be added - we don't care about new segments
136             // since our current request shouldn't be working on it - it will have to retrieve it later
137             beforeSegments.removeAll(endHash.getSegmentsForOwner(localAddress));
138             if (!beforeSegments.isEmpty()) {
139                // We have to make sure all current listeners get the newest hashes updated.  This has to occur for
140                // new nodes and nodes leaving as the hash segments will change in both cases.
141                for (Map.Entry<UUID, SegmentListener> entry : changeListener.entrySet()) {
142                   if (trace) {
143                      log.tracef("Notifying %s through SegmentChangeListener", entry.getKey());
144                   }
145                   entry.getValue().lostSegments(beforeSegments);
146                }
147             } else if (trace) {
148                log.tracef("No segments have been removed from data rehash, no notification required");
149             }
150          } else {
151             log.tracef("No change listeners present!");
152          }
153       }
154    }

156    private AdvancedCache<K, V> getCacheRespectingLoader(boolean includeLoader) {
157       // We only need to "skip" the loader if there is one and we were told to skip it
158       if (hasLoader && !includeLoader) {
159          return cache.getAdvancedCache().withFlags(Flag.SKIP_CACHE_LOAD);
160       }
161       return cache;
162    }

164    private Stream<CacheEntry<K, V>> getStream(CacheSet<CacheEntry<K, V>> cacheEntrySet, boolean parallelStream,
165            Set<Integer> segments, Set<K> keysToInclude, Set<K> keysToExclude) {
166       Stream<CacheEntry<K, V>> stream = (parallelStream ? cacheEntrySet.parallelStream() : cacheEntrySet.stream())
167               .filterKeys(keysToInclude).filterKeySegments(segments);
168       if (!keysToExclude.isEmpty()) {
169          return stream.filter(e -> !keysToExclude.contains(e.getKey()));
170       }
171       return stream;
172    }

174    private Stream<CacheEntry<K, V>> getRehashStream(CacheSet<CacheEntry<K, V>> cacheEntrySet, UUID requestId,
175            SegmentListener listener, boolean parallelStream, Set<Integer> segments, Set<K> keysToInclude,
176            Set<K> keysToExclude) {
177       CacheTopology topology = stm.getCacheTopology();
178       log.tracef("Topology for supplier is %s for id %s", topology, requestId);
179       ConsistentHash readCH = topology.getCurrentCH();
180       ConsistentHash pendingCH = topology.getPendingCH();
181       if (pendingCH != null) {
182          Set<Integer> lostSegments = new HashSet<>();
183          Iterator<Integer> iterator = segments.iterator();
184          while (iterator.hasNext()) {
185             Integer segment = iterator.next();
186             // If the segment is not owned by both CHs we can't use it during rehash
187             if (!pendingCH.locateOwnersForSegment(segment).contains(localAddress)
188                     || !readCH.locateOwnersForSegment(segment).contains(localAddress)) {
189                iterator.remove();
190                lostSegments.add(segment);
191             }
192          }
193          if (!lostSegments.isEmpty()) {
194             log.tracef("Lost segments %s during rehash for id %s", lostSegments, requestId);
195             listener.lostSegments(lostSegments);
196          } else {
197             log.tracef("Currently in the middle of a rehash for id %s", requestId);
198          }
199       } else {
200          Set<Integer> ourSegments = readCH.getSegmentsForOwner(localAddress);
201          if (segments.retainAll(ourSegments)) {
202             log.tracef("We found to be missing some segments requested for id %s", requestId);
203             listener.localSegments(ourSegments);
204          } else {
205             log.tracef("Hash %s for id %s", readCH, requestId);
206          }
207       }
208 
209       return getStream(cacheEntrySet, parallelStream, segments, keysToInclude, keysToExclude);
210    }

212    @Override
213    public <R> void streamOperation(UUID requestId, Address origin, boolean parallelStream, Set<Integer> segments,
214            Set<K> keysToInclude, Set<K> keysToExclude, boolean includeLoader, TerminalOperation<R> operation) {
215       log.tracef("Received operation request for id %s from %s for segments %s", requestId, origin, segments);
216       CacheSet<CacheEntry<K, V>> cacheEntrySet = getCacheRespectingLoader(includeLoader).cacheEntrySet();
217       operation.setSupplier(() -> getStream(cacheEntrySet, parallelStream, segments, keysToInclude, keysToExclude));
218       operation.handleInjection(registry);
219       R value = operation.performOperation();
220       rpc.invokeRemotely(Collections.singleton(origin), factory.buildStreamResponseCommand(requestId, true,
221               Collections.emptySet(), value), rpc.getDefaultRpcOptions(true));
222    }

224    @Override
225    public <R> void streamOperationRehashAware(UUID requestId, Address origin, boolean parallelStream,
226            Set<Integer> segments, Set<K> keysToInclude, Set<K> keysToExclude, boolean includeLoader,
227            TerminalOperation<R> operation) {
228       log.tracef("Received rehash aware operation request for id %s from %s for segments %s", requestId, origin, segments);
229       CacheSet<CacheEntry<K, V>> cacheEntrySet = getCacheRespectingLoader(includeLoader).cacheEntrySet();
230       SegmentListener listener = new SegmentListener(segments, operation);
231       R value;
232 
233       operation.handleInjection(registry);
234       // We currently only allow 1 request per id (we may change this later)
235       changeListener.put(requestId, listener);
236       log.tracef("Registered change listener for %s", requestId);
237       try {
238          operation.setSupplier(() -> getRehashStream(cacheEntrySet, requestId, listener, parallelStream, segments,
239                  keysToInclude, keysToExclude));
240          value = operation.performOperation();
241          log.tracef("Request %s completed for segments %s with %s suspected segments", requestId, segments,
242                  listener.segmentsLost);
243       } finally {
244          changeListener.remove(requestId);
245          log.tracef("UnRegistered change listener for %s", requestId);
246       }
247       if (cache.getStatus() != ComponentStatus.RUNNING) {
248          if (log.isTraceEnabled()) {
249             log.tracef("Cache status is no longer running, all segments are now suspect for %s", requestId);
250          }
251          listener.segmentsLost.addAll(segments);
252          value = null;
253       }
254 
255       log.tracef("Sending response for %s", requestId);
256       rpc.invokeRemotely(Collections.singleton(origin), factory.buildStreamResponseCommand(requestId, true,
257               listener.segmentsLost, value), rpc.getDefaultRpcOptions(true));
258       log.tracef("Sent response for %s", requestId);
259    }

261    @Override
262    public <R> void streamOperation(UUID requestId, Address origin, boolean parallelStream, Set<Integer> segments,
263            Set<K> keysToInclude, Set<K> keysToExclude, boolean includeLoader,
264            KeyTrackingTerminalOperation<K, R, ?> operation) {
265       log.tracef("Received key aware operation request for id %s from %s for segments %s", requestId, origin, segments);
266       CacheSet<CacheEntry<K, V>> cacheEntrySet = getCacheRespectingLoader(includeLoader).cacheEntrySet();
267       operation.setSupplier(() -> getStream(cacheEntrySet, parallelStream, segments, keysToInclude, keysToExclude));
268       operation.handleInjection(registry);
269       Collection<R> value = operation.performOperation(new NonRehashIntermediateCollector<>(origin, requestId,
270               parallelStream));
271       rpc.invokeRemotely(Collections.singleton(origin), factory.buildStreamResponseCommand(requestId, true,
272               Collections.emptySet(), value), rpc.getDefaultRpcOptions(true));
273    }

312    class NonRehashIntermediateCollector<R> implements KeyTrackingTerminalOperation.IntermediateCollector<R> {
313       private final Address origin;
314       private final UUID requestId;
315       private final boolean useManagedBlocker;
316 
317       NonRehashIntermediateCollector(Address origin, UUID requestId, boolean useManagedBlocker) {
318          this.origin = origin;
319          this.requestId = requestId;
320          this.useManagedBlocker = useManagedBlocker;
321       }
322 
323       @Override
324       public void sendDataResonse(R response) {
325          // If we know we were in a parallel stream we should use a managed blocker to not consume core fork join
326          // threads if applicable.
327          if (useManagedBlocker) {
328             try {
329                // We use a managed blocker just in case if this invoked in the common thread pool
330                ForkJoinPool.managedBlock(new ResponseBlocker(response));
331             } catch (InterruptedException e) {
332                throw new CacheException(e);
333             }
334          } else {
335             rpc.invokeRemotely(Collections.singleton(origin), new StreamResponseCommand<>(cache.getName(), localAddress,
336                     requestId, false, response), rpc.getDefaultRpcOptions(true));
337          }
338       }
339 
340       class ResponseBlocker implements ForkJoinPool.ManagedBlocker {
341          private final R response;
342          private boolean completed = false;
343 
344          ResponseBlocker(R response) {
345             this.response = response;
346          }
347 
348          @Override
349          public boolean block() throws InterruptedException {
350             if (!completed) {
351                // This way we don't send more than 1 response to the originating node but still inside managed blocker
352                // so we don't consume a thread
353                synchronized (NonRehashIntermediateCollector.this) {
354                   rpc.invokeRemotely(Collections.singleton(origin), new StreamResponseCommand<>(cache.getName(), localAddress,
355                           requestId, false, response), rpc.getDefaultRpcOptions(true));
356                }
357             }
358             completed = true;
359             return completed;
360          }
361 
362          @Override
363          public boolean isReleasable() {
364             return completed;
365          }
366       }
