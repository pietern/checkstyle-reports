76  /**
77   * A factory to build commands, initializing and injecting dependencies accordingly.  Commands built for a specific,
78   * named cache instance cannot be reused on a different cache instance since most commands contain the cache name it
79   * was built for along with references to other named-cache scoped components.
80   *
81   * @author Manik Surtani
82   * @author Mircea.Markus@jboss.com
83   * @author Galder Zamarreño
84   * @since 4.0
85   */
86  @Scope(Scopes.NAMED_CACHE)
87  public interface CommandsFactory {
88

90      * Builds a PutKeyValueCommand
91      * @param key key to put
92      * @param value value to put
93      * @param metadata metadata of entry
94      * @param flags Command flags provided by cache
95      * @return a PutKeyValueCommand
96      */
97     PutKeyValueCommand buildPutKeyValueCommand(Object key, Object value, Metadata metadata, Set<Flag> flags);
