3   /**
4    * Defines the logic of mapping a key object to a String. This is required by certain cache stores, in order
5    * to map each key to a String which the underlying store is capable of handling. It should generate a unique String
6    * based on the supplied key.
7    *
8    * @author Mircea.Markus@jboss.com
9    * @author Manik Surtani
10   */
11  public interface Key2StringMapper {
12  
13     /**
14      * Do we support this key type?
15      * @param keyType type to test
16      * @return true if the type is supported, false otherwise.
17      */
18     boolean isSupportedType(Class<?> keyType);
19  
20     /**
21      * Must return an unique String for the supplied key.
22      * @param key key to map to a String
23      * @return String representation of the key
24      */
25     String getStringMapping(Object key);
26  }
