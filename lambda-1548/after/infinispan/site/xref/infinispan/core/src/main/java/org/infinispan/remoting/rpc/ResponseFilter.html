1   package org.infinispan.remoting.rpc;
2   
3   import org.infinispan.remoting.responses.Response;
4   import org.infinispan.remoting.transport.Address;
5   
6   /**
7    * A mechanism of filtering RPC responses.  Used with the RPC manager.
8    *
9    * @author Manik Surtani
10   * @since 4.0
11   */
12  public interface ResponseFilter {
13     /**
14      * Determines whether a response from a given sender should be added to the response list of the request
15      *
16      * @param response The response (usually a serializable value)
17      * @param sender   The sender of response
18      * @return True if we should add the response to the response list of a request, otherwise false. In the latter case,
19      *         we don't add the response to the response list.
20      */
21     boolean isAcceptable(Response response, Address sender);
22  
23     /**
24      * Right after calling {@link #isAcceptable(Response, Address)}, this method is called to see whether we are done
25      * with the request and can unblock the caller
26      *
27      * @return False if the request is done, otherwise true
28      */
29     boolean needMoreResponses();
30  
31  }
