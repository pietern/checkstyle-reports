3   import org.infinispan.commons.util.concurrent.NotifyingNotifiableFuture;

5   import java.util.concurrent.CompletableFuture;
6   import java.util.concurrent.ExecutionException;
7   import java.util.concurrent.TimeUnit;

9   import static java.util.Objects.requireNonNull;
10  
11  /**
12   * Utility methods connecting {@link CompletableFuture} futures and our {@link NotifyingNotifiableFuture} futures.
13   *
14   * @author Dan Berindei
15   * @since 8.0
16   */
17  public class CompletableFutures {
18     public static <T> void connect(NotifyingNotifiableFuture<T> sink, CompletableFuture<T> source) {
19        CompletableFuture<T> compoundSource = source.whenComplete((value, throwable) -> {
20           if (throwable == null) {
21              sink.notifyDone(value);
22           } else {
23              sink.notifyException(throwable);
24           }
25        });
26        sink.setFuture(compoundSource);
27     }
28  
29     /**
30      * It waits until the {@link CompletableFuture} is completed.
31      * <p>
32      * It ignore if the {@link CompletableFuture} is completed normally or exceptionally.
33      *
34      * @param future the {@link CompletableFuture} to test.
35      * @param time   the timeout.
36      * @param unit   the timeout unit.
37      * @return {@code true} if completed, {@code false} if timed out.
38      * @throws InterruptedException if interrupted while waiting.
39      * @throws NullPointerException if {@code future} or {@code unit} is {@code null}.
40      */
41     public static boolean await(CompletableFuture<?> future, long time, TimeUnit unit) throws InterruptedException {
42        try {
43           requireNonNull(future, "Completable Future must be non-null.").get(time, requireNonNull(unit, "Time Unit must be non-null"));
44           return true;
45        } catch (ExecutionException e) {
46           return true;
47        } catch (java.util.concurrent.TimeoutException e) {
48           return false;
49        }
50     }
51  }
