1   package org.infinispan.batch;
2   
3   import net.jcip.annotations.NotThreadSafe;
4   import org.infinispan.commons.CacheConfigurationException;
5   import org.infinispan.configuration.cache.Configuration;
6   
7   /**
8    * Enables for automatic batching.
9    *
10   * @author Manik Surtani (<a href="mailto:manik AT jboss DOT org">manik AT jboss DOT org</a>)
11   * @since 4.0
12   */
13  @NotThreadSafe
14  public abstract class AutoBatchSupport {
15     protected BatchContainer batchContainer;
16  
17     protected static void assertBatchingSupported(Configuration c) {
18        if (!c.invocationBatching().enabled())
19           throw new CacheConfigurationException("Invocation batching not enabled in current configuration! Please enable it.");
20     }
21  
22     protected void startAtomic() {
23        batchContainer.startBatch(true);
24     }
25  
26     protected void endAtomic() {
27        batchContainer.endBatch(true, true);
28     }
29  
30     protected void failAtomic() {
31        batchContainer.endBatch(true, false);
32     }
33  }
