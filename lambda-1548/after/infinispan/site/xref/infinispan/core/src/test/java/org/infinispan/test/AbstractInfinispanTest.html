98     protected void eventually(String message, Condition ec, long timeout, long pollInterval, TimeUnit unit) {
99        if (pollInterval <= 0) {
100          throw new IllegalArgumentException("Check interval must be positive");
101       }
102       try {
103          long expectedEndTime = System.nanoTime() + TimeUnit.NANOSECONDS.convert(timeout, unit);
104          long sleepMillis = TimeUnit.MILLISECONDS.convert(pollInterval, unit);
105          while (expectedEndTime - System.nanoTime() > 0) {
106             if (ec.isSatisfied()) return;
107             Thread.sleep(sleepMillis);
108          }
109          assertTrue(message, ec.isSatisfied());
110       } catch (Exception e) {
111          throw new RuntimeException("Unexpected!", e);
112       }
113    }

152     * This should normally not be used, use the {@link AbstractInfinispanTest#completionService()}
153     * method when an executor is required.  Although if you want a limited set of threads this could
154     * still be useful for something like {@link java.util.concurrent.Executors#newFixedThreadPool(int, java.util.concurrent.ThreadFactory)} or
155     * {@link java.util.concurrent.Executors#newSingleThreadExecutor(java.util.concurrent.ThreadFactory)}
156     * @param prefix The prefix starting for the thread factory
157     * @return A thread factory that will use the same naming schema as the other methods
158     */
159    protected ThreadFactory getTestThreadFactory(final String prefix) {
160       final String className = getClass().getSimpleName();
161 
162       ThreadFactory threadFactory = new ThreadFactory() {
163          private final AtomicInteger counter = new AtomicInteger(0);
164 
165          @Override
166          public Thread newThread(Runnable r) {
167             String threadName = prefix + "-" + counter.incrementAndGet() + "," + className;
168             Thread thread = new Thread(r, threadName);
169             TestResourceTracker.addResource(AbstractInfinispanTest.this, new ThreadCleaner(thread));
170             return thread;
171          }
172       };
173       return threadFactory;
174    }

176    /**
177     * This will run the various callables at approximately the same time by ensuring they all start before
178     * allowing the callables to proceed.  The callables may not be running yet at the offset of the return of
179     * the completion service.
180     *
181     * This method doesn't allow distinction between which Future applies to which Callable.  If that is required
182     * it is suggested to use {@link AbstractInfinispanTest#completionService()} and retrieve the future
183     * directly when submitting the Callable while also wrapping your callables in something like
184     * {@link org.infinispan.test.AbstractInfinispanTest.ConcurrentCallable} to ensure your callables will be invoked
185     * at approximately the same time.
186     * @param task1 First task to add - required to ensure at least 2 callables are provided
187     * @param task2 Second task to add - required to ensure at least 2 callables are provided
188     * @param tasks The callables to run
189     * @param <V> The type of callables
190     * @return The completion service that can be used to query when a callable completes
191     */
192    protected <V> CompletionService<V> runConcurrentlyWithCompletionService(Callable<? extends V> task1,
193                                                                            Callable<? extends V> task2,
194                                                                            Callable<? extends V>... tasks) {
195       if (task1 == null) {
196          throw new NullPointerException("Task1 was not provided!");
197       }
198       if (task2 == null) {
199          throw new NullPointerException("Task2 was not provided!");
200       }
201       Callable[] callables = new Callable[tasks.length + 2];
202       int i = 0;
203       callables[i++] = task1;
204       callables[i++] = task2;
205       for (Callable<? extends V> callable : tasks) {
206          callables[i++] = callable;
207       }
208       return runConcurrentlyWithCompletionService(callables);
209    }

211    /**
212     * This will run the various callables at approximately the same time by ensuring they all start before
213     * allowing the callables to proceed.  The callables may not be running yet at the offset of the return of
214     * the completion service.
215     *
216     * This method doesn't allow distinction between which Future applies to which Callable.  If that is required
217     * it is suggested to use {@link AbstractInfinispanTest#completionService()} and retrieve the future
218     * directly when submitting the Callable while also wrapping your callables in something like
219     * {@link org.infinispan.test.AbstractInfinispanTest.ConcurrentCallable} to ensure your callables will be invoked
220     * at approximately the same time.
221     * @param tasks The callables to run.  Note the type isn't provided since it is difficult
222     * @param <V> The type of callables
223     * @return The completion service that can be used to query when a callable completes
224     */
225    protected <V> CompletionService<V> runConcurrentlyWithCompletionService(Callable[] tasks) {
226       if (tasks.length == 0) {
227          throw new IllegalArgumentException("Provided tasks array was empty");
228       }
229       CompletionService<V> completionService = completionService();
230       CyclicBarrier barrier = new CyclicBarrier(tasks.length);
231       for (Callable task : tasks) {
232          completionService.submit(new ConcurrentCallable<>(new LoggingCallable<V>(task), barrier));
233       }
234 
235       return completionService;
236    }

275    /**
276     * Waits for the desired numberOfTasks in the completion service to complete one at a time polling for each up to
277     * 10 seconds.  If one of the tasks produces an exception, the first one is rethrown wrapped in a ExecutionException
278     * after all tasks have completed.
279     * @param completionService The completion service that had tasks submitted to that total numberOfTasks
280     * @param numberOfTasks How many tasks have been sent to the service
281     * @throws InterruptedException If this thread is interruped while waiting for the tasks to complete
282     * @throws ExecutionException Thrown if one of the tasks produces an exception.  The first encounted exception will
283     *         be thrown
284     * @throws TimeoutException If a task is not found to complete within 10 seconds of the previously completed or the
285     *         first task.
286     */
287    protected void waitForCompletionServiceTasks(CompletionService<?> completionService, int numberOfTasks)
288          throws InterruptedException,ExecutionException, TimeoutException {
289       if (completionService == null) {
290          throw new NullPointerException("Provided completionService cannot be null");
291       }
292       if (numberOfTasks <= 0) {
293          throw new IllegalArgumentException("Provided numberOfTasks cannot be less than or equal to 0");
294       }
295       // check for any errors
296       ExecutionException exception = null;
297       for (int i = 0; i < numberOfTasks; i++) {
298          try {
299             Future<?> future = completionService.poll(10, TimeUnit.SECONDS);
300             if (future == null) {
301                throw new TimeoutException("Concurrent task didn't complete within 10 seconds!");
302             }
303             // No timeout since it is guaranteed to be done
304             future.get();
305          } catch (ExecutionException e) {
306             log.debug("Exception in concurrent task", e);
307             if (exception == null) {
308                exception = e;
309             }
310          }
311       }
312 
313       // If there was an exception throw the last one
314       if (exception != null) {
315          throw exception;
316       }
317    }

319    /**
320     * This will run the callable at approximately the same time in different thrads by ensuring they all start before
321     * allowing the callables to proceed.  All callables will be ensure to completion else a TimeoutException will be thrown
322     *
323     * The callable itself should be thread safe since it will be invoked from multiple threads.
324     * @param callable The callable to run multiple times
325     * @param <V> The type of callable
326     * @throws InterruptedException Thrown if this thread is interrupted
327     * @throws ExecutionException Thrown if one of the callables throws any kind of Throwable.  The
328     *         thrown Throwable will be wrapped by this exception
329     * @throws TimeoutException If one of the callables doesn't complete within 10 seconds
330     */
331    protected <V> void runConcurrently(Callable<V> callable, int invocationCount) throws InterruptedException,
332                                                                                         ExecutionException,
333                                                                                         TimeoutException {
334       if (callable == null) {
335          throw new NullPointerException("Provided callable cannot be null");
336       }
337       if (invocationCount <= 0) {
338          throw new IllegalArgumentException("Provided invocationCount cannot be less than or equal to 0");
339       }
340       CompletionService<V> completionService = completionService();
341       CyclicBarrier barrier = new CyclicBarrier(invocationCount);
342 
343       for (int i = 0; i < invocationCount; ++i) {
344          completionService.submit(new ConcurrentCallable<>(new LoggingCallable<>(callable), barrier));
345       }
346 
347       waitForCompletionServiceTasks(completionService, invocationCount);
348    }

350    /**
351     * A callable that will first await on the provided barrier before calling the provided callable.
352     * This is useful to have a better attempt at multiple threads ran at the same time, but still is
353     * no guarantee since this is controlled by the thread scheduler.
354     * @param <V>
355     */
356    public final class ConcurrentCallable<V> implements Callable<V> {
357       private final Callable<? extends V> callable;
358       private final CyclicBarrier barrier;
359 
360       public ConcurrentCallable(Callable<? extends V> callable, CyclicBarrier barrier) {
361          this.callable = callable;
362          this.barrier = barrier;
363       }
364 
365       @Override
366       public V call() throws Exception {
367          log.trace("About to wait on provided barrier");
368          try {
369             barrier.await(10, TimeUnit.SECONDS);
370             log.trace("Completed await on provided barrier");
371          } catch (InterruptedException | TimeoutException | BrokenBarrierException e) {
372             log.tracef("Barrier await was broken due to exception", e);
373          }
374          return callable.call();
375       }
376    }

378    public final class RunnableWrapper implements Runnable {
379 
380       final Runnable realOne;
381 
382       public RunnableWrapper(Runnable realOne) {
383          this.realOne = realOne;
384       }
385 
386       @Override
387       public void run() {
388          try {
389             log.trace("Started fork runnable..");
390             realOne.run();
391             log.debug("Exiting fork runnable.");
392          } catch (Throwable e) {
393             log.debug("Exiting fork runnable due to exception", e);
394          }
395       }

411    private class LoggingCallable<T> implements Callable<T> {
412       private final Callable<? extends T> c;
413 
414       public LoggingCallable(Callable<? extends T> c) {
415          this.c = c;
416       }
417 
418       @Override
419       public T call() throws Exception {
420          try {
421             log.trace("Started fork callable..");
422             T result = c.call();
423             log.debug("Exiting fork callable.");
424             return result;
425          } catch (Exception e) {
426             log.debug("Exiting fork callable due to exception", e);
427             throw e;
428          }
