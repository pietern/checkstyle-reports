1   package org.infinispan.remoting.inboundhandler;
2   
3   import org.infinispan.commands.remote.CacheRpcCommand;
4   import org.infinispan.remoting.responses.CacheNotFoundResponse;
5   import org.infinispan.remoting.responses.Response;
6   
7   import java.util.concurrent.TimeUnit;
8   
9   /**
10   * The default {@link Runnable} for the remote commands receives.
11   * <p/>
12   * It checks the command topology and ensures that the topology higher or equal is installed in this node.
13   *
14   * @author Pedro Ruivo
15   * @since 8.0
16   */
17  public class DefaultTopologyRunnable extends BaseBlockingRunnable {
18  
19     private final TopologyMode topologyMode;
20     private final int commandTopologyId;
21  
22     public DefaultTopologyRunnable(BasePerCacheInboundInvocationHandler handler, CacheRpcCommand command, Reply reply,
23                                    TopologyMode topologyMode, int commandTopologyId) {
24        super(handler, command, reply);
25        this.topologyMode = topologyMode;
26        this.commandTopologyId = commandTopologyId;
27     }
28  
29     @Override
30     public boolean isReady() {
31        switch (topologyMode) {
32           case READY_TOPOLOGY:
33              return handler.getStateTransferLock().topologyReceived(waitTopology());
34           case READY_TX_DATA:
35              return handler.getStateTransferLock().transactionDataReceived(waitTopology());
36           default:
37              return true;
38        }
39     }
40  
41     @Override
42     protected Response beforeInvoke() throws Exception {
43        switch (topologyMode) {
44           case WAIT_TOPOLOGY:
45              handler.getStateTransferLock().waitForTopology(waitTopology(), 1, TimeUnit.DAYS);
46              break;
47           case WAIT_TX_DATA:
48              handler.getStateTransferLock().waitForTransactionData(waitTopology(), 1, TimeUnit.DAYS);
49              break;
50           default:
51              break;
52        }
53        if (handler.isCommandSentBeforeFirstTopology(commandTopologyId)) {
54           return CacheNotFoundResponse.INSTANCE;
55        }
56        return super.beforeInvoke();
57     }
58  
59     private int waitTopology() {
60        // Always wait for the first topology (i.e. for the join to finish)
61        return Math.max(commandTopologyId, 0);
62     }
63  
64  }
