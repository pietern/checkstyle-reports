5   /**
6    * Control's key grouping.
7    *
8    * @author Pete Muir

10  public interface GroupManager {

18  	String getGroup(Object key);

20     /**

22      *
23      * @param group the group name.

25      */
26     boolean isOwner(String group);
27  
28     /**
29      * It returns the primary owner of the group.
30      *
31      * @param group the group name.
32      * @return the primary owner of the group.
33      */
34     Address getPrimaryOwner(String group);
35  
36     /**
37      * It checks if this node is the primary owner of the group.
38      *
39      * @param group the group name.
40      * @return {@code true} if this node is the primary owner of the group, {@code false} otherwise.
41      */
42     boolean isPrimaryOwner(String group);

44  }
