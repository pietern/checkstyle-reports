1   package org.infinispan.container.entries;
2   
3   import org.infinispan.io.ExpandableMarshalledValueByteStream;
4   import org.infinispan.io.ImmutableMarshalledValueByteStream;
5   import org.infinispan.io.MarshalledValueByteStream;
6   import org.infinispan.marshall.core.MarshalledValue;
7   import sun.misc.Unsafe;
8   
9   /**
10   * Entry Size calculator that returns an approximation of how much memory a marshalled value contains for a key
11   * and value.  Note that this also handles any types that MarshalledValue doesn't as well.
12   * @author wburns
13   * @since 8.0
14   */
15  public class MarshalledValueEntrySizeCalculator extends PrimitiveEntrySizeCalculator {
16     @Override
17     protected long handleObject(Object object) {
18        Class<?> objClass = object.getClass();
19        if (objClass == MarshalledValue.class) {
20           MarshalledValue marshalledValue = (MarshalledValue)object;
21           // Marshalled value is an object and reference to class
22           long size = OBJECT_SIZE + POINTER_SIZE;
23           // Marshalled value has MarshalledValueByteStream & StreamingMarshaller instance fields and 2 ints
24           size = roundUpToNearest8(size + POINTER_SIZE * 2 + 4 * 2);
25           // We ignored the StreamingMarshaller instance as it is shared
26           MarshalledValueByteStream mvbs = marshalledValue.getRaw();
27           long mvbsSize;
28           if (mvbs instanceof ImmutableMarshalledValueByteStream) {
29              // Each object has the normal Object overhead plus an additional overhead to have a reference to its
30              // class definition
31              mvbsSize = OBJECT_SIZE + POINTER_SIZE;
32              // 1 variable in ImmutableMarshalledValueByteStream, which is a pointer to a byte[]
33              mvbsSize += POINTER_SIZE;
34              mvbsSize = roundUpToNearest8(mvbsSize);
35           } else if (mvbs instanceof ExpandableMarshalledValueByteStream) {
36              // Each object has the normal Object overhead plus an additional overhead to have a reference to its
37              // class definition
38              mvbsSize = OBJECT_SIZE + POINTER_SIZE;
39              // 3 variables in ExpandableMarshalledValueByteStream, 1 is a pointer to a byte[]
40              mvbsSize += POINTER_SIZE;
41              // 1: the byte[] is calculated below
42              // 2: the count as an int of how many bytes are written in the byte[]
43              mvbsSize += 4;
44              // 3: the doubling max size as an int that tells what the cutoff is before doing 25% increments versus 100%
45              mvbsSize += 4;
46              mvbsSize = roundUpToNearest8(mvbsSize);
47           } else {
48              throw new IllegalArgumentException(getClass() + " doesn't support counting for " + mvbs.getClass());
49           }
50           // Both implementations have a byte[] inside, so we need to add that to the size
51           byte[] bytes = mvbs.getRaw();
52           // Note that the byte[] takes up its own 8 aligned worth of size since it is a different instance
53           size += mvbsSize + roundUpToNearest8(Unsafe.ARRAY_BYTE_BASE_OFFSET +
54                   Unsafe.ARRAY_BYTE_INDEX_SCALE * bytes.length);
55           return size;
56        }
57        return super.handleObject(object);
58     }
59  }
