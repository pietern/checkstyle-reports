1   package org.infinispan.distribution;

3   import org.infinispan.container.entries.InternalCacheEntry;
4   import org.infinispan.factories.scopes.Scope;
5   import org.infinispan.factories.scopes.Scopes;
6   
7   /**
8    * Listener that is notified when a remote value is looked up
9    *
10   * @author William Burns
11   * @since 6.0
12   */
13  @Scope(Scopes.NAMED_CACHE)
14  public interface RemoteValueRetrievedListener {
15     /**
16      * Invoked when a remote value is found from a remote source
17      * @param ice The cache entry that was found
18      */
19     public void remoteValueFound(InternalCacheEntry ice);
20  
21     /**
22      * Invoked when a remote value is not found from the remote source for the given key
23      * @param key The key for which there was no value found
24      */
25     public void remoteValueNotFound(Object key);
26  }
