1   package org.infinispan.factories;
2   
3   import org.infinispan.configuration.cache.ClusteringConfiguration;
4   import org.infinispan.factories.annotations.DefaultFactoryFor;
5   import org.infinispan.remoting.ReplicationQueue;
6   import org.infinispan.remoting.ReplicationQueueImpl;
7   
8   /**
9    * Factory for ReplicationQueue.
10   *
11   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
12   * @since 4.0
13   */
14  @DefaultFactoryFor(classes = ReplicationQueue.class)
15  public class ReplicationQueueFactory extends AbstractNamedCacheComponentFactory implements AutoInstantiableFactory {
16     @Override
17     @SuppressWarnings("unchecked")
18     public <T> T construct(Class<T> componentType) {
19        ClusteringConfiguration clustering = configuration.clustering();
20        if ((!clustering.cacheMode().isSynchronous()) && clustering.async().useReplQueue()) {
21           ReplicationQueue replQueue = clustering.async().replQueue();
22           return replQueue != null ? componentType.cast(replQueue) : (T) new ReplicationQueueImpl();
23        } else {
24           return null;
25        }
26     }
27  }
