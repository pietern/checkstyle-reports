1   package org.infinispan.affinity;
2   
3   import org.infinispan.affinity.impl.KeyAffinityServiceImpl;
4   import org.infinispan.notifications.Listener;
5   import org.infinispan.notifications.cachelistener.annotation.TopologyChanged;
6   import org.infinispan.notifications.cachelistener.event.TopologyChangedEvent;
7   import org.infinispan.notifications.cachemanagerlistener.annotation.CacheStopped;
8   import org.infinispan.notifications.cachemanagerlistener.event.CacheStoppedEvent;
9   
10  /**
11  * Used for registering various cache notifications.
12  *
13  * @author Mircea.Markus@jboss.com
14  * @since 4.1
15  */
16  @Listener(sync = true)
17  public class ListenerRegistration {
18     private final KeyAffinityServiceImpl<?> keyAffinityService;
19  
20     public ListenerRegistration(KeyAffinityServiceImpl<?> keyAffinityService) {
21        this.keyAffinityService = keyAffinityService;
22     }
23  
24     @TopologyChanged
25     public void handleViewChange(TopologyChangedEvent<?, ?> tce) {
26        if (!tce.isPre()) keyAffinityService.handleViewChange(tce);
27     }
28  
29     @CacheStopped
30     public void handleCacheStopped(CacheStoppedEvent cse) {
31        keyAffinityService.handleCacheStopped(cse);
32     }
33  }
