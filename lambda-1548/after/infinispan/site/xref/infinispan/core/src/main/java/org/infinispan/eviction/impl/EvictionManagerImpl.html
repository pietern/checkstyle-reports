3   import net.jcip.annotations.ThreadSafe;
4   
5   import org.infinispan.container.entries.InternalCacheEntry;
6   import org.infinispan.context.InvocationContext;
7   import org.infinispan.context.impl.ImmutableContext;
8   import org.infinispan.eviction.EvictionManager;
9   import org.infinispan.expiration.ExpirationManager;
10  import org.infinispan.factories.annotations.Inject;
11  import org.infinispan.notifications.cachelistener.CacheNotifier;
12  
13  import java.util.Map;
14  
15  @ThreadSafe
16  public class EvictionManagerImpl<K, V> implements EvictionManager<K, V> {
17     // components to be injected
18     private CacheNotifier<K, V> cacheNotifier;
19  
20     @Inject
21     public void initialize(CacheNotifier<K, V> cacheNotifier) {
22        this.cacheNotifier = cacheNotifier;
23     }
24  
25     @Override
26     public void onEntryEviction(Map<? extends K, InternalCacheEntry<? extends K, ? extends V>> evicted) {
27        // don't reuse the threadlocal context as we don't want to include eviction
28        // operations in any ongoing transaction, nor be affected by flags
29        // especially see ISPN-1154: it's illegal to acquire locks in a committing transaction
30        InvocationContext ctx = ImmutableContext.INSTANCE;
31        // This is important because we make no external guarantees on the thread
32        // that will execute this code, so it could be the user thread, or could
33        // be the eviction thread.
34        // However, when a user calls cache.evict(), you do want to carry over the
35        // contextual information, hence it makes sense for the notifyCacheEntriesEvicted()
36        // call to carry on taking an InvocationContext object.
37        cacheNotifier.notifyCacheEntriesEvicted(evicted.values(), ctx, null);
38     }
39  }
