8   import org.infinispan.remoting.responses.ExceptionResponse;
9   import org.infinispan.remoting.responses.Response;
10  import org.infinispan.remoting.responses.SuccessfulResponse;

14  
15  /**
16   * Common transport-related behaviour
17   *
18   * @author Manik Surtani
19   * @version 4.2
20   */
21  public abstract class AbstractTransport implements Transport {
22  
23     protected GlobalConfiguration configuration;
24  
25     @Inject
26     @SuppressWarnings("unused")
27     public void setConfiguration(GlobalConfiguration globalConfiguration) {
28        this.configuration = globalConfiguration;
29     }
30  
31     public Response checkResponse(Object responseObject, Address sender, boolean ignoreCacheNotFoundResponse) {
32        Log log = getLog();
33        if (responseObject == null) {
34           return SuccessfulResponse.SUCCESSFUL_EMPTY_RESPONSE;
35        } else if (responseObject instanceof Response) {
36           Response response = (Response) responseObject;
37           if (response instanceof ExceptionResponse) {
38              ExceptionResponse exceptionResponse = (ExceptionResponse) response;
39              Exception e = exceptionResponse.getException();
40              if (e instanceof SuspectException) throw log.thirdPartySuspected(sender, (SuspectException) e);
41              if (e instanceof AvailabilityException || e instanceof OutdatedTopologyException) throw (CacheException) e;
42  
43              // if we have any application-level exceptions make sure we throw them!!
44              throw log.remoteException(sender, e);
45           } else if (!ignoreCacheNotFoundResponse && response instanceof CacheNotFoundResponse) {
46              throw new SuspectException("Cache not running on node " + sender, sender);
47           }
48           return response;
49        } else {
50           // All other responses should trigger an exception
51           Class<?> responseClass = responseObject.getClass();
52           log.tracef("Unexpected response object type from %s: %s", sender, responseClass);
53           throw new CacheException(String.format("Unexpected response object type from %s: %s", sender, responseClass));
54        }
55     }
56  
57  }
