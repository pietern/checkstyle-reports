24     @Override
25     protected FilterProcessingChain<Descriptor> createFilterProcessingChain() throws IOException, DescriptorParserException {
26        SerializationContext serCtx = ProtobufUtil.newSerializationContext(new Configuration.Builder().build());
27        MarshallerRegistration.registerMarshallers(serCtx);
28        EntityNamesResolver entityNamesResolver = new ProtobufEntityNamesResolver(serCtx);
29        ProtobufPropertyHelper protobufPropertyHelper = new ProtobufPropertyHelper(entityNamesResolver, serCtx);
30        return FilterProcessingChain.build(entityNamesResolver, protobufPropertyHelper, null);
31     }
32  
33     @Test
34     public void testParsingResult() throws Exception {
35        String jpaQuery = "from org.infinispan.objectfilter.test.model.Person p where p.name is not null";
36        FilterParsingResult<Descriptor> result = queryParser.parseQuery(jpaQuery, createFilterProcessingChain());
37  
38        assertNotNull(result.getWhereClause());
39  
40        assertEquals("org.infinispan.objectfilter.test.model.Person", result.getTargetEntityName());
41        assertEquals("org.infinispan.objectfilter.test.model.Person", result.getTargetEntityMetadata().getFullName());
42  
43        assertNull(result.getProjectedPaths());
44        assertNull(result.getSortFields());
45     }
