3   import org.hibernate.hql.ast.spi.predicate.DisjunctionPredicate;
4   import org.infinispan.objectfilter.impl.syntax.BooleanExpr;
5   import org.infinispan.objectfilter.impl.syntax.OrExpr;
6   
7   /**
8    * @author anistor@redhat.com
9    * @since 7.0
10   */
11  class FilterDisjunctionPredicate extends DisjunctionPredicate<BooleanExpr> {
12  
13     public FilterDisjunctionPredicate() {
14     }
15  
16     @Override
17     public BooleanExpr getQuery() {
18        // we always expect children.size() >= 1
19        if (children.isEmpty()) {
20           throw new IllegalStateException("A disjunction must have at least one child");
21        }
22        BooleanExpr firstChild = children.get(0).getQuery();
23        if (children.size() == 1) {
24           return firstChild;
25        }
26  
27        OrExpr orExpr = new OrExpr(firstChild);
28  
29        for (int i = 1; i < children.size(); i++) {
30           BooleanExpr child = children.get(i).getQuery();
31           orExpr.getChildren().add(child);
32        }
33  
34        return orExpr;
35     }
36  }
