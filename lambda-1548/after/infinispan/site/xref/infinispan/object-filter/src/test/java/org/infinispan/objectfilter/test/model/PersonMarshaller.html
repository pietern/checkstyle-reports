1   package org.infinispan.objectfilter.test.model;
2   
3   import org.infinispan.protostream.MessageMarshaller;
4   
5   import java.io.IOException;
6   import java.util.ArrayList;
7   import java.util.Date;
8   
9   /**
10   * @author anistor@redhat.com
11   * @since 7.0
12   */
13  public class PersonMarshaller implements MessageMarshaller<Person> {
14  
15     @Override
16     public Person readFrom(ProtoStreamReader reader) throws IOException {
17        Person person = new Person();
18        person.setId(reader.readInt("id"));
19        person.setName(reader.readString("name"));
20        person.setSurname(reader.readString("surname"));
21        person.setAddress(reader.readObject("address", Address.class));
22        person.setPhoneNumbers(reader.readCollection("phoneNumbers", new ArrayList<PhoneNumber>(), PhoneNumber.class));
23        person.setAge(reader.readInt("age"));
24        person.setFavouriteNumbers(reader.readCollection("favouriteNumbers", new ArrayList<Integer>(), Integer.class));
25        person.setLicense(reader.readString("license"));
26        person.setGender(reader.readObject("gender", Person.Gender.class));
27        person.setLastUpdate(new Date(reader.readLong("lastUpdate")));
28        person.setDeleted(reader.readBoolean("deleted"));
29        return person;
30     }
31  
32     @Override
33     public void writeTo(ProtoStreamWriter writer, Person person) throws IOException {
34        writer.writeInt("id", person.getId());
35        writer.writeString("name", person.getName());
36        writer.writeString("surname", person.getSurname());
37        writer.writeObject("address", person.getAddress(), Address.class);
38        writer.writeCollection("phoneNumbers", person.getPhoneNumbers(), PhoneNumber.class);
39        writer.writeInt("age", person.getAge());
40        writer.writeCollection("favouriteNumbers", person.getFavouriteNumbers(), Integer.class);
41        writer.writeString("license", person.getLicense());
42        writer.writeObject("gender", person.getGender(), Person.Gender.class);
43        if (person.getLastUpdate() != null) {
44           writer.writeLong("lastUpdate", person.getLastUpdate().getTime());
45        }
46        writer.writeBoolean("deleted", person.isDeleted());
47     }
48  
49     @Override
50     public Class<Person> getJavaClass() {
51        return Person.class;
52     }
53  
54     @Override
55     public String getTypeName() {
56        return "org.infinispan.objectfilter.test.model.Person";
57     }
58  }
