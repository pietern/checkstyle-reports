1   package org.infinispan.objectfilter.impl.hql.predicate;
2   
3   import org.hibernate.hql.ast.spi.EntityNamesResolver;
4   import org.hibernate.hql.ast.spi.predicate.*;
5   import org.hibernate.hql.ast.spi.predicate.ComparisonPredicate.Type;
6   import org.infinispan.objectfilter.impl.hql.ObjectPropertyHelper;
7   import org.infinispan.objectfilter.impl.syntax.BooleanExpr;
8   import org.infinispan.objectfilter.impl.syntax.PropertyValueExpr;
9   import org.infinispan.objectfilter.impl.syntax.ValueExpr;
10  
11  import java.util.List;
12  
13  /**
14   * @author anistor@redhat.com
15   * @since 7.0
16   */
17  public final class FilterPredicateFactory implements PredicateFactory<BooleanExpr> {
18  
19     private final EntityNamesResolver entityNamesResolver;
20  
21     private final ObjectPropertyHelper propertyHelper;
22  
23     public FilterPredicateFactory(EntityNamesResolver entityNamesResolver, ObjectPropertyHelper propertyHelper) {
24        this.entityNamesResolver = entityNamesResolver;
25        this.propertyHelper = propertyHelper;
26     }
27  
28     @Override
29     public RootPredicate<BooleanExpr> getRootPredicate(String entityType) {
30        if (entityNamesResolver.getClassFromName(entityType) == null) {
31           throw new IllegalStateException("Unknown entity name " + entityType);
32        }
33        return new FilterRootPredicate();
34     }
35  
36     @Override
37     public ComparisonPredicate<BooleanExpr> getComparisonPredicate(String entityType, Type comparisonType,
38                                                                    List<String> propertyPath, Object comparisonValue) {
39        ValueExpr valueExpr = new PropertyValueExpr(propertyPath, propertyHelper.isRepeatedProperty(entityType, propertyPath));
40        return new FilterComparisonPredicate(valueExpr, comparisonType, comparisonValue);
41     }
42  
43     @Override
44     public InPredicate<BooleanExpr> getInPredicate(String entityType, List<String> propertyPath, List<Object> values) {
45        ValueExpr valueExpr = new PropertyValueExpr(propertyPath, propertyHelper.isRepeatedProperty(entityType, propertyPath));
46        return new FilterInPredicate(valueExpr, values);
47     }
48  
49     @Override
50     public RangePredicate<BooleanExpr> getRangePredicate(String entityType, List<String> propertyPath,
51                                                          Object lowerValue, Object upperValue) {
52        ValueExpr valueExpr = new PropertyValueExpr(propertyPath, propertyHelper.isRepeatedProperty(entityType, propertyPath));
53        return new FilterRangePredicate(valueExpr, lowerValue, upperValue);
54     }
55  
56     @Override
57     public NegationPredicate<BooleanExpr> getNegationPredicate() {
58        return new FilterNegationPredicate();
59     }
60  
61     @Override
62     public DisjunctionPredicate<BooleanExpr> getDisjunctionPredicate() {
63        return new FilterDisjunctionPredicate();
64     }
65  
66     @Override
67     public ConjunctionPredicate<BooleanExpr> getConjunctionPredicate() {
68        return new FilterConjunctionPredicate();
69     }
70  
71     @Override
72     public LikePredicate<BooleanExpr> getLikePredicate(String entityType, List<String> propertyPath,
73                                                        String patternValue, Character escapeCharacter) {
74        ValueExpr valueExpr = new PropertyValueExpr(propertyPath, propertyHelper.isRepeatedProperty(entityType, propertyPath));
75        return new FilterLikePredicate(valueExpr, patternValue, escapeCharacter);
76     }
77  
78     @Override
79     public IsNullPredicate<BooleanExpr> getIsNullPredicate(String entityType, List<String> propertyPath) {
80        ValueExpr valueExpr = new PropertyValueExpr(propertyPath, propertyHelper.isRepeatedProperty(entityType, propertyPath));
81        return new FilterIsNullPredicate(valueExpr);
82     }
83  }
