17  public final class ReflectionPropertyHelper extends ObjectPropertyHelper<Class<?>> {
18  
19     private static final Log log = Logger.getMessageLogger(Log.class, ReflectionPropertyHelper.class.getName());
20  
21     private static final Map<Class<?>, Class<?>> primitives = new HashMap<Class<?>, Class<?>>();
22  
23     static {
24        primitives.put(java.util.Date.class, java.util.Date.class);
25        primitives.put(String.class, String.class);
26        primitives.put(Character.class, Character.class);
27        primitives.put(char.class, Character.class);
28        primitives.put(Double.class, Double.class);
29        primitives.put(double.class, Double.class);
30        primitives.put(Float.class, Float.class);
31        primitives.put(float.class, Float.class);
32        primitives.put(Long.class, Long.class);
33        primitives.put(long.class, Long.class);
34        primitives.put(Integer.class, Integer.class);
35        primitives.put(int.class, Integer.class);
36        primitives.put(Short.class, Short.class);
37        primitives.put(short.class, Short.class);
38        primitives.put(Byte.class, Byte.class);
39        primitives.put(byte.class, Byte.class);
40        primitives.put(Boolean.class, Boolean.class);
41        primitives.put(boolean.class, Boolean.class);
42     }
43  
44     public ReflectionPropertyHelper(EntityNamesResolver entityNamesResolver) {
45        super(entityNamesResolver);
46     }
47  
48     @Override
49     public Class<?> getEntityMetadata(String targetTypeName) {
50        return entityNamesResolver.getClassFromName(targetTypeName);
51     }
52  
53     @Override
54     public Class<?> getPrimitivePropertyType(String entityType, List<String> propertyPath) {
55        Class<?> type = entityNamesResolver.getClassFromName(entityType);
56        if (type == null) {
57           throw new IllegalStateException("Unknown entity name " + entityType);
58        }
59  
60        Class<?> propType = getPropertyAccessor(type, propertyPath).getPropertyType();
61        if (propType.isEnum()) {
62           return propType;
63        }
64        if (primitives.containsKey(propType)) {
65           return primitives.get(propType);
66        }
67        return null;
68     }
69  
70     @Override
71     public boolean hasProperty(String entityType, List<String> propertyPath) {
72        return hasProperty(entityType, propertyPath.toArray(new String[propertyPath.size()]));
73     }
74  
75     @Override
76     public boolean hasEmbeddedProperty(String entityType, List<String> propertyPath) {
77        Class<?> entity = entityNamesResolver.getClassFromName(entityType);
78        if (entity == null) {
79           throw new IllegalStateException("Unknown entity name " + entityType);
80        }
81  
82        try {
83           Class<?> propType = getPropertyAccessor(entity, propertyPath).getPropertyType();
84           return propType != null && !propType.isEnum() && !primitives.containsKey(propType);
85        } catch (Exception e) {
86           return false; // todo [anistor] need clean solution
87        }
88     }
89  
90     @Override
91     public boolean isRepeatedProperty(String entityType, List<String> propertyPath) {
92        Class<?> entity = entityNamesResolver.getClassFromName(entityType);
93        if (entity == null) {
94           throw new IllegalStateException("Unknown entity name " + entityType);
95        }
96        ReflectionHelper.PropertyAccessor a = ReflectionHelper.getAccessor(entity, propertyPath.get(0));
97        if (a.isMultiple()) {
98           return true;
99        }
100       for (int i = 1; i < propertyPath.size(); i++) {
101          a = a.getAccessor(propertyPath.get(i));
102          if (a.isMultiple()) {
103             return true;
104          }
105       }
106       return false;
107    }
108 
109    private boolean hasProperty(String entityType, String... propertyPath) {
110       Class<?> entity = entityNamesResolver.getClassFromName(entityType);
111       if (entity == null) {
112          throw new IllegalStateException("Unknown entity name " + entityType);
113       }
114 
115       try {
116          Class<?> propType = getPropertyAccessor(entity, Arrays.asList(propertyPath)).getPropertyType();
117          return propType != null;
118       } catch (Exception e) {
119          return false; // todo [anistor] need clean solution
120       }
121    }
122 
123    private ReflectionHelper.PropertyAccessor getPropertyAccessor(Class<?> entityClass, List<String> propertyPath) {
124       ReflectionHelper.PropertyAccessor accessor = ReflectionHelper.getAccessor(entityClass, propertyPath.get(0));
125       for (int i = 1; i < propertyPath.size(); i++) {
126          accessor = accessor.getAccessor(propertyPath.get(i));
127       }
128       return accessor;
129    }
