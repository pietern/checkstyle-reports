6   /**
7    * An expression that applies a boolean operator (OR, AND) to a list of boolean sub-expressions.
8    *
9    * @author anistor@redhat.com
10   * @since 7.0
11   */
12  public abstract class BooleanOperatorExpr implements BooleanExpr {
13  
14     protected final List<BooleanExpr> children = new ArrayList<BooleanExpr>();
15  
16     protected BooleanOperatorExpr(BooleanExpr... children) {
17        for (BooleanExpr child : children) {
18           this.children.add(child);
19        }
20     }
21  
22     protected BooleanOperatorExpr(List<BooleanExpr> children) {
23        for (BooleanExpr child : children) {
24           this.children.add(child);
25        }
26     }
27  
28     public List<BooleanExpr> getChildren() {
29        return children;
30     }
31  }
