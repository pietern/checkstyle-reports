5   import java.util.Map;

7   /**
8    * An object matcher able to test a given object against multiple registered filters specified either as JPA queries or
9    * using the query DSL (see {@link org.infinispan.query.dsl}). The matching filters are notified via a callback supplied
10   * when registering the filter. The filter will have to specify the fully qualified type name of the matching
11   * object/entity because simple names cannot be resolved as it would happen in the case of an {@link
12   * javax.persistence.EntityManager} which has knowledge of all types in advance.
13   *
14   * @author anistor@redhat.com
15   * @since 7.0
16   */
17  public interface Matcher {
18  
19     FilterSubscription registerFilter(Query query, FilterCallback callback, Object... eventType);
20  
21     FilterSubscription registerFilter(String jpaQuery, Map<String, Object> namedParameters, FilterCallback callback, Object... eventType);
22  
23     FilterSubscription registerFilter(String jpaQuery, FilterCallback callback, Object... eventType);
24  
25     void unregisterFilter(FilterSubscription filterSubscription);
26  
27     /**
28      * Test the given instance against all the subscribed filters and notify all callbacks registered for instances of
29      * the same type.
30      *
31      * @param userContext an optional user provided object to be passed to matching subscribers along with the matching
32      *                    instance; can be {@code null}
33      * @param instance    the object to test against the registered filters; never {@code null}
34      * @param eventType   on optional event type discriminator that is matched against the even type specified when the
35      *                    filter was registered; can be {@code null}
36      */
37     void match(Object userContext, Object instance, Object eventType);
38  
39     /**
40      * Obtains an ObjectFilter instance that is capable of testing a single filter condition.
41      *
42      * @param filterSubscription a filter subscription previously registered with this Matcher; the newly created
43      *                           ObjectFilter will be based on the same filter condition
44      * @return the single-filter
45      */
46     ObjectFilter getObjectFilter(FilterSubscription filterSubscription);
47  
48     ObjectFilter getObjectFilter(Query query);
49  
50     ObjectFilter getObjectFilter(String jpaQuery);
51  
52     ObjectFilter getObjectFilter(String jpaQuery, Map<String, Object> namedParameters);
53  }
