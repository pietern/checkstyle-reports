1   package org.infinispan.objectfilter;
2   
3   import java.util.Comparator;
4   
5   /**
6    * A filter that tests if an object matches a pre-defined condition and returns either the original instance or the
7    * projection, depending on how the filter was created. The projection is represented as an Object[]. If the given
8    * instance does not match the filter will just return null.
9    *
10   * @author anistor@redhat.com
11   * @since 7.0
12   */
13  public interface ObjectFilter {
14  
15     /**
16      * The fully qualified entity type name accepted by this filter.
17      */
18     String getEntityTypeName();
19  
20     /**
21      * The array of '.' separated path names of the projected fields if any, or {@code null} otherwise.
22      */
23     String[] getProjection();
24  
25     /**
26      * The array of sort specifications if defined, or {@code null} otherwise.
27      */
28     SortField[] getSortFields();
29  
30     /**
31      * The comparator corresponding to the 'order by' clause, if any.
32      *
33      * @return the Comparator or {@code null} if no 'order by' was specified
34      */
35     Comparator<Comparable[]> getComparator();
36  
37     /**
38      * Tests if an object instance matches the filter.
39      *
40      * @param instance the instance to test; this is never {@code null}
41      * @return a {@code FilterResult} if there is a match or {@code null} otherwise
42      */
43     FilterResult filter(Object instance);
44  
45     /**
46      * The output of the {@link ObjectFilter#filter} method.
47      */
48     interface FilterResult {
49  
50        /**
51         * Returns the matched object. This is non-null unless projections are present.
52         */
53        Object getInstance();
54  
55        /**
56         * Returns the projection, if a projection was requested or {@code null} otherwise.
57         */
58        Object[] getProjection();
59  
60        /**
61         * Returns the projection of fields used for sorting, if sorting was requested or {@code null} otherwise.
62         */
63        Comparable[] getSortProjection();
64     }
65  }
