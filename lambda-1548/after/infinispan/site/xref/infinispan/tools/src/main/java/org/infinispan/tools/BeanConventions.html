14     public static String getPropertyFromBeanConvention(Method method) {
15        String getterOrSetter = method.getName();
16        if (getterOrSetter.startsWith("get") || getterOrSetter.startsWith("set")) {
17           String withoutGet = getterOrSetter.substring(4);
18           // not specifically BEAN convention, but this is what is bound in JMX.
19           return Character.toUpperCase(getterOrSetter.charAt(3)) + withoutGet;
20        } else if (getterOrSetter.startsWith("is")) {
21           String withoutIs = getterOrSetter.substring(3);
22           return Character.toUpperCase(getterOrSetter.charAt(2)) + withoutIs;
23        }
24        return getterOrSetter;
25     }
26  
27     public static String getPropertyFromBeanConvention(Field field) {
28        String fieldName = field.getName();
29        String withoutFirstChar = fieldName.substring(1);
30        return Character.toUpperCase(fieldName.charAt(0)) + withoutFirstChar;
31     }

33  }
