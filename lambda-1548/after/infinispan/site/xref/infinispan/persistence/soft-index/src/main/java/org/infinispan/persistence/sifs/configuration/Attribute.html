1   package org.infinispan.persistence.sifs.configuration;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   
6   /**
7    * Enumerates the attributes used by the LevelDB cache stores configuration
8    * 
9    * @author <a href="mailto:rtsang@redhat.com">Ray Tsang</a>
10   */
11  public enum Attribute {
12     // must be first
13     UNKNOWN(null),
14     COMPACTION_THRESHOLD("compaction-threshold"),
15     PATH("path"),
16     INDEX_QUEUE_LENGTH("max-queue-length"),
17     SEGMENTS("segments"),
18     MAX_FILE_SIZE("max-file-size"),
19     MAX_NODE_SIZE("max-node-size"),
20     MIN_NODE_SIZE("min-node-size"),
21     OPEN_FILES_LIMIT("open-files-limit"),
22     SYNC_WRITES("sync-writes")
23     ;
24  
25     private final String name;
26  
27     private Attribute(final String name) {
28        this.name = name;
29     }
30  
31     /**
32      * Get the local name of this element.
33      * 
34      * @return the local name
35      */
36     public String getLocalName() {
37        return name;
38     }
39  
40     private static final Map<String, Attribute> attributes;
41  
42     static {
43        final Map<String, Attribute> map = new HashMap<String, Attribute>(64);
44        for (Attribute attribute : values()) {
45           final String name = attribute.getLocalName();
46           if (name != null) {
47              map.put(name, attribute);
48           }
49        }
50        attributes = map;
51     }
52  
53     public static Attribute forName(final String localName) {
54        final Attribute attribute = attributes.get(localName);
55        return attribute == null ? UNKNOWN : attribute;
56     }
57  }
