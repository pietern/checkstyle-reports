1   package org.infinispan.cli;
2   
3   import java.util.Collections;
4   import java.util.HashMap;
5   import java.util.Map;
6   import java.util.Set;
7   
8   import org.infinispan.cli.commands.Command;
9   import org.infinispan.commons.util.ServiceFinder;
10  
11  public class CommandRegistry {
12     private Map<String, Command> commands;
13  
14     public CommandRegistry() {
15        commands = new HashMap<String, Command>();
16        for (Command cmd : ServiceFinder.load(Command.class)) {
17           String name = cmd.getName();
18           if (commands.containsKey(name)) {
19              throw new RuntimeException("Command " + cmd.getClass().getName() + " overrides "
20                    + commands.get(name).getClass().getName());
21           }
22           commands.put(name, cmd);
23        }
24     }
25  
26     public Set<String> getCommandNames() {
27        return Collections.unmodifiableSet(commands.keySet());
28     }
29  
30     public Command getCommand(String name) {
31        return commands.get(name);
32     }
33  
34  }
