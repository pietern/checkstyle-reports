1   package org.infinispan.stats.container;

3   import org.infinispan.stats.ExtendedStatisticNotFoundException;
4   
5   import java.io.PrintStream;
6   
7   /**
8    * Common logic for the {@link ExtendedStatisticsContainer}
9    *
10   * @author Pedro Ruivo
11   * @since 6.0
12   */
13  public abstract class BaseExtendedStatisticsContainer implements ExtendedStatisticsContainer {
14  
15     protected final double[] stats;
16  
17     protected BaseExtendedStatisticsContainer(int size) {
18        stats = new double[size];
19     }
20  
21     @Override
22     public final void addValue(ExtendedStatistic statistic, double value) {
23        int index = getIndex(statistic);
24        if (index != ExtendedStatistic.NO_INDEX) {
25           stats[index] += value;
26        }
27     }
28  
29     @Override
30     public final double getValue(ExtendedStatistic statistic) throws ExtendedStatisticNotFoundException {
31        int index = getIndex(statistic);
32        if (index != ExtendedStatistic.NO_INDEX) {
33           return stats[index];
34        }
35        throw new ExtendedStatisticNotFoundException(statistic + " not found in " + this);
36     }
37  
38     /**
39      * @return the index of the statistic in this container. It return {@link ExtendedStatistic#NO_INDEX} if the
40      *         statistic does not exist in this container
41      */
42     protected abstract int getIndex(ExtendedStatistic statistic);
43  }
