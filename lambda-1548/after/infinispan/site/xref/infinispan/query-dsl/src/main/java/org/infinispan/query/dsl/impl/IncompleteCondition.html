1   package org.infinispan.query.dsl.impl;
2   
3   import org.infinispan.query.dsl.Expression;
4   import org.infinispan.query.dsl.FilterConditionBeginContext;
5   import org.infinispan.query.dsl.FilterConditionContext;
6   import org.infinispan.query.dsl.FilterConditionEndContext;
7   import org.infinispan.query.dsl.QueryBuilder;
8   import org.infinispan.query.dsl.QueryFactory;
9   
10  /**
11   * @author anistor@redhat.com
12   * @since 6.0
13   */
14  class IncompleteCondition extends BaseCondition implements FilterConditionBeginContext {
15  
16     private boolean isNegated = false;
17  
18     private BaseCondition filterCondition;
19  
20     IncompleteCondition(QueryFactory queryFactory) {
21        super(queryFactory);
22     }
23  
24     @Override
25     void setQueryBuilder(QueryBuilder queryBuilder) {
26        super.setQueryBuilder(queryBuilder);
27        if (filterCondition != null) {
28           filterCondition.setQueryBuilder(queryBuilder);
29        }
30     }
31  
32     @Override
33     public <ReturnType> ReturnType accept(Visitor<ReturnType> visitor) {
34        if (filterCondition == null) {
35           throw new IllegalStateException("Cannot visit an incomplete condition.");
36        }
37  
38        return filterCondition.accept(visitor);
39     }
40  
41     @Override
42     public FilterConditionEndContext having(Expression expression) {
43        if (filterCondition != null) {
44           throw new IllegalStateException("Sentence already started. Cannot use 'having(..)' again.");
45        }
46        AttributeCondition attributeCondition = new AttributeCondition(queryFactory, expression);
47        attributeCondition.setNegated(isNegated);
48        attributeCondition.setQueryBuilder(queryBuilder);
49        attributeCondition.setParent(this);
50        filterCondition = attributeCondition;
51        return attributeCondition;
52     }
53  
54     @Override
55     public FilterConditionEndContext having(String attributePath) {
56        return having(Expression.property(attributePath));
57     }
58  
59     @Override
60     public FilterConditionBeginContext not() {
61        if (filterCondition != null) {
62           throw new IllegalStateException("Sentence already started. Cannot use 'not()' again.");
63        }
64  
65        isNegated = !isNegated;
66        return this;
67     }
68  
69     @Override
70     public FilterConditionContext not(FilterConditionContext fcc) {
71        if (fcc == null) {
72           throw new IllegalArgumentException("Argument cannot be null");
73        }
74  
75        if (filterCondition != null) {
76           throw new IllegalStateException("Sentence already started. Cannot use 'not(..)' again.");
77        }
78  
79        BaseCondition baseCondition = ((BaseCondition) fcc).getRoot();
80        if (baseCondition.queryFactory != queryFactory) {
81           throw new IllegalArgumentException("The given condition was created by a different factory");
82        }
83        if (baseCondition.queryBuilder != null) {
84           throw new IllegalArgumentException("The given condition is already in use by another builder");
85        }
86  
87        isNegated = !isNegated;
88        if (isNegated) {
89           NotCondition notCondition = new NotCondition(queryFactory, baseCondition);
90           notCondition.setQueryBuilder(queryBuilder);
91           filterCondition = notCondition;
92        } else {
93           baseCondition.setQueryBuilder(queryBuilder);
94           filterCondition = baseCondition;
95        }
96        return filterCondition;
97     }
98  }
