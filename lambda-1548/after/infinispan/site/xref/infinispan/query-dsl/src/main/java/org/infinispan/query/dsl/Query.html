6   /**
7    * An immutable object representing both the query and the result. The result is obtained lazily when one of the methods
8    * in this interface is executed first time. The query is executed only once. Further calls will just return the
9    * previously cached results. If you intend to re-execute the query to obtain fresh data you need to build another
10   * instance using a {@link QueryBuilder}.
11   *
12   * @author anistor@redhat.com
13   * @since 6.0
14   */
15  public interface Query {
16  
17     Query setParameter(String paramName, Object paramValue);
18  
19     Query setParameters(Map<String, Object> paramValues);
20  
21     /**
22      * Returns the results of a search as a list.
23      *
24      * @return list of objects that were found from the search.
25      */
26     <T> List<T> list();
27  
28     /**
29      * Gets the total number of results matching the query, ignoring pagination (firstResult, maxResult).
30      *
31      * @return total number of results.
32      */
33     int getResultSize(); //todo [anistor] this should probably be a long?

35     //todo [anistor] also add long getStartOffset() ?
