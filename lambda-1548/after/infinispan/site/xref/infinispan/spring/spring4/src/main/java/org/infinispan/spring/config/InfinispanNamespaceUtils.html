18     public static BeanComponentDefinition parseInnerBeanDefinition(Element element, ParserContext parserContext) {
19        List<Element> childElements = DomUtils.getChildElementsByTagName(element, "bean");
20        BeanComponentDefinition innerComponentDefinition = null;
21        if (childElements != null && childElements.size() == 1) {
22           Element beanElement = childElements.get(0);
23           if (!"http://www.springframework.org/schema/beans".equals(beanElement.getNamespaceURI())) {
24              throw new IllegalStateException("Illegal inner child element");
25           }
26           BeanDefinitionParserDelegate delegate = parserContext.getDelegate();
27           BeanDefinitionHolder beanDefinitionHolder = delegate.parseBeanDefinitionElement(beanElement);
28           beanDefinitionHolder = delegate.decorateBeanDefinitionIfRequired(beanElement, beanDefinitionHolder);
29           BeanDefinition beanDefinition = beanDefinitionHolder.getBeanDefinition();
30           innerComponentDefinition = new BeanComponentDefinition(beanDefinition, beanDefinitionHolder.getBeanName());
31        }
32        return innerComponentDefinition;
33     }
