35  public final class JPAContinuousQueryProtobufCacheEventFilterConverter extends AbstractCacheEventFilterConverter<Object, Object, byte[]> {
36  
37     /**
38      * The JPA query to execute.
39      */
40     private final String jpaQuery;
41  
42     private final Map<String, Object> namedParameters;
43  
44     /**
45      * The implementation class of the Matcher component to lookup and use.
46      */
47     private final Class<? extends Matcher> matcherImplClass;
48  
49     private transient SerializationContext serCtx;
50  
51     /**
52      * Optional cache for query objects.
53      */
54     private transient QueryCache queryCache;
55  
56     /**
57      * The Matcher, acquired via dependency injection.
58      */
59     private transient Matcher matcher;
60  
61     /**
62      * The ObjectFilter is created lazily.
63      */
64     private transient ObjectFilter objectFilter;
65  
66     private boolean usesValueWrapper;
67  
68     public JPAContinuousQueryProtobufCacheEventFilterConverter(String jpaQuery,  Map<String, Object> namedParameters, Class<? extends Matcher> matcherImplClass) {
69        if (jpaQuery == null || matcherImplClass == null) {
70           throw new IllegalArgumentException("Arguments cannot be null");
71        }
72        this.jpaQuery = jpaQuery;
73        this.namedParameters = namedParameters;
74        this.matcherImplClass = matcherImplClass;
75     }
76  
77     @Inject
78     protected void injectDependencies(Cache cache) {
79        serCtx = ProtobufMetadataManagerImpl.getSerializationContextInternal(cache.getCacheManager());
80        queryCache = cache.getCacheManager().getGlobalComponentRegistry().getComponent(QueryCache.class);
81        matcher = cache.getAdvancedCache().getComponentRegistry().getComponent(matcherImplClass);
82        if (matcher == null) {
83           throw new CacheException("Expected component not found in registry: " + matcherImplClass.getName());
84        }
85        Configuration cfg = cache.getCacheConfiguration();
86        usesValueWrapper = cfg.indexing().index().isEnabled() && !cfg.compatibility().enabled();
87     }
88  
89     private ObjectFilter getObjectFilter() {
90        if (objectFilter == null) {
91           // if parameters are present caching cannot be currently performed due to internal implementation limitations
92           if (queryCache != null && (namedParameters == null || namedParameters.isEmpty())) {
93              KeyValuePair<String, Class> queryCacheKey = new KeyValuePair<String, Class>(jpaQuery, matcherImplClass);
94              ObjectFilter objectFilter = queryCache.get(queryCacheKey);
95              if (objectFilter == null) {
96                 objectFilter = matcher.getObjectFilter(jpaQuery, namedParameters);
97                 queryCache.put(queryCacheKey, objectFilter);
98              }
99              this.objectFilter = objectFilter;
100          } else {
101             objectFilter = matcher.getObjectFilter(jpaQuery, namedParameters);
102          }
103       }
104       return objectFilter;
105    }
106 
107    private ObjectFilter.FilterResult filterAndConvert(Object value) {
108       if (value == null) {
109          return null;
110       }
111       return getObjectFilter().filter(value);
112    }
113 
114    @Override
115    public byte[] filterAndConvert(Object key, Object oldValue, Metadata oldMetadata, Object newValue, Metadata newMetadata, EventType eventType) {
116       if (usesValueWrapper) {
117          if (oldValue instanceof ProtobufValueWrapper) {
118             oldValue = ((ProtobufValueWrapper) oldValue).getBinary();
119          }
120          if (newValue instanceof ProtobufValueWrapper) {
121             newValue = ((ProtobufValueWrapper) newValue).getBinary();
122          }
123       }
124 
125       if (eventType.isExpired()) {  // expired events have the expired value as newValue
126          oldValue = newValue;
127          newValue = null;
128       }
129 
130       ObjectFilter.FilterResult f1 = filterAndConvert(oldValue);
131       ObjectFilter.FilterResult f2 = filterAndConvert(newValue);
132       ContinuousQueryResult result;
133       if (f1 == null && f2 != null) {
134          result = new ContinuousQueryResult(true, (byte[]) key, (byte[]) newValue);
135       }
136       else if (f1 != null && f2 == null) {
137          result = new ContinuousQueryResult(false, (byte[]) key, null);
138       } else {
139          return null;
140       }
141       try {
142          return ProtobufUtil.toByteArray(serCtx, result);
143       } catch (IOException e) {
144          throw new RuntimeException(e);
145       }
146    }

148    public static final class Externalizer extends AbstractExternalizer<JPAContinuousQueryProtobufCacheEventFilterConverter> {
149 
150       @Override
151       public void writeObject(ObjectOutput output, JPAContinuousQueryProtobufCacheEventFilterConverter filterAndConverter) throws IOException {
152          output.writeUTF(filterAndConverter.jpaQuery);
153          Map<String, Object> namedParameters = filterAndConverter.namedParameters;
154          if (namedParameters != null) {
155             UnsignedNumeric.writeUnsignedInt(output, namedParameters.size());
156             for (Map.Entry<String, Object> e : namedParameters.entrySet()) {
157                output.writeUTF(e.getKey());
158                output.writeObject(e.getValue());
159             }
160          } else {
161             UnsignedNumeric.writeUnsignedInt(output, 0);
162          }
163          output.writeObject(filterAndConverter.matcherImplClass);
164       }
165 
166       @Override
167       public JPAContinuousQueryProtobufCacheEventFilterConverter readObject(ObjectInput input) throws IOException, ClassNotFoundException {
168          String jpaQuery = input.readUTF();
169          int paramsSize = UnsignedNumeric.readUnsignedInt(input);
170          Map<String, Object> namedParameters = null;
171          if (paramsSize != 0) {
172             namedParameters = new HashMap<String, Object>(paramsSize);
173             for (int i = 0; i < paramsSize; i++) {
174                String paramName = input.readUTF();
175                Object paramValue = input.readObject();
176                namedParameters.put(paramName, paramValue);
177             }
178          }
179          Class<? extends Matcher> matcherImplClass = (Class<? extends Matcher>) input.readObject();
180          return new JPAContinuousQueryProtobufCacheEventFilterConverter(jpaQuery, namedParameters, matcherImplClass);
181       }
182 
183       @Override
184       public Integer getId() {
185          return ExternalizerIds.JPA_CONTINUOUS_QUERY_CACHE_EVENT_FILTER_CONVERTER;
186       }
187 
188       @Override
189       public Set<Class<? extends JPAContinuousQueryProtobufCacheEventFilterConverter>> getTypeClasses() {
190          return Collections.<Class<? extends JPAContinuousQueryProtobufCacheEventFilterConverter>>singleton(JPAContinuousQueryProtobufCacheEventFilterConverter.class);
191       }
192    }
