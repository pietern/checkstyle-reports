3   import org.infinispan.AdvancedCache;

5   import org.infinispan.factories.ComponentRegistry;

10  import java.security.AccessController;
11  import java.security.PrivilegedAction;
12  
13  /**
14   * SecurityActions for the org.infinispan.query.remote package. Do not move and do not change class and method
15   * visibility!
16   *
17   * @author anistor@redhat.com
18   * @since 7.2
19   */
20  final class SecurityActions {
21  
22     static <T> T doPrivileged(PrivilegedAction<T> action) {
23        return System.getSecurityManager() != null ?
24              AccessController.doPrivileged(action) : Security.doPrivileged(action);
25     }
26  
27     static Configuration getCacheConfiguration(AdvancedCache<?, ?> cache) {
28        return doPrivileged(new GetCacheConfigurationAction(cache));
29     }
30  
31     static ComponentRegistry getCacheComponentRegistry(AdvancedCache<?, ?> cache) {
32        return doPrivileged(new GetCacheComponentRegistryAction(cache));
33     }
34  }
