27  /**
28   * A query facade implementation for both Lucene based queries and non-indexed queries.
29   *
30   * @author anistor@redhat.com
31   * @since 6.0
32   */
33  @MetaInfServices
34  public final class QueryFacadeImpl implements QueryFacade {
35  
36     private static final Log log = LogFactory.getLog(QueryFacadeImpl.class, Log.class);
37  
38     /**
39      * A special hidden Lucene document field that holds the actual protobuf type name.
40      */
41     public static final String TYPE_FIELD_NAME = "$type$";
42  
43     /**
44      * A special placeholder value that is indexed if the actual field value is null. This placeholder is needed because
45      * Lucene does not index null values.
46      */
47     public static final String NULL_TOKEN = "_null_";
48  
49     public static final NullMarkerCodec NULL_TOKEN_CODEC = new KeywordBasedNullCodec(NULL_TOKEN);
50  
51     @Override
52     public byte[] query(AdvancedCache<byte[], byte[]> cache, byte[] query) {
53        try {
54           SerializationContext serCtx = ProtobufMetadataManagerImpl.getSerializationContextInternal(cache.getCacheManager());
55           QueryRequest request = ProtobufUtil.fromByteArray(serCtx, query, 0, query.length, QueryRequest.class);
56  
57           Configuration cacheConfiguration = SecurityActions.getCacheConfiguration(cache);
58           boolean isIndexed = cacheConfiguration.indexing().index().isEnabled();
59           boolean isCompatMode = cacheConfiguration.compatibility().enabled();
60           SearchManager searchManager = isIndexed ? Search.getSearchManager(cache) : null;  // this also checks access permissions
61           RemoteQueryEngine queryEngine = new RemoteQueryEngine(cache, searchManager, isCompatMode, serCtx);
62  
63           long startOffset = request.getStartOffset() == null ? -1 : request.getStartOffset();
64           int maxResults = request.getMaxResults() == null ? -1 : request.getMaxResults();
65           Map<String, Object> namedParameters = getNamedParameters(request);
66           Query q = queryEngine.buildQuery(null, request.getJpqlString(), namedParameters, startOffset, maxResults);
67  
68           QueryResponse response = makeResponse(q);
69           return ProtobufUtil.toByteArray(serCtx, response);
70        } catch (IOException e) {
71           throw log.errorExecutingQuery(e);
72        }
73     }
74  
75     private Map<String, Object> getNamedParameters(QueryRequest request) {
76        List<QueryRequest.NamedParameter> namedParameters = request.getNamedParameters();
77        if (namedParameters == null || namedParameters.isEmpty()) {
78           return null;
79        }
80        Map<String, Object> params = new HashMap<String, Object>(namedParameters.size());
81        for (QueryRequest.NamedParameter p : namedParameters) {
82           params.put(p.getName(), p.getValue());
83        }
84        return params;
85     }
86  
87     private QueryResponse makeResponse(Query q) {
88        List<?> list = q.list();
89        int numResults = list.size();
90        String[] projection = ((BaseQuery) q).getProjection();
91        int projSize = projection != null ? projection.length : 0;
92        List<WrappedMessage> results = new ArrayList<WrappedMessage>(projSize == 0 ? numResults : numResults * projSize);
93  
94        for (Object o : list) {
95           if (projSize == 0) {
96              results.add(new WrappedMessage(o));
97           } else {
98              Object[] row = (Object[]) o;
99              for (int j = 0; j < projSize; j++) {
100                results.add(new WrappedMessage(row[j]));
101             }
102          }
103       }
104 
105       QueryResponse response = new QueryResponse();
106       response.setTotalResults(q.getResultSize());
107       response.setNumResults(numResults);
108       response.setProjectionSize(projSize);
109       response.setResults(results);
110       return response;
111    }
