22  package org.infinispan.server.jgroups.subsystem;
23  
24  /**
25   * Enumeration of the supported subsystem xml schemas.
26   * @author Paul Ferraro
27   */
28  public enum JGroupsSchema {
29  
30      VERSION_1_0("jboss:domain:jgroups", 1, 0),
31      VERSION_1_1("jboss:domain:jgroups", 1, 1),
32      VERSION_2_0("jboss:domain:jgroups", 2, 0),
33      VERSION_3_0("jboss:domain:jgroups", 3, 0),
34  
35      INFINISPAN_SERVER_JGROUPS_7_0("infinispan:server:jgroups", 7, 0),
36      INFINISPAN_SERVER_JGROUPS_8_0("infinispan:server:jgroups", 8, 0),
37      ;
38      public static final JGroupsSchema CURRENT = INFINISPAN_SERVER_JGROUPS_8_0;
39  
40      private static final String URN_PATTERN = "urn:%s:%d.%d";
41  
42      private final String domain;
43      private final int major;
44      private final int minor;
45  
46      private JGroupsSchema(String domain, int major, int minor) {
47          this.domain = domain;
48          this.major = major;
49          this.minor = minor;
50      }
51  
52      /**
53       * Indicates whether this version of the schema is greater than or equal to the version of the specified schema.
54       * @param a schema
55       * @return true, if this version of the schema is greater than or equal to the version of the specified schema, false otherwise.
56       */
57      public boolean since(JGroupsSchema... schemas) {
58          for(JGroupsSchema schema : schemas) {
59              if ((this.domain.equals(schema.domain) && ((this.major > schema.major) || ((this.major == schema.major) && (this.minor >= schema.minor)))))
60                  return true;
61          }
62          return false;
63      }
64  
65      /**
66       * Get the namespace URI for this schema version.
67       * @return the namespace URI
68       */
69      public String getNamespaceUri() {
70          return this.format(URN_PATTERN);
71      }
72  
73      /**
74       * Formats a string using the specified pattern.
75       * @param pattern a formatter pattern
76       * @return a formatted string
77       */
78      String format(String pattern) {
79          return String.format(pattern, this.domain, this.major, this.minor);
80      }
81  }
