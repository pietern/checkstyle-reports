49  public class InfinispanSubsystemXMLWriter implements XMLElementWriter<SubsystemMarshallingContext> {
50      public static final XMLElementWriter<SubsystemMarshallingContext> INSTANCE = new InfinispanSubsystemXMLWriter();

52      /**
53       * {@inheritDoc}
54       * @see org.jboss.staxmapper.XMLElementWriter#writeContent(org.jboss.staxmapper.XMLExtendedStreamWriter, java.lang.Object)
55       */
56      @Override
57      public void writeContent(XMLExtendedStreamWriter writer, SubsystemMarshallingContext context) throws XMLStreamException {
58          context.startSubsystemElement(Namespace.CURRENT.getUri(), false);
59          ModelNode model = context.getModelNode();
60          if (model.isDefined()) {
61              for (Property entry: model.get(ModelKeys.CACHE_CONTAINER).asPropertyList()) {
62  
63                  String containerName = entry.getName();
64                  ModelNode container = entry.getValue();
65  
66                  writer.writeStartElement(Element.CACHE_CONTAINER.getLocalName());
67                  writer.writeAttribute(Attribute.NAME.getLocalName(), containerName);
68                  // AS7-3488 make default-cache a non required attribute
69                  // this.writeRequired(writer, Attribute.DEFAULT_CACHE, container, ModelKeys.DEFAULT_CACHE);
70                  this.writeListAsAttribute(writer, Attribute.ALIASES, container, ModelKeys.ALIASES);
71                  this.writeOptional(writer, Attribute.DEFAULT_CACHE, container, ModelKeys.DEFAULT_CACHE);
72                  this.writeOptional(writer, Attribute.JNDI_NAME, container, ModelKeys.JNDI_NAME);
73                  this.writeOptional(writer, Attribute.START, container, ModelKeys.START);
74                  this.writeOptional(writer, Attribute.MODULE, container, ModelKeys.MODULE);
75                  this.writeOptional(writer, Attribute.STATISTICS, container, ModelKeys.STATISTICS);
76  
77                  if (container.hasDefined(ModelKeys.TRANSPORT)) {
78                      writer.writeStartElement(Element.TRANSPORT.getLocalName());
79                      ModelNode transport = container.get(ModelKeys.TRANSPORT, ModelKeys.TRANSPORT_NAME);
80                      this.writeOptional(writer, Attribute.CHANNEL, transport, ModelKeys.CHANNEL);
81                      this.writeOptional(writer, Attribute.LOCK_TIMEOUT, transport, ModelKeys.LOCK_TIMEOUT);
82                      this.writeOptional(writer, Attribute.STRICT_PEER_TO_PEER, transport, ModelKeys.STRICT_PEER_TO_PEER);
83                      writer.writeEndElement();
84                  }
85  
86                  if (container.hasDefined(ModelKeys.SECURITY)) {
87                      writer.writeStartElement(Element.SECURITY.getLocalName());
88                      ModelNode security = container.get(ModelKeys.SECURITY, ModelKeys.SECURITY_NAME);
89                      if (security.hasDefined(ModelKeys.AUTHORIZATION)) {
90                          writer.writeStartElement(Element.AUTHORIZATION.getLocalName());
91                          ModelNode authorization = security.get(ModelKeys.AUTHORIZATION, ModelKeys.AUTHORIZATION_NAME);
92                          if (authorization.hasDefined(ModelKeys.MAPPER)) {
93                              String mapper = authorization.get(ModelKeys.MAPPER).asString();
94                              if (CommonNameRoleMapper.class.getName().equals(mapper)) {
95                                  writer.writeEmptyElement(Element.COMMON_NAME_ROLE_MAPPER.getLocalName());
96                              } else if (ClusterRoleMapper.class.getName().equals(mapper)) {
97                                  writer.writeEmptyElement(Element.CLUSTER_ROLE_MAPPER.getLocalName());
98                              } else if (IdentityRoleMapper.class.getName().equals(mapper)) {
99                                  writer.writeEmptyElement(Element.IDENTITY_ROLE_MAPPER.getLocalName());
100                             } else {
101                                 writer.writeStartElement(Element.CUSTOM_ROLE_MAPPER.getLocalName());
102                                 writer.writeAttribute(Attribute.CLASS.getLocalName(), mapper);
103                                 writer.writeEndElement();
104                             }
105                         }
106 
107                         ModelNode roles = authorization.get(ModelKeys.ROLE);
108                         for(ModelNode roleNode : roles.asList()) {
109                             ModelNode role = roleNode.get(0);
110                             writer.writeStartElement(Element.ROLE.getLocalName());
111                             AuthorizationRoleResource.NAME.marshallAsAttribute(role, writer);
112                             this.writeListAsAttribute(writer, Attribute.PERMISSIONS, role, ModelKeys.PERMISSIONS);
113                             writer.writeEndElement();
114                         }
115 
116                         writer.writeEndElement();
117                     }
118 
119                     writer.writeEndElement();
120                 }
121 
122                 if (container.hasDefined(ModelKeys.STATE_PERSISTENCE)) {
123                     writer.writeStartElement(Element.STATE_PERSISTENCE.getLocalName());
124                     ModelNode statePersistence = container.get(ModelKeys.STATE_PERSISTENCE, ModelKeys.STATE_PERSISTENCE_NAME);
125                     this.writeOptional(writer, Attribute.RELATIVE_TO, statePersistence, ModelKeys.RELATIVE_TO);
126                     this.writeOptional(writer, Attribute.PATH, statePersistence, ModelKeys.PATH);
127                     writer.writeEndElement();
128                 }
129 
130                 // write any configured thread pools
131                 if (container.hasDefined(ThreadPoolResource.WILDCARD_PATH.getKey())) {
132                     writeThreadPoolElements(Element.ASYNC_OPERATIONS_THREAD_POOL, ThreadPoolResource.ASYNC_OPERATIONS, writer, container);
133                     writeThreadPoolElements(Element.LISTENER_THREAD_POOL, ThreadPoolResource.LISTENER, writer, container);
134                     writeThreadPoolElements(Element.PERSISTENCE_THREAD_POOL, ThreadPoolResource.PERSISTENCE, writer, container);
135                     writeThreadPoolElements(Element.REMOTE_COMMAND_THREAD_POOL, ThreadPoolResource.REMOTE_COMMAND, writer, container);
136                     writeThreadPoolElements(Element.STATE_TRANSFER_THREAD_POOL, ThreadPoolResource.STATE_TRANSFER, writer, container);
137                     writeThreadPoolElements(Element.TRANSPORT_THREAD_POOL, ThreadPoolResource.TRANSPORT, writer, container);
138                     writeScheduledThreadPoolElements(Element.EXPIRATION_THREAD_POOL, ScheduledThreadPoolResource.EXPIRATION, writer, container);
139                     writeScheduledThreadPoolElements(Element.REPLICATION_QUEUE_THREAD_POOL, ScheduledThreadPoolResource.REPLICATION_QUEUE, writer, container);
140                 }
141 
142                 ModelNode configurations = container.get(ModelKeys.CONFIGURATIONS, ModelKeys.CONFIGURATIONS_NAME);
143 
144                 // write any existent cache types
145                 processCacheConfiguration(writer, container, configurations, ModelKeys.LOCAL_CACHE);
146                 processCacheConfiguration(writer, container, configurations, ModelKeys.INVALIDATION_CACHE);
147                 processCacheConfiguration(writer, container, configurations, ModelKeys.REPLICATED_CACHE);
148                 processCacheConfiguration(writer, container, configurations, ModelKeys.DISTRIBUTED_CACHE);
149 
150                 writer.writeEndElement();
151             }
152         }
153         writer.writeEndElement();
154     }

156     private static void writeThreadPoolElements(Element element, ThreadPoolResource pool, XMLExtendedStreamWriter writer, ModelNode container) throws XMLStreamException {
157         if (container.get(pool.getPathElement().getKey()).hasDefined(pool.getPathElement().getValue())) {
158             ModelNode threadPool = container.get(pool.getPathElement().getKeyValuePair());
159             if (hasDefined(threadPool, pool.getAttributes())) {
160                 writer.writeStartElement(element.getLocalName());
161                 writeAttributes(writer, threadPool, pool.getAttributes());
162                 writer.writeEndElement();
163             }
164         }
165     }

168         if (container.get(pool.getPathElement().getKey()).hasDefined(pool.getPathElement().getValue())) {
169             ModelNode threadPool = container.get(pool.getPathElement().getKeyValuePair());
170             if (hasDefined(threadPool, pool.getAttributes())) {
171                 writer.writeStartElement(element.getLocalName());
172                 writeAttributes(writer, threadPool, pool.getAttributes());
173                 writer.writeEndElement();
174             }
175         }

178     private void processCacheConfiguration(XMLExtendedStreamWriter writer, ModelNode container, ModelNode configurations, String cacheType)
179             throws XMLStreamException {
180         String cacheConfigurationType = cacheType + ModelKeys.CONFIGURATION_SUFFIX;
181         Map<String, List<String>> configurationMappings = new HashMap<>();
182         if (container.get(cacheType).isDefined()) {
183             for (Property cacheEntry : container.get(cacheType).asPropertyList()) {
184                 String cacheName = cacheEntry.getName();
185                 String configurationName = cacheEntry.getValue().get(ModelKeys.CONFIGURATION).asString();
186 
187                 configurationMappings.compute(configurationName, (k, v) -> {
188                     if (v == null) {
189                         v = new ArrayList<>();
190                     }
191                     v.add(cacheName);
192                     return v;
193                 });
194             }
195         }
196         if (configurations.get(cacheConfigurationType).isDefined()) {
197             for (Property cacheEntry : configurations.get(cacheConfigurationType).asPropertyList()) {
198                 String name = cacheEntry.getName();
199                 ModelNode cacheConfiguration = cacheEntry.getValue();
200 
201                 Element element;
202                 boolean identity = false;
203                 List<String> caches = configurationMappings.get(name);
204                 if (caches != null && caches.size() == 1 && caches.get(0).equals(name)) {
205                     element = Element.forName(cacheType);
206                     identity = true;
207                 } else {
208                     element = Element.forName(cacheConfigurationType);
209                 }
210 
211                 writer.writeStartElement(element.getLocalName());
212                 // write identifier before other attributes
213                 writer.writeAttribute(Attribute.NAME.getLocalName(), name);
214 
215                 switch(cacheType) {
216                     case ModelKeys.DISTRIBUTED_CACHE:
217                         processDistributedCacheAttributes(writer, cacheConfiguration);
218                     case ModelKeys.REPLICATED_CACHE:
219                     case ModelKeys.INVALIDATION_CACHE:
220                         processCommonClusteredCacheAttributes(writer, cacheConfiguration);
221                     default:
222                         processCommonCacheConfigurationAttributesElements(writer, cacheConfiguration);
223                 }
224 
225                 writer.writeEndElement();
226 
227                 // Now the concrete instances
228                 if (!identity && caches!= null) {
229                     for (String cache : caches) {
230                         writer.writeStartElement(Element.forName(cacheType).getLocalName());
231                         writer.writeAttribute(Attribute.NAME.getLocalName(), cache);
232                         writer.writeAttribute(Attribute.CONFIGURATION.getLocalName(), name);
233                         writer.writeEndElement();
234                     }
235                 }
236             }
237         }
238     }

240     private void processDistributedCacheAttributes(XMLExtendedStreamWriter writer, ModelNode distributedCache) throws XMLStreamException {
241         this.writeOptional(writer, Attribute.OWNERS, distributedCache, ModelKeys.OWNERS);
242         this.writeOptional(writer, Attribute.SEGMENTS, distributedCache, ModelKeys.SEGMENTS);
243         this.writeOptional(writer, Attribute.CAPACITY_FACTOR, distributedCache, ModelKeys.CAPACITY_FACTOR);
244         this.writeOptional(writer, Attribute.L1_LIFESPAN, distributedCache, ModelKeys.L1_LIFESPAN);
245     }

265         this.writeOptional(writer, Attribute.STATISTICS_AVAILABLE, cache, ModelKeys.STATISTICS_AVAILABLE);

272             this.writeOptional(writer, Attribute.ACQUIRE_TIMEOUT, locking, ModelKeys.ACQUIRE_TIMEOUT);
273             this.writeOptional(writer, Attribute.CONCURRENCY_LEVEL, locking, ModelKeys.CONCURRENCY_LEVEL);
274             writer.writeEndElement();
275         }
276 
277         if (cache.get(ModelKeys.TRANSACTION, ModelKeys.TRANSACTION_NAME).isDefined()) {
278             writer.writeStartElement(Element.TRANSACTION.getLocalName());
279             ModelNode transaction = cache.get(ModelKeys.TRANSACTION, ModelKeys.TRANSACTION_NAME);
280             this.writeOptional(writer, Attribute.STOP_TIMEOUT, transaction, ModelKeys.STOP_TIMEOUT);
281             this.writeOptional(writer, Attribute.MODE, transaction, ModelKeys.MODE);
282             this.writeOptional(writer, Attribute.LOCKING, transaction, ModelKeys.LOCKING);
283             this.writeOptional(writer, Attribute.NOTIFICATIONS, transaction, ModelKeys.NOTIFICATIONS);
284             writer.writeEndElement();
285         }
286 
287         if (cache.get(ModelKeys.EVICTION, ModelKeys.EVICTION_NAME).isDefined()) {
288             writer.writeStartElement(Element.EVICTION.getLocalName());

290             this.writeOptional(writer, Attribute.STRATEGY, eviction, ModelKeys.STRATEGY);
291             this.writeOptional(writer, Attribute.MAX_ENTRIES, eviction, ModelKeys.MAX_ENTRIES);
292             this.writeOptional(writer, Attribute.TYPE, eviction, ModelKeys.TYPE);
293             this.writeOptional(writer, Attribute.SIZE, eviction, ModelKeys.SIZE);
294             writer.writeEndElement();
295         }
296 
297         if (cache.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME).isDefined()) {
298             writer.writeStartElement(Element.EXPIRATION.getLocalName());
299             ModelNode expiration = cache.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME);
300             this.writeOptional(writer, Attribute.MAX_IDLE, expiration, ModelKeys.MAX_IDLE);
301             this.writeOptional(writer, Attribute.LIFESPAN, expiration, ModelKeys.LIFESPAN);
302             this.writeOptional(writer, Attribute.INTERVAL, expiration, ModelKeys.INTERVAL);
303             writer.writeEndElement();
304         }
305 
306         if (cache.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME).isDefined()) {
307             ModelNode stateTransfer = cache.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME);
308             writer.writeStartElement(Element.STATE_TRANSFER.getLocalName());
309             this.writeOptional(writer, Attribute.AWAIT_INITIAL_TRANSFER, stateTransfer, ModelKeys.AWAIT_INITIAL_TRANSFER);
310             this.writeOptional(writer, Attribute.ENABLED, stateTransfer, ModelKeys.ENABLED);
311             this.writeOptional(writer, Attribute.TIMEOUT, stateTransfer, ModelKeys.TIMEOUT);
312             this.writeOptional(writer, Attribute.CHUNK_SIZE, stateTransfer, ModelKeys.CHUNK_SIZE);
313             writer.writeEndElement();
314         }
315 
316         if (cache.get(ModelKeys.PARTITION_HANDLING, ModelKeys.PARTITION_HANDLING_NAME).isDefined()) {
317             ModelNode partitionHandling = cache.get(ModelKeys.PARTITION_HANDLING, ModelKeys.PARTITION_HANDLING_NAME);
318             writer.writeStartElement(Element.PARTITION_HANDLING.getLocalName());
319             this.writeOptional(writer, Attribute.ENABLED, partitionHandling, ModelKeys.ENABLED);
320             writer.writeEndElement();
321         }
322 
323         if (cache.get(ModelKeys.COMPATIBILITY).isDefined()) {
324             ModelNode compatibility = cache.get(ModelKeys.COMPATIBILITY, ModelKeys.COMPATIBILITY_NAME);
325             writer.writeStartElement(Element.COMPATIBILITY.getLocalName());

327             CompatibilityConfigurationResource.MARSHALLER.marshallAsAttribute(compatibility, writer);
328             writer.writeEndElement();
329         }
330 
331         if (cache.hasDefined(ModelKeys.SECURITY)) {
332             writer.writeStartElement(Element.SECURITY.getLocalName());
333             ModelNode security = cache.get(ModelKeys.SECURITY, ModelKeys.SECURITY_NAME);
334             if (security.hasDefined(ModelKeys.AUTHORIZATION)) {
335                 writer.writeStartElement(Element.AUTHORIZATION.getLocalName());

417                 this.writeJdbcStoreAttributes(writer, store);
418                 this.writeStoreWriteBehind(writer, store);
419                 this.writeStoreProperties(writer, store);
420                 this.writeJDBCStoreTable(writer, Element.STRING_KEYED_TABLE, store, ModelKeys.STRING_KEYED_TABLE);
421                 writer.writeEndElement();
422             }
423         }
424 
425         if (cache.get(ModelKeys.BINARY_KEYED_JDBC_STORE).isDefined()) {
426             for (Property binaryKeyedJDBCStoreEntry : cache.get(ModelKeys.BINARY_KEYED_JDBC_STORE).asPropertyList()) {
427                 ModelNode store = binaryKeyedJDBCStoreEntry.getValue();
428                 writer.writeStartElement(Element.BINARY_KEYED_JDBC_STORE.getLocalName());
429                 // write identifier before other attributes
430                 ModelNode name = new ModelNode();
431                 name.get(ModelKeys.NAME).set(binaryKeyedJDBCStoreEntry.getName());
432                 BinaryKeyedJDBCStoreConfigurationResource.NAME.marshallAsAttribute(name, false, writer);
433                 this.writeJdbcStoreAttributes(writer, store);
434                 this.writeStoreWriteBehind(writer, store);
435                 this.writeStoreProperties(writer, store);
436                 this.writeJDBCStoreTable(writer, Element.BINARY_KEYED_TABLE, store, ModelKeys.BINARY_KEYED_TABLE);

438             }
439         }
440 
441         if (cache.get(ModelKeys.MIXED_KEYED_JDBC_STORE).isDefined()) {
442             for (Property mixedKeyedJDBCStoreEntry : cache.get(ModelKeys.MIXED_KEYED_JDBC_STORE).asPropertyList()) {
443                 ModelNode store = mixedKeyedJDBCStoreEntry.getValue();
444                 writer.writeStartElement(Element.MIXED_KEYED_JDBC_STORE.getLocalName());
445                 // write identifier before other attributes
446                 ModelNode name = new ModelNode();
447                 name.get(ModelKeys.NAME).set(mixedKeyedJDBCStoreEntry.getName());
448                 MixedKeyedJDBCStoreConfigurationResource.NAME.marshallAsAttribute(name, false, writer);
449                 this.writeJdbcStoreAttributes(writer, store);

456         }
457 
458         if (cache.get(ModelKeys.REMOTE_STORE).isDefined()) {
459             for (Property remoteStoreEntry : cache.get(ModelKeys.REMOTE_STORE).asPropertyList()) {
460                 ModelNode store = remoteStoreEntry.getValue();
461                 writer.writeStartElement(Element.REMOTE_STORE.getLocalName());
462                 // write identifier before other attributes
463                 ModelNode name = new ModelNode();

481         }

483         if (cache.get(ModelKeys.REST_STORE).isDefined()) {
484             for (Property restStoreEntry : cache.get(ModelKeys.REST_STORE).asPropertyList()) {
485                 ModelNode store = restStoreEntry.getValue();
486                 writer.writeStartElement(Element.REST_STORE.getLocalName());
487                 // write identifier before other attributes
488                 ModelNode name = new ModelNode();
489                 name.get(ModelKeys.NAME).set(restStoreEntry.getName());
490                 RestStoreConfigurationResource.NAME.marshallAsAttribute(name, false, writer);
491                 this.writeOptional(writer, Attribute.APPEND_CACHE_NAME_TO_PATH, store, ModelKeys.APPEND_CACHE_NAME_TO_PATH);
492                 this.writeOptional(writer, Attribute.PATH, store, ModelKeys.PATH);
493 
494                 this.writeStoreAttributes(writer, store);
495                 this.writeStoreWriteBehind(writer, store);
496                 this.writeStoreProperties(writer, store);
497 
498                 if (store.hasDefined(ModelKeys.CONNECTION_POOL)) {
499                     ModelNode pool = store.get(ModelKeys.CONNECTION_POOL);
500                     writer.writeStartElement(Element.CONNECTION_POOL.getLocalName());
501                     this.writeOptional(writer, Attribute.CONNECTION_TIMEOUT, pool, ModelKeys.CONNECTION_TIMEOUT);
502                     this.writeOptional(writer, Attribute.MAX_CONNECTIONS_PER_HOST, pool, ModelKeys.MAX_CONNECTIONS_PER_HOST);
503                     this.writeOptional(writer, Attribute.MAX_TOTAL_CONNECTIONS, pool, ModelKeys.MAX_TOTAL_CONNECTIONS);
504                     this.writeOptional(writer, Attribute.BUFFER_SIZE, pool, ModelKeys.BUFFER_SIZE);
505                     this.writeOptional(writer, Attribute.SOCKET_TIMEOUT, pool, ModelKeys.SOCKET_TIMEOUT);
506                     this.writeOptional(writer, Attribute.TCP_NO_DELAY, pool, ModelKeys.TCP_NO_DELAY);
507                     writer.writeEndElement();
508                 }
509 
510                 for (ModelNode remoteServer: store.require(ModelKeys.REMOTE_SERVERS).asList()) {
511                     writer.writeStartElement(Element.REMOTE_SERVER.getLocalName());
512                     writer.writeAttribute(Attribute.OUTBOUND_SOCKET_BINDING.getLocalName(), remoteServer.get(ModelKeys.OUTBOUND_SOCKET_BINDING).asString());
513                     writer.writeEndElement();
514                 }
515 
516                 writer.writeEndElement();
517             }
518         }
519 
520         if (cache.get(ModelKeys.INDEXING).isDefined()|| cache.get(ModelKeys.INDEXING_PROPERTIES).isDefined()){
521             writer.writeStartElement(Element.INDEXING.getLocalName());
522             CacheConfigurationResource.INDEXING.marshallAsAttribute(cache, writer);
523             CacheConfigurationResource.INDEXING_AUTO_CONFIG.marshallAsAttribute(cache, writer);
524             CacheConfigurationResource.INDEXING_PROPERTIES.marshallAsElement(cache,writer);
525             writer.writeEndElement();
526         }
527 
528         if (cache.get(ModelKeys.BACKUP).isDefined()) {
529             writer.writeStartElement(Element.BACKUPS.getLocalName());
530             for (Property property : cache.get(ModelKeys.BACKUP).asPropertyList()) {
531                 writer.writeStartElement(Element.BACKUP.getLocalName());
532                 writer.writeAttribute(Attribute.SITE.getLocalName(), property.getName());
533                 ModelNode backup = property.getValue();
534                 BackupSiteConfigurationResource.FAILURE_POLICY.marshallAsAttribute(backup, writer);
535                 BackupSiteConfigurationResource.STRATEGY.marshallAsAttribute(backup, writer);
536                 BackupSiteConfigurationResource.REPLICATION_TIMEOUT.marshallAsAttribute(backup, writer);
537                 BackupSiteConfigurationResource.ENABLED.marshallAsAttribute(backup, writer);
538                 if (backup.hasDefined(ModelKeys.TAKE_BACKUP_OFFLINE_AFTER_FAILURES)
539                         || backup.hasDefined(ModelKeys.TAKE_BACKUP_OFFLINE_MIN_WAIT)) {
540                     writer.writeStartElement(Element.TAKE_OFFLINE.getLocalName());
541                     BackupSiteConfigurationResource.TAKE_OFFLINE_AFTER_FAILURES.marshallAsAttribute(backup, writer);
542                     BackupSiteConfigurationResource.TAKE_OFFLINE_MIN_WAIT.marshallAsAttribute(backup, writer);
543                     writer.writeEndElement();
544                 }
545                 if (backup.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME).isDefined()) {
546                     ModelNode stateTransfer = backup.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME);
547                     if (stateTransfer.hasDefined(ModelKeys.CHUNK_SIZE)
548                           || stateTransfer.hasDefined(ModelKeys.TIMEOUT)
549                           || stateTransfer.hasDefined(ModelKeys.MAX_RETRIES)
550                           || stateTransfer.hasDefined(ModelKeys.WAIT_TIME)) {
551                        writer.writeStartElement(Element.STATE_TRANSFER.getLocalName());
552                        BackupSiteStateTransferConfigurationResource.STATE_TRANSFER_CHUNK_SIZE.marshallAsAttribute(stateTransfer, writer);
553                        BackupSiteStateTransferConfigurationResource.STATE_TRANSFER_TIMEOUT.marshallAsAttribute(stateTransfer, writer);
554                        BackupSiteStateTransferConfigurationResource.STATE_TRANSFER_MAX_RETRIES.marshallAsAttribute(stateTransfer, writer);
555                        BackupSiteStateTransferConfigurationResource.STATE_TRANSFER_WAIT_TIME.marshallAsAttribute(stateTransfer, writer);
556                        writer.writeEndElement();
557                     }
558                 }
559                 writer.writeEndElement();
560             }
561             writer.writeEndElement();
562         }
563 
564         if (cache.get(ModelKeys.REMOTE_CACHE).isDefined() || cache.get(ModelKeys.REMOTE_SITE).isDefined()) {
565             writer.writeStartElement(Element.BACKUP_FOR.getLocalName());
566             CacheConfigurationResource.REMOTE_CACHE.marshallAsAttribute(cache, writer);
567             CacheConfigurationResource.REMOTE_SITE.marshallAsAttribute(cache, writer);
568             writer.writeEndElement();
569         }
570 
571         if (cache.get(ModelKeys.LEVELDB_STORE).isDefined()) {
572             for (Property levelDbStoreEntry : cache.get(ModelKeys.LEVELDB_STORE).asPropertyList()) {
573                 ModelNode store = levelDbStoreEntry.getValue();
574                 writer.writeStartElement(Element.LEVELDB_STORE.getLocalName());
575                 // write identifier before other attributes
576                 ModelNode name = new ModelNode();
577                 name.get(ModelKeys.NAME).set(levelDbStoreEntry.getName());
578                 LevelDBStoreConfigurationResource.NAME.marshallAsAttribute(name, false, writer);
579                 this.writeOptional(writer, Attribute.RELATIVE_TO, store, ModelKeys.RELATIVE_TO);
580                 this.writeOptional(writer, Attribute.PATH, store, ModelKeys.PATH);
581                 this.writeOptional(writer, Attribute.BLOCK_SIZE, store, ModelKeys.BLOCK_SIZE);
582                 this.writeOptional(writer, Attribute.CACHE_SIZE, store, ModelKeys.CACHE_SIZE);
583                 this.writeOptional(writer, Attribute.CLEAR_THRESHOLD, store, ModelKeys.CLEAR_THRESHOLD);
584                 this.writeStoreAttributes(writer, store);
585                 this.writeStoreExpiration(writer, store);
586                 this.writeStoreCompression(writer, store);
587                 this.writeStoreImplementation(writer, store);
588                 this.writeStoreProperties(writer, store);
589                 writer.writeEndElement();
590             }
591         }
592 
593     }

595     private void writeListAsAttribute(XMLExtendedStreamWriter writer, Attribute attribute, ModelNode node, String key) throws XMLStreamException {
596         if (node.hasDefined(key)) {
597             StringBuilder result = new StringBuilder() ;
598             ModelNode list = node.get(key);
599             if (list.isDefined() && list.getType() == ModelType.LIST) {
600                 List<ModelNode> nodeList = list.asList();
601                 for (int i = 0; i < nodeList.size(); i++) {
602                     result.append(nodeList.get(i).asString());
603                     if (i < nodeList.size()-1) {
604                         result.append(" ");
605                     }
606                 }
607                 writer.writeAttribute(attribute.getLocalName(), result.toString());
608             }
609         }
610     }
