23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import org.jboss.as.controller.AttributeDefinition;
26  import org.jboss.as.controller.OperationStepHandler;
27  import org.jboss.as.controller.PathElement;
28  import org.jboss.as.controller.SimpleAttributeDefinition;
29  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
30  import org.jboss.as.controller.client.helpers.MeasurementUnit;
31  import org.jboss.as.controller.registry.AttributeAccess;
32  import org.jboss.as.controller.registry.ManagementResourceRegistration;
33  import org.jboss.as.controller.services.path.ResolvePathHandler;
34  import org.jboss.dmr.ModelNode;
35  import org.jboss.dmr.ModelType;
36  
37  /**
38   * Resource description for the addressable resource /subsystem=infinispan/cache-container=X/distributed-cache=*
39   *
40   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
41   * @author Radoslav Husar
42   */
43  public class DistributedCacheConfigurationResource extends SharedCacheConfigurationResource {
44  
45      public static final PathElement PATH = PathElement.pathElement(ModelKeys.DISTRIBUTED_CACHE_CONFIGURATION);
46  
47      // attributes
48      static final SimpleAttributeDefinition L1_LIFESPAN =
49              new SimpleAttributeDefinitionBuilder(ModelKeys.L1_LIFESPAN, ModelType.LONG, true)
50                      .setXmlName(Attribute.L1_LIFESPAN.getLocalName())
51                      .setMeasurementUnit(MeasurementUnit.MILLISECONDS)
52                      .setAllowExpression(true)
53                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
54                      .setDefaultValue(new ModelNode().set(0))
55                      .build();
56  
57      static final SimpleAttributeDefinition OWNERS =
58              new SimpleAttributeDefinitionBuilder(ModelKeys.OWNERS, ModelType.INT, true)
59                      .setXmlName(Attribute.OWNERS.getLocalName())
60                      .setAllowExpression(true)
61                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
62                      .setDefaultValue(new ModelNode().set(2))
63                      .build();
64  
65      static final SimpleAttributeDefinition SEGMENTS =
66              new SimpleAttributeDefinitionBuilder(ModelKeys.SEGMENTS, ModelType.INT, true)
67                      .setXmlName(Attribute.SEGMENTS.getLocalName())
68                      .setAllowExpression(true)
69                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
70                      .setDefaultValue(new ModelNode().set(80)) // Recommended value is 10 * max_cluster_size.
71                      .build();
72  
73      static final SimpleAttributeDefinition CAPACITY_FACTOR =
74              new SimpleAttributeDefinitionBuilder(ModelKeys.CAPACITY_FACTOR, ModelType.DOUBLE, true)
75                      .setXmlName(Attribute.CAPACITY_FACTOR.getLocalName())
76                      .setAllowExpression(true)
77                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
78                      .setDefaultValue(new ModelNode().set(1.0))
79                      .build();
80  
81      static final AttributeDefinition[] ATTRIBUTES = {OWNERS, SEGMENTS, CAPACITY_FACTOR, L1_LIFESPAN};
82  
83      public DistributedCacheConfigurationResource(final ResolvePathHandler resolvePathHandler, boolean runtimeRegistration) {
84          super(PATH,
85                  new InfinispanResourceDescriptionResolver(ModelKeys.DISTRIBUTED_CACHE_CONFIGURATION),
86                  DistributedCacheConfigurationAdd.INSTANCE,
87                  new CacheConfigurationRemove(), resolvePathHandler, runtimeRegistration);
88      }
89  
90      @Override
91      public void registerAttributes(ManagementResourceRegistration resourceRegistration) {
92          super.registerAttributes(resourceRegistration);
93  
94          final OperationStepHandler restartWriteHandler = new RestartCacheWriteAttributeHandler(getPathElement().getKey(), getServiceInstaller(), ATTRIBUTES);
95          for (AttributeDefinition attr : ATTRIBUTES) {
96              resourceRegistration.registerReadWriteAttribute(attr, CacheReadAttributeHandler.INSTANCE, restartWriteHandler);
97          }
98      }
99  }
