117     /*
118      * Two constraints need to be dealt with here:
119      * 1. There may be no started cache container instance available to interrogate. Because of lazy deployment,
120      * a cache container instance is only started upon deployment of an application which uses that cache instance.
121      * 2. The attribute name passed in may not correspond to a defined metric
122      *
123      * Read-only attributes have no easy way to throw an exception without negatively impacting other parts
124      * of the system. Therefore in such cases, as message will be logged and a ModelNode of undefined will be returned.
125      */
126     @Override
127     protected void executeRuntimeStep(OperationContext context, ModelNode operation) throws OperationFailedException {
128         final PathAddress address = PathAddress.pathAddress(operation.require(OP_ADDR));
129         final String cacheContainerName = address.getLastElement().getValue();
130         final String attrName = operation.require(ModelDescriptionConstants.NAME).asString();
131         final ServiceController<?> controller = context.getServiceRegistry(false).getService(CacheContainerServiceName.CACHE_CONTAINER.getServiceName(cacheContainerName));
132         DefaultCacheContainer cacheManager = (DefaultCacheContainer) controller.getValue();
133 
134         CacheManagerMetrics metric = CacheManagerMetrics.getStat(attrName);
135         ModelNode result = new ModelNode();
136 
137         if (metric == null) {
138             context.getFailureDescription().set(String.format("Unknown metric %s", attrName));
139         } else if (cacheManager == null) {
140             context.getFailureDescription().set(String.format("Unavailable cache container %s", attrName));
141         } else {
142             CacheContainerStats stats = cacheManager.getStats();
143             switch (metric) {
144                 case CACHE_MANAGER_STATUS:
145                     result.set(SecurityActions.getCacheManagerStatus(cacheManager).toString());
146                     break;
147                 case IS_COORDINATOR:
148                     result.set(SecurityActions.getCacheManagerIsCoordinator(cacheManager));
149                     break;
150                 case LOCAL_ADDRESS:
151                     Address localAddress = SecurityActions.getCacheManagerLocalAddress(cacheManager);
152                     result.set(localAddress != null ? localAddress.toString() : "N/A");
153                     break;
154                 case COORDINATOR_ADDRESS:
155                     Address coordinatorAddress = SecurityActions.getCacheManagerCoordinatorAddress(cacheManager);
156                     result.set(coordinatorAddress != null ? coordinatorAddress.toString() : "N/A");
157                     break;
158                 case CLUSTER_AVAILABILITY:
159                     result.set(SecurityActions.getCacheManagerClusterAvailability(cacheManager));
160                     break;
161                 case CLUSTER_NAME:
162                     String clusterName = SecurityActions.getCacheManagerClusterName(cacheManager);
163                     result.set(clusterName != null ? clusterName : "N/A");
164                     break;
165                 case DEFINED_CACHE_NAMES:
166                     String definedCacheNames = SecurityActions.getDefinedCacheNames(cacheManager);
167                     result.set(definedCacheNames != null ? definedCacheNames : "N/A");
168                     break;
169                 case CLUSTER_SIZE:
170                     List<Address> members = SecurityActions.getMembers(cacheManager);
171                     result.set(members != null ? Integer.toString(members.size()) : "N/A");
172                     break;
173                 case CREATED_CACHE_COUNT:
174                     result.set(SecurityActions.getCacheCreatedCount(cacheManager));
175                     break;
176                 case DEFINED_CACHE_COUNT:
177                     result.set(SecurityActions.getDefinedCacheCount(cacheManager));
178                     break;
179                 case MEMBERS:
180                     members = SecurityActions.getMembers(cacheManager);
181                     result.set(members != null ? members.toString() : "N/A");
182                     break;
183                 case RUNNING_CACHE_COUNT:
184                     result.set(SecurityActions.getRunningCacheCount(cacheManager));
185                     break;
186                 case VERSION:
187                     result.set(Version.getVersion());
188                     break;
189                 case AVERAGE_READ_TIME:
190                    result.set(stats.getAverageReadTime());
191                    break;
192                 case AVERAGE_WRITE_TIME:
193                    result.set(stats.getAverageWriteTime());
194                    break;
195                 case AVERAGE_REMOVE_TIME:
196                    result.set(stats.getAverageRemoveTime());
197                    break;
198                 case TIME_SINCE_START:
199                    result.set(stats.getTimeSinceStart());
200                    break;
201                 case EVICTIONS:
202                    result.set(stats.getEvictions());
203                    break;
204                 case HIT_RATIO:
205                    result.set(stats.getHitRatio());
206                    break;
207                 case HITS:
208                    result.set(stats.getHits());
209                    break;
210                 case MISSES:
211                    result.set(stats.getMisses());
212                    break;
213                 case NUMBER_OF_ENTRIES:
214                    result.set(stats.getCurrentNumberOfEntries());
215                    break;
216                 case READ_WRITE_RATIO:
217                    result.set(stats.getReadWriteRatio());
218                    break;
219                 case REMOVE_HITS:
220                    result.set(stats.getHits());
221                    break;
222                 case REMOVE_MISSES:
223                    result.set(stats.getRemoveMisses());
224                    break;
225                 case STORES:
226                    result.set(stats.getStores());
227                    break;
228                 case TIME_SINCE_RESET:
229                    result.set(stats.getTimeSinceStart());
230                    break;
231                 default:
232                     context.getFailureDescription().set(String.format("Unknown metric %s", metric));
233                     break;
234             }
235             context.getResult().set(result);
236         }
237         context.stepCompleted();
238     }

240     public void registerMetrics(ManagementResourceRegistration container) {
241         for (CacheManagerMetrics metric : CacheManagerMetrics.values()) {
242             container.registerMetric(metric.definition, this);
243         }
244     }
