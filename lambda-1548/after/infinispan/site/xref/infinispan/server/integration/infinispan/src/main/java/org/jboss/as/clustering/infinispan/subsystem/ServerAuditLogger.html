23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import javax.security.auth.Subject;
26  
27  import org.infinispan.security.AuditContext;
28  import org.infinispan.security.AuditLogger;
29  import org.infinispan.security.AuditResponse;
30  import org.infinispan.security.AuthorizationPermission;
31  import org.jboss.security.audit.AuditEvent;
32  import org.jboss.security.audit.AuditLevel;
33  import org.jboss.security.audit.AuditManager;
34  
35  /**
36   * ServerAuditLogger.
37   *
38   * @author Tristan Tarrant
39   * @since 7.0
40   */
41  public class ServerAuditLogger implements AuditLogger {
42  
43     private final AuditManager auditManager;
44  
45     ServerAuditLogger(AuditManager auditManager) {
46        this.auditManager = auditManager;
47     }
48  
49     @Override
50     public void audit(Subject subject, AuditContext context, String contextName, AuthorizationPermission permission, AuditResponse response) {
51        String level;
52        switch (response) {
53        case ALLOW:
54           level = AuditLevel.SUCCESS;
55           break;
56        case DENY:
57           level = AuditLevel.FAILURE;
58           break;
59        case ERROR:
60           level = AuditLevel.ERROR;
61           break;
62        default:
63           level = AuditLevel.INFO;
64           break;
65        }
66        AuditEvent ae = new AuditEvent(level);
67        auditManager.audit(ae);
68     }
69  
70  }
