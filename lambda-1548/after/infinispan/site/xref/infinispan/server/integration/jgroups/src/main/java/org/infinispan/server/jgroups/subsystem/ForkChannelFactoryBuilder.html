24  import java.util.ArrayList;
25  import java.util.LinkedList;
26  import java.util.List;

28  import org.infinispan.server.commons.service.Builder;
29  import org.infinispan.server.commons.service.InjectedValueDependency;
30  import org.infinispan.server.commons.service.ValueDependency;
31  import org.infinispan.server.jgroups.ForkChannelFactory;
32  import org.infinispan.server.jgroups.spi.ChannelFactory;
33  import org.infinispan.server.jgroups.spi.ProtocolConfiguration;
34  import org.infinispan.server.jgroups.spi.service.ChannelServiceName;
35  import org.infinispan.server.jgroups.spi.service.ProtocolStackServiceName;
36  import org.jboss.msc.service.ServiceBuilder;
37  import org.jboss.msc.service.ServiceController;
38  import org.jboss.msc.service.ServiceName;
39  import org.jboss.msc.service.ServiceTarget;
40  import org.jboss.msc.service.ValueService;
41  import org.jboss.msc.value.InjectedValue;
42  import org.jboss.msc.value.Value;
43  import org.jgroups.Channel;
44  
45  /**
46   * Builder for a service that provides a {@link ChannelFactory} for creating fork channels.
47   * @author Paul Ferraro
48   */
49  public class ForkChannelFactoryBuilder implements Builder<ChannelFactory>, Value<ChannelFactory> {
50  
51      private final String channelName;
52      private final InjectedValue<Channel> parentChannel = new InjectedValue<>();
53      private final InjectedValue<ChannelFactory> parentFactory = new InjectedValue<>();
54      private final List<ValueDependency<ProtocolConfiguration>> protocols = new LinkedList<>();
55  
56      public ForkChannelFactoryBuilder(String channelName) {
57          this.channelName = channelName;
58      }
59  
60      @Override
61      public ServiceName getServiceName() {
62          return ProtocolStackServiceName.CHANNEL_FACTORY.getServiceName(this.channelName);
63      }
64  
65      @Override
66      public ServiceBuilder<ChannelFactory> build(ServiceTarget target) {
67          return target.addService(this.getServiceName(), new ValueService<>(this))
68                  .addDependency(ChannelServiceName.CONNECTOR.getServiceName(this.channelName), Channel.class, this.parentChannel)
69                  .addDependency(ChannelServiceName.FACTORY.getServiceName(this.channelName), ChannelFactory.class, this.parentFactory)
70                  .setInitialMode(ServiceController.Mode.ON_DEMAND)
71          ;
72      }
73  
74      @Override
75      public ChannelFactory getValue() {
76          List<ProtocolConfiguration> protocols = new ArrayList<>(this.protocols.size());
77          for (Value<ProtocolConfiguration> protocol : this.protocols) {
78              protocols.add(protocol.getValue());
79          }
80          return new ForkChannelFactory(this.parentChannel.getValue(), this.parentFactory.getValue(), protocols);
81      }
82  
83      public ProtocolConfigurationBuilder addProtocol(String type) {
84          ProtocolConfigurationBuilder builder = new ProtocolConfigurationBuilder(this.channelName, type);
85          this.protocols.add(new InjectedValueDependency<>(builder, ProtocolConfiguration.class));
86          return builder;
87      }
88  }
