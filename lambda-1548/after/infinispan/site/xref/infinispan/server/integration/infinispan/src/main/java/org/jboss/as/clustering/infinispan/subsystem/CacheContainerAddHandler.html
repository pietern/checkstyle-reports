58  /**
59   * Add operation handler for /subsystem=infinispan/cache-container=*
60   * @author Paul Ferraro
61   * @author Tristan Tarrant
62   * @author Richard Achmatowicz
63   */
64  public class CacheContainerAddHandler extends AbstractAddStepHandler {
65  
66      CacheContainerAddHandler() {
67          super(CacheContainerResource.CACHE_CONTAINER_ATTRIBUTES);
68      }
69  
70      @Override
71      protected void performRuntime(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
72          // Because we use child resources in a read-only manner to configure the cache container, replace the local model with the full model
73          installRuntimeServices(context, operation, Resource.Tools.readModel(context.readResource(PathAddress.EMPTY_ADDRESS)));
74      }
75  
76      static void installRuntimeServices(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
77          String name = context.getCurrentAddressValue();
78  
79          ServiceTarget target = context.getServiceTarget();
80  
81          // pick up the attribute values from the model
82          // make default cache non required (AS7-3488)
83          String defaultCache = ModelNodes.asString(CacheContainerResource.DEFAULT_CACHE.resolveModelAttribute(context, model));
84          String jndiName = ModelNodes.asString(CacheContainerResource.JNDI_NAME.resolveModelAttribute(context, model));
85          ModuleIdentifier module = ModelNodes.asModuleIdentifier(CacheContainerResource.CACHE_CONTAINER_MODULE.resolveModelAttribute(context, model));
86  
87          CacheContainerConfigurationBuilder configBuilder = new CacheContainerConfigurationBuilder(name)
88                  .setModule(module)
89                  .setStatisticsEnabled(CacheContainerResource.STATISTICS.resolveModelAttribute(context, model).asBoolean());
90  
91          if (model.hasDefined(TransportResource.TRANSPORT_PATH.getKey())) {
92              ModelNode transport = model.get(TransportResource.TRANSPORT_PATH.getKeyValuePair());
93              String channel = ModelNodes.asString(TransportResource.CHANNEL.resolveModelAttribute(context, transport), ChannelServiceNameFactory.DEFAULT_CHANNEL);
94  
95              configBuilder.setTransport()
96                      .setLockTimeout(TransportResource.LOCK_TIMEOUT.resolveModelAttribute(context, transport).asLong(), TimeUnit.MILLISECONDS)
97                      .setStrictPeerToPeer(TransportResource.STRICT_PEER_TO_PEER.resolveModelAttribute(context, transport).asBoolean())
98                      .build(target).install();
99  
100             if (!name.equals(channel)) {
101                 new BinderServiceBuilder<>(JGroupsBindingFactory.createChannelBinding(name), ChannelServiceName.CHANNEL.getServiceName(name), Channel.class).build(target).install();
102 
103                 new ChannelBuilder(name).build(target).install();
104                 new ChannelConnectorBuilder(name).build(target).install();
105                 new AliasServiceBuilder<>(ChannelServiceName.FACTORY.getServiceName(name), ProtocolStackServiceName.CHANNEL_FACTORY.getServiceName(channel), ChannelFactory.class).build(target).install();
106             }
107         }
108 
109         if (model.hasDefined(StatePersistenceResource.STATE_PERSISTENCE_PATH.getKey())) {
110             ModelNode statePersistence = model.get(StatePersistenceResource.STATE_PERSISTENCE_PATH.getKeyValuePair());
111             final String path = ModelNodes.asString(StatePersistenceResource.PATH.resolveModelAttribute(context, statePersistence), InfinispanExtension.SUBSYSTEM_NAME + File.separatorChar + name);
112             final String relativeTo = ModelNodes.asString(StatePersistenceResource.RELATIVE_TO.resolveModelAttribute(context, statePersistence));
113             configBuilder.setStatePersistence().setPath(path).setRelativeTo(relativeTo);
114         }
115 
116         AuthorizationConfigurationBuilder authorizationConfig = null;
117         if (model.hasDefined(ModelKeys.SECURITY) && model.get(ModelKeys.SECURITY).hasDefined(ModelKeys.SECURITY_NAME)) {
118             ModelNode securityModel = model.get(ModelKeys.SECURITY, ModelKeys.SECURITY_NAME);
119 
120             if (securityModel.hasDefined(ModelKeys.AUTHORIZATION) && securityModel.get(ModelKeys.AUTHORIZATION).hasDefined(ModelKeys.AUTHORIZATION_NAME)) {
121                 ModelNode authzModel = securityModel.get(ModelKeys.AUTHORIZATION, ModelKeys.AUTHORIZATION_NAME);
122 
123                 authorizationConfig = configBuilder.setAuthorization();
124                 authorizationConfig.setPrincipalMapper(ModelNodes.asString(CacheContainerAuthorizationResource.MAPPER.resolveModelAttribute(context, authzModel)));
125 
126                 for(ModelNode roleNode : authzModel.get(ModelKeys.ROLE).asList()) {
127                     ModelNode role = roleNode.get(0);
128                     String roleName = AuthorizationRoleResource.NAME.resolveModelAttribute(context, role).asString();
129                     List<String> permissions = new ArrayList<>();
130                     for(ModelNode permission : AuthorizationRoleResource.PERMISSIONS.resolveModelAttribute(context, role).asList()) {
131                         permissions.add(permission.asString());
132                     }
133                     authorizationConfig.getRoles().put(roleName, permissions);
134                 }
135 
136             }
137         }
138 
139         // Install cache container configuration service
140         configBuilder.build(target).install();
141 
142         // Install cache container service
143         CacheContainerBuilder containerBuilder = new CacheContainerBuilder(name, defaultCache);
144 
145         if (model.hasDefined(CacheContainerResource.ALIASES.getName())) {
146             for (ModelNode alias : operation.get(CacheContainerResource.ALIASES.getName()).asList()) {
147                 containerBuilder.addAlias(alias.asString());
148             }
149         }
150 
151         containerBuilder.build(target).install();
152 
153         // Install cache container jndi binding
154         ContextNames.BindInfo binding = InfinispanBindingFactory.createCacheContainerBinding(name);
155         BinderServiceBuilder<CacheContainer> bindingBuilder = new BinderServiceBuilder<>(binding, CacheContainerServiceName.CACHE_CONTAINER.getServiceName(name), CacheContainer.class);
156         if (jndiName != null) {
157             bindingBuilder.alias(ContextNames.bindInfoFor(JndiNameFactory.parse(jndiName).getAbsoluteName()));
158         }
159         bindingBuilder.build(target).install();
160 
161         if ((defaultCache != null) && !defaultCache.equals(CacheServiceNameFactory.DEFAULT_CACHE)) {
162 
163             for (CacheServiceNameFactory nameFactory : CacheServiceName.values()) {
164                 new AliasServiceBuilder<>(nameFactory.getServiceName(name), nameFactory.getServiceName(name, defaultCache), Object.class).build(target).install();
165             }
166 
167             new BinderServiceBuilder<>(InfinispanBindingFactory.createCacheBinding(name, CacheServiceNameFactory.DEFAULT_CACHE), CacheServiceName.CACHE.getServiceName(name), Cache.class).build(target).install();
168         }
169     }
170 
171     static void removeRuntimeServices(OperationContext context, ModelNode operation, ModelNode model) throws OperationFailedException {
172         String name = context.getCurrentAddressValue();
173 
174         // remove the BinderService entry
175         context.removeService(InfinispanBindingFactory.createCacheContainerBinding(name).getBinderServiceName());
176 
177         for (CacheContainerServiceNameFactory factory : CacheContainerServiceName.values()) {
178             context.removeService(factory.getServiceName(name));
179         }
180 
181         if (model.hasDefined(TransportResource.TRANSPORT_PATH.getKey())) {
182             context.removeService(new TransportConfigurationBuilder(name).getServiceName());
183 
184             context.removeService(JGroupsBindingFactory.createChannelBinding(name).getBinderServiceName());
185             for (ChannelServiceNameFactory factory : ChannelServiceName.values()) {
186                 context.removeService(factory.getServiceName(name));
187             }
188         }
189     }
