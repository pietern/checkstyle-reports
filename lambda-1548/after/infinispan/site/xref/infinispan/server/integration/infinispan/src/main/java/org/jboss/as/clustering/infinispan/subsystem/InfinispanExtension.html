22  package org.jboss.as.clustering.infinispan.subsystem;
23  
24  import org.infinispan.server.infinispan.spi.InfinispanSubsystem;
25  import org.jboss.as.controller.Extension;
26  import org.jboss.as.controller.ExtensionContext;
27  import org.jboss.as.controller.PathElement;
28  import org.jboss.as.controller.SubsystemRegistration;
29  import org.jboss.as.controller.descriptions.ModelDescriptionConstants;
30  import org.jboss.as.controller.parsing.ExtensionParsingContext;
31  import org.jboss.as.controller.services.path.ResolvePathHandler;
32  import org.kohsuke.MetaInfServices;
33  
34  /**
35   * Defines the Infinispan subsystem and its addressable resources.
36   *
37   * @author Paul Ferraro
38   * @author Richard Achmatowicz
39   * @author Tristan Tarrant
40   */
41  @MetaInfServices
42  public class InfinispanExtension implements Extension {
43      public static final String SUBSYSTEM_NAME = InfinispanSubsystem.SUBSYSTEM_NAME;
44      static final PathElement SUBSYSTEM_PATH = PathElement.pathElement(ModelDescriptionConstants.SUBSYSTEM, SUBSYSTEM_NAME);
45      public static final String RESOURCE_NAME = InfinispanExtension.class.getPackage().getName() + ".LocalDescriptions";
46  
47      /**
48       * {@inheritDoc}
49       * @see org.jboss.as.controller.Extension#initialize(org.jboss.as.controller.ExtensionContext)
50       */
51      @Override
52      public void initialize(ExtensionContext context) {
53          // IMPORTANT: Management API version != xsd version! Not all Management API changes result in XSD changes
54          SubsystemRegistration subsystem = context.registerSubsystem(SUBSYSTEM_NAME, Namespace.CURRENT.getVersion());
55          // Create the path resolver handler
56          final ResolvePathHandler resolvePathHandler;
57          if (context.getProcessType().isServer()) {
58              resolvePathHandler = ResolvePathHandler.Builder.of(context.getPathManager())
59                      .setPathAttribute(FileStoreResource.PATH)
60                      .setRelativeToAttribute(FileStoreResource.RELATIVE_TO)
61                      .build();
62          } else {
63              resolvePathHandler = null;
64          }
65  
66          subsystem.registerSubsystemModel(new InfinispanSubsystemRootResource(resolvePathHandler, context.isRuntimeOnlyRegistrationValid()));
67          subsystem.registerXMLElementWriter(new InfinispanSubsystemXMLWriter());
68      }
69  
70      /**
71       * {@inheritDoc}
72       * @see org.jboss.as.controller.Extension#initializeParsers(org.jboss.as.controller.parsing.ExtensionParsingContext)
73       */
74      @Override
75      public void initializeParsers(ExtensionParsingContext context) {
76          for (Namespace namespace: Namespace.values()) {
77              context.setSubsystemXmlMapping(SUBSYSTEM_NAME, namespace.getUri(), new InfinispanSubsystemXMLReader(namespace));
78          }
79      }
80  }
