23  package org.jboss.as.clustering.infinispan.subsystem;
24  
25  import org.jboss.as.controller.AttributeDefinition;
26  import org.jboss.as.controller.ObjectListAttributeDefinition;
27  import org.jboss.as.controller.ObjectTypeAttributeDefinition;
28  import org.jboss.as.controller.PathElement;
29  import org.jboss.as.controller.SimpleAttributeDefinition;
30  import org.jboss.as.controller.SimpleAttributeDefinitionBuilder;
31  import org.jboss.as.controller.client.helpers.MeasurementUnit;
32  import org.jboss.as.controller.registry.AttributeAccess;
33  import org.jboss.dmr.ModelNode;
34  import org.jboss.dmr.ModelType;
35  
36  /**
37   * Resource description for the addressable resource /subsystem=infinispan/cache-container=X/cache=Y/remote-store=REMOTE_STORE
38   *
39   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
40   */
41  public class RemoteStoreConfigurationResource extends BaseStoreConfigurationResource {
42  
43      public static final PathElement REMOTE_STORE_PATH = PathElement.pathElement(ModelKeys.REMOTE_STORE);
44  
45      // attributes
46      static final SimpleAttributeDefinition CACHE =
47              new SimpleAttributeDefinitionBuilder(ModelKeys.CACHE, ModelType.STRING, true)
48                      .setXmlName(Attribute.CACHE.getLocalName())
49                      .setAllowExpression(true)
50                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
51                      .build();
52      static final SimpleAttributeDefinition HOTROD_WRAPPING =
53              new SimpleAttributeDefinitionBuilder(ModelKeys.HOTROD_WRAPPING, ModelType.BOOLEAN, true)
54                      .setXmlName(Attribute.HOTROD_WRAPPING.getLocalName())
55                      .setAllowExpression(true)
56                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
57                      .setDefaultValue(new ModelNode().set(false))
58                      .build();
59      static final SimpleAttributeDefinition RAW_VALUES =
60              new SimpleAttributeDefinitionBuilder(ModelKeys.RAW_VALUES, ModelType.BOOLEAN, true)
61                      .setXmlName(Attribute.RAW_VALUES.getLocalName())
62                      .setAllowExpression(false)
63                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
64                      .setDefaultValue(new ModelNode().set(false))
65                      .build();
66      static final SimpleAttributeDefinition TCP_NO_DELAY =
67              new SimpleAttributeDefinitionBuilder(ModelKeys.TCP_NO_DELAY, ModelType.BOOLEAN, true)
68                      .setXmlName(Attribute.TCP_NO_DELAY.getLocalName())
69                      .setAllowExpression(true)
70                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
71                      .setDefaultValue(new ModelNode().set(true))
72                      .build();
73      static final SimpleAttributeDefinition SOCKET_TIMEOUT =
74              new SimpleAttributeDefinitionBuilder(ModelKeys.SOCKET_TIMEOUT, ModelType.LONG, true)
75                      .setXmlName(Attribute.SOCKET_TIMEOUT.getLocalName())
76                      .setMeasurementUnit(MeasurementUnit.MILLISECONDS)
77                      .setAllowExpression(true)
78                      .setFlags(AttributeAccess.Flag.RESTART_RESOURCE_SERVICES)
79                      .setDefaultValue(new ModelNode().set(60000))
80                      .build();
81  
82      static final SimpleAttributeDefinition OUTBOUND_SOCKET_BINDING = new SimpleAttributeDefinition("outbound-socket-binding", ModelType.STRING, true);
83  
84      static final SimpleAttributeDefinition NAME =
85              new SimpleAttributeDefinitionBuilder(BaseStoreConfigurationResource.NAME)
86                     .setDefaultValue(new ModelNode().set(ModelKeys.REMOTE_STORE_NAME))
87                     .build();
88  
89      static final ObjectTypeAttributeDefinition REMOTE_SERVER = ObjectTypeAttributeDefinition.
90              Builder.of(ModelKeys.REMOTE_SERVER, OUTBOUND_SOCKET_BINDING).
91              setAllowNull(true).
92              setSuffix("remote-server").
93              build();
94  
95      static final ObjectListAttributeDefinition REMOTE_SERVERS = ObjectListAttributeDefinition.Builder.of(ModelKeys.REMOTE_SERVERS, REMOTE_SERVER).
96              setAllowNull(false).
97              build();
98  
99      static final AttributeDefinition[] REMOTE_STORE_ATTRIBUTES = {CACHE, HOTROD_WRAPPING, TCP_NO_DELAY, RAW_VALUES, SOCKET_TIMEOUT, REMOTE_SERVERS};
100 
101     public RemoteStoreConfigurationResource(CacheConfigurationResource parent) {
102         super(REMOTE_STORE_PATH, ModelKeys.REMOTE_STORE, parent, REMOTE_STORE_ATTRIBUTES);
103     }
104 
105 }
