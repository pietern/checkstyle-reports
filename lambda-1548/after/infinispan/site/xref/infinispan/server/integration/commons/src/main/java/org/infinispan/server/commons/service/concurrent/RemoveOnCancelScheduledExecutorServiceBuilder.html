22  package org.infinispan.server.commons.service.concurrent;
23  
24  import java.util.concurrent.ScheduledExecutorService;
25  import java.util.concurrent.ScheduledThreadPoolExecutor;
26  import java.util.concurrent.ThreadFactory;
27  
28  import org.infinispan.server.commons.service.AsynchronousServiceBuilder;
29  import org.infinispan.server.commons.service.Builder;
30  import org.jboss.msc.service.Service;
31  import org.jboss.msc.service.ServiceBuilder;
32  import org.jboss.msc.service.ServiceController;
33  import org.jboss.msc.service.ServiceName;
34  import org.jboss.msc.service.ServiceTarget;
35  import org.jboss.msc.service.StartContext;
36  import org.jboss.msc.service.StopContext;
37  import org.jboss.threads.JBossExecutors;
38  
39  /**
40   * Service that provides a {@link ScheduledThreadPoolExecutor} that removes tasks from the task queue upon cancellation.
41   * @author Paul Ferraro
42   */
43  public class RemoveOnCancelScheduledExecutorServiceBuilder implements Builder<ScheduledExecutorService>, Service<ScheduledExecutorService> {
44  
45      private final ServiceName name;
46      private final ThreadFactory factory;
47      private volatile int size = 1;
48  
49      private volatile ScheduledExecutorService executor;
50  
51      public RemoveOnCancelScheduledExecutorServiceBuilder(ServiceName name, ThreadFactory factory) {
52          this.name = name;
53          this.factory = factory;
54      }
55  
56      @Override
57      public ServiceName getServiceName() {
58          return this.name;
59      }
60  
61      @Override
62      public ServiceBuilder<ScheduledExecutorService> build(ServiceTarget target) {
63          return new AsynchronousServiceBuilder<>(this.name, this).startSynchronously().build(target).setInitialMode(ServiceController.Mode.ON_DEMAND);
64      }
65  
66      public RemoveOnCancelScheduledExecutorServiceBuilder size(int size) {
67          this.size = size;
68          return this;
69      }
70  
71      @Override
72      public ScheduledExecutorService getValue() {
73          return JBossExecutors.protectedScheduledExecutorService(this.executor);
74      }
75  
76      @Override
77      public void start(StartContext context) {
78          ScheduledThreadPoolExecutor executor = new ScheduledThreadPoolExecutor(this.size, this.factory);
79          executor.setRemoveOnCancelPolicy(true);
80          executor.setExecuteExistingDelayedTasksAfterShutdownPolicy(false);
81          this.executor = executor;
82      }
83  
84      @Override
85      public void stop(StopContext context) {
86          this.executor.shutdown();
87          this.executor = null;
88      }
89  }
