37  /**
38   * RestartCacheResourceAdd. Restarts a cache when a child resource is added
39   *
40   * @author Tristan Tarrant
41   * @since 7.2
42   */
43  public class RestartCacheResourceAdd extends RestartParentResourceAddHandler {
44      private final AttributeDefinition[] attributes;
45      private final RestartableServiceHandler parentServiceHandler;
46  
47      RestartCacheResourceAdd(String parentKeyName, RestartableServiceHandler parentServiceHandler, AttributeDefinition[] attributes) {
48          super(parentKeyName);
49          this.attributes = attributes;
50          this.parentServiceHandler = parentServiceHandler;
51      }
52  
53      @Override
54      protected void populateModel(ModelNode operation, ModelNode model) throws OperationFailedException {
55          for (AttributeDefinition attr : attributes) {
56              attr.validateAndSet(operation, model);
57          }
58      }
59  
60      @Override
61      protected void recreateParentService(OperationContext context, PathAddress cacheAddress, ModelNode cacheModel) throws OperationFailedException {
62          PathAddress containerAddress = cacheAddress.subAddress(0, cacheAddress.size() - 1);
63          ModelNode containerModel = context.readResourceFromRoot(containerAddress).getModel();
64          ModelNode operation = Util.createAddOperation(cacheAddress);
65  
66          parentServiceHandler.installRuntimeServices(context, operation, containerModel, cacheModel);
67      }
68  
69      @Override
70      protected ServiceName getParentServiceName(PathAddress parentAddress) {
71          int position = parentAddress.size();
72          PathAddress cacheAddress = parentAddress.subAddress(position - 1);
73          PathAddress containerAddress = parentAddress.subAddress(position - 2, position - 1);
74          return CacheServiceName.CACHE.getServiceName(containerAddress.getLastElement().getValue(), cacheAddress.getLastElement()
75                  .getValue());
76      }
77

79      protected void removeServices(OperationContext context, ServiceName parentService, ModelNode parentModel)
80              throws OperationFailedException {

85                  .isDefined() ? resolvedValue.asString() : null;
86          ContextNames.BindInfo bindInfo;
87          bindInfo = ContextNames.bindInfoFor(InfinispanJndiName.createCacheJndiName(jndiName, containerName, cacheName));
88          context.removeService(bindInfo.getBinderServiceName());*/
89          super.removeServices(context, parentService, parentModel);
90          //context.removeService(CacheServiceName.CONFIGURATION.getServiceName(containerName, cacheName));
91      }
92  
93      @Override
94      protected boolean isResourceServiceRestartAllowed(OperationContext context, ServiceController<?> service) {
95          return true;
96      }
