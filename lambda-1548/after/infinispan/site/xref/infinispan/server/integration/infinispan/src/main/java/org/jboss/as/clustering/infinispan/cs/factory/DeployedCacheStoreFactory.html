3   import org.infinispan.configuration.cache.PersistenceConfigurationBuilder;
4   import org.infinispan.configuration.cache.StoreConfiguration;
5   import org.infinispan.configuration.cache.StoreConfigurationBuilder;
6   import org.infinispan.persistence.factory.CacheStoreFactory;
7   import org.jboss.as.clustering.infinispan.InfinispanLogger;
8   import org.jboss.as.clustering.infinispan.cs.configuration.DeployedStoreConfiguration;
9   
10  import java.util.concurrent.CompletableFuture;
11  import java.util.concurrent.ConcurrentHashMap;
12  import java.util.concurrent.ExecutionException;
13  import java.util.concurrent.TimeUnit;
14  import java.util.concurrent.TimeoutException;
15  
16  /**
17   * Cache Store factory designed for deployed instances.
18   *
19   * @author Sebastian Laskawiec
20   * @since 7.2
21   */
22  public class DeployedCacheStoreFactory implements CacheStoreFactory {
23  
24     private static final int TIMEOUT_SECONDS = 60;
25  
26     private ConcurrentHashMap<String, CompletableFuture<DeployedCacheStoreMetadata>> deployedCacheStores = new ConcurrentHashMap<>();
27  
28     @Override
29     public <T> T createInstance(StoreConfiguration cfg) {
30        if (cfg instanceof DeployedStoreConfiguration) {
31           DeployedStoreConfiguration deployedConfiguration = (DeployedStoreConfiguration) cfg;
32           DeployedCacheStoreMetadata deployedCacheStoreMetadata = null;
33           try {
34              deployedCacheStoreMetadata = getPromise(deployedConfiguration.getCustomStoreClassName()).get(TIMEOUT_SECONDS, TimeUnit.SECONDS);
35              return (T) deployedCacheStoreMetadata.getLoaderWriterRawInstance();
36           } catch (InterruptedException | ExecutionException e) {
37              throw new IllegalStateException("An error occurred while processing the deployment", e);
38           } catch (TimeoutException e) {
39              InfinispanLogger.ROOT_LOGGER.error("Could not get Custom Cache Store metadata (" + deployedConfiguration.getCustomStoreClassName() + ") within given time.", e);
40           }
41        }
42        return null;
43     }
44  
45     @Override
46     public StoreConfiguration processConfiguration(StoreConfiguration storeConfiguration) {
47        if (storeConfiguration instanceof DeployedStoreConfiguration) {
48           DeployedStoreConfiguration deployedConfiguration = (DeployedStoreConfiguration) storeConfiguration;
49           PersistenceConfigurationBuilder replacedBuilder = deployedConfiguration.getPersistenceConfigurationBuilder();
50  
51           try {
52              DeployedCacheStoreMetadata deployedCacheStoreMetadata = getPromise(deployedConfiguration.getCustomStoreClassName()).get(TIMEOUT_SECONDS, TimeUnit.SECONDS);
53  
54              StoreConfigurationBuilder replacedStoreBuilder = replacedBuilder.addStore(deployedCacheStoreMetadata.getStoreBuilderClass());
55              replacedStoreBuilder.fetchPersistentState(deployedConfiguration.fetchPersistentState());
56              replacedStoreBuilder.ignoreModifications(deployedConfiguration.ignoreModifications());
57              replacedStoreBuilder.preload(deployedConfiguration.preload());
58              replacedStoreBuilder.purgeOnStartup(deployedConfiguration.purgeOnStartup());
59              replacedStoreBuilder.shared(deployedConfiguration.shared());
60              replacedStoreBuilder.withProperties(deployedConfiguration.properties());
61  
62              return (StoreConfiguration) replacedStoreBuilder.create();
63           } catch (InterruptedException | ExecutionException e) {
64              throw new IllegalStateException("An error occurred while processing the deployment", e);
65           } catch (TimeoutException e) {
66              InfinispanLogger.ROOT_LOGGER.error("Could not get Custom Cache Store metadata (" + deployedConfiguration.getCustomStoreClassName() + ") within given time.", e);
67           }
68        }
69        return null;
70     }
71  
72     public void addInstance(Object instance) {
73        DeployedCacheStoreMetadata deployedCacheStoreMetadata = DeployedCacheStoreMetadata.fromDeployedStoreInstance(instance);
74        getPromise(deployedCacheStoreMetadata.getDeployedCacheClassName()).complete(deployedCacheStoreMetadata);
75     }
76  
77     public void removeInstance(Object instance) {
78        deployedCacheStores.remove(instance.getClass().getName());
79     }
80  
81     private CompletableFuture<DeployedCacheStoreMetadata> getPromise(String key) {
82        return deployedCacheStores.computeIfAbsent(key, mappedKey -> new CompletableFuture<>());
83     }
84  }
