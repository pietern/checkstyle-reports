4   import org.infinispan.Cache;

10  import java.lang.invoke.MethodHandles;
11  
12  /**
13   * Cache "get" operation handler.
14   *
15   * @author <a href="mailto:tom.fennelly@gmail.com">tom.fennelly@gmail.com</a>
16   */
17  public class PutHandler implements OpHandler {
18  
19     private static final Log logger = LogFactory.getLog(MethodHandles.lookup().lookupClass(), Log.class);
20  
21     @Override
22     public void handleOp(JsonObject opPayload, Cache<Object, Object> cache, ChannelHandlerContext ctx) {
23        String key = (String) opPayload.get(OpHandler.KEY);
24        String value = (String) opPayload.get(OpHandler.VALUE);
25        String mimeType = (String) opPayload.get(OpHandler.MIME);
26  
27        if (mimeType.equals("application/json")) {
28           // Decode the payload to a JSON string...
29  
30           // TODO:  Need some way to populate the JSON object string to an Object graph.
31           // Something plugable... allowing JAXB, Smooks etc
32  
33           throw logger.complexGraphObjectAreNotYetSupported(value);
34        } else {
35           // Put the raw value into the cache...
36           cache.put(key, value);
37        }
38     }
39  }
