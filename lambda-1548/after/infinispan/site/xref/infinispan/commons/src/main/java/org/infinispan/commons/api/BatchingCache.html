1   package org.infinispan.commons.api;

3   /**
4    * The BatchingCache is implemented by all caches which support batching
5    *
6    * @author Tristan Tarrant
7    * @since 6.0
8    */
9   public interface BatchingCache {
10     /**
11      * Starts a batch.  All operations on the current client thread are performed as a part of this batch, with locks
12      * held for the duration of the batch and any remote calls delayed till the end of the batch.
13      * <p/>
14      *
15      * @return true if a batch was successfully started; false if one was available and already running.
16      */
17     boolean startBatch();
18  
19     /**
20      * Completes a batch if one has been started using {@link #startBatch()}.  If no batch has been started, this is a
21      * no-op.
22      * <p/>
23      *
24      * @param successful if true, the batch completes, otherwise the batch is aborted and changes are not committed.
25      */
26     void endBatch(boolean successful);
27  }
