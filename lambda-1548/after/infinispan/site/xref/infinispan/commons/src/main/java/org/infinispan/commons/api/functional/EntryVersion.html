5   import org.infinispan.commons.util.Experimental;
6   
7   /**
8    * Entry version.
9    *
10   * @since 8.0
11   */
12  @Experimental
13  public interface EntryVersion<T> {
14  
15     /**
16      * Get the underlying representation of the entry's version.
17      */
18     T get();
19  
20     /**
21      * Compare the entry version.
22      */
23     CompareResult compareTo(EntryVersion<T> other);
24  
25     enum CompareResult {
26        /**
27         * Denotes a version that was created temporally <i>before</i> another version.
28         */
29        BEFORE,
30        /**
31         * Denotes a version that was created temporally <i>after</i> another version.
32         */
33        AFTER,
34        /**
35         * Denotes that the two versions being compared are equal.
36         */
37        EQUAL,
38        /**
39         * Denotes a version that was created at the same time as another version, but is not equal.  This is only really
40         * useful when using a partition-aware versioning scheme, such as vector or Lamport clocks.
41         */
42        CONFLICTING
43     }
44  
45     final class NumericEntryVersion implements EntryVersion<Long> {
46        private final long version;
47  
48        public NumericEntryVersion(long version) {
49           this.version = version;
50        }
51  
52        @Override
53        public Long get() {
54           return version;
55        }
56  
57        @Override
58        public CompareResult compareTo(EntryVersion<Long> other) {
59           if (version < other.get()) return CompareResult.BEFORE;
60           else if (version > other.get()) return CompareResult.AFTER;
61           else return CompareResult.EQUAL;
62        }
63  
64        @Override
65        public boolean equals(Object o) {
66           if (this == o) return true;
67           if (o == null || getClass() != o.getClass()) return false;
68  
69           NumericEntryVersion that = (NumericEntryVersion) o;
70  
71           return version == that.version;
72        }
73  
74        @Override
75        public int hashCode() {
76           return (int) (version ^ (version >>> 32));
77        }
78  
79        @Override
80        public String toString() {
81           return "NumericEntryVersion=" + version;
82        }
83     }
84  }
