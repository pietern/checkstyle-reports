1   package org.infinispan.commons.util;

3   import java.util.function.Function;
4   
5   /**
6    * A iterator that maps each value to the output of the Function.  Note that the remove is supported if the iterator
7    * originally supported remove.
8    * @author William Burns
9    * @since 8.0
10   */
11  public class CloseableIteratorMapper<E, S> extends IteratorMapper<E, S> implements CloseableIterator<S> {
12     private final CloseableIterator<? extends E> iterator;
13  
14     public CloseableIteratorMapper(CloseableIterator<? extends E> iterator, Function<? super E, ? extends S> function) {
15        super(iterator, function);
16        this.iterator = iterator;
17     }
18  
19     @Override
20     public void close() {
21        iterator.close();
22     }
23  }
