3   /**
4    * Abstract class that provides a method to round up to the nearest value of 8 which is important for most jvm when
5    * doing size calculations.  This is due to the fact that most JVMs align to the nearest 8 bytes for addressing
6    * purposes.
7    * {@inheritDoc}
8    */
9   public abstract class AbstractEntrySizeCalculatorHelper<K, V> implements EntrySizeCalculator<K, V> {
10     // This is how large the object header info is
11     public static final int OBJECT_SIZE = sun.misc.Unsafe.ADDRESS_SIZE;
12     // This is how large an object pointer is - note that each object
13     // has to reference its class
14     public static final int POINTER_SIZE = sun.misc.Unsafe.ARRAY_OBJECT_INDEX_SCALE;
15  
16     public long roundUpToNearest8(long size) {
17        return BoundedEquivalentConcurrentHashMapV8.roundUpToNearest8(size);
18     }
19  }
