25     /**
26      * Run a task defined by the {@link Runnable} instance with a {@link CachingProvider} making sure that the caching
27      * provider is closed after use.
28      *
29      * @param r task to execute with caching provider
30      */
31     public static void withCachingProvider(JCacheRunnable r) {
32        ClassLoader tccl = Thread.currentThread().getContextClassLoader();
33        CachingProvider p = Caching.getCachingProvider(new TestClassLoader(tccl));
34        try {
35           r.run(p);
36        } finally {
37           p.close();
38        }
39     }
40  
41     public static class TestClassLoader extends ClassLoader {
42        public TestClassLoader(ClassLoader parent) {
43           super(parent);
44        }
45     }
46  
47     public static void sleep(long duration) {
48        try {
49           Thread.currentThread().sleep(duration);
50        } catch (InterruptedException e) {
51           Thread.currentThread().interrupt();
52           throw new IllegalStateException(e);
53        }
54     }
55  
56     public static int getEntryCount(Iterator iterator) {
57        int entryCount = 0;
58        while (iterator.hasNext()) {
59           iterator.next();
60           entryCount++;
61        }
62        return entryCount;
63     }
64  
65     public static Cache createCacheWithProperties(CachingProvider provider, Class invoker, String cacheName, Properties properties) {
66        CacheManager manager = provider.getCacheManager(URI.create(invoker.getName()), new TestClassLoader(Thread.currentThread().getContextClassLoader()), properties);
67        return manager.createCache(cacheName, new MutableConfiguration());
68     }
