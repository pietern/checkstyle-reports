28  @Test(testName = "org.infinispan.jcache.JCacheTwoCachesAnnotationsTest", groups = "functional")
29  public class JCacheTwoCachesAnnotationsTest extends AbstractTwoCachesAnnotationsTest {
30  
31     private EmbeddedCacheManager cacheManager1;
32     private EmbeddedCacheManager cacheManager2;
33  
34     @Deployment
35     public static JavaArchive createDeployment() {
36        return ShrinkWrap.create(JavaArchive.class).addPackage(JCacheAnnotatedClass.class.getPackage()).addPackage(JCache.class.getPackage())
37              .addPackage(InjectedCachePutInterceptor.class.getPackage()).addPackage(CacheProducer.class.getPackage()).addPackage(JCacheTestingUtil.class.getPackage())
38              .addAsResource(JCacheTwoCachesAnnotationsTest.class.getResource("/beans.xml"), "beans.xml");
39     }
40  
41     @Inject
42     private JCacheAnnotatedClass jCacheAnnotatedClass;
43  
44     @Override
45     public JCacheAnnotatedClass getJCacheAnnotatedClass() {
46        return jCacheAnnotatedClass;
47     }
48  
49     @Override
50     public Cache getCache1(Method m) {
51        JCacheManager jCacheManager = new JCacheManager(URI.create(m.getName()), cacheManager1, null);
52        return jCacheManager.getCache("annotation");
53     }
54  
55     @Override
56     public Cache getCache2(Method m) {
57        JCacheManager jCacheManager = new JCacheManager(URI.create(m.getName()), cacheManager2, null);
58        return jCacheManager.getCache("annotation");
59     }
60  
61     @BeforeMethod
62     public void initCacheManagers() {
63        cacheManager1 = TestCacheManagerFactory.createClusteredCacheManager(getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC));
64        cacheManager1.defineConfiguration("annotation", getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC).build());
65        cacheManager2 = TestCacheManagerFactory.createClusteredCacheManager(getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC));
66        cacheManager2.defineConfiguration("annotation", getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC).build());
67     }
68  
69     @AfterMethod
70     public void killCacheManagers() {
71        cacheManager1.getCache("annotation", false).clear();
72        cacheManager2.getCache("annotation", false).clear();
73        TestingUtil.killCacheManagers(cacheManager1, cacheManager2);
74     }
