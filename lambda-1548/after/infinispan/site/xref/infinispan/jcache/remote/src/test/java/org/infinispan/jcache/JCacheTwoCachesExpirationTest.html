28     private HotRodServer hotRodServer1;
29     private HotRodServer hotRodServer2;
30     private Cache cache1;
31     private Cache cache2;
32     private ClassLoader testSpecificClassLoader;
33  
34     // Enable once implemented
35     @Test(enabled = false, description = "ISPN-5482")
36     @Override
37     public void testExpiration(Method m) {
38        super.testExpiration(m);
39     }
40  
41     @Override
42     protected void createCacheManagers() throws Throwable {
43        createClusteredCaches(2, "expiry", getExpiryCacheConfig());
44  
45        hotRodServer1 = HotRodClientTestingUtil.startHotRodServer(cacheManagers.get(0));
46        hotRodServer2 = HotRodClientTestingUtil.startHotRodServer(cacheManagers.get(1));
47        testSpecificClassLoader = new JCacheTestingUtil.TestClassLoader(JCacheTwoCachesExpirationTest.class.getClassLoader());
48  
49        Properties properties = new Properties();
50        properties.put("infinispan.client.hotrod.server_list", hotRodServer1.getHost() + ":" + hotRodServer1.getPort());
51        cache1 = createCacheWithProperties(Caching.getCachingProvider(testSpecificClassLoader), JCacheTwoCachesExpirationTest.class, "expiry", properties);
52  
53        properties = new Properties();
54        properties.put("infinispan.client.hotrod.server_list", hotRodServer2.getHost() + ":" + hotRodServer2.getPort());
55        cache2 = createCacheWithProperties(Caching.getCachingProvider(testSpecificClassLoader), JCacheTwoCachesExpirationTest.class, "expiry", properties);
56  
57        waitForClusterToForm("expiry");
58     }
59  
60     protected static org.infinispan.configuration.cache.ConfigurationBuilder getExpiryCacheConfig() {
61        ConfigurationBuilder builder = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC);
62        builder.expiration().lifespan(2000);
63        return hotRodCacheConfiguration(builder);
64     }
65  
66     @AfterClass
67     @Override
68     protected void destroy() {
69        super.destroy();
70        killServers(hotRodServer1, hotRodServer2);
71        Caching.getCachingProvider(testSpecificClassLoader).close();
72     }
73  
74     @Override
75     public Cache getCache1(Method m) {
76        return cache1;
77     }
78
