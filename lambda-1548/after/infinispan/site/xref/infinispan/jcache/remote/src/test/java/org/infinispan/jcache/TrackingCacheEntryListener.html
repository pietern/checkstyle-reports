1   package org.infinispan.jcache;

3   import java.io.Serializable;
4   import java.util.concurrent.atomic.AtomicInteger;
5   
6   import javax.cache.event.CacheEntryCreatedListener;
7   import javax.cache.event.CacheEntryEvent;
8   import javax.cache.event.CacheEntryExpiredListener;
9   import javax.cache.event.CacheEntryListenerException;
10  import javax.cache.event.CacheEntryRemovedListener;
11  import javax.cache.event.CacheEntryUpdatedListener;
12  import javax.cache.event.EventType;
13  
14  import org.testng.Assert;
15  
16  /**
17   * Borrowed from TCK tests
18   *
19   * @author gustavonalle
20   */
21  public class TrackingCacheEntryListener<K, V> implements CacheEntryCreatedListener<K, V>, CacheEntryUpdatedListener<K, V>, CacheEntryExpiredListener<K, V>, CacheEntryRemovedListener<K, V>, Serializable, AutoCloseable {
22     final AtomicInteger created = new AtomicInteger();
23     final AtomicInteger updated = new AtomicInteger();
24     final AtomicInteger removed = new AtomicInteger();
25  
26     public TrackingCacheEntryListener() {
27     }
28  
29     public int getCreated() {
30        return this.created.get();
31     }
32  
33     public int getUpdated() {
34        return this.updated.get();
35     }
36  
37     public int getRemoved() {
38        return this.removed.get();
39     }
40  
41     public void onCreated(Iterable<CacheEntryEvent<? extends K, ? extends V>> events) throws CacheEntryListenerException {
42  
43        for (CacheEntryEvent<? extends K, ? extends V> event : events) {
44           Assert.assertEquals(EventType.CREATED, event.getEventType());
45           this.created.incrementAndGet();
46        }
47  
48     }
49  
50     public void onExpired(Iterable<CacheEntryEvent<? extends K, ? extends V>> events) throws CacheEntryListenerException {
51     }
52  
53     public void onRemoved(Iterable<CacheEntryEvent<? extends K, ? extends V>> events) throws CacheEntryListenerException {
54  
55        for (CacheEntryEvent<? extends K, ? extends V> event : events) {
56           Assert.assertEquals(EventType.REMOVED, event.getEventType());
57           this.removed.incrementAndGet();
58           if (event.isOldValueAvailable()) {
59              event.getOldValue();
60           }
61        }
62  
63     }
64  
65     public void onUpdated(Iterable<CacheEntryEvent<? extends K, ? extends V>> events) throws CacheEntryListenerException {
66  
67        for (CacheEntryEvent<? extends K, ? extends V> event : events) {
68           Assert.assertEquals(EventType.UPDATED, event.getEventType());
69           this.updated.incrementAndGet();
70           if (event.isOldValueAvailable()) {
71              event.getOldValue();
72           }
73        }
74  
75     }
76  
77     public void close() {
78     }
79  }
