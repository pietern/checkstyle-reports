58     public Map<String, AuthorizationPermission[]> getRolePermissionMap() {
59        Map<String, AuthorizationPermission[]> roles = new HashMap<String, AuthorizationPermission[]>();
60        roles.put(ADMIN_ROLE, new AuthorizationPermission[]{AuthorizationPermission.ALL});
61        roles.put(WRITER_ROLE, new AuthorizationPermission[]{AuthorizationPermission.WRITE});
62        roles.put(READER_ROLE, new AuthorizationPermission[]{AuthorizationPermission.READ});
63        roles.put(UNPRIVILEGED_ROLE, new AuthorizationPermission[]{AuthorizationPermission.NONE});
64        return roles;
65     }

126    /**
127     * A {@link ServerSetupTask} instance which creates security domains for this test case.
128     *
129     * @author jcacek@redhat,com
130     * @author vchepeli@redhat,com
131     */
132    static class SecurityDomainsSetupTask extends AbstractSecurityDomainsServerSetupTask {
133 
134       /**
135        * Returns SecurityDomains configuration for this testcase.
136        */
137       @Override
138       protected SecurityDomain[] getSecurityDomains() {
139          final String hostname = Utils.getCannonicalHost(managementClient);
140          final String ldapUrl = "ldap://" + hostname + ":" + "10389";
141          final SecurityDomain sd = new SecurityDomain.Builder()
142                .name(SECURITY_DOMAIN_NAME)
143                .cacheType("default")
144                .loginModules(
145                      new SecurityModule.Builder()
146                            .name("org.jboss.security.auth.spi.LdapLoginModule")
147                            .flag("required")
148 
149                            .putOption(Context.INITIAL_CONTEXT_FACTORY, "com.sun.jndi.ldap.LdapCtxFactory")
150                            .putOption("java.naming.provider.url", ldapUrl)
151                            .putOption(Context.SECURITY_AUTHENTICATION, "simple")
152 
153                            .putOption("principalDNPrefix", "uid=")
154                            .putOption("principalDNSuffix", ",ou=People,dc=infinispan,dc=org")
155 
156                            .putOption("rolesCtxDN", "ou=Roles,dc=infinispan,dc=org")
157                            .putOption("uidAttributeID", "member")
158                            .putOption("matchOnUserDN", "true")
159                            .putOption("roleAttributeID", "cn")
160                            .putOption("roleAttributeIsDN", "false")
161                            .putOption("searchScope", "ONELEVEL_SCOPE")
162                            .build())
163                .build();
164          return new SecurityDomain[]{sd};
165       }
166    }
