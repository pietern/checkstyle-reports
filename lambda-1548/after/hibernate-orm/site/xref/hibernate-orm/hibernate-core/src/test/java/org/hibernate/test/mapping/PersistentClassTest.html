13  	public void testGetMappedClass() {

15  		pc.setClassName(String.class.getName());
16  		Assert.assertEquals(String.class.getName(), pc.getClassName());
17  		Assert.assertEquals(String.class, pc.getMappedClass());
18  		pc.setClassName(Integer.class.getName());
19  		Assert.assertEquals(Integer.class, pc.getMappedClass());
20  	}
21  	
22  	@Test
23  	public void testGetProxyInterface() {
24  		RootClass pc = new RootClass();

26  		Assert.assertEquals(String.class.getName(), pc.getProxyInterfaceName());
27  		Assert.assertEquals(String.class, pc.getProxyInterface());
28  		pc.setProxyInterfaceName(Integer.class.getName());

30  	}
31  	
32  	@Test
33  	public void testGetProperty() {
34  		RootClass pc = new RootClass();
35  		Property p = new Property();
36  		p.setName("name");
37  		pc.addProperty(p);
38  		Assert.assertEquals(p, pc.getProperty("name"));
39  		Assert.assertEquals(p, pc.getProperty("name.test"));
40  		try {
41  			Assert.assertNull(pc.getProperty("test"));
42  			Assert.fail("MappingException expected");

44  			// expected
45  		}
46  	}
