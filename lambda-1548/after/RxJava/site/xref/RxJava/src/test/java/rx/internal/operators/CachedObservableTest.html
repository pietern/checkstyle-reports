36      @Test
37      public void testColdReplayNoBackpressure() {
38          CachedObservable<Integer> source = CachedObservable.from(Observable.range(0, 1000));
39          
40          assertFalse("Source is connected!", source.isConnected());
41          
42          TestSubscriber<Integer> ts = new TestSubscriber<Integer>();
43          
44          source.subscribe(ts);
45  
46          assertTrue("Source is not connected!", source.isConnected());
47          assertFalse("Subscribers retained!", source.hasObservers());
48          
49          ts.assertNoErrors();
50          ts.assertTerminalEvent();
51          List<Integer> onNextEvents = ts.getOnNextEvents();
52          assertEquals(1000, onNextEvents.size());
53  
54          for (int i = 0; i < 1000; i++) {
55              assertEquals((Integer)i, onNextEvents.get(i));
56          }
57      }
58      @Test
59      public void testColdReplayBackpressure() {
60          CachedObservable<Integer> source = CachedObservable.from(Observable.range(0, 1000));
61          
62          assertFalse("Source is connected!", source.isConnected());
63          
64          TestSubscriber<Integer> ts = new TestSubscriber<Integer>();
65          ts.requestMore(10);
66          
67          source.subscribe(ts);
68  
69          assertTrue("Source is not connected!", source.isConnected());
70          assertTrue("Subscribers not retained!", source.hasObservers());
71          
72          ts.assertNoErrors();
73          assertTrue(ts.getOnCompletedEvents().isEmpty());
74          List<Integer> onNextEvents = ts.getOnNextEvents();
75          assertEquals(10, onNextEvents.size());
76  
77          for (int i = 0; i < 10; i++) {
78              assertEquals((Integer)i, onNextEvents.get(i));
79          }
80          
81          ts.unsubscribe();
82          assertFalse("Subscribers retained!", source.hasObservers());
83      }

85      @Test
86      public void testCache() throws InterruptedException {
87          final AtomicInteger counter = new AtomicInteger();
88          Observable<String> o = Observable.create(new Observable.OnSubscribe<String>() {
89  
90              @Override
91              public void call(final Subscriber<? super String> observer) {
92                  new Thread(new Runnable() {
93  
94                      @Override
95                      public void run() {
96                          counter.incrementAndGet();
97                          System.out.println("published observable being executed");
98                          observer.onNext("one");
99                          observer.onCompleted();
100                     }
101                 }).start();
102             }
103         }).cache();
104 
105         // we then expect the following 2 subscriptions to get that same value
106         final CountDownLatch latch = new CountDownLatch(2);
107 
108         // subscribe once
109         o.subscribe(new Action1<String>() {
110 
111             @Override
112             public void call(String v) {
113                 assertEquals("one", v);
114                 System.out.println("v: " + v);
115                 latch.countDown();
116             }
117         });
118 
119         // subscribe again
120         o.subscribe(new Action1<String>() {
121 
122             @Override
123             public void call(String v) {
124                 assertEquals("one", v);
125                 System.out.println("v: " + v);
126                 latch.countDown();
127             }
128         });
129 
130         if (!latch.await(1000, TimeUnit.MILLISECONDS)) {
131             fail("subscriptions did not receive values");
132         }
133         assertEquals(1, counter.get());
134     }

184     @Test
185     public void testAsyncComeAndGo() {
186         Observable<Long> source = Observable.interval(1, 1, TimeUnit.MILLISECONDS)
187                 .take(1000)
188                 .subscribeOn(Schedulers.io());
189         CachedObservable<Long> cached = CachedObservable.from(source);
190         
191         Observable<Long> output = cached.observeOn(Schedulers.computation());
192         
193         List<TestSubscriber<Long>> list = new ArrayList<TestSubscriber<Long>>(100);
194         for (int i = 0; i < 100; i++) {
195             TestSubscriber<Long> ts = new TestSubscriber<Long>();
196             list.add(ts);
197             output.skip(i * 10).take(10).subscribe(ts);
198         }
199 
200         List<Long> expected = new ArrayList<Long>();
201         for (int i = 0; i < 10; i++) {
202             expected.add((long)(i - 10));
203         }
204         int j = 0;
205         for (TestSubscriber<Long> ts : list) {
206             ts.awaitTerminalEvent(3, TimeUnit.SECONDS);
207             ts.assertNoErrors();
208             ts.assertTerminalEvent();
209             
210             for (int i = j * 10; i < j * 10 + 10; i++) {
211                 expected.set(i - j * 10, (long)i);
212             }
213             
214             ts.assertReceivedOnNext(expected);
215             
216             j++;
217         }
218     }

220     @Test
221     public void testNoMissingBackpressureException() {
222         final int m = 4 * 1000 * 1000;
223         Observable<Integer> firehose = Observable.create(new OnSubscribe<Integer>() {
224             @Override
225             public void call(Subscriber<? super Integer> t) {
226                 for (int i = 0; i < m; i++) {
227                     t.onNext(i);
228                 }
229                 t.onCompleted();
230             }
231         });
232         
233         TestSubscriber<Integer> ts = new TestSubscriber<Integer>();
234         firehose.cache().observeOn(Schedulers.computation()).takeLast(100).subscribe(ts);
235         
236         ts.awaitTerminalEvent(3, TimeUnit.SECONDS);
237         ts.assertNoErrors();
238         ts.assertTerminalEvent();
239         
240         assertEquals(100, ts.getOnNextEvents().size());
241     }
