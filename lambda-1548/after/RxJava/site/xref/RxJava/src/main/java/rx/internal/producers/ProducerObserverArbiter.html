16  package rx.internal.producers;
17  
18  import java.util.*;
19  
20  import rx.*;
21  import rx.Observer;
22  import rx.exceptions.*;
23  
24  /**
25   * Producer that serializes any event emission with requesting and producer changes.
26   * <p>
27   * The implementation shortcuts on error and overwrites producers that got delayed, similar
28   * to ProducerArbiter.
29   *
30   * @param <T> the value type
31   */
32  public final class ProducerObserverArbiter<T> implements Producer, Observer<T> {
33      final Subscriber<? super T> child;
34       
35      boolean emitting;
36       
37      List<T> queue;
38      
39      Producer currentProducer;
40      long requested;
41      
42      long missedRequested;
43      Producer missedProducer;
44      Object missedTerminal;
45      
46      volatile boolean hasError;
47      
48      static final Producer NULL_PRODUCER = new Producer() {
49          @Override
50          public void request(long n) {
51              
52          }
53      };
54       
55      public ProducerObserverArbiter(Subscriber<? super T> child) {
56          this.child = child;
57      }
58       
59      @Override
60      public void onNext(T t) {
61          synchronized (this) {
62              if (emitting) {
63                  List<T> q = queue;
64                  if (q == null) {
65                      q = new ArrayList<T>(4);
66                      queue = q;
67                  }
68                  q.add(t);
69                  return;
70              }
71          }
72          boolean skipFinal = false;
73          try {
74              child.onNext(t);
75              
76              long r = requested;
77              if (r != Long.MAX_VALUE) {
78                  requested = r - 1;
79              }
80              
81              emitLoop();
82              skipFinal = true;
83          } finally {
84              if (!skipFinal) {
85                  synchronized (this) {
86                      emitting = false;
87                  }
88              }
89          }
90      }
91       
92      @Override
93      public void onError(Throwable e) {
94          boolean emit;
95          synchronized (this) {
96              if (emitting) {
97                  missedTerminal = e;
98                  emit = false;
99              } else {
100                 emitting = true;
101                 emit = true;
102             }
103         }
104         if (emit) {
105             child.onError(e);
106         } else {
107             hasError = true;
108         }
109     }
110     
111     @Override
112     public void onCompleted() {
113         synchronized (this) {
114             if (emitting) {
115                 missedTerminal = true;
116                 return;
117             }
118             emitting = true;
119         }
120         child.onCompleted();
121     }
122     
123     @Override
124     public void request(long n) {
125         if (n < 0) {
126             throw new IllegalArgumentException("n >= 0 required");
127         }
128         if (n == 0) {
129             return;
130         }
131         synchronized (this) {
132             if (emitting) {
133                 missedRequested += n;
134                 return;
135             }
136             emitting = true;
137         }
138         boolean skipFinal = false;
139         try {
140             long r = requested;
141             long u = r + n;
142             if (u < 0) {
143                 u = Long.MAX_VALUE;
144             }
145             requested = u;
146             
147             Producer p = currentProducer;
148             if (p != null) {
149                 p.request(n);
150             }
151             
152             emitLoop();
153             skipFinal = true;
154         } finally {
155             if (!skipFinal) {
156                 synchronized (this) {
157                     emitting = false;
158                 }
159             }
160         }
161     }
162     
163     public void setProducer(Producer p) {
164         synchronized (this) {
165             if (emitting) {
166                 missedProducer = p != null ? p : NULL_PRODUCER;
167                 return;
168             }
169             emitting = true;
170         }
171         boolean skipFinal = false;
172         try {
173             currentProducer = p;
174             long r = requested;
175             if (p != null && r != 0) {
176                 p.request(r);
177             }
178             emitLoop();
179             skipFinal = true;
180         } finally {
181             if (!skipFinal) {
182                 synchronized (this) {
183                     emitting = false;
184                 }
185             }
186         }
187     }
188     
189     void emitLoop() {
190         final Subscriber<? super T> c = child;
191 
192         outer:
193         for (;;) {
194             long localRequested;
195             Producer localProducer;
196             Object localTerminal;
197             List<T> q;
198             synchronized (this) {
199                 localRequested = missedRequested;
200                 localProducer = missedProducer;
201                 localTerminal = missedTerminal;
202                 q = queue;
203                 if (localRequested == 0L && localProducer == null && q == null
204                         && localTerminal == null) {
205                     emitting = false;
206                     return;
207                 }
208                 missedRequested = 0L;
209                 missedProducer = null;
210                 queue = null;
211                 missedTerminal = null;
212             }
213             boolean empty = q == null || q.isEmpty();
214             if (localTerminal != null) {
215                 if (localTerminal != Boolean.TRUE) {
216                     c.onError((Throwable)localTerminal);
217                     return;
218                 } else
219                 if (empty) {
220                     c.onCompleted();
221                     return;
222                 }
223             }
224             long e = 0;
225             if (q != null) {
226                 for (T v : q) {
227                     if (c.isUnsubscribed()) {
228                         return;
229                     } else
230                     if (hasError) {
231                         continue outer; // if an error has been set, shortcut the loop and act on it
232                     }
233                     try {
234                         c.onNext(v);
235                     } catch (Throwable ex) {
236                         Exceptions.throwOrReport(ex, c, v);
237                         return;
238                     }
239                 }
240                 e += q.size();
241             }
242             long r = requested;
243             // if requested is max, we don't do any accounting
244             if (r != Long.MAX_VALUE) {
245                 // if there were missing requested, add it up
246                 if (localRequested != 0L) {
247                     long u = r + localRequested;
248                     if (u < 0) {
249                         u = Long.MAX_VALUE;
250                     }
251                     r = u;
252                 }
253                 // if there were emissions and we don't run on max since the last check, subtract
254                 if (e != 0L && r != Long.MAX_VALUE) {
255                     long u = r - e;
256                     if (u < 0) {
257                         throw new IllegalStateException("More produced than requested");
258                     }
259                     r = u;
260                 }
261                 requested = r;
262             }
263             if (localProducer != null) {
264                 if (localProducer == NULL_PRODUCER) {
265                     currentProducer = null;
266                 } else {
267                     currentProducer = localProducer;
268                     if (r != 0L) {
269                         localProducer.request(r);
270                     }
271                 }
272             } else {
273                 Producer p = currentProducer;
274                 if (p != null && localRequested != 0L) {
275                     p.request(localRequested);
276                 }
277             }
278         }
279     }
