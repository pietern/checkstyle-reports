17  package rx.internal.util.atomic;
18  
19  /**
20   * This is a direct Java port of the MPSC algorithm as presented <a
21   * href="http://www.1024cores.net/home/lock-free-algorithms/queues/non-intrusive-mpsc-node-based-queue"> on 1024
22   * Cores</a> by D. Vyukov. The original has been adapted to Java and it's quirks with regards to memory model and
23   * layout:
24   * <ol>
25   * <li>Use XCHG functionality provided by AtomicReference (which is better in JDK 8+).
26   * </ol>
27   * The queue is initialized with a stub node which is set to both the producer and consumer node references. From this
28   * point follow the notes on offer/poll.
29   * 
30   * @author nitsanw
31   * 
32   * @param <E>
33   */
34  public final class MpscLinkedAtomicQueue<E> extends BaseLinkedAtomicQueue<E> {
35  
36      public MpscLinkedAtomicQueue() {
37          super();
38          LinkedQueueNode<E> node = new LinkedQueueNode<E>();
39          spConsumerNode(node);
40          xchgProducerNode(node);// this ensures correct construction: StoreLoad
41      }
42      /**
43       * {@inheritDoc} <br>
44       * <p>
45       * IMPLEMENTATION NOTES:<br>
46       * Offer is allowed from multiple threads.<br>
47       * Offer allocates a new node and:
48       * <ol>
49       * <li>Swaps it atomically with current producer node (only one producer 'wins')
50       * <li>Sets the new node as the node following from the swapped producer node
51       * </ol>
52       * This works because each producer is guaranteed to 'plant' a new node and link the old node. No 2 producers can
53       * get the same producer node as part of XCHG guarantee.
54       * 
55       * @see MessagePassingQueue#offer(Object)
56       * @see java.util.Queue#offer(java.lang.Object)
57       */
58      @Override
59      public final boolean offer(final E nextValue) {
60          if (nextValue == null) {
61              throw new NullPointerException("null elements not allowed");
62          }
63          final LinkedQueueNode<E> nextNode = new LinkedQueueNode<E>(nextValue);
64          final LinkedQueueNode<E> prevProducerNode = xchgProducerNode(nextNode);
65          // Should a producer thread get interrupted here the chain WILL be broken until that thread is resumed
66          // and completes the store in prev.next.
67          prevProducerNode.soNext(nextNode); // StoreStore
68          return true;
69      }
70  
71      /**
72       * {@inheritDoc} <br>
73       * <p>
74       * IMPLEMENTATION NOTES:<br>
75       * Poll is allowed from a SINGLE thread.<br>
76       * Poll reads the next node from the consumerNode and:
77       * <ol>
78       * <li>If it is null, the queue is assumed empty (though it might not be).
79       * <li>If it is not null set it as the consumer node and return it's now evacuated value.
80       * </ol>
81       * This means the consumerNode.value is always null, which is also the starting point for the queue. Because null
82       * values are not allowed to be offered this is the only node with it's value set to null at any one time.
83       * 
84       * @see MessagePassingQueue#poll()
85       * @see java.util.Queue#poll()
86       */
87      @Override
88      public final E poll() {
89          LinkedQueueNode<E> currConsumerNode = lpConsumerNode(); // don't load twice, it's alright
90          LinkedQueueNode<E> nextNode = currConsumerNode.lvNext();
91          if (nextNode != null) {
92              // we have to null out the value because we are going to hang on to the node
93              final E nextValue = nextNode.getAndNullValue();
94              spConsumerNode(nextNode);
95              return nextValue;
96          }
97          else if (currConsumerNode != lvProducerNode()) {
98              // spin, we are no longer wait free
99              while((nextNode = currConsumerNode.lvNext()) == null);
100             // got the next node...
101             
102             // we have to null out the value because we are going to hang on to the node
103             final E nextValue = nextNode.getAndNullValue();
104             spConsumerNode(nextNode);
105             return nextValue;
106         }
107         return null;
108     }
109 
110     @Override
111     public final E peek() {
112         LinkedQueueNode<E> currConsumerNode = lpConsumerNode(); // don't load twice, it's alright
113         LinkedQueueNode<E> nextNode = currConsumerNode.lvNext();
114         if (nextNode != null) {
115             return nextNode.lpValue();
116         }
117         else if (currConsumerNode != lvProducerNode()) {
118             // spin, we are no longer wait free
119             while((nextNode = currConsumerNode.lvNext()) == null);
120             // got the next node...
121             return nextNode.lpValue();
122         }
123         return null;
124     }
125 
126 }
